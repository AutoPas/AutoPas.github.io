47template <class... T>
48void printHelpMessage(std::ostream &ostream, const std::string &relPathOfExecutable,
49 const std::tuple<T...> &relevantOptions) {
50 // print header
51 ostream << "Usage: " << relPathOfExecutable << "\n";
52 ostream << "A simple molecular dynamics simulation program showcasing AutoPas.\n\n";
53 ostream << "Non-mandatory options to define a simulation:\n";
54
55 // vector that will hold the lines for options
56 std::vector<std::string> outputLines;
57 // reserve one vector entry for each tuple element
58 outputLines.reserve(std::tuple_size<std::tuple<T...>>{});
59
60 autopas::utils::TupleUtils::for_each(relevantOptions, [&](auto &elem) {
61 std::stringstream ss;
62 ss << " --" << std::setw(MDFlexConfig::valueOffset + 2) << std::left << elem.name;
63 ss << elem.description;
64 ss << '\n';
65 outputLines.push_back(ss.str());
66 });
67
68 // sort lines alphabetically
69 std::sort(std::begin(outputLines), std::end(outputLines));
70 // print all lines
71 for_each(std::begin(outputLines), std::end(outputLines), [&](auto l) { ostream << l; });
72
73 // print footer
74 ostream << '\n';
75 ostream << "md-flexible documentation locally available via: 'make doc_doxygen_md-flexible'\n";
76 ostream << "and online at: https://autopas.github.io/doxygen_documentation_md-flexible/git-master/\n";
77 ostream << "Report bugs to: https://github.com/AutoPas/AutoPas/issues\n";
78 ostream << "Full AutoPas documentation at: https://autopas.github.io/doxygen_documentation/git-master/index.html\n";
79}

87template <class... T>
88void createZSHCompletionFile(const std::tuple<T...> &cliOptions) {
89 constexpr auto filename = "_md-flexible";
90 std::ofstream fileStream;
91 fileStream.open(filename);
92
93 // some header
94 fileStream << "#compdef md-flexible\n"
95 "\n"
96 "typeset -A opt_args\n"
97 "local context state line\n"
98 "\n"
99 "# doc: \n"
100 "# THIS FILE IS AUTO GENERATED BY md-flexible --zsh-completions\n"
101 "# http://zsh.sourceforge.net/Doc/Release/Completion-System.html#Completion-System\n"
102 "# format : \"--arg[description.]: : (list of options)\"\n"
103 "\n"
104 "_arguments -s -S \\\n";
105
106 // go through all cli options and create the corresponding completion line
107 // also, parse possible values from descriptions
108 autopas::utils::TupleUtils::for_each(cliOptions, [&](auto &option) {
109 // the keyword to look for value suggestions
110 constexpr auto keywordForValues{"Possible Values:"};
111 constexpr std::pair<char, char> bracketsForValues = std::make_pair('(', ')');
112 constexpr auto keywordForPaths{"Path to"};
113
114 // basic name and description for every option.
115 fileStream << " \"--" << option.name << "[" << option.description << "]";
116
117 // If the option requires arguments and the description suggests some, parse them.
118 if (option.requiresArgument) {
119 std::string choices;
120 if (const auto posKeywordForValues = option.description.find(keywordForValues);
121 posKeywordForValues != std::string::npos) {
122 // start after (+1) first occurrence of '[' after "Possible Values:"
123 const auto startPossibleValues{option.description.find(bracketsForValues.first, posKeywordForValues) + 1};
124 // end is the first ']' after the start. The length of the substr is the index difference to start.
125 const auto strLengthPossibleValues{option.description.find(bracketsForValues.second, startPossibleValues) -
126 startPossibleValues};
127 choices = "(" + option.description.substr(startPossibleValues, strLengthPossibleValues) + ")";
128 } else if (const auto posKeywordForPaths = option.description.find(keywordForPaths);
129 posKeywordForPaths != std::string::npos) {
130 choices = "_files";
131 // look if a filetype is given
132 std::regex rgxFileType{"\\.[a-zA-Z]+"};
133 std::smatch stringMatch;
134 std::regex_search(option.description, stringMatch, rgxFileType);
135 if (not stringMatch.empty()) {
136 choices += " -g '*" + stringMatch[0].str() + "'";
137 }
138 }
139 // append completion for options
140 fileStream << ": :" << choices;
141 }
142
143 // closing " and end the line
144 fileStream << "\"\\\n";
145 });
146
147 fileStream.close();
148
149 std::cout << "Created file: " << filename << std::endl;
150}
