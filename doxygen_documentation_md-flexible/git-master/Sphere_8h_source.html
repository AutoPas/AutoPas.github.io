53  // generate regular grid for 1/8th of the sphere
54  for (int z = 0; z <= radius; ++z) {
55  for (int y = 0; y <= radius; ++y) {
56  for (int x = 0; x <= radius; ++x) {

58  std::array<double, 3> relativePos = {(double)x, (double)y, (double)z};
59  // mirror to rest of sphere
60  for (int i = -1; i <= 1; i += 2) {
61  for (int k = -1; k <= 1; k += 2) {
62  for (int l = -1; l <= 1; l += 2) {
63  std::array<double, 3> mirrorMultipliers = {(double)i, (double)k, (double)l};
64  // position mirrored, scaled and absolute
65  std::array<double, 3> posVector = autopas::utils::ArrayMath::add(
66  center, autopas::utils::ArrayMath::mulScalar(
67  autopas::utils::ArrayMath::mul(relativePos, mirrorMultipliers), particleSpacing));
68

71  autopas::utils::ArrayMath::sub(posVector, center));
