56 void iteratePositions(const std::function<void(std::array<double, 3>)> &f) const {
57 // generate regular grid for 1/8th of the sphere
58 for (int z = 0; z <= _radius; ++z) {
59 for (int y = 0; y <= _radius; ++y) {
60 for (int x = 0; x <= _radius; ++x) {
61 // position relative to the center
62 std::array<double, 3> relativePos = {(double)x, (double)y, (double)z};
63 // mirror to rest of sphere
64 for (int i = -1; i <= 1; i += 2) {
65 for (int k = -1; k <= 1; k += 2) {
66 for (int l = -1; l <= 1; l += 2) {
67 std::array<double, 3> mirrorMultipliers = {(double)i, (double)k, (double)l};
68 // position mirrored, scaled and absolute
69 std::array<double, 3> posVector = autopas::utils::ArrayMath::add(
70 _center, autopas::utils::ArrayMath::mulScalar(
71 autopas::utils::ArrayMath::mul(relativePos, mirrorMultipliers), _particleSpacing));
72
73 double distFromCentersSquare =
74 autopas::utils::ArrayMath::dot(autopas::utils::ArrayMath::sub(posVector, _center),
75 autopas::utils::ArrayMath::sub(posVector, _center));
76 const auto r = (_radius + 1) * _particleSpacing;
77 const auto rSquare = r * r;
78 // since the loops create a cubic grid only apply f for positions inside the sphere
79 if (distFromCentersSquare <= rSquare) {
80 f(posVector);
81 }
82 // avoid duplicates
83 if (z == 0) break;
84 }
85 if (y == 0) break;
86 }
87 if (x == 0) break;
88 }
89 }
90 }
