50 void iteratePositions(const std::function<void(std::array<double, 3>)> &f) const {
51 using namespace autopas::utils::ArrayMath::literals;
52
53 // generate regular grid for 1/8th of the sphere
54 for (int z = 0; z <= _radius; ++z) {
55 for (int y = 0; y <= _radius; ++y) {
56 for (int x = 0; x <= _radius; ++x) {
57 // position relative to the center
58 const std::array<double, 3> relativePos = {(double)x, (double)y, (double)z};
59 // mirror to rest of sphere
60 for (int i = -1; i <= 1; i += 2) {
61 for (int k = -1; k <= 1; k += 2) {
62 for (int l = -1; l <= 1; l += 2) {
63 const std::array<double, 3> mirrorMultipliers = {(double)i, (double)k, (double)l};
64 // position mirrored, scaled and absolute
65 const std::array<double, 3> posVector =
66 _center + ((relativePos * mirrorMultipliers) * _particleSpacing);
67
68 double distFromCentersSquare = autopas::utils::ArrayMath::dot(posVector - _center, posVector - _center);
69 const auto r = (_radius + 1) * _particleSpacing;
70 const auto rSquare = r * r;
71 // since the loops create a cubic grid only apply f for positions inside the sphere
72 if (distFromCentersSquare <= rSquare) {
73 f(posVector);
74 }
75 // avoid duplicates
76 if (z == 0) break;
77 }
78 if (y == 0) break;
79 }
80 if (x == 0) break;
81 }
82 }
83 }
