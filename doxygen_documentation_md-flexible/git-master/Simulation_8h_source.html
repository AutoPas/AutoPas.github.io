578 template <class Particle>
579 double Simulation<Particle>::calculateHomogeneity(autopas::AutoPas<Particle> &autopas) {
580  int numberOfParticles = autopas.getNumberOfParticles();
581  // approximately the resolution we want to get.
582  int numberOfCells = ceil(numberOfParticles / 10.);
583 
584  std::array<double, 3> startCorner = autopas.getBoxMin();
585  std::array<double, 3> endCorner = autopas.getBoxMax();
586  std::array<double, 3> domainSizePerDimension = {};
587  for (int i = 0; i < 3; ++i) {
588  domainSizePerDimension[i] = endCorner[i] - startCorner[i];
589  }
590 
591  // get cellLength which is equal in each direction, derived from the domainsize and the requested number of cells
592  double volume = domainSizePerDimension[0] * domainSizePerDimension[1] * domainSizePerDimension[2];
593  double cellVolume = volume / numberOfCells;
594  double cellLength = cbrt(cellVolume);
595 
596  // calculate the size of the boundary cells, which might be smaller then the other cells
597  std::array<size_t, 3> cellsPerDimension = {};
598  // size of the last cell layer per dimension. This cell might get truncated to fit in the domain.
599  std::array<double, 3> outerCellSizePerDimension = {};
600  for (int i = 0; i < 3; ++i) {
601  outerCellSizePerDimension[i] =
602  domainSizePerDimension[i] - (floor(domainSizePerDimension[i] / cellLength) * cellLength);
603  cellsPerDimension[i] = ceil(domainSizePerDimension[i] / cellLength);
604  }
605  // Actual number of cells we end up with
606  numberOfCells = cellsPerDimension[0] * cellsPerDimension[1] * cellsPerDimension[2];
607 
608  std::vector<size_t> particlesPerCell(numberOfCells, 0);
609  std::vector<double> allVolumes(numberOfCells, 0);
610 
611  // add particles accordingly to their cell to get the amount of particles in each cell
612  for (auto particleItr = autopas.begin(autopas::IteratorBehavior::ownedOnly); particleItr.isValid(); ++particleItr) {
613  std::array<double, 3> particleLocation = particleItr->getR();
614  std::array<size_t, 3> index = {};
615  for (int i = 0; i < particleLocation.size(); i++) {
616  index[i] = particleLocation[i] / cellLength;
617  }
618  const size_t cellIndex = autopas::utils::ThreeDimensionalMapping::threeToOneD(index, cellsPerDimension);
619  particlesPerCell[cellIndex] += 1;
620  // calculate the size of the current cell
621  allVolumes[cellIndex] = 1;
622  for (int i = 0; i < cellsPerDimension.size(); ++i) {
623  // the last cell layer has a special size
624  if (index[i] == cellsPerDimension[i] - 1) {
625  allVolumes[cellIndex] *= outerCellSizePerDimension[i];
626  } else {
627  allVolumes[cellIndex] *= cellLength;
628  }
629  }
630  }
631 
632  // calculate density for each cell
633  std::vector<double> densityPerCell(numberOfCells, 0.0);
634  for (int i = 0; i < particlesPerCell.size(); i++) {
635  densityPerCell[i] = (particlesPerCell[i] == 0)
636  ? 0

638  }
639

641  double mean = numberOfParticles / volume;

643

648  }
