127template <typename T>
128const std::vector<T> parseComplexTypeValueSequence(const YAML::Node node, const std::string &key,
129 std::vector<std::string> &complexTypeErrors) {
130 std::vector<T> value;
131 try {
132 YAML::Node n = node[key];
133 const auto vecLength = n.size();
134 value.reserve(vecLength);
135 for (int i = 0; i < vecLength; i++) {
136 value.emplace_back(n[i].as<T>());
137 }
138
139 } catch (const std::exception &e) {
140 std::stringstream ss;
141 ss << "Error parsing " << key << ". Make sure that key \"" << key << "\" exists and has the expected value: "
142 << "YAML-sequence of " << typeToStr<T>() << " values.";
143 complexTypeErrors.push_back(ss.str());
144 }
145 return value;
146}
