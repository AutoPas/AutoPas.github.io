51template <class AutoPasTemplate, class ParticlePropertiesLibraryTemplate>
52double calcTemperature(const AutoPasTemplate &autopas, ParticlePropertiesLibraryTemplate &particlePropertiesLibrary) {
53 // kinetic energy times 2
54 double kineticEnergyMul2 = 0;
55#ifdef AUTOPAS_OPENMP
56#pragma omp parallel reduction(+ : kineticEnergyMul2) default(none) shared(autopas, particlePropertiesLibrary)
57#endif
58 for (auto iter = autopas.begin(); iter.isValid(); ++iter) {
59 auto vel = iter->getV();
60 kineticEnergyMul2 +=
61 particlePropertiesLibrary.getMass(iter->getTypeId()) * autopas::utils::ArrayMath::dot(vel, vel);
62 }
63 // AutoPas works always on 3 dimensions
64 constexpr unsigned int dimensions{3};
65 return kineticEnergyMul2 / (autopas.getNumberOfParticles() * dimensions);
66}

77template <class AutoPasTemplate, class ParticlePropertiesLibraryTemplate>
78auto calcTemperatureComponent(const AutoPasTemplate &autopas,
79 ParticlePropertiesLibraryTemplate &particlePropertiesLibrary) {
80 // map of: particle typeID -> kinetic energy times 2 for this type
81 std::map<size_t, double> kineticEnergyMul2Map;
82 // map of: particle typeID -> number of particles of this type
83 std::map<size_t, size_t> numParticleMap;
84
85 for (const auto &typeID : particlePropertiesLibrary.getTypes()) {
86 kineticEnergyMul2Map[typeID] = 0.;
87 numParticleMap[typeID] = 0ul;
88 }
89
90#ifdef AUTOPAS_OPENMP
91#pragma omp parallel
92#endif
93 {
94 // create aggregators for each thread
95 std::map<size_t, double> kineticEnergyMul2MapThread;
96 std::map<size_t, size_t> numParticleMapThread;
97 for (const auto &typeID : particlePropertiesLibrary.getTypes()) {
98 kineticEnergyMul2MapThread[typeID] = 0.;
99 numParticleMapThread[typeID] = 0ul;
100 }
101 // parallel iterators
102 for (auto iter = autopas.begin(); iter.isValid(); ++iter) {
103 auto vel = iter->getV();
104 kineticEnergyMul2MapThread.at(iter->getTypeId()) +=
105 particlePropertiesLibrary.getMass(iter->getTypeId()) * autopas::utils::ArrayMath::dot(vel, vel);
106 numParticleMapThread.at(iter->getTypeId())++;
107 }
108 // manual reduction
109#ifdef AUTOPAS_OPENMP
110#pragma omp critical
111#endif
112 {
113 for (const auto &typeID : particlePropertiesLibrary.getTypes()) {
114 kineticEnergyMul2Map[typeID] += kineticEnergyMul2MapThread[typeID];
115 numParticleMap[typeID] += numParticleMapThread[typeID];
116 }
117 }
118 }
119 // AutoPas works always on 3 dimensions
120 constexpr unsigned int dimensions{3};
121
122 auto kineticEnergyAndParticleMaps = std::make_tuple(kineticEnergyMul2Map.begin(), numParticleMap.begin());
123
124 for (auto [kinEIter, numParIter] = kineticEnergyAndParticleMaps; kinEIter != kineticEnergyMul2Map.end();
125 ++kinEIter, ++numParIter) {
126 kinEIter->second /= numParIter->second * dimensions;
127 }
128 return kineticEnergyMul2Map;
129}

144template <class AutoPasTemplate, class ParticlePropertiesLibraryTemplate>
145void addBrownianMotion(AutoPasTemplate &autopas, ParticlePropertiesLibraryTemplate &particlePropertiesLibrary,
146 const double targetTemperature) {
147 // factors for the brownian motion per particle type.
148 std::map<size_t, double> factors;
149 // brownian motion with disturbance depending on current temperature and mass
150 for (auto typeID : particlePropertiesLibrary.getTypes()) {
151 factors.emplace(typeID, std::sqrt(targetTemperature / particlePropertiesLibrary.getMass(typeID)));
152 }
153#ifdef AUTOPAS_OPENMP
154#pragma omp parallel default(none) shared(autopas, factors)
155#endif
156 {
157 // we use a constant seed for repeatability.
158 // we need one random engine and distribution per thread
159 std::default_random_engine randomEngine(42 + autopas::autopas_get_thread_num());
160 std::normal_distribution<double> normalDistribution{0, 1};
161 for (auto iter = autopas.begin(); iter.isValid(); ++iter) {
162 addMaxwellBoltzmannDistributedVelocity(*iter, factors[iter->getTypeId()], randomEngine, normalDistribution);
163 }
164 }
165}

176template <class AutoPasTemplate, class ParticlePropertiesLibraryTemplate>
177void apply(AutoPasTemplate &autopas, ParticlePropertiesLibraryTemplate &particlePropertiesLibrary,
178 const double targetTemperature, const double deltaTemperature) {
179 auto currentTemperatureMap = calcTemperatureComponent(autopas, particlePropertiesLibrary);
180 // make sure we work with a positive delta
181 const double deltaTemperaturePositive = std::abs(deltaTemperature);
182 decltype(currentTemperatureMap) scalingMap;
183
184 for (const auto &[particleTypeID, currentTemperature] : currentTemperatureMap) {
185 double nextTargetTemperature;
186 // check if we are already in the vicinity of our target or if we still need full steps
187 if (std::abs(currentTemperature - targetTemperature) < std::abs(deltaTemperature)) {
188 nextTargetTemperature = targetTemperature;
189 } else {
190 // make sure we scale in the right direction
191 nextTargetTemperature = currentTemperature < targetTemperature ? currentTemperature + deltaTemperaturePositive
192 : currentTemperature - deltaTemperaturePositive;
193 }
194 scalingMap[particleTypeID] = std::sqrt(nextTargetTemperature / currentTemperature);
195 }
196#ifdef AUTOPAS_OPENMP
197#pragma omp parallel default(none) shared(autopas, scalingMap)
198#endif
199 for (auto iter = autopas.begin(); iter.isValid(); ++iter) {
200 iter->setV(autopas::utils::ArrayMath::mulScalar(iter->getV(), scalingMap[iter->getTypeId()]));
201 }
202}
