52template <class AutoPasTemplate, class ParticlePropertiesLibraryTemplate>
53double calcTemperature(const AutoPasTemplate &autopas, ParticlePropertiesLibraryTemplate &particlePropertiesLibrary) {
54 // kinetic energy times 2
55 double kineticEnergyMul2 = 0;
56#ifdef AUTOPAS_OPENMP
57#pragma omp parallel reduction(+ : kineticEnergyMul2) default(none) shared(autopas, particlePropertiesLibrary)
58#endif
59 for (auto iter = autopas.begin(); iter.isValid(); ++iter) {
60 auto vel = iter->getV();
61 kineticEnergyMul2 +=
62 particlePropertiesLibrary.getMass(iter->getTypeId()) * autopas::utils::ArrayMath::dot(vel, vel);
63 }
64 // AutoPas works always on 3 dimensions
65 constexpr unsigned int dimensions{3};
66 return kineticEnergyMul2 / (autopas.getNumberOfParticles() * dimensions);
67}

78template <class AutoPasTemplate, class ParticlePropertiesLibraryTemplate>
79auto calcTemperatureComponent(const AutoPasTemplate &autopas,
80 ParticlePropertiesLibraryTemplate &particlePropertiesLibrary) {
81 // map of: particle typeID -> kinetic energy times 2 for this type
82 std::map<size_t, double> kineticEnergyMul2Map;
83 // map of: particle typeID -> number of particles of this type
84 std::map<size_t, size_t> numParticleMap;
85
86 for (const auto &typeID : particlePropertiesLibrary.getTypes()) {
87 kineticEnergyMul2Map[typeID] = 0.;
88 numParticleMap[typeID] = 0ul;
89 }
90
91#ifdef AUTOPAS_OPENMP
92#pragma omp parallel
93#endif
94 {
95 // create aggregators for each thread
96 std::map<size_t, double> kineticEnergyMul2MapThread;
97 std::map<size_t, size_t> numParticleMapThread;
98 for (const auto &typeID : particlePropertiesLibrary.getTypes()) {
99 kineticEnergyMul2MapThread[typeID] = 0.;
100 numParticleMapThread[typeID] = 0ul;
101 }
102 // parallel iterators
103 for (auto iter = autopas.begin(); iter.isValid(); ++iter) {
104 const auto &vel = iter->getV();
105 kineticEnergyMul2MapThread.at(iter->getTypeId()) +=
106 particlePropertiesLibrary.getMass(iter->getTypeId()) * autopas::utils::ArrayMath::dot(vel, vel);
107 numParticleMapThread.at(iter->getTypeId())++;
108 }
109 // manual reduction
110#ifdef AUTOPAS_OPENMP
111#pragma omp critical
112#endif
113 {
114 for (const auto &typeID : particlePropertiesLibrary.getTypes()) {
115 kineticEnergyMul2Map[typeID] += kineticEnergyMul2MapThread[typeID];
116 numParticleMap[typeID] += numParticleMapThread[typeID];
117 }
118 }
119 }
120 // AutoPas works always on 3 dimensions
121 constexpr unsigned int dimensions{3};
122
123 for (const auto &typeID : particlePropertiesLibrary.getTypes()) {
124 // workaround for MPICH: send and receive buffer must not be the same.
125 autopas::AutoPas_MPI_Allreduce(AUTOPAS_MPI_IN_PLACE, &kineticEnergyMul2Map[typeID], 1, AUTOPAS_MPI_DOUBLE,
126 AUTOPAS_MPI_SUM, AUTOPAS_MPI_COMM_WORLD);
127
128 autopas::AutoPas_MPI_Allreduce(AUTOPAS_MPI_IN_PLACE, &numParticleMap[typeID], 1, AUTOPAS_MPI_UNSIGNED_LONG,
129 AUTOPAS_MPI_SUM, AUTOPAS_MPI_COMM_WORLD);
130 }
131
132 auto kineticEnergyAndParticleMaps = std::make_tuple(kineticEnergyMul2Map.begin(), numParticleMap.begin());
133
134 for (auto [kinEIter, numParIter] = kineticEnergyAndParticleMaps; kinEIter != kineticEnergyMul2Map.end();
135 ++kinEIter, ++numParIter) {
136 kinEIter->second /= static_cast<double>(numParIter->second) * dimensions;
137 }
138 return kineticEnergyMul2Map;
139}

154template <class AutoPasTemplate, class ParticlePropertiesLibraryTemplate>
155void addBrownianMotion(AutoPasTemplate &autopas, ParticlePropertiesLibraryTemplate &particlePropertiesLibrary,
156 const double targetTemperature) {
157 // factors for the brownian motion per particle type.
158 std::map<size_t, double> factors;
159 // brownian motion with disturbance depending on current temperature and mass
160 for (auto typeID : particlePropertiesLibrary.getTypes()) {
161 factors.emplace(typeID, std::sqrt(targetTemperature / particlePropertiesLibrary.getMass(typeID)));
162 }
163#ifdef AUTOPAS_OPENMP
164#pragma omp parallel default(none) shared(autopas, factors)
165#endif
166 {
167 // we use a constant seed for repeatability.
168 // we need one random engine and distribution per thread
169 std::default_random_engine randomEngine(42 + autopas::autopas_get_thread_num());
170 std::normal_distribution<double> normalDistribution{0, 1};
171 for (auto iter = autopas.begin(); iter.isValid(); ++iter) {
172 addMaxwellBoltzmannDistributedVelocity(*iter, factors[iter->getTypeId()], randomEngine, normalDistribution);
173 }
174 }
175}

186template <class AutoPasTemplate, class ParticlePropertiesLibraryTemplate>
187void apply(AutoPasTemplate &autopas, ParticlePropertiesLibraryTemplate &particlePropertiesLibrary,
188 const double targetTemperature, const double deltaTemperature) {
189 const auto currentTemperatureMap = calcTemperatureComponent(autopas, particlePropertiesLibrary);
190
191 // make sure we work with a positive delta
192 const double deltaTemperaturePositive = std::abs(deltaTemperature);
193 std::remove_const_t<decltype(currentTemperatureMap)> scalingMap;
194
195 for (const auto &[particleTypeID, currentTemperature] : currentTemperatureMap) {
196 double nextTargetTemperature{};
197 // check if we are already in the vicinity of our target or if we still need full steps
198 if (std::abs(currentTemperature - targetTemperature) < std::abs(deltaTemperature)) {
199 nextTargetTemperature = targetTemperature;
200 } else {
201 // make sure we scale in the right direction
202 nextTargetTemperature = currentTemperature < targetTemperature ? currentTemperature + deltaTemperaturePositive
203 : currentTemperature - deltaTemperaturePositive;
204 }
205 scalingMap[particleTypeID] = std::sqrt(nextTargetTemperature / currentTemperature);
206 }
207#ifdef AUTOPAS_OPENMP
208#pragma omp parallel default(none) shared(autopas, scalingMap)
209#endif
210 for (auto iter = autopas.begin(); iter.isValid(); ++iter) {
211 iter->setV(autopas::utils::ArrayMath::mulScalar(iter->getV(), scalingMap[iter->getTypeId()]));
212 }
213}
