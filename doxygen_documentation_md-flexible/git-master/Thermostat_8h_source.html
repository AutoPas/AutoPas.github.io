51 double calcTemperature(const AutoPasTemplate &autopas, ParticlePropertiesLibraryTemplate &particlePropertiesLibrary) {
52  // kinetic energy times 2
53  double kineticEnergyMul2 = 0;
54 #ifdef AUTOPAS_OPENMP
55 #pragma omp parallel reduction(+ : kineticEnergyMul2) default(none) shared(autopas, particlePropertiesLibrary)

63  constexpr unsigned int dimensions{3};
64  return kineticEnergyMul2 / (autopas.getNumberOfParticles() * dimensions);

174 void apply(AutoPasTemplate &autopas, ParticlePropertiesLibraryTemplate &particlePropertiesLibrary,
175  const double targetTemperature, const double deltaTemperature) {
176  auto currentTemperatureMap = calcTemperatureComponent(autopas, particlePropertiesLibrary);
177  // make sure we work with a positive delta
178  const double deltaTemperaturePositive = std::abs(deltaTemperature);
179  decltype(currentTemperatureMap) scalingMap;
180 
181  for (const auto &[particleTypeID, currentTemperature] : currentTemperatureMap) {
182  double nextTargetTemperature;
183  // check if we are already in the vicinity of our target or if we still need full steps
184  if (std::abs(currentTemperature - targetTemperature) < std::abs(deltaTemperature)) {
185  nextTargetTemperature = targetTemperature;
186  } else {
187  // make sure we scale in the right direction
188  nextTargetTemperature = currentTemperature < targetTemperature ? currentTemperature + deltaTemperaturePositive
189  : currentTemperature - deltaTemperaturePositive;
190  }
191  scalingMap[particleTypeID] = std::sqrt(nextTargetTemperature / currentTemperature);
192  }
193 #ifdef AUTOPAS_OPENMP
194 #pragma omp parallel default(none) shared(autopas, scalingMap)
195 #endif
196  for (auto iter = autopas.begin(); iter.isValid(); ++iter) {
197  iter->setV(autopas::utils::ArrayMath::mulScalar(iter->getV(), scalingMap[iter->getTypeId()]));
