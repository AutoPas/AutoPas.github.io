99  autopas::AutoPas<ParticleType, ParticleCell> &autoPas) {

102  for (short x : {-1, 0, 1}) {
103  for (short y : {-1, 0, 1}) {
104  for (short z : {-1, 0, 1}) {
105  if (x == 0 and y == 0 and z == 0) continue;
106  std::array<short, 3> direction{x, y, z};
107  std::array<double, 3> min{}, max{}, shiftVec{};
108  for (size_t dim = 0; dim < 3; ++dim) {
109  // The search domain has to be enlarged as the position of the particles is not certain.
110  bool needsShift = false;
111  if (direction[dim] == -1) {
112  min[dim] = autoPas.getBoxMin()[dim] - autoPas.getVerletSkin();
113  max[dim] = autoPas.getBoxMin()[dim] + autoPas.getCutoff() + autoPas.getVerletSkin();
114  needsShift = true;
115  } else if (direction[dim] == 1) {
116  min[dim] = autoPas.getBoxMax()[dim] - autoPas.getCutoff() - autoPas.getVerletSkin();
117  max[dim] = autoPas.getBoxMax()[dim] + autoPas.getVerletSkin();
118  needsShift = true;
119  } else { // 0
120  min[dim] = autoPas.getBoxMin()[dim] - autoPas.getVerletSkin();
121  max[dim] = autoPas.getBoxMax()[dim] + autoPas.getVerletSkin();
122  }
123  if (needsShift) {
124  shiftVec[dim] = -(autoPas.getBoxMax()[dim] - autoPas.getBoxMin()[dim]) * direction[dim];
125  } else {
126  shiftVec[dim] = 0;
127  }
128  }
129  // here it is important to only iterate over the owned particles!
130  for (auto iter = autoPas.getRegionIterator(min, max, autopas::IteratorBehavior::ownedOnly); iter.isValid();
