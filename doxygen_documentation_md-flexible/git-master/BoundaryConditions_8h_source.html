30  const auto &boxMin = autoPas.getBoxMin();
31  const auto &boxMax = autoPas.getBoxMax();
32  const auto boxLength = std::array<double, 3>{
33  boxMax[0] - boxMin[0],
34  boxMax[1] - boxMin[1],
35  boxMax[2] - boxMin[2],
36  };
37 
38  for (auto &p : particles) {
39  auto pos = p.getR();
40  for (auto dim = 0ul; dim < pos.size(); dim++) {
41  if (pos[dim] < boxMin[dim]) {
42  // has to be smaller than boxMax
43  pos[dim] = std::min(std::nextafter(boxMax[dim], boxMin[dim]), pos[dim] + boxLength[dim]);
44  } else if (pos[dim] >= boxMax[dim]) {
45  // should at least be boxMin
46  pos[dim] = std::max(boxMin[dim], pos[dim] - boxLength[dim]);
47  }
48  }
49  p.setR(pos);
50  }

62 std::vector<Particle> identifyNewHaloParticles(autopas::AutoPas<Particle> &autoPas) {
63  std::vector<Particle> haloParticles;
64 
65  const auto &boxMin = autoPas.getBoxMin();
66  const auto &boxMax = autoPas.getBoxMax();
67  const auto boxLength = std::array<double, 3>{
68  boxMax[0] - boxMin[0],
69  boxMax[1] - boxMin[1],
70  boxMax[2] - boxMin[2],
71  };
72 
73  // look in every direction
74  for (auto x : {-1, 0, 1}) {
75  for (auto y : {-1, 0, 1}) {
76  for (auto z : {-1, 0, 1}) {
77  if (x == 0 and y == 0 and z == 0) continue;
78  std::array<int, 3> direction{x, y, z};
79  // Define a region where to search for particles that need to be copied to the halo.
80  std::array<double, 3> min{}, max{}, shiftVec{};
81  for (auto dim = 0ul; dim < direction.size(); ++dim) {
82  // Flag for the periodic wrap. Will occur in at least one direction.
83  bool needsShift = false;
84  // The search domain has to be enlarged by the skin as the position of the particles is not certain.
85  if (direction[dim] == -1) {
86  // just inside of boxMin
87  min[dim] = boxMin[dim] - autoPas.getVerletSkin();
88  max[dim] = boxMin[dim] + autoPas.getCutoff() + autoPas.getVerletSkin();
89  needsShift = true;
90  } else if (direction[dim] == 1) {
91  // just inside of boxMax
92  min[dim] = boxMax[dim] - autoPas.getCutoff() - autoPas.getVerletSkin();
93  max[dim] = boxMax[dim] + autoPas.getVerletSkin();
94  needsShift = true;
95  } else { // direction[dim] == 0
96  min[dim] = boxMin[dim] - autoPas.getVerletSkin();
97  max[dim] = boxMax[dim] + autoPas.getVerletSkin();
98  }
99  if (needsShift) {
100  // particles left of min (direction == -1) need a shift by +boxLength. For right of max the opposite.
101  shiftVec[dim] = -boxLength[dim] * direction[dim];
102  } else {
103  shiftVec[dim] = 0;
104  }
105  }
106  // Find non-halo particles in the designated region, create periodic copies of them and insert those as halos.
107  for (auto iter = autoPas.getRegionIterator(min, max, autopas::IteratorBehavior::owned); iter.isValid();
108  ++iter) {
109  auto particleCopy = *iter;
110  particleCopy.addR(shiftVec);
111  haloParticles.push_back(particleCopy);
112  }
113  }
114  }
115  }
116  return haloParticles;
117 }

158 void applyPeriodic(autopas::AutoPas<Particle> &autoPas, bool forceUpdate) {
159  // 1. update Container; return value is a vector of the particles leaving the domain box
160  // and a flag whether an update occured.
161  auto [leavingParticles, updated] = autoPas.updateContainer(forceUpdate);
162  if (updated) {
163  // 2. apply periodic wrap by shifting positions of leaving particles to positions of periodic images.
164  wrapPositionsAroundBoundaries(autoPas, leavingParticles);
165  // 2b. re-insert shifted particles
166  addEnteringParticles(autoPas, leavingParticles);
167  }
168  // 3. identify inner particles for which a periodic copy in the opposing halo region is needed.
169  auto haloParticles = identifyNewHaloParticles(autoPas);
170  addHaloParticles(autoPas, haloParticles);
