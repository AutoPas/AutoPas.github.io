45 [[nodiscard]] size_t getParticlesTotal() const override {
46 // Number of particles in the first row.
47 const size_t xNumRow = std::ceil(_boxLength[0] / _particleSpacing);
48 // True if the total number of x-positions is odd.
49 const bool xOdd = static_cast<int>(std::ceil(_boxLength[0] / _xOffset)) % 2 == 1;
50
51 // Distance between layers and rows
52 const auto spacingLayer = _particleSpacing * sqrt(2. / 3.);
53 const auto spacingRow = _particleSpacing * sqrt(3. / 4.);
54
55 // Number of rows in an even layer.
56 const size_t yNumEven = std::ceil(_boxLength[1] / spacingRow);
57 // Number of rows in an odd layer.
58 const size_t yNumOdd = std::ceil((_boxLength[1] - _yOffset) / spacingRow);
59
60 // Number of particles in an even layer.
61 const size_t evenLayer = xNumRow * yNumEven - std::floor(xOdd * yNumEven * 0.5);
62 // Number of particles in an odd layer.
63 const size_t oddLayer = xNumRow * yNumOdd - std::ceil(xOdd * yNumOdd * 0.5);
64
65 // Total number of layers.
66 const double numLayers = std::ceil(_boxLength[2] / spacingLayer);
67 // Add up all even and odd layers.
68 return evenLayer * std::ceil(numLayers / 2.) + oddLayer * std::floor(numLayers / 2.);
69 }
