50 [[nodiscard]] size_t getParticlesTotal() const override {
51 // Number of particles in the first row.
52 const size_t xNumRow = std::ceil(_boxLength[0] / _particleSpacing);
53 // True if the total number of x-positions is odd.
54 const bool xOdd = static_cast<int>(std::ceil(_boxLength[0] / _xOffset)) % 2 == 1;
55
56 // Number of rows in an even layer.
57 const size_t yNumEven = std::ceil(_boxLength[1] / _spacingRow);
58 // Number of rows in an odd layer.
59 const size_t yNumOdd = std::ceil((_boxLength[1] - _yOffset) / _spacingRow);
60
61 // Number of particles in an even layer.
62 const size_t evenLayer = xNumRow * yNumEven - std::floor(xOdd * yNumEven * 0.5);
63 // Number of particles in an odd layer.
64 const size_t oddLayer = xNumRow * yNumOdd - std::ceil(xOdd * yNumOdd * 0.5);
65
66 // Total number of layers.
67 const double numLayers = std::ceil(_boxLength[2] / _spacingLayer);
68 // Add up all even and odd layers.
69 return evenLayer * std::ceil(numLayers / 2.) + oddLayer * std::floor(numLayers / 2.);
