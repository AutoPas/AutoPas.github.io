180  // clear the aos 2 soa map

184  size_t index = 0;
185 
186  // Here we have to iterate over all particles, as particles might be later on marked for deletion, and we cannot
187  // differentiate them from particles already marked for deletion.
188  for (auto iter = this->begin(IteratorBehavior::ownedOrHaloOrDummy); iter.isValid(); ++iter, ++index) {
189  // set the map
190  _particlePtr2indexMap[&(*iter)] = index;
191  }
192  size_t accumulatedListSize = 0;
193  for (auto &[particlePtr, neighborPtrVector] : _aosNeighborLists) {
194  accumulatedListSize += neighborPtrVector.size();
195  size_t i_id = _particlePtr2indexMap[particlePtr];
196  // each soa neighbor list should be of the same size as for aos
197  _soaNeighborLists[i_id].resize(neighborPtrVector.size());
198  size_t j = 0;
199  for (auto &neighborPtr : neighborPtrVector) {
200  _soaNeighborLists[i_id][j] = _particlePtr2indexMap[neighborPtr];
201  j++;
202  }
203  }
