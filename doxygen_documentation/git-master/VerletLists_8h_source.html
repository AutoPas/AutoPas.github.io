163 size_t generateAoSNeighborLists() {
164 size_t numParticles = 0;
165 _aosNeighborLists.clear();
166 // DON'T simply parallelize this loop!!! this needs modifications if you want to parallelize it!
167 // We have to iterate also over dummy particles here to ensure a correct size of the arrays.
168 for (auto iter = this->begin(IteratorBehavior::ownedOrHaloOrDummy); iter.isValid(); ++iter, ++numParticles) {
169 // create the verlet list entries for all particles
170 _aosNeighborLists[&(*iter)];

179 void generateSoAListFromAoSVerletLists() {
180 // resize the list to the size of the aos neighborlist
181 _soaNeighborLists.resize(_aosNeighborLists.size());
182 // clear the aos 2 soa map
183 _particlePtr2indexMap.clear();
184
185 _particlePtr2indexMap.reserve(_aosNeighborLists.size());
186 size_t index = 0;
187
188 // Here we have to iterate over all particles, as particles might be later on marked for deletion, and we cannot
189 // differentiate them from particles already marked for deletion.
190 for (auto iter = this->begin(IteratorBehavior::ownedOrHaloOrDummy); iter.isValid(); ++iter, ++index) {
191 // set the map
192 _particlePtr2indexMap[&(*iter)] = index;
193 }
194 size_t accumulatedListSize = 0;
195 for (const auto &[particlePtr, neighborPtrVector] : _aosNeighborLists) {
196 accumulatedListSize += neighborPtrVector.size();
197 size_t i_id = _particlePtr2indexMap[particlePtr];
198 // each soa neighbor list should be of the same size as for aos
199 _soaNeighborLists[i_id].resize(neighborPtrVector.size());
200 size_t j = 0;
201 for (auto &neighborPtr : neighborPtrVector) {
202 _soaNeighborLists[i_id][j] = _particlePtr2indexMap[neighborPtr];
203 j++;
204 }
205 }
206
207 AutoPasLog(debug,
208 "VerletLists::generateSoAListFromAoSVerletLists: average verlet list "
209 "size is {}",
210 static_cast<double>(accumulatedListSize) / _aosNeighborLists.size());
211 _soaListIsValid = true;
212 }
