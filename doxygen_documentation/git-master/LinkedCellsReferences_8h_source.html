194  std::vector<ParticleType> updateContainer() override {
195  this->deleteHaloParticles();
196 
197  std::vector<ParticleType> invalidParticles;
198 #ifdef AUTOPAS_OPENMP
199 #pragma omp parallel
200 #endif // AUTOPAS_OPENMP
201  {
202  // private for each thread!
203  std::vector<ParticleType> myInvalidParticles, myInvalidNotOwnedParticles;
204 #ifdef AUTOPAS_OPENMP
205 #pragma omp for
206 #endif // AUTOPAS_OPENMP
207  for (size_t cellId = 0; cellId < this->getCells().size(); ++cellId) {
208  // Delete dummy particles of each cell.
209  this->getCells()[cellId].deleteDummyParticles();
210 
211  // if empty
212  if (not this->getCells()[cellId].isNotEmpty()) continue;
213 
214  auto [cellLowerCorner, cellUpperCorner] = this->getCellBlock().getCellBoundingBox(cellId);
215 
216  for (auto &&pIter = this->getCells()[cellId].begin(); pIter.isValid(); ++pIter) {
217  // if not in cell
218  if (utils::notInBox(pIter->getR(), cellLowerCorner, cellUpperCorner)) {
219  myInvalidParticles.push_back(*pIter);
220  internal::deleteParticle(pIter);
221  }
222  }
223  }
224  // implicit barrier here
225  // the barrier is needed because iterators are not threadsafe w.r.t. addParticle()
226 
227  // this loop is executed for every thread and thus parallel. Don't use #pragma omp for here!
228  for (auto &&p : myInvalidParticles) {
229  // if not in halo
230  if (utils::inBox(p.getR(), this->getBoxMin(), this->getBoxMax())) {
231  this->template addParticle<false>(p);
232  } else {
233  myInvalidNotOwnedParticles.push_back(p);
234  }
235  }
236 #ifdef AUTOPAS_OPENMP
237 #pragma omp critical
238 #endif
239  {
240  // merge private vectors to global one.
241  invalidParticles.insert(invalidParticles.end(), myInvalidNotOwnedParticles.begin(),
242  myInvalidNotOwnedParticles.end());
243  }
244  }
245  _particleList.deleteDummyParticles();
246  updateDirtyParticleReferences();
247  return invalidParticles;
248  }

270  ParticleIteratorWrapper<ParticleType, true> getRegionIterator(
271  const std::array<double, 3> &lowerCorner, const std::array<double, 3> &higherCorner,
272  IteratorBehavior behavior = IteratorBehavior::haloAndOwned) override {
273  // We increase the search region by skin, as particles can move over cell borders.
274  auto startIndex3D =
275  this->_cellBlock.get3DIndexOfPosition(utils::ArrayMath::subScalar(lowerCorner, this->getSkin()));
276  auto stopIndex3D =
277  this->_cellBlock.get3DIndexOfPosition(utils::ArrayMath::addScalar(higherCorner, this->getSkin()));
278 
279  size_t numCellsOfInterest = (stopIndex3D[0] - startIndex3D[0] + 1) * (stopIndex3D[1] - startIndex3D[1] + 1) *
280  (stopIndex3D[2] - startIndex3D[2] + 1);
281  std::vector<size_t> cellsOfInterest(numCellsOfInterest);
282 
283  int i = 0;
284  for (size_t z = startIndex3D[2]; z <= stopIndex3D[2]; ++z) {
285  for (size_t y = startIndex3D[1]; y <= stopIndex3D[1]; ++y) {
286  for (size_t x = startIndex3D[0]; x <= stopIndex3D[0]; ++x) {
287  cellsOfInterest[i++] =
288  utils::ThreeDimensionalMapping::threeToOneD({x, y, z}, this->_cellBlock.getCellsPerDimensionWithHalo());
289  }
290  }
291  }
292 
293  return ParticleIteratorWrapper<ParticleType, true>(
294  new internal::RegionParticleIterator<ParticleType, ReferenceCell, true>(
295  &this->_cells, lowerCorner, higherCorner, cellsOfInterest, &_cellBlock, behavior));
296  }
