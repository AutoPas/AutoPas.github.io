229 template <bool regionIter>
230 std::tuple<const Particle *, size_t, size_t> getParticleImpl(size_t cellIndex, size_t particleIndex,
231 IteratorBehavior iteratorBehavior,
232 const std::array<double, 3> &boxMin,
233 const std::array<double, 3> &boxMax) const {
234 // first and last relevant cell index
235 const auto [startCellIndex, endCellIndex] = [&]() -> std::tuple<size_t, size_t> {
236 if constexpr (regionIter) {
237 return {_cellBlock.get1DIndexOfPosition(boxMin), _cellBlock.get1DIndexOfPosition(boxMax)};
238 } else {
239 if (not(iteratorBehavior & IteratorBehavior::halo)) {
240 // only potentially owned region
241 return {_cellBlock.getFirstOwnedCellIndex(), _cellBlock.getLastOwnedCellIndex()};
242 } else {
243 // whole range of cells
244 return {0, this->_cells.size() - 1};
245 }
246 }
247 }();
248
249 // if we are at the start of an iteration ...
250 if (cellIndex == 0 and particleIndex == 0) {
251 cellIndex =
252 startCellIndex + ((iteratorBehavior & IteratorBehavior::forceSequential) ? 0 : autopas_get_thread_num());
253 }
254 // abort if the start index is already out of bounds
255 if (cellIndex >= this->_cells.size()) {
256 return {nullptr, 0, 0};
257 }
258 // check the data behind the indices
259 if (particleIndex >= this->_cells[cellIndex].size() or
260 not containerIteratorUtils::particleFulfillsIteratorRequirements<regionIter>(
261 this->_cells[cellIndex][particleIndex], iteratorBehavior, boxMin, boxMax)) {
262 // either advance them to something interesting or invalidate them.
263 std::tie(cellIndex, particleIndex) =
264 advanceIteratorIndices<regionIter>(cellIndex, particleIndex, iteratorBehavior, boxMin, boxMax, endCellIndex);
265 }
266
267 // shortcut if the given index doesn't exist
268 if (cellIndex > endCellIndex) {
269 return {nullptr, 0, 0};
270 }
271 const Particle *retPtr = &this->_cells[cellIndex][particleIndex];
272
273 return {retPtr, cellIndex, particleIndex};
274 }

293 std::vector<ParticleType> invalidParticles;
294
295 // for exception handling in parallel region
296 bool exceptionCaught{false};
297 std::string exceptionMsg{""};
298
299 AUTOPAS_OPENMP(parallel) {
300 // private for each thread!
301 std::vector<ParticleType> myInvalidParticles, myInvalidNotOwnedParticles;
302 AUTOPAS_OPENMP(for)
303 for (size_t cellId = 0; cellId < this->getCells().size(); ++cellId) {
304 // Delete dummy particles of each cell.
305 this->getCells()[cellId].deleteDummyParticles();
306
307 // if empty
308 if (this->getCells()[cellId].isEmpty()) continue;
309
310 auto [cellLowerCorner, cellUpperCorner] = this->getCellBlock().getCellBoundingBox(cellId);
311
312 auto &particleVec = this->getCells()[cellId]._particles;
313 for (auto pIter = particleVec.begin(); pIter != particleVec.end();) {
314 if ((*pIter)->isOwned() and utils::notInBox((*pIter)->getR(), cellLowerCorner, cellUpperCorner)) {
315 myInvalidParticles.push_back(**pIter);
316
317 // multi layer swap-delete
318 // we copy the particle behind the last pointer in the cell over the particle we want to delete
319 **pIter = *particleVec.back();
320 // since we will not actually delete the particle we just copied mark it as dummy.
321 particleVec.back()->setOwnershipState(OwnershipState::dummy);
322 // then we pop the last pointer from the cell.
323 particleVec.pop_back();
324
325 } else {
326 ++pIter;
327 }
328 }
329 }
330 // implicit barrier here
331 // the barrier is needed because iterators are not thread safe w.r.t. addParticle()
332
333 // this loop is executed for every thread and thus parallel. Don't use #pragma omp for here!
334 // addParticle might throw. Set exceptionCaught to true, so we can handle that after the OpenMP region
335 try {
336 for (auto &&p : myInvalidParticles) {
337 // if not in halo
338 if (utils::inBox(p.getR(), this->getBoxMin(), this->getBoxMax())) {
339 this->template addParticle<false>(p);
340 } else {
341 myInvalidNotOwnedParticles.push_back(p);
342 }
343 }
344 } catch (const std::exception &e) {
345 exceptionCaught = true;
346 AUTOPAS_OPENMP(critical)
347 exceptionMsg.append(e.what());
348 }
349 AUTOPAS_OPENMP(critical) {
350 // merge private vectors to global one.
351 invalidParticles.insert(invalidParticles.end(), myInvalidNotOwnedParticles.begin(),
352 myInvalidNotOwnedParticles.end());
353 }
354 }
355
356 if (exceptionCaught) {
357 throw autopas::utils::ExceptionHandler::AutoPasException(exceptionMsg);
358 }
359
360 // we have to remove halo particles after the above for-loop since removing halo particles changes the underlying
361 // datastructure (_particleList) which invalidates the references in the cells.
362 // Note: removing halo particles before the loop and do a call to updateDirtyParticleReferences() right after won't
363 // work since there are owned particles in _particleList which fall into the halo area (they are not yet filtered
364 // out by the above loop) and can therefore not added to halo cells during particle insert in
365 // updateDirtyParticleReferences()
366 this->deleteHaloParticles();
367
368 _particleList.deleteDummyParticles();
369 updateDirtyParticleReferences();
370 return invalidParticles;
371 }

444 template <typename Lambda>
445 void forEachInRegion(Lambda forEachLambda, const std::array<double, 3> &lowerCorner,
446 const std::array<double, 3> &higherCorner, IteratorBehavior behavior) {
447 using namespace autopas::utils::ArrayMath::literals;
448
449 // We increase the search region by skin, as particles can move over cell borders.
450 const auto startIndex3D = this->_cellBlock.get3DIndexOfPosition(lowerCorner - this->getVerletSkin());
451 const auto stopIndex3D = this->_cellBlock.get3DIndexOfPosition(higherCorner + this->getVerletSkin());
452
453 size_t numCellsOfInterest = (stopIndex3D[0] - startIndex3D[0] + 1) * (stopIndex3D[1] - startIndex3D[1] + 1) *
454 (stopIndex3D[2] - startIndex3D[2] + 1);
455 std::vector<size_t> cellsOfInterest(numCellsOfInterest);
456
457 int i = 0;
458 for (size_t z = startIndex3D[2]; z <= stopIndex3D[2]; ++z) {
459 for (size_t y = startIndex3D[1]; y <= stopIndex3D[1]; ++y) {
460 for (size_t x = startIndex3D[0]; x <= stopIndex3D[0]; ++x) {
461 cellsOfInterest[i++] =
462 utils::ThreeDimensionalMapping::threeToOneD({x, y, z}, this->_cellBlock.getCellsPerDimensionWithHalo());
463 }
464 }
465 }
466
467 for (size_t index : cellsOfInterest) {
468 if (!_cellBlock.ignoreCellForIteration(index, behavior)) {
469 getCells()[index].forEach(forEachLambda, lowerCorner, higherCorner, behavior);
470 }
471 }
472 }

478 void reduceInRegion(Lambda reduceLambda, A &result, const std::array<double, 3> &lowerCorner,
479 const std::array<double, 3> &higherCorner, IteratorBehavior behavior) {
480 using namespace autopas::utils::ArrayMath::literals;
481
482 // We increase the search region by skin, as particles can move over cell borders.
483 const auto startIndex3D = this->_cellBlock.get3DIndexOfPosition(lowerCorner - this->getVerletSkin());
484 const auto stopIndex3D = this->_cellBlock.get3DIndexOfPosition(higherCorner + this->getVerletSkin());
485
486 size_t numCellsOfInterest = (stopIndex3D[0] - startIndex3D[0] + 1) * (stopIndex3D[1] - startIndex3D[1] + 1) *
487 (stopIndex3D[2] - startIndex3D[2] + 1);
488 std::vector<size_t> cellsOfInterest(numCellsOfInterest);
489
490 int i = 0;
491 for (size_t z = startIndex3D[2]; z <= stopIndex3D[2]; ++z) {
492 for (size_t y = startIndex3D[1]; y <= stopIndex3D[1]; ++y) {

494 cellsOfInterest[i++] =

496 }
497 }
498 }

538 template <bool regionIter>
539 std::tuple<size_t, size_t> advanceIteratorIndices(size_t cellIndex, size_t particleIndex,
540 IteratorBehavior iteratorBehavior,
541 const std::array<double, 3> &boxMin,
542 const std::array<double, 3> &boxMax, size_t endCellIndex) const {
543 // Finding the indices for the next particle
544 const size_t stride = (iteratorBehavior & IteratorBehavior::forceSequential) ? 1 : autopas_get_num_threads();
545
546 // helper function to determine if the cell can even contain particles of interest to the iterator
547 auto cellIsRelevant = [&]() -> bool {
548 bool isRelevant =
549 // behavior matches possible particle ownership
550 (iteratorBehavior & IteratorBehavior::owned and _cellBlock.cellCanContainOwnedParticles(cellIndex)) or
551 (iteratorBehavior & IteratorBehavior::halo and _cellBlock.cellCanContainHaloParticles(cellIndex));
552 if constexpr (regionIter) {
553 // short circuit if already false
554 if (isRelevant) {
555 // is the cell in the region?
556 const auto [cellLowCorner, cellHighCorner] = _cellBlock.getCellBoundingBox(cellIndex);
557 // particles can move over cell borders. Calculate the volume this cell's particles can be.
558 const auto cellLowCornerSkin = utils::ArrayMath::subScalar(cellLowCorner, this->getVerletSkin() * 0.5);
559 const auto cellHighCornerSkin = utils::ArrayMath::addScalar(cellHighCorner, this->getVerletSkin() * 0.5);
560 isRelevant = utils::boxesOverlap(cellLowCornerSkin, cellHighCornerSkin, boxMin, boxMax);
561 }
562 }
563 return isRelevant;
564 };
565
566 do {
567 // advance to the next particle
568 ++particleIndex;
569 // If this breaches the end of a cell, find the next non-empty cell and reset particleIndex.
570
571 // If cell has wrong type, or there are no more particles in this cell jump to the next
572 while (not cellIsRelevant() or particleIndex >= this->_cells[cellIndex].size()) {
573 // TODO: can this jump be done more efficient if behavior is only halo or owned?
574 // TODO: can this jump be done more efficient for region iters if the cell is outside the region?
575 cellIndex += stride;
576 particleIndex = 0;
577
578 // If we notice that there is nothing else to look at set invalid values, so we get a nullptr next time and
579 // break.
580 if (cellIndex > endCellIndex) {
581 return {std::numeric_limits<decltype(cellIndex)>::max(), std::numeric_limits<decltype(particleIndex)>::max()};
582 }
583 }
584 } while (not containerIteratorUtils::particleFulfillsIteratorRequirements<regionIter>(
585 this->_cells[cellIndex][particleIndex], iteratorBehavior, boxMin, boxMax));
586
587 // the indices returned at this point should always be valid
588 return {cellIndex, particleIndex};
589 }
