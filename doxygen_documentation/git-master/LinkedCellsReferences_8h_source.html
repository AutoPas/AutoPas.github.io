158 void updateDirtyParticleReferences() {
159 if (_particleList.isDirty()) {
160 if (_particleList.needsRebuild()) {
161 for (auto &cell : this->_cells) {
162 cell.clear();
163 }
164 }
165
166 for (auto it = _particleList.beginDirty(); it < _particleList.endDirty(); it++) {
167 if (it->isDummy()) {
168 continue;
169 }
170 ReferenceCell &cell = _cellBlock.getContainingCell(it->getR());
171 auto address = &(*it);
172 cell.addParticleReference(address);
173 }
174 _particleList.markAsClean();
175 }

179 // Check if traversal is allowed for this container and give it the data it needs.

181 auto *cellPairTraversal = dynamic_cast<CellPairTraversal<ReferenceCell> *>(traversal);

198 std::vector<ParticleType> updateContainer(bool keepNeighborListsValid) override {
199 if (keepNeighborListsValid) {
200 return autopas::LeavingParticleCollector::collectParticlesAndMarkNonOwnedAsDummy(*this);
201 }
202 this->deleteHaloParticles();
203
204 std::vector<ParticleType> invalidParticles;
205#ifdef AUTOPAS_OPENMP
206#pragma omp parallel
207#endif // AUTOPAS_OPENMP
208 {
209 // private for each thread!
210 std::vector<ParticleType> myInvalidParticles, myInvalidNotOwnedParticles;
211#ifdef AUTOPAS_OPENMP
212#pragma omp for
213#endif // AUTOPAS_OPENMP
214 for (size_t cellId = 0; cellId < this->getCells().size(); ++cellId) {
215 // Delete dummy particles of each cell.
216 this->getCells()[cellId].deleteDummyParticles();
217
218 // if empty
219 if (this->getCells()[cellId].isEmpty()) continue;
220
221 auto [cellLowerCorner, cellUpperCorner] = this->getCellBlock().getCellBoundingBox(cellId);
222
223 for (auto &&pIter = this->getCells()[cellId].begin(); pIter.isValid(); ++pIter) {
224 // if not in cell
225 if (utils::notInBox(pIter->getR(), cellLowerCorner, cellUpperCorner)) {
226 myInvalidParticles.push_back(*pIter);
227 internal::deleteParticle(pIter);
228 }
229 }
230 }
231 // implicit barrier here
232 // the barrier is needed because iterators are not threadsafe w.r.t. addParticle()
233
234 // this loop is executed for every thread and thus parallel. Don't use #pragma omp for here!
235 for (auto &&p : myInvalidParticles) {
236 // if not in halo
237 if (utils::inBox(p.getR(), this->getBoxMin(), this->getBoxMax())) {
238 this->template addParticle<false>(p);
239 } else {
240 myInvalidNotOwnedParticles.push_back(p);
241 }
242 }
243#ifdef AUTOPAS_OPENMP
244#pragma omp critical
245#endif
246 {
247 // merge private vectors to global one.
248 invalidParticles.insert(invalidParticles.end(), myInvalidNotOwnedParticles.begin(),
249 myInvalidNotOwnedParticles.end());
250 }
251 }
252 _particleList.deleteDummyParticles();
253 updateDirtyParticleReferences();
254 return invalidParticles;
255 }

313 ParticleIteratorWrapper<ParticleType, true> getRegionIterator(const std::array<double, 3> &lowerCorner,
314 const std::array<double, 3> &higherCorner,
315 IteratorBehavior behavior) override {
316 // We increase the search region by skin, as particles can move over cell borders.
317 auto startIndex3D =
318 this->_cellBlock.get3DIndexOfPosition(utils::ArrayMath::subScalar(lowerCorner, this->getVerletSkin()));
319 auto stopIndex3D =
320 this->_cellBlock.get3DIndexOfPosition(utils::ArrayMath::addScalar(higherCorner, this->getVerletSkin()));
321
322 size_t numCellsOfInterest = (stopIndex3D[0] - startIndex3D[0] + 1) * (stopIndex3D[1] - startIndex3D[1] + 1) *
323 (stopIndex3D[2] - startIndex3D[2] + 1);
324 std::vector<size_t> cellsOfInterest(numCellsOfInterest);
325
326 int i = 0;
327 for (size_t z = startIndex3D[2]; z <= stopIndex3D[2]; ++z) {
328 for (size_t y = startIndex3D[1]; y <= stopIndex3D[1]; ++y) {
329 for (size_t x = startIndex3D[0]; x <= stopIndex3D[0]; ++x) {
330 cellsOfInterest[i++] =
331 utils::ThreeDimensionalMapping::threeToOneD({x, y, z}, this->_cellBlock.getCellsPerDimensionWithHalo());
332 }
333 }
334 }
335
336 return ParticleIteratorWrapper<ParticleType, true>(
337 new internal::RegionParticleIterator<ParticleType, ReferenceCell, true>(
338 &this->_cells, lowerCorner, higherCorner, cellsOfInterest, &_cellBlock, behavior, nullptr));
339 }
