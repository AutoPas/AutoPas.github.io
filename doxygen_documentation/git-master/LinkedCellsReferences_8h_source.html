155 void updateDirtyParticleReferences() {
156 if (_particleList.isDirty()) {
157 if (_particleList.needsRebuild()) {
158 for (auto &cell : this->_cells) {
159 cell.clear();
160 }
161 }
162
163 for (auto it = _particleList.beginDirty(); it < _particleList.endDirty(); it++) {
164 if (it->isDummy()) {
165 continue;
166 }
167 ReferenceCell &cell = _cellBlock.getContainingCell(it->getR());
168 auto address = &(*it);
169 cell.addParticleReference(address);
170 }
171 _particleList.markAsClean();
172 }

175 void iteratePairwise(TraversalInterface *traversal) override {
176 // Check if traversal is allowed for this container and give it the data it needs.
177 auto *traversalInterface = dynamic_cast<LCTraversalInterface<ReferenceCell> *>(traversal);
178 auto *cellPairTraversal = dynamic_cast<CellPairTraversal<ReferenceCell> *>(traversal);
179 if (auto *balancedTraversal = dynamic_cast<BalancedTraversal *>(traversal)) {
180 balancedTraversal->setLoadEstimator(getLoadEstimatorFunction());
181 }
182 if (traversalInterface && cellPairTraversal) {
183 cellPairTraversal->setCellsToTraverse(this->_cells);
184 } else {
185 autopas::utils::ExceptionHandler::exception(
186 "Trying to use a traversal of wrong type in LinkedCellsReferences::iteratePairwise. TraversalID: {}",
187 traversal->getTraversalType());
188 }
189
190 traversal->initTraversal();
191 traversal->traverseParticlePairs();
192 traversal->endTraversal();

195 std::vector<ParticleType> updateContainer() override {
196 this->deleteHaloParticles();
197
198 std::vector<ParticleType> invalidParticles;
199#ifdef AUTOPAS_OPENMP
200#pragma omp parallel
201#endif // AUTOPAS_OPENMP
202 {
203 // private for each thread!
204 std::vector<ParticleType> myInvalidParticles, myInvalidNotOwnedParticles;
205#ifdef AUTOPAS_OPENMP
206#pragma omp for
207#endif // AUTOPAS_OPENMP
208 for (size_t cellId = 0; cellId < this->getCells().size(); ++cellId) {
209 // Delete dummy particles of each cell.
210 this->getCells()[cellId].deleteDummyParticles();
211
212 // if empty
213 if (not this->getCells()[cellId].isNotEmpty()) continue;
214
215 auto [cellLowerCorner, cellUpperCorner] = this->getCellBlock().getCellBoundingBox(cellId);
216
217 for (auto &&pIter = this->getCells()[cellId].begin(); pIter.isValid(); ++pIter) {
218 // if not in cell
219 if (utils::notInBox(pIter->getR(), cellLowerCorner, cellUpperCorner)) {
220 myInvalidParticles.push_back(*pIter);
221 internal::deleteParticle(pIter);
222 }
223 }
224 }
225 // implicit barrier here
226 // the barrier is needed because iterators are not threadsafe w.r.t. addParticle()
227
228 // this loop is executed for every thread and thus parallel. Don't use #pragma omp for here!
229 for (auto &&p : myInvalidParticles) {
230 // if not in halo
231 if (utils::inBox(p.getR(), this->getBoxMin(), this->getBoxMax())) {
232 this->template addParticle<false>(p);
233 } else {
234 myInvalidNotOwnedParticles.push_back(p);
235 }
236 }
237#ifdef AUTOPAS_OPENMP
238#pragma omp critical
239#endif
240 {
241 // merge private vectors to global one.
242 invalidParticles.insert(invalidParticles.end(), myInvalidNotOwnedParticles.begin(),
243 myInvalidNotOwnedParticles.end());
244 }
245 }
246 _particleList.deleteDummyParticles();
247 updateDirtyParticleReferences();
248 return invalidParticles;
249 }

273 ParticleIteratorWrapper<ParticleType, true> getRegionIterator(const std::array<double, 3> &lowerCorner,
274 const std::array<double, 3> &higherCorner,
275 IteratorBehavior behavior) override {
276 // We increase the search region by skin, as particles can move over cell borders.
277 auto startIndex3D =
278 this->_cellBlock.get3DIndexOfPosition(utils::ArrayMath::subScalar(lowerCorner, this->getSkin()));
279 auto stopIndex3D =
280 this->_cellBlock.get3DIndexOfPosition(utils::ArrayMath::addScalar(higherCorner, this->getSkin()));
281
282 size_t numCellsOfInterest = (stopIndex3D[0] - startIndex3D[0] + 1) * (stopIndex3D[1] - startIndex3D[1] + 1) *
283 (stopIndex3D[2] - startIndex3D[2] + 1);
284 std::vector<size_t> cellsOfInterest(numCellsOfInterest);
285
286 int i = 0;
287 for (size_t z = startIndex3D[2]; z <= stopIndex3D[2]; ++z) {
288 for (size_t y = startIndex3D[1]; y <= stopIndex3D[1]; ++y) {
289 for (size_t x = startIndex3D[0]; x <= stopIndex3D[0]; ++x) {
290 cellsOfInterest[i++] =
291 utils::ThreeDimensionalMapping::threeToOneD({x, y, z}, this->_cellBlock.getCellsPerDimensionWithHalo());
292 }
293 }
294 }
295
296 return ParticleIteratorWrapper<ParticleType, true>(
297 new internal::RegionParticleIterator<ParticleType, ReferenceCell, true>(
298 &this->_cells, lowerCorner, higherCorner, cellsOfInterest, &_cellBlock, behavior, nullptr));
299 }
