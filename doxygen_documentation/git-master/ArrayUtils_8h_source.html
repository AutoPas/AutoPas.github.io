178template <class OuterContainerT, class F>
179void balanceVectors(OuterContainerT &vecvec, F innerContainerToVec) {
180 using InnerContainerT = typename OuterContainerT::value_type;
181 using ElemT = typename std::remove_reference_t<
182 std::invoke_result_t<decltype(innerContainerToVec), InnerContainerT &>>::value_type;
183 // calculate vecvec statistics
184 const auto vecvecSize = vecvec.size();
185 if (vecvecSize == 0) {
186 // nothing to do
187 return;
188 }
189 const size_t numElem = std::transform_reduce(vecvec.begin(), vecvec.end(), 0, std::plus<>(),
190 [&](auto &vec) { return innerContainerToVec(vec).size(); });
191 const auto targetSize = static_cast<long>(numElem / vecvecSize);
192 const auto remainder = numElem % vecvecSize;
193
194 std::vector<ElemT> tmpStorage;
195 // index of the first subvec that has too few elements
196 size_t firstTooFew = 0;
197 // repeat as long as there is something in the buffer but at least 2 iterations.
198 for (int pass = 0; pass == 0 or not tmpStorage.empty(); ++pass) {
199 // scan all subvecs that are not known to have the desired size
200 for (size_t i = firstTooFew; i < vecvecSize; ++i) {
201 auto &vec = innerContainerToVec(vecvec[i]);
202 const auto thisTargetSize = i < remainder ? targetSize + 1 : targetSize;
203 if (vec.size() > thisTargetSize) {
204 // move what is too much to tmpStorage
205 const auto startOfTooMuch = vec.begin() + thisTargetSize;
206 tmpStorage.insert(tmpStorage.end(), std::make_move_iterator(startOfTooMuch),
207 std::make_move_iterator(vec.end()));
208 vec.erase(startOfTooMuch, vec.end());
209 // if firstTooFew points here bump it, since this vector is now satisfied
210 if (firstTooFew == i) {
211 ++firstTooFew;
212 }
213 } else if (vec.size() < thisTargetSize) {
214 // fill too small vectors from storage
215 vec.reserve(thisTargetSize);
216 const long numMissing = thisTargetSize - vec.size();
217 const auto startOfInsertion =
218 tmpStorage.begin() + (std::max(0l, static_cast<long>(tmpStorage.size()) - numMissing));
219 vec.insert(vec.end(), std::make_move_iterator(startOfInsertion), std::make_move_iterator(tmpStorage.end()));
220 tmpStorage.erase(startOfInsertion, tmpStorage.end());
221
222 } else if (firstTooFew == i) {
223 // if firstTooFew points here bump it, since this vector is already satisfied
224 ++firstTooFew;
225 }
226 // after the first pass all excess elements are collected. So as soon as everything is distributed we are done.
227 if (pass > 0 and tmpStorage.empty()) {
228 break;
229 }
230 }
231 }
232}
