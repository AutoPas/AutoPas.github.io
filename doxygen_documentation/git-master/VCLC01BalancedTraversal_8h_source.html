70 auto &towers = *VCLTraversalInterface<Particle>::_towers;
71 size_t clusterCount = 0;
72 for (size_t towerIndex = clusterRange.startTowerIndex;
73 clusterCount < clusterRange.numClusters and towerIndex < towers.size(); towerIndex++) {
74 auto &currentTower = towers[towerIndex];
75 auto startIndexInTower = clusterCount == 0 ? clusterRange.startIndexInTower : 0;
76 for (size_t clusterIndex = startIndexInTower;
77 clusterIndex < currentTower.getNumClusters() && clusterCount < clusterRange.numClusters;
78 clusterIndex++, clusterCount++) {

80 _clusterFunctor.traverseCluster(currentCluster);
81 for (auto *neighborCluster : currentCluster.getNeighbors()) {
82 _clusterFunctor.traverseClusterPair(currentCluster, *neighborCluster);
83 }
84 }
