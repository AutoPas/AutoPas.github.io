113  _cellOffsets.resize(2 * this->_overlap[1] + 1, std::vector<offsetArray_t>(2 * this->_overlap[0] + 1));
114  const std::array<long, 3> _overlap_s = utils::ArrayUtils::static_cast_array<long>(this->_overlap);
115 
116  const auto interactionLengthSquare(this->_interactionLength * this->_interactionLength);
117 
118  for (long z = 0l; z <= _overlap_s[2]; ++z) {
119  for (long y = -_overlap_s[1]; y <= _overlap_s[1]; ++y) {
120  for (long x = -_overlap_s[0]; x <= _overlap_s[0]; ++x) {
121  const long offset = utils::ThreeDimensionalMapping::threeToOneD(
122  x, y, z, utils::ArrayUtils::static_cast_array<long>(this->_cellsPerDimension));
123 
124  if (offset < 0l) {
125  continue;
126  }
127  // add to each applicable special case
128  for (long yArray = -_overlap_s[1]; yArray <= _overlap_s[1]; ++yArray) {
129  if (std::abs(yArray + y) <= _overlap_s[1]) {
130  for (long xArray = -_overlap_s[0]; xArray <= _overlap_s[0]; ++xArray) {
131  if (std::abs(xArray + x) <= _overlap_s[0]) {
132  std::array<double, 3> pos = {};
133  pos[0] = std::max(0l, (std::abs(x) - 1l)) * this->_cellLength[0];
134  pos[1] = std::max(0l, (std::abs(y) - 1l)) * this->_cellLength[1];
135  pos[2] = std::max(0l, (std::abs(z) - 1l)) * this->_cellLength[2];
