127 bool RandomSearch::tune(bool currentInvalid) {
128  // We can overwrite the cellSizeFactor, because _currentConfig will be changed anyways.
129  if (currentInvalid) {
130  _currentConfig.cellSizeFactor = 1;
131  if (_invalidConfigs.find(_currentConfig) == _invalidConfigs.end()) {
132  ++_numTestedConfigsNoCSF;
133  }
134  _invalidConfigs.emplace(_currentConfig);
135  }
136 
137  if (searchSpaceIsEmpty()) {
138  // no valid configuration
139  _currentConfig = Configuration();
140  return false;
141  }
142 
143  if (_traversalTimes.size() >= _maxEvidence or _numTestedConfigsNoCSF >= _searchSpaceSizeNoCSF) {
144  // select best config
145  selectOptimalConfiguration();
146  return false;
147  }
148 
149  // needed to test if a new configuration has been found for infinite cellSizeFactors.
150  Configuration testingConfig;
151  do {
152  // select random config
153  _currentConfig.container = _rng.pickRandom(_containerOptions);
154  _currentConfig.cellSizeFactor = _cellSizeFactors->getRandom(_rng);
155  _currentConfig.traversal = _rng.pickRandom(_traversalOptions);
156  auto applicableLoadEstimators = loadEstimators::getApplicableLoadEstimators(
157  _currentConfig.container, _currentConfig.traversal, _loadEstimatorOptions);
158  _currentConfig.loadEstimator = _rng.pickRandom(applicableLoadEstimators);
159  _currentConfig.dataLayout = _rng.pickRandom(_dataLayoutOptions);
160  _currentConfig.newton3 = _rng.pickRandom(_newton3Options);
161 
162  testingConfig = Configuration(_currentConfig);
163  if (not _cellSizeFactors->isFinite()) {
164  testingConfig.cellSizeFactor = 1;
165  }
166  } while (_invalidConfigs.find(testingConfig) != _invalidConfigs.end() or
167  _traversalTimes.find(_currentConfig) != _traversalTimes.end());
168  return true;
169 }

171 void RandomSearch::selectOptimalConfiguration() {
172  if (searchSpaceIsTrivial()) {
173  _currentConfig.container = *_containerOptions.begin();
174  _currentConfig.cellSizeFactor = _cellSizeFactors->getMin();
175  _currentConfig.traversal = *_traversalOptions.begin();
176  auto applicableLoadEstimators = loadEstimators::getApplicableLoadEstimators(
177  _currentConfig.container, _currentConfig.traversal, _loadEstimatorOptions);
178  _currentConfig.loadEstimator = *applicableLoadEstimators.begin();
179  _currentConfig.dataLayout = *_dataLayoutOptions.begin();
180  _currentConfig.newton3 = *_newton3Options.begin();
181  AutoPasLog(debug, "Selected optimal configuration {}", _currentConfig.toString());
182  return;
183  }
184 
185  // Time measure strategy
186  if (_traversalTimes.empty()) {
187  utils::ExceptionHandler::exception(
188  "RandomSearch: Trying to determine fastest configuration without any measurements! "
189  "Either selectOptimalConfiguration was called too early or no applicable configurations were found");
190  }
191 
192  auto optimum = std::min_element(_traversalTimes.begin(), _traversalTimes.end(),
193  [](std::pair<Configuration, size_t> a, std::pair<Configuration, size_t> b) -> bool {
194  return a.second < b.second;
195  });
196 
197  _currentConfig = optimum->first;
198 
199  AutoPasLog(debug, "Selected Configuration {}", _currentConfig.toString());
200 }
