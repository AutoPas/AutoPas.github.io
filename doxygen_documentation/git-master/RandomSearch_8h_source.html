129bool RandomSearch::tune(bool currentInvalid) {
130 // We can overwrite the cellSizeFactor, because _currentConfig will be changed anyways.
131 if (currentInvalid) {
132 _currentConfig.cellSizeFactor = 1;
133 if (_invalidConfigs.find(_currentConfig) == _invalidConfigs.end()) {
134 ++_numTestedConfigsNoCSF;
135 }
136 _invalidConfigs.emplace(_currentConfig);
137 }
138
139 if (searchSpaceIsEmpty()) {
140 // no valid configuration
141 _currentConfig = Configuration();
142 return false;
143 }
144
145 if (_traversalTimes.size() >= _maxEvidence or _numTestedConfigsNoCSF >= _searchSpaceSizeNoCSF) {
146 // select best config
147 selectOptimalConfiguration();
148 return false;
149 }
150
151 // needed to test if a new configuration has been found for infinite cellSizeFactors.
152 Configuration testingConfig;
153 do {
154 // select random config
155 _currentConfig.container = _rng.pickRandom(_containerOptions);
156 _currentConfig.cellSizeFactor = _cellSizeFactors->getRandom(_rng);
157 _currentConfig.traversal = _rng.pickRandom(_traversalOptions);
158 auto applicableLoadEstimators = loadEstimators::getApplicableLoadEstimators(
159 _currentConfig.container, _currentConfig.traversal, _loadEstimatorOptions);
160 _currentConfig.loadEstimator = _rng.pickRandom(applicableLoadEstimators);
161 _currentConfig.dataLayout = _rng.pickRandom(_dataLayoutOptions);
162 _currentConfig.newton3 = _rng.pickRandom(_newton3Options);
163
164 testingConfig = Configuration(_currentConfig);
165 if (not _cellSizeFactors->isFinite()) {
166 testingConfig.cellSizeFactor = 1;
167 }
168 } while (_invalidConfigs.find(testingConfig) != _invalidConfigs.end() or
169 _traversalTimes.find(_currentConfig) != _traversalTimes.end());
170 return true;
171}

173void RandomSearch::selectOptimalConfiguration() {
174 if (searchSpaceIsTrivial()) {
175 _currentConfig.container = *_containerOptions.begin();
176 _currentConfig.cellSizeFactor = _cellSizeFactors->getMin();
177 _currentConfig.traversal = *_traversalOptions.begin();
178 auto applicableLoadEstimators = loadEstimators::getApplicableLoadEstimators(
179 _currentConfig.container, _currentConfig.traversal, _loadEstimatorOptions);
180 _currentConfig.loadEstimator = *applicableLoadEstimators.begin();
181 _currentConfig.dataLayout = *_dataLayoutOptions.begin();
182 _currentConfig.newton3 = *_newton3Options.begin();
183 AutoPasLog(debug, "Selected optimal configuration {}", _currentConfig.toString());
184 return;
185 }
186
187 // Time measure strategy
188 if (_traversalTimes.empty()) {
189 utils::ExceptionHandler::exception(
190 "RandomSearch: Trying to determine fastest configuration without any measurements! "
191 "Either selectOptimalConfiguration was called too early or no applicable configurations were found");
192 }
193
194 auto optimum = std::min_element(_traversalTimes.begin(), _traversalTimes.end(),
195 [](std::pair<Configuration, size_t> a, std::pair<Configuration, size_t> b) -> bool {
196 return a.second < b.second;
197 });
198
199 _currentConfig = optimum->first;
200
201 AutoPasLog(debug, "Selected Configuration {}", _currentConfig.toString());
202}

204bool RandomSearch::searchSpaceIsTrivial() const {
205 if (searchSpaceIsEmpty()) {
206 return false;

209 // if no load estimators are specified, none is automatically added.
