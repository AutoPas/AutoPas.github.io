113void VCLC06Traversal<ParticleCell, PairwiseFunctor, dataLayout, useNewton3>::processColorCell(
114 unsigned long xColorCell, unsigned long yColorCell, unsigned long zColorCell, int towersPerColoringCell) {
115 // We are only doing a 2D coloring.
116 if (zColorCell != 0) {
117 autopas::utils::ExceptionHandler::exception("Coloring should only be 2D, not in z-direction!");
118 }
119
120 auto &clusterList = *VCLTraversalInterface<Particle>::_verletClusterLists;
121 const auto towersPerDim = clusterList.getTowersPerDimension();
122
123 for (int yInner = 0; yInner < towersPerColoringCell; yInner++) {
124 for (int xInner = 0; xInner < towersPerColoringCell; xInner++) {
125 const auto y = yColorCell * towersPerColoringCell + yInner;
126 const auto x = xColorCell * towersPerColoringCell + xInner;
127
128 // Not every coloring cell has to have gridsPerColoringCell grids in every direction.
129 if (x >= towersPerDim[0] or y >= towersPerDim[1]) {
130 continue;
131 }
132
133 auto &currentTower = clusterList.getTowerAtCoordinates(x, y);
134 for (auto &cluster : currentTower.getClusters()) {
135 _clusterFunctor.traverseCluster(cluster);
136
137 for (auto *neighborClusterPtr : cluster.getNeighbors()) {
138 _clusterFunctor.traverseClusterPair(cluster, *neighborClusterPtr);
139 }
140 }
141 }
