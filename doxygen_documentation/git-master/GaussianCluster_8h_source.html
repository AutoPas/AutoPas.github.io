130 void addEvidence(const GaussianModelTypes::VectorDiscrete &inputDiscrete,
131 const GaussianModelTypes::VectorContinuous &inputContinuous, double output) {
132 size_t clusterIdx = getIndex(inputDiscrete);
133 if (static_cast<size_t>(inputContinuous.size()) != _continuousDims) {
134 utils::ExceptionHandler::exception(
135 "GaussianCluster: size of continuous input {} does not match specified dimensions {}", inputContinuous.size(),
136 _continuousDims);
137 }
138
139 if (_numEvidence == 0) {
140 // first evidence
141 _evidenceMinValue = _evidenceMaxValue = output;
142 _evidenceMaxVector = std::make_pair(inputDiscrete, inputContinuous);
143 } else if (output < _evidenceMinValue) {
144 _evidenceMinValue = output;
145 } else if (output > _evidenceMaxValue) {
146 _evidenceMaxValue = output;
147 _evidenceMaxVector = std::make_pair(inputDiscrete, inputContinuous);
148 }
149
150 _clusters[clusterIdx].addEvidence(inputContinuous, output, false);
151 ++_numEvidence;
152
153 auto [sample_means, sample_thetas, sample_dimScales] = GaussianProcess::generateHyperparameterSamples(
154 hp_sample_size, _rng, _continuousDims, _sigma, _evidenceMinValue, _evidenceMaxValue);
155
156 // set hyperparameter of each cluster
157 for (auto &cluster : _clusters) {
158 cluster.setHyperparameters(sample_means, sample_thetas, sample_dimScales);
159 }
160
161 // combine score of each cluster
162 for (size_t i = 0; i < hp_sample_size; ++i) {
163 // set the score of a hyperparameter sample to the product of all scores
164 double combinedScore = 0.;
165 for (auto &cluster : _clusters) {
166 combinedScore += cluster.getHyperparameters()[i].score;
167 }
168
169 if (std::isnan(combinedScore) or std::isinf(combinedScore)) {
170 utils::ExceptionHandler::exception("GaussianCluster: Score of hyperparameter is {}", combinedScore);
171 }
172
173 for (auto &cluster : _clusters) {
174 cluster.getHyperparameters()[i].score = combinedScore;
175 }
176 }
177
178 // normalize all hyperparameters
179 for (auto &cluster : _clusters) {
180 cluster.normalizeHyperparameters();
181 }
182 }

214 [[nodiscard]] std::vector<GaussianModelTypes::VectorAcquisition> sampleAcquisition(
215 AcquisitionFunctionOption af, const GaussianModelTypes::NeighbourFunction &neighbourFun,
216 const std::vector<GaussianModelTypes::VectorContinuous> &continuousSamples) const {
217 std::vector<GaussianModelTypes::VectorAcquisition> acquisitions;
218 // pair up all discrete with all continuous samples
219 acquisitions.reserve(_clusters.size() * continuousSamples.size());
220
221 auto neighbourWeights = initNeighbourWeights(neighbourFun);
222
223 for (const auto &continuousSample : continuousSamples) {
224 auto [means, vars, stddevs] = precalculateDistributions(continuousSample);
225 updateNeighbourWeights(neighbourWeights, means, vars, stddevs);
226 auto currentAcquisisitions = precalculateAcquisitions(af, means, vars);
227
228 _logger->add(_clusters, _discreteVectorMap, continuousSample, means, vars, neighbourWeights);
229
230 // get acquisition considering neighbours
231 for (size_t i = 0; i < _clusters.size(); ++i) {
232 // target cluster gets weight 1.
233 double mixAcquisition = currentAcquisisitions[i];
234 double weightSum = 1.;
235
236 // weighted sum over neighbours
237 for (const auto &[n, _, weight] : neighbourWeights[i]) {
238 mixAcquisition += currentAcquisisitions[n] * weight;
239 weightSum += weight;
240 }
241
242 // normalize
243 mixAcquisition /= weightSum;
244
245 acquisitions.emplace_back(std::make_pair(_discreteVectorMap[i], continuousSample), mixAcquisition);
246 }
247 }
248
249 _logger->flush();
250
251 return acquisitions;
252 }

396 [[nodiscard]] size_t getIndex(const GaussianModelTypes::VectorDiscrete &x) const {
397 if (static_cast<size_t>(x.size()) != _dimRestriction.size()) {
398 utils::ExceptionHandler::exception(
399 "GaussianCluster: size of discrete input {} does not match specified dimensions {}", x.size(),
400 _dimRestriction.size());
401 }
402
403 size_t result = 0;
404 for (long i = x.size() - 1; i >= 0; --i) {
405 if (x[i] < 0 or x[i] >= _dimRestriction[i]) {
406 utils::ExceptionHandler::exception("GaussianCluster: The {}th dimension is {} but is restricted to [0,{})", i,
407 x[i], _dimRestriction[i]);
408 }
409
410 result = result * _dimRestriction[i] + x[i];
411 }
412
413 return result;
414 }

438 [[nodiscard]] std::tuple<std::vector<double>, std::vector<double>, std::vector<double>> precalculateDistributions(
439 const GaussianModelTypes::VectorContinuous &continuousTuple) const {
440 std::vector<double> means;
441 std::vector<double> vars;
442 std::vector<double> stddevs;
443 means.reserve(_clusters.size());
444 vars.reserve(_clusters.size());
445 stddevs.reserve(_clusters.size());
446 for (const auto &cluster : _clusters) {
447 means.push_back(cluster.predictMean(continuousTuple));
448
449 double var = cluster.predictVar(continuousTuple);
450 vars.push_back(var);
451 stddevs.push_back(std::sqrt(var));
452 }
453
454 return std::make_tuple(means, vars, stddevs);

482 [[nodiscard]] GaussianModelTypes::NeighboursWeights initNeighbourWeights(
483 const GaussianModelTypes::NeighbourFunction &neighbourFun) const {
484 GaussianModelTypes::NeighboursWeights result;
485 result.reserve(_clusters.size());
486
487 // for each cluster create a neighbour list
488 for (size_t i = 0; i < _clusters.size(); ++i) {
489 auto neighbours = neighbourFun(_discreteVectorMap[i]);
490
491 // calculate initial weight for each neighbour
492 std::vector<std::tuple<size_t, double, double>> neighbourWeights;
493 neighbourWeights.reserve(neighbours.size());
494 for (const auto &[n, priorWeight] : neighbours) {
495 size_t n_index = getIndex(n);
496 // ignore clusters without evidence
497 if (_clusters[n_index].numEvidence() > 0) {
498 // calculate initial value for given weight function
499 double weight = 0.;
500 switch (_weightFun) {
501 case evidenceMatchingProbabilityGM: {
502 const auto &[inputs, outputs] = _clusters[n_index].getEvidence();
503 weight = 1.;
504 // product of probability densitiy over all evidence in neighbouring cluster if provided to the target
505 // cluster
506 for (size_t e = 0; e < inputs.size(); ++e) {
507 weight *= _clusters[i].predictOutputPDF(inputs[e], outputs[e]);
508 }
509 // geometric mean
510 weight = std::pow(weight, 1.0 / inputs.size());
511 break;
512 }
513 case evidenceMatchingScaledProbabilityGM: {
514 const auto &[inputs, outputs] = _clusters[n_index].getEvidence();
515 weight = 1.;
516 // product of probability densitiy over all evidence in neighbouring cluster if provided to the target
517 // cluster
518 for (size_t e = 0; e < inputs.size(); ++e) {
519 weight *= _clusters[i].predictOutputScaledPDF(inputs[e], outputs[e]);
520 }
521 // geometric mean
522 weight = std::pow(weight, 1.0 / inputs.size());
523 break;
524 }
525 case wasserstein2:
526 break;
527 }
528 neighbourWeights.emplace_back(n_index, priorWeight, priorWeight * weight);
529 }
530 }
531 result.push_back(std::move(neighbourWeights));
532 }
533
534 return result;
535 }

545 void updateNeighbourWeights(GaussianModelTypes::NeighboursWeights &neighbourWeights, const std::vector<double> &means,
546 const std::vector<double> &vars, const std::vector<double> &stddevs) const {
547 // for each cluster update neighbour-weight list
548 GaussianModelTypes::VectorDiscrete currentDiscrete = Eigen::VectorXi::Zero(_dimRestriction.size());
549 for (size_t i = 0; i < _clusters.size(); ++i) {
550 // for each neighbour update weight
551 for (auto &[n, priorWeight, weight] : neighbourWeights[i]) {
552 switch (_weightFun) {
553 case evidenceMatchingProbabilityGM:
554 case evidenceMatchingScaledProbabilityGM:
555 // keep inital value
556 break;
557 case wasserstein2:
558 weight = vars[i] / (vars[i] + std::pow(means[n] - means[i], 2) + std::pow(stddevs[n] - stddevs[i], 2));
559 weight *= priorWeight;
560 break;
561 }
562 }
563 }
