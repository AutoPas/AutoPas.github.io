125 // find the next non-empty cell
126 const int stride = (this->_behavior & IteratorBehavior::forceSequential) ? 1 : autopas_get_num_threads();
127 if (_currentRegionIndex + stride >= _indicesInRegion.size()) {
128 // make the iterator invalid!
129 this->_iteratorAcrossCells = this->_vectorOfCells->end();
130 return;
131 }
132 size_t iteratorInc = _indicesInRegion[_currentRegionIndex + stride] - _indicesInRegion[_currentRegionIndex];
133
134 for (this->_iteratorAcrossCells += iteratorInc; this->getCurrentCellId() <= *(_indicesInRegion.end() - 1);
135 this->_iteratorAcrossCells += iteratorInc) {
136 _currentRegionIndex += stride;

141 break;
142 }
143 if (_currentRegionIndex + stride >= _indicesInRegion.size()) {
144 // make the iterator invalid!

146 break;
147 }
