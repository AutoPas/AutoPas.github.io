86 auto needles = autopas::utils::StringUtils::tokenize(optionsString, autopas::utils::StringUtils::delimiters);

88 if (needles.size() == 1 and needles[0] == "all") {
89 return actualOption::getAllOptions();
90 }

92 // create a map of enum -> string with lowercase enums as a lookup and fill strings in the haystack
93 std::map<std::string, actualOption> allOptionNamesLower;
94 for (auto &[optionEnum, optionString] : actualOption::getOptionNames()) {
95 std::transform(optionString.begin(), optionString.end(), optionString.begin(), ::tolower);
96 allOptionNamesLower.emplace(optionString, optionEnum);
97 haystack.push_back(optionString);
98 }
99
100 for (auto &needle : needles) {
101 // first find the best matching string
102 auto matchingString = autopas::utils::StringUtils::matchStrings(haystack, needle);
103 // then find the corresponding enum and add it to the return set
104 foundOptions.insert(allOptionNamesLower[matchingString]);
105 }
106
107 return foundOptions;
