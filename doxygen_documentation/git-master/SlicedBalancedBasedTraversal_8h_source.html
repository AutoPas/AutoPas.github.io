53 void initSliceThickness(unsigned long minSliceThickness) override {
54 // make thicknesses are empty
55 this->_sliceThickness.clear();
56
57 // estimate loads along longest axis
58 auto maxDimension = this->_dimsPerLength[0];
59 auto maxDimensionLength = this->_cellsPerDimension[this->_dimsPerLength[0]];
60
61 std::vector<unsigned long> loads;
62 utils::Timer timer;
63 timer.start();
64 loads.resize(maxDimensionLength);
65#ifdef AUTOPAS_OPENMP
66#pragma omp parallel for schedule(static, 1)
67#endif
68 for (auto x = 0; x < maxDimensionLength; x++) {
69 std::array<unsigned long, 3> lowerCorner = {0, 0, 0};
70 std::array<unsigned long, 3> upperCorner = this->_cellsPerDimension;
71 // upper corner is inclusive, so subtract 1 from each coordinate
72 upperCorner[0]--;
73 upperCorner[1]--;
74 upperCorner[2]--;
75 lowerCorner[maxDimension] = x;
76 upperCorner[maxDimension] = x;
77 if (not this->_loadEstimator) {
78 autopas::utils::ExceptionHandler::exception(
79 "AutoPas internal error: SlicedBalancedBasedTraversal's _loadEstimator is null.");
80 }
81 auto load = this->_loadEstimator(this->_cellsPerDimension, lowerCorner, upperCorner);
82 loads[x] = load;
83 }
84 for (auto i = 1; i < loads.size(); i++) {
85 loads[i] += loads[i - 1];
86 }
87 auto fullLoad = loads.back();
88 auto loadEstimationTime = timer.stop();
89 AutoPasLog(debug, "load estimation took {} nanoseconds", loadEstimationTime);
90
91 auto numSlices = (size_t)autopas_get_max_threads();
92 AutoPasLog(debug, "{} threads available.", numSlices);
93 // using greedy algorithm to assign slice thicknesses. May lead to less slices being used.
94 unsigned int totalThickness = 0;
95 // avg load per slice
96 auto avg = fullLoad / numSlices;
97 auto lastLoad = 0;
98 for (auto s = 0; s < numSlices; s++) {
99 unsigned int thickness;
100 if (s == numSlices - 1) {
101 thickness = maxDimensionLength - totalThickness;
102 } else {
103 thickness = minSliceThickness;
104 while (totalThickness + thickness + 1 < maxDimensionLength and
105 loads[totalThickness + thickness - 1] - lastLoad < avg) {
106 auto load1 = loads[totalThickness + thickness - 1] - lastLoad;
107 auto load2 = loads[totalThickness + thickness] - lastLoad;
108 // if (abs(avg-load1) < abs(avg-load2))
109 // doing this manually as we are using unsigned longs and would have to cast otherwise
110 if (((avg > load1) ? (avg - load1) : (load1 - avg)) < ((avg > load2) ? (avg - load2) : (load2 - avg))) {
111 break;
112 }
113 thickness++;
114 }
115 }
116 if (totalThickness + thickness > maxDimensionLength || thickness < minSliceThickness) {
117 // if minSlicethickness can no longer be satisfied, add remaining space to last slice
118 this->_sliceThickness[s - 1] += maxDimensionLength - totalThickness;
119 AutoPasLog(debug, "Balanced Sliced traversal only using {} threads because of greedy algorithm.", s);
120 numSlices = s;
121 break;
122
123 } else {
124 totalThickness += thickness;
126 this->_sliceThickness.push_back(thickness);
127 if (s != numSlices - 1) {
128 // avg of remaining load over remaining threads
129 avg = (fullLoad - loads[totalThickness - 1]) / (numSlices - s - 1);
130 lastLoad = loads[totalThickness - 1];
131 }
132 }
133 }

137 totalThickness = 0;
138 for (auto t : this->_sliceThickness) {
139 thicknessStr += std::to_string(t) + ", ";
140 totalThickness += t;
141 loadStr += std::to_string(loads[totalThickness - 1] - lastLoad) + ", ";
142 lastLoad = loads[totalThickness - 1];
143 }
144
145 // some analysis output that is only relevant when logger is set to debug
146 if (autopas::Logger::get()->level() <= autopas::Logger::LogLevel::debug) {
147 std::string thicknessStr;
148 std::string loadStr;
149 auto lastLoad = 0;
150 totalThickness = 0;
151 for (auto t : this->_sliceThickness) {
152 thicknessStr += std::to_string(t) + ", ";
153 totalThickness += t;
154 loadStr += std::to_string(loads[totalThickness - 1] - lastLoad) + ", ";

156 }
157

160 AutoPasLog(debug, "Slice loads: [{}]", loadStr);
161 }
