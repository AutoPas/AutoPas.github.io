43 void AoSFunctor(Particle &i, Particle &j, bool /*newton3*/) override {
44 if (i.isDummy() or j.isDummy()) {
45 return;
46 }
47 auto dr = utils::ArrayMath::sub(i.getR(), j.getR());
48 double dr2 = utils::ArrayMath::dot(dr, dr);
49 _distanceCalculations.fetch_add(1, std::memory_order_relaxed);
50
51 if (dr2 <= _cutoffSquare) {
52 _kernelCalls.fetch_add(1, std::memory_order_relaxed);
53 }
54 }

60 void SoAFunctorSingle(SoAView<typename Particle::SoAArraysType> soa, bool /*newton3*/) override {
61 if (soa.getNumParticles() == 0) return;
62
63 double *const __restrict x1ptr = soa.template begin<Particle::AttributeNames::posX>();
64 double *const __restrict y1ptr = soa.template begin<Particle::AttributeNames::posY>();
65 double *const __restrict z1ptr = soa.template begin<Particle::AttributeNames::posZ>();
66
67 for (size_t i = 0; i < soa.getNumParticles(); ++i) {
68 size_t distanceCalculationsAcc = 0;
69 size_t kernelCallsAcc = 0;
70
71// icpc vectorizes this.
72// g++ only with -ffast-math or -funsafe-math-optimizations
73#pragma omp simd reduction(+ : kernelCallsAcc, distanceCalculationsAcc)
74 for (size_t j = i + 1; j < soa.getNumParticles(); ++j) {
75 ++distanceCalculationsAcc;
76
77 const double drx = x1ptr[i] - x1ptr[j];
78 const double dry = y1ptr[i] - y1ptr[j];
79 const double drz = z1ptr[i] - z1ptr[j];
80
81 const double drx2 = drx * drx;
82 const double dry2 = dry * dry;
83 const double drz2 = drz * drz;
84
85 const double dr2 = drx2 + dry2 + drz2;
86
87 if (dr2 <= _cutoffSquare) {
88 ++kernelCallsAcc;
89 }
90 }
91 _distanceCalculations.fetch_add(distanceCalculationsAcc, std::memory_order_relaxed);
92 _kernelCalls.fetch_add(kernelCallsAcc, std::memory_order_relaxed);
93 }

145 void SoAFunctorVerlet(SoAView<typename Particle::SoAArraysType> soa, const size_t indexFirst,
146 const std::vector<size_t, autopas::AlignedAllocator<size_t>> &neighborList,
147 bool /*newton3*/) override {
148 auto numParts = soa.getNumParticles();
149
150 if (numParts == 0) return;
151
152 double *const __restrict xptr = soa.template begin<Particle::AttributeNames::posX>();
153 double *const __restrict yptr = soa.template begin<Particle::AttributeNames::posY>();
154 double *const __restrict zptr = soa.template begin<Particle::AttributeNames::posZ>();
155
156 const size_t listSizeI = neighborList.size();
157 const size_t *const __restrict currentList = neighborList.data();
158
159 // this is a magic number, that should correspond to at least
160 // vectorization width*N have testet multiple sizes:
161 // 4: small speedup compared to AoS
162 // 8: small speedup compared to AoS
163 // 12: small but best speedup compared to Aos
164 // 16: smaller speedup
165 // in theory this is a variable, we could auto-tune over...
166#ifdef __AVX512F__
167 // use a multiple of 8 for avx
168 const size_t vecsize = 16;
169#else
170 // for everything else 12 is faster
171 const size_t vecsize = 12;
172#endif
173 size_t joff = 0;
174
175 // if the size of the verlet list is larger than the given size vecsize,
176 // we will use a vectorized version.
177 if (listSizeI >= vecsize) {
178 alignas(64) std::array<double, vecsize> xtmp{}, ytmp{}, ztmp{}, xArr{}, yArr{}, zArr{};
179 // broadcast of the position of particle i
180 for (size_t tmpj = 0; tmpj < vecsize; tmpj++) {
181 xtmp[tmpj] = xptr[indexFirst];
182 ytmp[tmpj] = yptr[indexFirst];
183 ztmp[tmpj] = zptr[indexFirst];
184 }
185 // loop over the verlet list from 0 to x*vecsize
186 for (; joff < listSizeI - vecsize + 1; joff += vecsize) {
187 size_t distanceCalculationsAcc = 0;
188 size_t kernelCallsAcc = 0;
189 // in each iteration we calculate the interactions of particle i with
190 // vecsize particles in the neighborlist of particle i starting at
191 // particle joff
192
193 // gather position of particle j
194#pragma omp simd safelen(vecsize)
195 for (size_t tmpj = 0; tmpj < vecsize; tmpj++) {
196 xArr[tmpj] = xptr[currentList[joff + tmpj]];
197 yArr[tmpj] = yptr[currentList[joff + tmpj]];
198 zArr[tmpj] = zptr[currentList[joff + tmpj]];
199 }
200
201 // do omp simd with reduction of the interaction
202#pragma omp simd reduction(+ : kernelCallsAcc, distanceCalculationsAcc) safelen(vecsize)
203 for (size_t j = 0; j < vecsize; j++) {
204 ++distanceCalculationsAcc;
205 const double drx = xtmp[j] - xArr[j];
206 const double dry = ytmp[j] - yArr[j];
207 const double drz = ztmp[j] - zArr[j];
208
209 const double drx2 = drx * drx;
210 const double dry2 = dry * dry;
211 const double drz2 = drz * drz;
212
213 const double dr2 = drx2 + dry2 + drz2;
214
215 const unsigned long mask = (dr2 <= _cutoffSquare) ? 1 : 0;
216
217 kernelCallsAcc += mask;
218 }
219 _distanceCalculations.fetch_add(distanceCalculationsAcc, std::memory_order_relaxed);
220 _kernelCalls.fetch_add(kernelCallsAcc, std::memory_order_relaxed);
221 }
222 }
223 size_t distanceCalculationsAcc = 0;
224 size_t kernelCallsAcc = 0;
225 // this loop goes over the remainder and uses no optimizations
226 for (size_t jNeighIndex = joff; jNeighIndex < listSizeI; ++jNeighIndex) {
227 size_t j = neighborList[jNeighIndex];
228 if (indexFirst == j) continue;
229
230 ++distanceCalculationsAcc;
231 const double drx = xptr[indexFirst] - xptr[j];
232 const double dry = yptr[indexFirst] - yptr[j];
233 const double drz = zptr[indexFirst] - zptr[j];
234
235 const double drx2 = drx * drx;
236 const double dry2 = dry * dry;
237 const double drz2 = drz * drz;
238
239 const double dr2 = drx2 + dry2 + drz2;
240
241 if (dr2 <= _cutoffSquare) {
242 ++kernelCallsAcc;
243 }
244 }
245 _distanceCalculations.fetch_add(distanceCalculationsAcc, std::memory_order_relaxed);
246 _kernelCalls.fetch_add(kernelCallsAcc, std::memory_order_relaxed);
247 }
