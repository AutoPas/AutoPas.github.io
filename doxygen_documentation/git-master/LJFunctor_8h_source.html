153 
154  if (dr2 > _cutoffsquare) {
155  return;
156  }
157 
158  double invdr2 = 1. / dr2;
159  double lj6 = sigmasquare * invdr2;
160  lj6 = lj6 * lj6 * lj6;
161  double lj12 = lj6 * lj6;
162  double lj12m6 = lj12 - lj6;
163  double fac = epsilon24 * (lj12 + lj12m6) * invdr2;
164  auto f = utils::ArrayMath::mulScalar(dr, fac);
165  i.addF(f);
166  if (newton3) {
167  // only if we use newton 3 here, we want to
168  j.subF(f);
169  }
170  if (calculateGlobals) {
171  auto virial = utils::ArrayMath::mul(dr, f);
172  double upot = epsilon24 * lj12m6 + shift6;
173 
174  const int threadnum = autopas_get_thread_num();
175  // for non-newton3 the division is in the post-processing step.
176  if (newton3) {
177  upot *= 0.5;
178  virial = utils::ArrayMath::mulScalar(virial, (double)0.5);
179  }
180  if (i.isOwned()) {
181  _aosThreadData[threadnum].upotSum += upot;
182  _aosThreadData[threadnum].virialSum = utils::ArrayMath::add(_aosThreadData[threadnum].virialSum, virial);

185  if (newton3 and j.isOwned()) {
186  _aosThreadData[threadnum].upotSum += upot;

864  SoAFloatPrecision virialSumZ = 0.;

867  SoAFloatPrecision fyacc = 0;
868  SoAFloatPrecision fzacc = 0;
869  const size_t neighborListSize = neighborList.size();
870  const size_t *const __restrict__ neighborListPtr = neighborList.data();

872  // checks whether particle i is owned.
873  const auto ownedStateI = ownedStatePtr[indexFirst];
874  if (ownedStateI == OwnershipState::dummy) {
875  return;
876  }

879  // vectorization width*N have testet multiple sizes:
880  // 4: does not give a speedup, slower than original AoSFunctor
881  // 8: small speedup compared to AoS
882  // 12: highest speedup compared to Aos
883  // 16: smaller speedup
884  // in theory this is a variable, we could auto-tune over...
885 #ifdef __AVX512F__
886  // use a multiple of 8 for avx
887  constexpr size_t vecsize = 16;
888 #else
889  // for everything else 12 is faster
890  constexpr size_t vecsize = 12;
891 #endif
892  size_t joff = 0;
893 
894  // if the size of the verlet list is larger than the given size vecsize,
895  // we will use a vectorized version.
896  if (neighborListSize >= vecsize) {
897  alignas(64) std::array<SoAFloatPrecision, vecsize> xtmp, ytmp, ztmp, xArr, yArr, zArr, fxArr, fyArr, fzArr;
898  alignas(64) std::array<OwnershipState, vecsize> ownedStateArr{};
899  // broadcast of the position of particle i
900  for (size_t tmpj = 0; tmpj < vecsize; tmpj++) {
901  xtmp[tmpj] = xptr[indexFirst];
902  ytmp[tmpj] = yptr[indexFirst];
903  ztmp[tmpj] = zptr[indexFirst];
904  }
905  // loop over the verlet list from 0 to x*vecsize
906  for (; joff < neighborListSize - vecsize + 1; joff += vecsize) {
907  // in each iteration we calculate the interactions of particle i with
908  // vecsize particles in the neighborlist of particle i starting at
909  // particle joff
910 
911  [[maybe_unused]] alignas(DEFAULT_CACHE_LINE_SIZE) std::array<SoAFloatPrecision, vecsize> sigmaSquares;
912  [[maybe_unused]] alignas(DEFAULT_CACHE_LINE_SIZE) std::array<SoAFloatPrecision, vecsize> epsilon24s;
913  [[maybe_unused]] alignas(DEFAULT_CACHE_LINE_SIZE) std::array<SoAFloatPrecision, vecsize> shift6s;
914  if constexpr (useMixing) {
915  for (size_t j = 0; j < vecsize; j++) {
916  sigmaSquares[j] = _PPLibrary->mixingSigmaSquare(typeptr1[indexFirst], typeptr2[neighborListPtr[joff + j]]);
917  epsilon24s[j] = _PPLibrary->mixing24Epsilon(typeptr1[indexFirst], typeptr2[neighborListPtr[joff + j]]);
918  if constexpr (applyShift) {
919  shift6s[j] = _PPLibrary->mixingShift6(typeptr1[indexFirst], typeptr2[neighborListPtr[joff + j]]);
920  }
921  }
922  }
923 
924  // gather position of particle j
925 #pragma omp simd safelen(vecsize)
926  for (size_t tmpj = 0; tmpj < vecsize; tmpj++) {
927  xArr[tmpj] = xptr[neighborListPtr[joff + tmpj]];
928  yArr[tmpj] = yptr[neighborListPtr[joff + tmpj]];
929  zArr[tmpj] = zptr[neighborListPtr[joff + tmpj]];
930  ownedStateArr[tmpj] = ownedStatePtr[neighborListPtr[joff + tmpj]];
931  }
932  // do omp simd with reduction of the interaction
933 #pragma omp simd reduction(+ : fxacc, fyacc, fzacc, upotSum, virialSumX, virialSumY, virialSumZ) safelen(vecsize)
934  for (size_t j = 0; j < vecsize; j++) {
935  if constexpr (useMixing) {
936  sigmasquare = sigmaSquares[j];
937  epsilon24 = epsilon24s[j];
938  if constexpr (applyShift) {
939  shift6 = shift6s[j];
940  }
941  }
942  // const size_t j = currentList[jNeighIndex];
943 
944  const auto ownedStateJ = ownedStateArr[j];
945 
946  const SoAFloatPrecision drx = xtmp[j] - xArr[j];
947  const SoAFloatPrecision dry = ytmp[j] - yArr[j];
948  const SoAFloatPrecision drz = ztmp[j] - zArr[j];
949 
950  const SoAFloatPrecision drx2 = drx * drx;
951  const SoAFloatPrecision dry2 = dry * dry;
952  const SoAFloatPrecision drz2 = drz * drz;
953 
954  const SoAFloatPrecision dr2 = drx2 + dry2 + drz2;
955 
956  // Mask away if distance is too large or any particle is a dummy.
957  // Particle ownedStateI was already checked previously.
958  const bool mask = dr2 <= cutoffsquare and ownedStateJ != OwnershipState::dummy;
959 
960  const SoAFloatPrecision invdr2 = 1. / dr2;
961  const SoAFloatPrecision lj2 = sigmasquare * invdr2;
962  const SoAFloatPrecision lj6 = lj2 * lj2 * lj2;
963  const SoAFloatPrecision lj12 = lj6 * lj6;
964  const SoAFloatPrecision lj12m6 = lj12 - lj6;
965  const SoAFloatPrecision fac = mask ? epsilon24 * (lj12 + lj12m6) * invdr2 : 0.;
966 
967  const SoAFloatPrecision fx = drx * fac;
968  const SoAFloatPrecision fy = dry * fac;
969  const SoAFloatPrecision fz = drz * fac;
970

972  fyacc += fy;
973  fzacc += fz;
974  if (newton3) {
975  fxArr[j] = fx;
976  fyArr[j] = fy;
977  fzArr[j] = fz;
978  }
979  if (calculateGlobals) {
980  SoAFloatPrecision virialx = drx * fx;
981  SoAFloatPrecision virialy = dry * fy;
982  SoAFloatPrecision virialz = drz * fz;
983  SoAFloatPrecision upot = mask ? (epsilon24 * lj12m6 + shift6) : 0.;
984

986  if constexpr (newton3) {
987  energyFactor += (ownedStateJ == OwnershipState::owned ? 1. : 0.);
988  }

990  virialSumX += virialx * energyFactor;
991  virialSumY += virialy * energyFactor;
992  virialSumZ += virialz * energyFactor;
993  }
994  }
995  // scatter the forces to where they belong, this is only needed for newton3
996  if (newton3) {
997 #pragma omp simd safelen(vecsize)
998  for (size_t tmpj = 0; tmpj < vecsize; tmpj++) {
999  const size_t j = neighborListPtr[joff + tmpj];
1000  fxptr[j] -= fxArr[tmpj];
1001  fyptr[j] -= fyArr[tmpj];
1002  fzptr[j] -= fzArr[tmpj];

1004  }

1048

1051  fzacc += fz;
1052  if (newton3) {
1053  fxptr[j] -= fx;
1054  fyptr[j] -= fy;
1055  fzptr[j] -= fz;
1056  }
1057  if (calculateGlobals) {
1058  SoAFloatPrecision virialx = drx * fx;
1059  SoAFloatPrecision virialy = dry * fy;
1060  SoAFloatPrecision virialz = drz * fz;
