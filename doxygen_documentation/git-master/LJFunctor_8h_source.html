116 void AoSFunctor(Particle &i, Particle &j, bool newton3) final {
117 using namespace autopas::utils::ArrayMath::literals;
118
119 if (i.isDummy() or j.isDummy()) {
120 return;
121 }
122 auto sigmaSquared = _sigmaSquared;
123 auto epsilon24 = _epsilon24;
124 auto shift6 = _shift6;
125 if constexpr (useMixing) {
126 sigmaSquared = _PPLibrary->getMixingSigmaSquared(i.getTypeId(), j.getTypeId());
127 epsilon24 = _PPLibrary->getMixing24Epsilon(i.getTypeId(), j.getTypeId());
128 if constexpr (applyShift) {
129 shift6 = _PPLibrary->getMixingShift6(i.getTypeId(), j.getTypeId());
130 }
131 }
132 auto dr = i.getR() - j.getR();
133 double dr2 = autopas::utils::ArrayMath::dot(dr, dr);
134
135 if (dr2 > _cutoffSquared) {
136 return;
137 }
138
139 double invdr2 = 1. / dr2;
140 double lj6 = sigmaSquared * invdr2;
141 lj6 = lj6 * lj6 * lj6;
142 double lj12 = lj6 * lj6;
143 double lj12m6 = lj12 - lj6;
144 double fac = epsilon24 * (lj12 + lj12m6) * invdr2;
145 auto f = dr * fac;
146 i.addF(f);
147 if (newton3) {
148 // only if we use newton 3 here, we want to
149 j.subF(f);
150 }
151 if (calculateGlobals) {
152 auto virial = dr * f;
153 // Here we calculate either the potential energy * 6.
154 // For newton3, this potential energy contribution is distributed evenly to the two molecules.
155 // For non-newton3, the full potential energy is added to the one molecule.
156 // The division by 6 is handled in endTraversal, as well as the division by two needed if newton3 is not used.
157 double potentialEnergy6 = epsilon24 * lj12m6 + shift6;
158
159 const int threadnum = autopas::autopas_get_thread_num();
160 if (i.isOwned()) {
161 if (newton3) {
162 _aosThreadData[threadnum].potentialEnergySumN3 += potentialEnergy6 * 0.5;
163 _aosThreadData[threadnum].virialSumN3 += virial * 0.5;
164 } else {
165 // for non-newton3 the division is in the post-processing step.
166 _aosThreadData[threadnum].potentialEnergySumNoN3 += potentialEnergy6;
167 _aosThreadData[threadnum].virialSumNoN3 += virial;
168 }
169 }
170 // for non-newton3 the second particle will be considered in a separate calculation
171 if (newton3 and j.isOwned()) {
172 _aosThreadData[threadnum].potentialEnergySumN3 += potentialEnergy6 * 0.5;
173 _aosThreadData[threadnum].virialSumN3 += virial * 0.5;
174 }
175 }

292 fyptr[j] -= fy;
293 fzptr[j] -= fz;

314 fyptr[i] += fyacc;

324 _aosThreadData[threadnum].virialSumN3[0] += virialSumX * 0.5;
325 _aosThreadData[threadnum].virialSumN3[1] += virialSumY * 0.5;

605 _virialSum = {0., 0., 0.};

616 void endTraversal(bool newton3) final {
617 using namespace autopas::utils::ArrayMath::literals;
618
619 if (_postProcessed) {
620 throw autopas::utils::ExceptionHandler::AutoPasException(
621 "Already postprocessed, endTraversal(bool newton3) was called twice without calling initTraversal().");
622 }
623 if (calculateGlobals) {
624 // We distinguish between non-newton3 and newton3 functor calls. Newton3 calls are accumulated directly.
625 // Non-newton3 calls are accumulated temporarily and later divided by 2.
626 double potentialEnergySumNoN3Acc = 0;
627 std::array<double, 3> virialSumNoN3Acc = {0, 0, 0};
628 for (size_t i = 0; i < _aosThreadData.size(); ++i) {
629 potentialEnergySumNoN3Acc += _aosThreadData[i].potentialEnergySumNoN3;
630 _potentialEnergySum += _aosThreadData[i].potentialEnergySumN3;
631
632 virialSumNoN3Acc += _aosThreadData[i].virialSumNoN3;
633 _virialSum += _aosThreadData[i].virialSumN3;
634 }
635 // if the newton3 optimization is disabled we have added every energy contribution twice, so we divide by 2
636 // here.
637 potentialEnergySumNoN3Acc *= 0.5;
638 virialSumNoN3Acc *= 0.5;
639
640 _potentialEnergySum += potentialEnergySumNoN3Acc;
641 _virialSum += virialSumNoN3Acc;
642
643 // we have always calculated 6*potentialEnergy, so we divide by 6 here!
644 _potentialEnergySum /= 6.;
645 _postProcessed = true;
646
647 AutoPasLog(TRACE, "Final potential energy {}", _potentialEnergySum);
648 AutoPasLog(TRACE, "Final virial {}", _virialSum[0] + _virialSum[1] + _virialSum[2]);
649 }
650 }

688 template <bool newton3>
689 void SoAFunctorVerletImpl(autopas::SoAView<SoAArraysType> soa, const size_t indexFirst,
690 const std::vector<size_t, autopas::AlignedAllocator<size_t>> &neighborList) {
691 const auto *const __restrict xptr = soa.template begin<Particle::AttributeNames::posX>();
692 const auto *const __restrict yptr = soa.template begin<Particle::AttributeNames::posY>();
693 const auto *const __restrict zptr = soa.template begin<Particle::AttributeNames::posZ>();
694
695 auto *const __restrict fxptr = soa.template begin<Particle::AttributeNames::forceX>();
696 auto *const __restrict fyptr = soa.template begin<Particle::AttributeNames::forceY>();
697 auto *const __restrict fzptr = soa.template begin<Particle::AttributeNames::forceZ>();
698 [[maybe_unused]] auto *const __restrict typeptr1 = soa.template begin<Particle::AttributeNames::typeId>();
699 [[maybe_unused]] auto *const __restrict typeptr2 = soa.template begin<Particle::AttributeNames::typeId>();
700
701 const auto *const __restrict ownedStatePtr = soa.template begin<Particle::AttributeNames::ownershipState>();
702
703 const SoAFloatPrecision cutoffSquared = _cutoffSquared;
704 SoAFloatPrecision shift6 = _shift6;
705 SoAFloatPrecision sigmaSquared = _sigmaSquared;
706 SoAFloatPrecision epsilon24 = _epsilon24;
707
708 SoAFloatPrecision potentialEnergySum = 0.;
709 SoAFloatPrecision virialSumX = 0.;
710 SoAFloatPrecision virialSumY = 0.;
711 SoAFloatPrecision virialSumZ = 0.;
712
713 SoAFloatPrecision fxacc = 0;
714 SoAFloatPrecision fyacc = 0;
715 SoAFloatPrecision fzacc = 0;
716 const size_t neighborListSize = neighborList.size();
717 const size_t *const __restrict neighborListPtr = neighborList.data();
718
719 // checks whether particle i is owned.
720 const auto ownedStateI = ownedStatePtr[indexFirst];
721 if (ownedStateI == autopas::OwnershipState::dummy) {
722 return;
723 }
724
725 // this is a magic number, that should correspond to at least
726 // vectorization width*N have testet multiple sizes:
727 // 4: does not give a speedup, slower than original AoSFunctor
728 // 8: small speedup compared to AoS
729 // 12: highest speedup compared to Aos
730 // 16: smaller speedup
731 // in theory this is a variable, we could auto-tune over...
732#ifdef __AVX512F__
733 // use a multiple of 8 for avx
734 constexpr size_t vecsize = 16;
735#else
736 // for everything else 12 is faster
737 constexpr size_t vecsize = 12;
738#endif
739 size_t joff = 0;
740
741 // if the size of the verlet list is larger than the given size vecsize,
742 // we will use a vectorized version.
743 if (neighborListSize >= vecsize) {
744 alignas(64) std::array<SoAFloatPrecision, vecsize> xtmp, ytmp, ztmp, xArr, yArr, zArr, fxArr, fyArr, fzArr;
745 alignas(64) std::array<autopas::OwnershipState, vecsize> ownedStateArr{};
746 // broadcast of the position of particle i
747 for (size_t tmpj = 0; tmpj < vecsize; tmpj++) {
748 xtmp[tmpj] = xptr[indexFirst];
749 ytmp[tmpj] = yptr[indexFirst];
750 ztmp[tmpj] = zptr[indexFirst];
751 }
752 // loop over the verlet list from 0 to x*vecsize
753 for (; joff < neighborListSize - vecsize + 1; joff += vecsize) {
754 // in each iteration we calculate the interactions of particle i with
755 // vecsize particles in the neighborlist of particle i starting at
756 // particle joff
757
758 [[maybe_unused]] alignas(autopas::DEFAULT_CACHE_LINE_SIZE) std::array<SoAFloatPrecision, vecsize> sigmaSquareds;
759 [[maybe_unused]] alignas(autopas::DEFAULT_CACHE_LINE_SIZE) std::array<SoAFloatPrecision, vecsize> epsilon24s;
760 [[maybe_unused]] alignas(autopas::DEFAULT_CACHE_LINE_SIZE) std::array<SoAFloatPrecision, vecsize> shift6s;
761 if constexpr (useMixing) {
762 for (size_t j = 0; j < vecsize; j++) {
763 sigmaSquareds[j] =
764 _PPLibrary->getMixingSigmaSquared(typeptr1[indexFirst], typeptr2[neighborListPtr[joff + j]]);
765 epsilon24s[j] = _PPLibrary->getMixing24Epsilon(typeptr1[indexFirst], typeptr2[neighborListPtr[joff + j]]);
766 if constexpr (applyShift) {
767 shift6s[j] = _PPLibrary->getMixingShift6(typeptr1[indexFirst], typeptr2[neighborListPtr[joff + j]]);
768 }
769 }
770 }
771
772 // gather position of particle j
773#pragma omp simd safelen(vecsize)
774 for (size_t tmpj = 0; tmpj < vecsize; tmpj++) {
775 xArr[tmpj] = xptr[neighborListPtr[joff + tmpj]];
776 yArr[tmpj] = yptr[neighborListPtr[joff + tmpj]];
777 zArr[tmpj] = zptr[neighborListPtr[joff + tmpj]];
778 ownedStateArr[tmpj] = ownedStatePtr[neighborListPtr[joff + tmpj]];
779 }
780 // do omp simd with reduction of the interaction
781#pragma omp simd reduction(+ : fxacc, fyacc, fzacc, potentialEnergySum, virialSumX, virialSumY, virialSumZ) safelen(vecsize)
782 for (size_t j = 0; j < vecsize; j++) {
783 if constexpr (useMixing) {
784 sigmaSquared = sigmaSquareds[j];
785 epsilon24 = epsilon24s[j];
786 if constexpr (applyShift) {
787 shift6 = shift6s[j];
788 }
789 }
790 // const size_t j = currentList[jNeighIndex];
791
792 const auto ownedStateJ = ownedStateArr[j];
793
794 const SoAFloatPrecision drx = xtmp[j] - xArr[j];
795 const SoAFloatPrecision dry = ytmp[j] - yArr[j];
796 const SoAFloatPrecision drz = ztmp[j] - zArr[j];
797
798 const SoAFloatPrecision drx2 = drx * drx;
799 const SoAFloatPrecision dry2 = dry * dry;
800 const SoAFloatPrecision drz2 = drz * drz;
801
802 const SoAFloatPrecision dr2 = drx2 + dry2 + drz2;
803
804 // Mask away if distance is too large or any particle is a dummy.
805 // Particle ownedStateI was already checked previously.
806 const bool mask = dr2 <= cutoffSquared and ownedStateJ != autopas::OwnershipState::dummy;
807
808 const SoAFloatPrecision invdr2 = 1. / dr2;
809 const SoAFloatPrecision lj2 = sigmaSquared * invdr2;
810 const SoAFloatPrecision lj6 = lj2 * lj2 * lj2;
811 const SoAFloatPrecision lj12 = lj6 * lj6;
812 const SoAFloatPrecision lj12m6 = lj12 - lj6;
813 const SoAFloatPrecision fac = mask ? epsilon24 * (lj12 + lj12m6) * invdr2 : 0.;
814
815 const SoAFloatPrecision fx = drx * fac;
816 const SoAFloatPrecision fy = dry * fac;
817 const SoAFloatPrecision fz = drz * fac;
818
819 fxacc += fx;
820 fyacc += fy;
821 fzacc += fz;
822 if (newton3) {
823 fxArr[j] = fx;
824 fyArr[j] = fy;
825 fzArr[j] = fz;
826 }
827 if (calculateGlobals) {
828 SoAFloatPrecision virialx = drx * fx;
829 SoAFloatPrecision virialy = dry * fy;
830 SoAFloatPrecision virialz = drz * fz;
831 SoAFloatPrecision potentialEnergy6 = mask ? (epsilon24 * lj12m6 + shift6) : 0.;
832
833 const SoAFloatPrecision energyFactor =
834 (ownedStateI == autopas::OwnershipState::owned ? 1. : 0.) +
835 (newton3 ? (ownedStateJ == autopas::OwnershipState::owned ? 1. : 0.) : 0.);
836 potentialEnergySum += potentialEnergy6 * energyFactor;
837 virialSumX += virialx * energyFactor;
838 virialSumY += virialy * energyFactor;
839 virialSumZ += virialz * energyFactor;
840 }
841 }
842 // scatter the forces to where they belong, this is only needed for newton3
843 if (newton3) {
844#pragma omp simd safelen(vecsize)
845 for (size_t tmpj = 0; tmpj < vecsize; tmpj++) {
846 const size_t j = neighborListPtr[joff + tmpj];
847 fxptr[j] -= fxArr[tmpj];
848 fyptr[j] -= fyArr[tmpj];
849 fzptr[j] -= fzArr[tmpj];
850 }
851 }
852 }
853 }
854 // this loop goes over the remainder and uses no optimizations
855 for (size_t jNeighIndex = joff; jNeighIndex < neighborListSize; ++jNeighIndex) {
856 size_t j = neighborList[jNeighIndex];
857 if (indexFirst == j) continue;
858 if constexpr (useMixing) {
859 sigmaSquared = _PPLibrary->getMixingSigmaSquared(typeptr1[indexFirst], typeptr2[j]);
860 epsilon24 = _PPLibrary->getMixing24Epsilon(typeptr1[indexFirst], typeptr2[j]);
861 if constexpr (applyShift) {
862 shift6 = _PPLibrary->getMixingShift6(typeptr1[indexFirst], typeptr2[j]);
863 }
864 }
865
866 const auto ownedStateJ = ownedStatePtr[j];
867 if (ownedStateJ == autopas::OwnershipState::dummy) {
868 continue;
869 }
870
871 const SoAFloatPrecision drx = xptr[indexFirst] - xptr[j];
872 const SoAFloatPrecision dry = yptr[indexFirst] - yptr[j];
873 const SoAFloatPrecision drz = zptr[indexFirst] - zptr[j];
874
875 const SoAFloatPrecision drx2 = drx * drx;
876 const SoAFloatPrecision dry2 = dry * dry;
877 const SoAFloatPrecision drz2 = drz * drz;
878
879 const SoAFloatPrecision dr2 = drx2 + dry2 + drz2;
880
881 if (dr2 > cutoffSquared) {
882 continue;
883 }
884
885 const SoAFloatPrecision invdr2 = 1. / dr2;
886 const SoAFloatPrecision lj2 = sigmaSquared * invdr2;
887 const SoAFloatPrecision lj6 = lj2 * lj2 * lj2;
888 const SoAFloatPrecision lj12 = lj6 * lj6;
889 const SoAFloatPrecision lj12m6 = lj12 - lj6;
890 const SoAFloatPrecision fac = epsilon24 * (lj12 + lj12m6) * invdr2;
891
892 const SoAFloatPrecision fx = drx * fac;
893 const SoAFloatPrecision fy = dry * fac;
894 const SoAFloatPrecision fz = drz * fac;
895
896 fxacc += fx;
897 fyacc += fy;
898 fzacc += fz;
899 if (newton3) {
900 fxptr[j] -= fx;
901 fyptr[j] -= fy;
902 fzptr[j] -= fz;
903 }
904 if (calculateGlobals) {
905 SoAFloatPrecision virialx = drx * fx;
906 SoAFloatPrecision virialy = dry * fy;
907 SoAFloatPrecision virialz = drz * fz;
908 SoAFloatPrecision potentialEnergy6 = (epsilon24 * lj12m6 + shift6);
909
910 // Add to the potential energy sum for each particle which is owned.
911 // This results in obtaining 12 * the potential energy for the SoA.
912 const SoAFloatPrecision energyFactor =
913 (ownedStateI == autopas::OwnershipState::owned ? 1. : 0.) +
914 (newton3 ? (ownedStateJ == autopas::OwnershipState::owned ? 1. : 0.) : 0.);
915 potentialEnergySum += potentialEnergy6 * energyFactor;
916 virialSumX += virialx * energyFactor;
917 virialSumY += virialy * energyFactor;
918 virialSumZ += virialz * energyFactor;
919 }
920 }
921
922 if (fxacc != 0 or fyacc != 0 or fzacc != 0) {
923 fxptr[indexFirst] += fxacc;
924 fyptr[indexFirst] += fyacc;
925 fzptr[indexFirst] += fzacc;
926 }
927
928 if (calculateGlobals) {
929 const int threadnum = autopas::autopas_get_thread_num();
930
931 // SoAFunctorVerlet obtains the potential energy * 12. For non-newton3, this sum is divided by 12 in
932 // post-processing. For newton3, this sum is only divided by 6 in post-processing, so must be divided by 2 here.
933
934 if (newton3) {
935 _aosThreadData[threadnum].potentialEnergySumN3 += potentialEnergySum * 0.5;
936 _aosThreadData[threadnum].virialSumN3[0] += virialSumX * 0.5;
937 _aosThreadData[threadnum].virialSumN3[1] += virialSumY * 0.5;
938 _aosThreadData[threadnum].virialSumN3[2] += virialSumZ * 0.5;
939 } else {
940 _aosThreadData[threadnum].potentialEnergySumNoN3 += potentialEnergySum;
941 _aosThreadData[threadnum].virialSumNoN3[0] += virialSumX;
942 _aosThreadData[threadnum].virialSumNoN3[1] += virialSumY;
943 _aosThreadData[threadnum].virialSumNoN3[2] += virialSumZ;
944 }
945 }
946 }
