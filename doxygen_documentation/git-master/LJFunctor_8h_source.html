115 void AoSFunctor(Particle &i, Particle &j, bool newton3) final {
116 if (i.isDummy() or j.isDummy()) {
117 return;
118 }
119 auto sigmasquare = _sigmasquare;
120 auto epsilon24 = _epsilon24;
121 auto shift6 = _shift6;
122 if constexpr (useMixing) {
123 sigmasquare = _PPLibrary->mixingSigmaSquare(i.getTypeId(), j.getTypeId());
124 epsilon24 = _PPLibrary->mixing24Epsilon(i.getTypeId(), j.getTypeId());
125 if constexpr (applyShift) {
126 shift6 = _PPLibrary->mixingShift6(i.getTypeId(), j.getTypeId());
127 }
128 }
129 auto dr = utils::ArrayMath::sub(i.getR(), j.getR());
130 double dr2 = utils::ArrayMath::dot(dr, dr);
131
132 if (dr2 > _cutoffsquare) {
133 return;
134 }
135
136 double invdr2 = 1. / dr2;
137 double lj6 = sigmasquare * invdr2;
138 lj6 = lj6 * lj6 * lj6;
139 double lj12 = lj6 * lj6;
140 double lj12m6 = lj12 - lj6;
141 double fac = epsilon24 * (lj12 + lj12m6) * invdr2;
142 auto f = utils::ArrayMath::mulScalar(dr, fac);
143 i.addF(f);
144 if (newton3) {
145 // only if we use newton 3 here, we want to
146 j.subF(f);
147 }
148 if (calculateGlobals) {
149 auto virial = utils::ArrayMath::mul(dr, f);
150 double upot = epsilon24 * lj12m6 + shift6;
151
152 const int threadnum = autopas_get_thread_num();
153 // for non-newton3 the division is in the post-processing step.
154 if (newton3) {
155 upot *= 0.5;
156 virial = utils::ArrayMath::mulScalar(virial, (double)0.5);
157 }
158 if (i.isOwned()) {
159 _aosThreadData[threadnum].upotSum += upot;
160 _aosThreadData[threadnum].virialSum = utils::ArrayMath::add(_aosThreadData[threadnum].virialSum, virial);
161 }
162 // for non-newton3 the second particle will be considered in a separate calculation
163 if (newton3 and j.isOwned()) {
164 _aosThreadData[threadnum].upotSum += upot;
165 _aosThreadData[threadnum].virialSum = utils::ArrayMath::add(_aosThreadData[threadnum].virialSum, virial);
166 }
167 }

278 fyacc += fy;
279 fzacc += fz;

283 fyptr[j] -= fy;
284 fzptr[j] -= fz;

287 const SoAFloatPrecision virialx = drx * fx;
288 const SoAFloatPrecision virialy = dry * fy;

298 virialSumY += virialy * energyFactor;

300 }
301 }
302

304 fyptr[i] += fyacc;
305 fzptr[i] += fzacc;
306 }

308 const int threadnum = autopas_get_thread_num();

310 // we assume newton3 to be enabled in this function call, thus we multiply by two if the value of newton3 is
311 // false, since for newton3 disabled we divide by two later on.
312 factor *= newton3 ? .5 : 1.;
313 _aosThreadData[threadnum].upotSum += upotSum * factor;
314 _aosThreadData[threadnum].virialSum[0] += virialSumX * factor;
315 _aosThreadData[threadnum].virialSum[1] += virialSumY * factor;
316 _aosThreadData[threadnum].virialSum[2] += virialSumZ * factor;
317 }

589 void endTraversal(bool newton3) final {
590 if (_postProcessed) {
591 throw utils::ExceptionHandler::AutoPasException(
592 "Already postprocessed, endTraversal(bool newton3) was called twice without calling initTraversal().");
593 }
594 if (calculateGlobals) {
595 for (size_t i = 0; i < _aosThreadData.size(); ++i) {
596 _upotSum += _aosThreadData[i].upotSum;
597 _virialSum = utils::ArrayMath::add(_virialSum, _aosThreadData[i].virialSum);
598 }
599 if (not newton3) {
600 // if the newton3 optimization is disabled we have added every energy contribution twice, so we divide by 2
601 // here.
602 _upotSum *= 0.5;
603 _virialSum = utils::ArrayMath::mulScalar(_virialSum, 0.5);
604 }
605 // we have always calculated 6*upot, so we divide by 6 here!
606 _upotSum /= 6.;
607 _postProcessed = true;
608 }

666 [[maybe_unused]] auto *const __restrict typeptr1 = soa.template begin<Particle::AttributeNames::typeId>();
667 [[maybe_unused]] auto *const __restrict typeptr2 = soa.template begin<Particle::AttributeNames::typeId>();
668
669 const auto *const __restrict ownedStatePtr = soa.template begin<Particle::AttributeNames::ownershipState>();
670
671 const SoAFloatPrecision cutoffsquare = _cutoffsquare;
672 SoAFloatPrecision shift6 = _shift6;
673 SoAFloatPrecision sigmasquare = _sigmasquare;
674 SoAFloatPrecision epsilon24 = _epsilon24;
675
676 SoAFloatPrecision upotSum = 0.;
677 SoAFloatPrecision virialSumX = 0.;
678 SoAFloatPrecision virialSumY = 0.;
679 SoAFloatPrecision virialSumZ = 0.;
680
681 SoAFloatPrecision fxacc = 0;
682 SoAFloatPrecision fyacc = 0;
683 SoAFloatPrecision fzacc = 0;
684 const size_t neighborListSize = neighborList.size();
685 const size_t *const __restrict neighborListPtr = neighborList.data();
686
687 // checks whether particle i is owned.
688 const auto ownedStateI = ownedStatePtr[indexFirst];
689 if (ownedStateI == OwnershipState::dummy) {
690 return;
691 }
692
693 // this is a magic number, that should correspond to at least
694 // vectorization width*N have testet multiple sizes:
695 // 4: does not give a speedup, slower than original AoSFunctor
696 // 8: small speedup compared to AoS
697 // 12: highest speedup compared to Aos
698 // 16: smaller speedup
699 // in theory this is a variable, we could auto-tune over...
700#ifdef __AVX512F__
701 // use a multiple of 8 for avx
702 constexpr size_t vecsize = 16;
703#else
704 // for everything else 12 is faster
705 constexpr size_t vecsize = 12;
706#endif
707 size_t joff = 0;
708
709 // if the size of the verlet list is larger than the given size vecsize,
710 // we will use a vectorized version.
711 if (neighborListSize >= vecsize) {
712 alignas(64) std::array<SoAFloatPrecision, vecsize> xtmp, ytmp, ztmp, xArr, yArr, zArr, fxArr, fyArr, fzArr;
713 alignas(64) std::array<OwnershipState, vecsize> ownedStateArr{};
714 // broadcast of the position of particle i
715 for (size_t tmpj = 0; tmpj < vecsize; tmpj++) {
716 xtmp[tmpj] = xptr[indexFirst];
717 ytmp[tmpj] = yptr[indexFirst];
718 ztmp[tmpj] = zptr[indexFirst];
719 }
720 // loop over the verlet list from 0 to x*vecsize
721 for (; joff < neighborListSize - vecsize + 1; joff += vecsize) {
722 // in each iteration we calculate the interactions of particle i with
723 // vecsize particles in the neighborlist of particle i starting at
724 // particle joff
725
726 [[maybe_unused]] alignas(DEFAULT_CACHE_LINE_SIZE) std::array<SoAFloatPrecision, vecsize> sigmaSquares;
727 [[maybe_unused]] alignas(DEFAULT_CACHE_LINE_SIZE) std::array<SoAFloatPrecision, vecsize> epsilon24s;
728 [[maybe_unused]] alignas(DEFAULT_CACHE_LINE_SIZE) std::array<SoAFloatPrecision, vecsize> shift6s;
729 if constexpr (useMixing) {
730 for (size_t j = 0; j < vecsize; j++) {
731 sigmaSquares[j] = _PPLibrary->mixingSigmaSquare(typeptr1[indexFirst], typeptr2[neighborListPtr[joff + j]]);
732 epsilon24s[j] = _PPLibrary->mixing24Epsilon(typeptr1[indexFirst], typeptr2[neighborListPtr[joff + j]]);
733 if constexpr (applyShift) {
734 shift6s[j] = _PPLibrary->mixingShift6(typeptr1[indexFirst], typeptr2[neighborListPtr[joff + j]]);
735 }
736 }
737 }
738
739 // gather position of particle j
740#pragma omp simd safelen(vecsize)
741 for (size_t tmpj = 0; tmpj < vecsize; tmpj++) {
742 xArr[tmpj] = xptr[neighborListPtr[joff + tmpj]];
743 yArr[tmpj] = yptr[neighborListPtr[joff + tmpj]];
744 zArr[tmpj] = zptr[neighborListPtr[joff + tmpj]];
745 ownedStateArr[tmpj] = ownedStatePtr[neighborListPtr[joff + tmpj]];
746 }
747 // do omp simd with reduction of the interaction
748#pragma omp simd reduction(+ : fxacc, fyacc, fzacc, upotSum, virialSumX, virialSumY, virialSumZ) safelen(vecsize)
749 for (size_t j = 0; j < vecsize; j++) {
750 if constexpr (useMixing) {
751 sigmasquare = sigmaSquares[j];
752 epsilon24 = epsilon24s[j];
753 if constexpr (applyShift) {
754 shift6 = shift6s[j];
755 }
756 }
757 // const size_t j = currentList[jNeighIndex];
758
759 const auto ownedStateJ = ownedStateArr[j];
760
761 const SoAFloatPrecision drx = xtmp[j] - xArr[j];
762 const SoAFloatPrecision dry = ytmp[j] - yArr[j];
763 const SoAFloatPrecision drz = ztmp[j] - zArr[j];
764
765 const SoAFloatPrecision drx2 = drx * drx;
766 const SoAFloatPrecision dry2 = dry * dry;
767 const SoAFloatPrecision drz2 = drz * drz;
768
769 const SoAFloatPrecision dr2 = drx2 + dry2 + drz2;
770
771 // Mask away if distance is too large or any particle is a dummy.
772 // Particle ownedStateI was already checked previously.
773 const bool mask = dr2 <= cutoffsquare and ownedStateJ != OwnershipState::dummy;
774
775 const SoAFloatPrecision invdr2 = 1. / dr2;
776 const SoAFloatPrecision lj2 = sigmasquare * invdr2;
777 const SoAFloatPrecision lj6 = lj2 * lj2 * lj2;
778 const SoAFloatPrecision lj12 = lj6 * lj6;
779 const SoAFloatPrecision lj12m6 = lj12 - lj6;
780 const SoAFloatPrecision fac = mask ? epsilon24 * (lj12 + lj12m6) * invdr2 : 0.;
781
782 const SoAFloatPrecision fx = drx * fac;
783 const SoAFloatPrecision fy = dry * fac;
784 const SoAFloatPrecision fz = drz * fac;
785
786 fxacc += fx;
787 fyacc += fy;
788 fzacc += fz;
789 if (newton3) {
790 fxArr[j] = fx;
791 fyArr[j] = fy;
792 fzArr[j] = fz;
793 }
794 if (calculateGlobals) {
795 SoAFloatPrecision virialx = drx * fx;
796 SoAFloatPrecision virialy = dry * fy;
797 SoAFloatPrecision virialz = drz * fz;
798 SoAFloatPrecision upot = mask ? (epsilon24 * lj12m6 + shift6) : 0.;
799
800 SoAFloatPrecision energyFactor = (ownedStateI == OwnershipState::owned ? 1. : 0.);
801 if constexpr (newton3) {
802 energyFactor += (ownedStateJ == OwnershipState::owned ? 1. : 0.);
803 }
804 upotSum += upot * energyFactor;
805 virialSumX += virialx * energyFactor;
806 virialSumY += virialy * energyFactor;
807 virialSumZ += virialz * energyFactor;
808 }
809 }
810 // scatter the forces to where they belong, this is only needed for newton3
811 if (newton3) {
812#pragma omp simd safelen(vecsize)
813 for (size_t tmpj = 0; tmpj < vecsize; tmpj++) {
814 const size_t j = neighborListPtr[joff + tmpj];
815 fxptr[j] -= fxArr[tmpj];
816 fyptr[j] -= fyArr[tmpj];
817 fzptr[j] -= fzArr[tmpj];
818 }
819 }
820 }
821 }
822 // this loop goes over the remainder and uses no optimizations
823 for (size_t jNeighIndex = joff; jNeighIndex < neighborListSize; ++jNeighIndex) {
824 size_t j = neighborList[jNeighIndex];
825 if (indexFirst == j) continue;
826 if constexpr (useMixing) {
827 sigmasquare = _PPLibrary->mixingSigmaSquare(typeptr1[indexFirst], typeptr2[j]);
828 epsilon24 = _PPLibrary->mixing24Epsilon(typeptr1[indexFirst], typeptr2[j]);
829 if constexpr (applyShift) {
830 shift6 = _PPLibrary->mixingShift6(typeptr1[indexFirst], typeptr2[j]);
831 }
832 }
833
834 const auto ownedStateJ = ownedStatePtr[j];
835 if (ownedStateJ == OwnershipState::dummy) {
836 continue;
837 }
838
839 const SoAFloatPrecision drx = xptr[indexFirst] - xptr[j];
840 const SoAFloatPrecision dry = yptr[indexFirst] - yptr[j];
841 const SoAFloatPrecision drz = zptr[indexFirst] - zptr[j];
842
843 const SoAFloatPrecision drx2 = drx * drx;
844 const SoAFloatPrecision dry2 = dry * dry;
845 const SoAFloatPrecision drz2 = drz * drz;
846
847 const SoAFloatPrecision dr2 = drx2 + dry2 + drz2;
848
849 if (dr2 > cutoffsquare) {
850 continue;
851 }
852
853 const SoAFloatPrecision invdr2 = 1. / dr2;
854 const SoAFloatPrecision lj2 = sigmasquare * invdr2;
855 const SoAFloatPrecision lj6 = lj2 * lj2 * lj2;
856 const SoAFloatPrecision lj12 = lj6 * lj6;
857 const SoAFloatPrecision lj12m6 = lj12 - lj6;
858 const SoAFloatPrecision fac = epsilon24 * (lj12 + lj12m6) * invdr2;
859
860 const SoAFloatPrecision fx = drx * fac;
861 const SoAFloatPrecision fy = dry * fac;
862 const SoAFloatPrecision fz = drz * fac;
863
864 fxacc += fx;
865 fyacc += fy;
866 fzacc += fz;
867 if (newton3) {
868 fxptr[j] -= fx;
869 fyptr[j] -= fy;
870 fzptr[j] -= fz;
871 }
872 if (calculateGlobals) {
873 SoAFloatPrecision virialx = drx * fx;
874 SoAFloatPrecision virialy = dry * fy;
875 SoAFloatPrecision virialz = drz * fz;
876 SoAFloatPrecision upot = (epsilon24 * lj12m6 + shift6);
877
878 SoAFloatPrecision energyFactor = (ownedStateI == OwnershipState::owned ? 1. : 0.);
879 if constexpr (newton3) {
880 energyFactor += (ownedStateJ == OwnershipState::owned ? 1. : 0.);
881 }
882 upotSum += upot * energyFactor;
883 virialSumX += virialx * energyFactor;
884 virialSumY += virialy * energyFactor;
885 virialSumZ += virialz * energyFactor;
886 }
887 }
888
889 if (fxacc != 0 or fyacc != 0 or fzacc != 0) {
890 fxptr[indexFirst] += fxacc;
891 fyptr[indexFirst] += fyacc;
892 fzptr[indexFirst] += fzacc;
893 }
894
895 if (calculateGlobals) {
896 const int threadnum = autopas_get_thread_num();
897
898 SoAFloatPrecision energyMul = newton3 ? 0.5 : 1.;
899
900 _aosThreadData[threadnum].upotSum += upotSum * energyMul;
901 _aosThreadData[threadnum].virialSum[0] += virialSumX * energyMul;
902 _aosThreadData[threadnum].virialSum[1] += virialSumY * energyMul;
903 _aosThreadData[threadnum].virialSum[2] += virialSumZ * energyMul;
904 }
905 }

911 public:
912 AoSThreadData() : virialSum{0., 0., 0.}, upotSum{0.}, __remainingTo64{} {}
913 void setZero() {
914 virialSum = {0., 0., 0.};
915 upotSum = 0.;

925 };
