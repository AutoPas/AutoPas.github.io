81  using utils::ThreeDimensionalMapping::threeToOneD;
82  std::array<long, 3> index{};
83  const std::array<long, 3> signedDims = utils::ArrayUtils::static_cast_array<long>(this->_cellsPerDimension);
84 
85  // go through the six cells
86  for (long z = 0; z < 3; ++z) {
87  for (long x = 0; x < 2; ++x) {
88  index[0] = base3DIndex[0] + x;
89  index[1] = base3DIndex[1];
90  index[2] = base3DIndex[2] + z;
91 
92  bool isIn = true;
93  for (int d = 0; d < 3; ++d) {

137 template <class ParticleCell, class PairwiseFunctor, DataLayoutOption::Value dataLayout, bool useNewton3>
138 void LCC04HCPTraversal<ParticleCell, PairwiseFunctor, dataLayout, useNewton3>::traverseSingleColor(
139  std::vector<ParticleCell> &cells, int color) {
140  // determine a starting point of one of the grids
141  std::array<long, 3> startOfThisColor{}; // coordinates: {x,y,z}
142 
143  // different starting points for different colors
144  // some colors are starting outside the grid because only part of their cuboids are part of the grid
145  // this way the starting points of sticking out cuboids can be determined as well
146  switch (color) {
147  case 0:
148  startOfThisColor = {0l, 0l, 0l};
149  break;
150  case 1:
151  startOfThisColor = {-4l, 0l, 1l};
152  break;
153  case 2:
154  startOfThisColor = {-4l, 0l, -2l};
155  break;
156  case 3:
157  startOfThisColor = {-2l, 0l, -1l};
158  break;
159  default:
160  autopas::utils::ExceptionHandler::exception("LCC04HCPTraversal::traverseSingleColor: invalid color ({})", color);
161  }
162 
163  // to fix intel64 icpc compiler complaints about perfectly nested loop.
164  const long startX = startOfThisColor[0], endX = _end[0];
165  const long startY = startOfThisColor[1], endY = _end[1];
166  const long startZ = startOfThisColor[2], endZ = _end[2];
167 
168  // iterate over cartesian grid
169 #if defined(AUTOPAS_OPENMP)
170 #pragma omp for schedule(dynamic, 1) collapse(3) nowait
171 #endif
172  for (long z = startZ; z < endZ; z += 4) {
173  for (long y = startY; y < endY; y++) {
174  /* color starts every 6th column again, the +4 is needed to prevent ending too early, since it
175  will be shifted back inside the loop */
176  for (long x = startX; x < (endX + 4); x += 6) {
177  long x_index = x;
178  /* shift on x-axis according to z-value: shift two times and then go back to original x-value
179  first: no shift
180  second: -4 shift
181  third: -2 shift
182  fourth: go back to first
183  every 12th z, the shifting pattern repeats again at the origin of x without shift,
184  because z is shifted by 4 in every loop run and every third z-shift the pattern repeats
185  */
186  switch ((z - startZ) % 12 / 4) {
187  case 0:
188  break;
189  case 1:
190  x_index -= 4;
191  break;
192  case 2:
193  x_index -= 2;
194  break;
195  default:
196  break;
197  }
198  // shift x-axis every second y-row
199  if ((y - startY) % 2 != 0) {
200  x_index += 3;
201  }
202  processBasePack6(cells, {x_index, y, z});
203  }
204  }
205  }
206 }
