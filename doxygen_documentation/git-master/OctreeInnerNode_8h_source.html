34 OctreeInnerNode(std::array<double, 3> boxMin, std::array<double, 3> boxMax, OctreeNodeInterface<Particle> *parent,
35 int unsigned treeSplitThreshold, double interactionLength, double cellSizeFactor)
36 : OctreeNodeInterface<Particle>(boxMin, boxMax, parent, treeSplitThreshold, interactionLength, cellSizeFactor) {
37 using namespace autopas::utils;
38
39 // The inner node is initialized with 8 leaves.
40 const auto center = ArrayMath::mulScalar(ArrayMath::add(boxMin, boxMax), 0.5);
41 for (auto i = 0; i < _children.size(); ++i) {
42 // Subdivide the bounding box of the parent.
43 std::array<double, 3> newBoxMin = {};
44 std::array<double, 3> newBoxMax = {};
45 for (auto d = 0; d < 3; ++d) {
46 // `i`, `d` and `mask` are used to generate minimum and maximum coordinates for every octree leaf's bounding
47 // box. `i` represents a 3-bit wide number, where each bit corresponds to an axis. The following table
48 // visualizes this layout:
49 //
50 // <-- msb lsb -->
51 // +------++-----+---+---+---+
52 // | bit || ... | 2 | 1 | 0 |
53 // +------+------+---+---+---+
54 // | axis || x | y | z |
55 // +------++---------+---+---+
56 //
57 // When an axis bit is not set, a leaf is expected to range from the minimum coordinate to the center of the
58 // enclosing box on the respective axis `d`. If the axis bit is set, the leaf's region should start at the
59 // center coordinate and end at the maximum coordinate of the enclosing box.
60 //
61 // `mask` is used to extract the individual axis components from `i`: Using the shift operation, `mask` can
62 // either become 0b100, 0b010 or 0b001 since `d` ranges from 0 to 2 inclusively. This covers all available axis
63 // in 3 dimensions.
64 const auto mask = 4 >> d;
65 newBoxMin[d] = (not(i & mask)) ? boxMin[d] : center[d];
66 newBoxMax[d] = (not(i & mask)) ? center[d] : boxMax[d];
67 }
68
69 // Assign new leaves as the children.
70 _children[i] = std::make_unique<OctreeLeafNode<Particle>>(newBoxMin, newBoxMax, this, treeSplitThreshold,
71 interactionLength, cellSizeFactor);
72 }
73 }
