91 [[nodiscard]] std::vector<Particle> collectLeavingParticlesFromBuffer(bool insertOwnedParticlesToContainer) {
92 const auto &boxMin = _containerSelector.getCurrentContainer().getBoxMin();
93 const auto &boxMax = _containerSelector.getCurrentContainer().getBoxMax();
94 std::vector<Particle> leavingBufferParticles{};
95 for (auto &cell : _particleBuffer) {
96 auto &buffer = cell._particles;
97 if (insertOwnedParticlesToContainer) {
98 for (const auto &p : buffer) {
99 if (p.isDummy()) {
100 continue;
101 }
102 if (utils::inBox(p.getR(), boxMin, boxMax)) {
103 _containerSelector.getCurrentContainer().addParticle(p);
104 } else {
105 leavingBufferParticles.push_back(p);
106 }
107 }
108 buffer.clear();
109 } else {
110 for (auto iter = buffer.begin(); iter != buffer.end();) {
111 auto &p = *iter;
112
113 auto fastRemoveP = [&]() {
114 // Fast remove of particle, i.e., swap with last entry && pop.
115 std::swap(p, buffer.back());
116 buffer.pop_back();
117 // Do not increment the iter afterwards!
118 };
119 if (p.isDummy()) {
120 // We remove dummies!
121 fastRemoveP();
122 }
123 if (utils::notInBox(p.getR(), boxMin, boxMax)) {
124 leavingBufferParticles.push_back(p);
125 fastRemoveP();
126 } else {
127 ++iter;
128 }
129 }
130 }
131 }
132 return leavingBufferParticles;

138 [[nodiscard]] std::vector<Particle> updateContainer() {
139 bool doDataStructureUpdate = not neighborListsAreValid();

141 if (doDataStructureUpdate) {

150
151 // Substract the amount of leaving particles from the number of owned particles.

165 std::vector<Particle> resizeBox(const std::array<double, 3> &boxMin, const std::array<double, 3> &boxMax) {
166 using namespace autopas::utils::ArrayMath::literals;
167 const auto &oldMin = _containerSelector.getCurrentContainer().getBoxMin();
168 const auto &oldMax = _containerSelector.getCurrentContainer().getBoxMax();
169
170 // if nothing changed do nothing
171 if (oldMin == boxMin and oldMax == boxMax) {
172 return {};
173 }
174
175 // sanity check that new size is actually positive
176 for (size_t i = 0; i < boxMin.size(); ++i) {
177 if (boxMin[i] >= boxMax[i]) {
178 utils::ExceptionHandler::exception(
179 "New box size in dimension {} is not positive!\nboxMin[{}] = {}\nboxMax[{}] = {}", i, i, boxMin[i], i,
180 boxMax[i]);
181 }
182 }
183
184 // warn if domain changes too drastically
185 const auto newLength = boxMax - boxMin;
186 const auto oldLength = oldMax - oldMin;
187 const auto relDiffLength = newLength / oldLength;
188 for (size_t i = 0; i < newLength.size(); ++i) {
189 // warning threshold is set arbitrary and up for change if needed
190 if (relDiffLength[i] > 1.3 or relDiffLength[i] < 0.7) {
191 AutoPasLog(WARN,
192 "LogicHandler.resize(): Domain size changed drastically in dimension {}! Gathered AutoTuning "
193 "information might not be applicable anymore!\n"
194 "Size old box : {}\n"
195 "Size new box : {}\n"
196 "Relative diff: {}",
197 i, utils::ArrayUtils::to_string(oldLength), utils::ArrayUtils::to_string(newLength),
198 utils::ArrayUtils::to_string(relDiffLength));
199 }
200 }
201
202 // check all particles
203 std::vector<Particle> particlesNowOutside;
204 for (auto pIter = _containerSelector.getCurrentContainer().begin(); pIter.isValid(); ++pIter) {
205 // make sure only owned ones are present
206 if (not pIter->isOwned()) {
207 utils::ExceptionHandler::exception(
208 "LogicHandler::resizeBox() encountered non owned particle. "
209 "When calling resizeBox() these should be already deleted. "
210 "This could be solved by calling updateContainer() before resizeBox().");
211 }
212 // owned particles that are now outside are removed from the container and returned
213 if (not utils::inBox(pIter->getR(), boxMin, boxMax)) {
214 particlesNowOutside.push_back(*pIter);
215 decreaseParticleCounter(*pIter);
216 internal::markParticleAsDeleted(*pIter);
217 }
218 }
219
220 // actually resize the container
221 _containerSelector.resizeBox(boxMin, boxMax);
222 // The container might have changed sufficiently enough so that we need more or less spacial locks
223 const auto boxLength = boxMax - boxMin;
224 const auto interactionLengthInv = 1. / (_containerSelector.getCurrentContainer().getInteractionLength());
225 initSpacialLocks(boxLength, interactionLengthInv);
226
227 // Set this flag, s.t., the container is rebuilt!
228 _neighborListsAreValid.store(false, std::memory_order_relaxed);
229
230 return particlesNowOutside;
231 }

339 std::tuple<bool, bool> deleteParticleFromBuffers(Particle &particle) {
340 // find the buffer the particle belongs to
341 auto &bufferCollection = particle.isOwned() ? _particleBuffer : _haloParticleBuffer;
342 for (auto &cell : bufferCollection) {
343 auto &buffer = cell._particles;
344 // if the address of the particle is between start and end of the buffer it is in this buffer
345 if (not buffer.empty() and &(buffer.front()) <= &particle and &particle <= &(buffer.back())) {
346 const bool isRearParticle = &particle == &buffer.back();
347 // swap-delete
348 particle = buffer.back();
349 buffer.pop_back();
350 return {true, not isRearParticle};
351 }
352 }
353 return {false, true};
354 }

522 void setParticleBuffers(const std::vector<FullParticleCell<Particle>> &particleBuffers,
523 const std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
524 auto exchangeBuffer = [](const auto &newBuffers, auto &oldBuffers, auto &particleCounter) {
525 // sanity check
526 if (oldBuffers.size() < newBuffers.size()) {
527 autopas::utils::ExceptionHandler::exception(
528 "The number of new buffers ({}) is larger than number of existing buffers ({})!", newBuffers.size(),
529 oldBuffers.size());
530 }
531
532 // we will clear the old buffers so subtract the particles from the counters.
533 const auto numParticlesInOldBuffers =
534 std::transform_reduce(oldBuffers.begin(), std::next(oldBuffers.begin(), newBuffers.size()), 0, std::plus<>(),
535 [](const auto &cell) { return cell.numParticles(); });
536 particleCounter.fetch_sub(numParticlesInOldBuffers, std::memory_order_relaxed);
537
538 // clear the old buffers and copy the content of the new buffers over.
539 size_t numParticlesInNewBuffers = 0;
540 for (size_t i = 0; i < newBuffers.size(); ++i) {
541 oldBuffers[i].clear();
542 for (const auto &p : newBuffers[i]) {
543 ++numParticlesInNewBuffers;
544 oldBuffers[i].addParticle(p);
545 }
546 }
547 // update the counters.
548 particleCounter.fetch_add(numParticlesInNewBuffers, std::memory_order_relaxed);
549 };
550
551 exchangeBuffer(particleBuffers, _particleBuffer, _numParticlesOwned);
552 exchangeBuffer(haloParticleBuffers, _haloParticleBuffer, _numParticlesHalo);
553 }

851void LogicHandler<Particle>::doRemainderTraversal(PairwiseFunctor *f, ContainerType &container,
852 std::vector<FullParticleCell<Particle>> &particleBuffers,
853 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
854 // Sanity check. If this is violated feel free to add some logic here that adapts the number of locks.
855 if (_bufferLocks.size() < particleBuffers.size()) {
856 utils::ExceptionHandler::exception("Not enough locks for non-halo buffers! Num Locks: {}, Buffers: {}",
857 _bufferLocks.size(), particleBuffers.size());
858 }
859
860 // Balance buffers. This makes processing them with static scheduling quite efficient.
861 // Also, if particles were not inserted in parallel, this enables us to process them in parallel now.
862 // Cost is at max O(2N) worst O(N) per buffer collection and negligible compared to interacting them.
863 auto cellToVec = [](auto &cell) -> std::vector<Particle> & { return cell._particles; };
864 utils::ArrayUtils::balanceVectors(particleBuffers, cellToVec);
865 utils::ArrayUtils::balanceVectors(haloParticleBuffers, cellToVec);
866
867 // The following part performes the main remainder traversal. The actual calculation is done in 4 steps carried out in
868 // three helper functions.
869
870 // only activate time measurements if it will actually be logged
871#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
872 autopas::utils::Timer timerBufferContainer;
873 autopas::utils::Timer timerPBufferPBuffer;
874 autopas::utils::Timer timerPBufferHBuffer;
875
876 timerBufferContainer.start();
877#endif
878 // steps 1 & 2. particleBuffer with all close particles in container and haloParticleBuffer with owned, close
879 // particles in container
880 remainderHelperBufferContainer<newton3>(f, container, particleBuffers, haloParticleBuffers);
881
882#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
883 timerBufferContainer.stop();
884 timerPBufferPBuffer.start();
885#endif
886
887 // step 3. particleBuffer with itself and all other buffers
888 remainderHelperBufferBuffer<newton3>(f, particleBuffers, haloParticleBuffers);
889
890#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
891 timerPBufferPBuffer.stop();
892 timerPBufferHBuffer.start();
893#endif
894
895 // step 4. particleBuffer with haloParticleBuffer
896 remainderHelperBufferHaloBuffer<newton3>(f, particleBuffers, haloParticleBuffers);
897
898#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
899 timerPBufferHBuffer.stop();
900#endif
901
902 // unpack particle SoAs. Halo data is not interesting
903 for (auto &buffer : particleBuffers) {
904 f->SoAExtractor(buffer, buffer._particleSoABuffer, 0);
905 }
906
907 AutoPasLog(TRACE, "Timer Buffers <-> Container (1+2): {}", timerBufferContainer.getTotalTime());
908 AutoPasLog(TRACE, "Timer PBuffers<-> PBuffer ( 3): {}", timerPBufferPBuffer.getTotalTime());
909 AutoPasLog(TRACE, "Timer PBuffers<-> HBuffer ( 4): {}", timerPBufferHBuffer.getTotalTime());
910
911 // Note: haloParticleBuffer with itself is NOT needed, as interactions between halo particles are unneeded!
912}

914template <class Particle>
915template <bool newton3, class ContainerType, class PairwiseFunctor>
916void LogicHandler<Particle>::remainderHelperBufferContainer(
917 PairwiseFunctor *f, ContainerType &container, std::vector<FullParticleCell<Particle>> &particleBuffers,
918 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
919 using autopas::utils::ArrayUtils::static_cast_copy_array;
920 using namespace autopas::utils::ArrayMath::literals;
921
922 const auto haloBoxMin = container.getBoxMin() - container.getInteractionLength();
923 const auto interactionLengthInv = 1. / container.getInteractionLength();
924
925 const double cutoff = container.getCutoff();
926#ifdef AUTOPAS_OPENMP
927// one halo and particle buffer pair per thread
928#pragma omp parallel for schedule(static, 1), shared(f, _spacialLocks, haloBoxMin, interactionLengthInv)
929#endif
930 for (int bufferId = 0; bufferId < particleBuffers.size(); ++bufferId) {
931 auto &particleBuffer = particleBuffers[bufferId];
932 auto &haloParticleBuffer = haloParticleBuffers[bufferId];
933 // 1. particleBuffer with all close particles in container
934 for (auto &&p1 : particleBuffer) {
935 const auto pos = p1.getR();
936 const auto min = pos - cutoff;
937 const auto max = pos + cutoff;
938 container.forEachInRegion(
939 [&](auto &p2) {
940 const auto lockCoords = static_cast_copy_array<size_t>((p2.getR() - haloBoxMin) * interactionLengthInv);
941 if constexpr (newton3) {
942 const std::lock_guard<std::mutex> lock(*_spacialLocks[lockCoords[0]][lockCoords[1]][lockCoords[2]]);
943 f->AoSFunctor(p1, p2, true);
944 } else {
945 f->AoSFunctor(p1, p2, false);
946 // no need to calculate force enacted on a halo
947 if (not p2.isHalo()) {
948 const std::lock_guard<std::mutex> lock(*_spacialLocks[lockCoords[0]][lockCoords[1]][lockCoords[2]]);
949 f->AoSFunctor(p2, p1, false);
950 }
951 }
952 },
953 min, max, IteratorBehavior::ownedOrHalo);
954 }
955
956 // 2. haloParticleBuffer with owned, close particles in container
957 for (auto &&p1halo : haloParticleBuffer) {
958 const auto pos = p1halo.getR();
959 const auto min = pos - cutoff;
960 const auto max = pos + cutoff;
961 container.forEachInRegion(
962 [&](auto &p2) {
963 const auto lockCoords = static_cast_copy_array<size_t>((p2.getR() - haloBoxMin) * interactionLengthInv);
964 // No need to apply anything to p1halo
965 // -> AoSFunctor(p1, p2, false) not needed as it neither adds force nor Upot (potential energy)
966 // -> newton3 argument needed for correct globals
967 const std::lock_guard<std::mutex> lock(*_spacialLocks[lockCoords[0]][lockCoords[1]][lockCoords[2]]);
968 f->AoSFunctor(p2, p1halo, newton3);
969 },
970 min, max, IteratorBehavior::owned);
971 }
972 }
973}

975template <class Particle>
976template <bool newton3, class PairwiseFunctor>
977void LogicHandler<Particle>::remainderHelperBufferBuffer(PairwiseFunctor *f,
978 std::vector<FullParticleCell<Particle>> &particleBuffers,
979 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
980 // All (halo-)buffer interactions shall happen vectorized, hence, load all buffer data into SoAs
981 for (auto &buffer : particleBuffers) {
982 f->SoALoader(buffer, buffer._particleSoABuffer, 0);
983 }
984 for (auto &buffer : haloParticleBuffers) {
985 f->SoALoader(buffer, buffer._particleSoABuffer, 0);
986 }
987
988#ifdef AUTOPAS_OPENMP
989#pragma omp parallel
990#endif
991 {
992 // For buffer interactions where bufferA == bufferB we can always enable newton3. For all interactions between
993 // different buffers we turn newton3 always off, which ensures that only one thread at a time is writing to a
994 // buffer. This saves expensive locks.
995#ifdef AUTOPAS_OPENMP
996 // we can not use collapse here without locks, otherwise races would occur.
997#pragma omp for
998#endif
999 for (size_t i = 0; i < particleBuffers.size(); ++i) {
1000 for (size_t jj = 0; jj < particleBuffers.size(); ++jj) {
1001 auto *particleBufferSoAA = &particleBuffers[i]._particleSoABuffer;
1002 const auto j = (i + jj) % particleBuffers.size();
1003 if (i == j) {
1004 f->SoAFunctorSingle(*particleBufferSoAA, true);
1005 } else {
1006 auto *particleBufferSoAB = &particleBuffers[j]._particleSoABuffer;
1007 f->SoAFunctorPair(*particleBufferSoAA, *particleBufferSoAB, false);
1008 }
1009 }
1010 }
1011 }
1012}

1016void LogicHandler<Particle>::remainderHelperBufferHaloBuffer(
1017 PairwiseFunctor *f, std::vector<FullParticleCell<Particle>> &particleBuffers,
1018 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
1019#ifdef AUTOPAS_OPENMP
1020 // Here, phase / color based parallelism turned out to be more efficient than tasks
1021#pragma omp parallel
1022#endif
1023 for (int interactionOffset = 0; interactionOffset < haloParticleBuffers.size(); ++interactionOffset) {
1024#ifdef AUTOPAS_OPENMP
1025#pragma omp for
1026#endif

1028 auto &particleBufferSoA = particleBuffers[i]._particleSoABuffer;
1029 auto &haloBufferSoA =
1030 haloParticleBuffers[(i + interactionOffset) % haloParticleBuffers.size()]._particleSoABuffer;
1031 f->SoAFunctorPair(particleBufferSoA, haloBufferSoA, false);
1032 }
1033 }

1036template <typename Particle>
1037template <class Functor>
1038std::tuple<Configuration, std::unique_ptr<TraversalInterface>, bool> LogicHandler<Particle>::selectConfiguration(
1039 Functor &functor) {
1040 bool stillTuning = false;
1041 Configuration configuration{};
1042 std::optional<std::unique_ptr<TraversalInterface>> traversalPtrOpt{};
1043 // if this iteration is not relevant take the same algorithm config as before.
1044 if (not functor.isRelevantForTuning()) {
1045 stillTuning = false;
1046 configuration = _autoTuner.getCurrentConfig();
1047 const auto &container = _containerSelector.getCurrentContainer();
1048 traversalPtrOpt = autopas::utils::withStaticCellType<Particle>(
1049 container.getParticleCellTypeEnum(), [&](const auto &particleCellDummy) -> decltype(traversalPtrOpt) {
1050 // Can't make this unique_ptr const otherwise we can't move it later.
1051 auto traversalPtr =
1052 TraversalSelector<std::decay_t<decltype(particleCellDummy)>>::template generateTraversal<Functor>(
1053 configuration.traversal, functor, container.getTraversalSelectorInfo(), configuration.dataLayout,
1054 configuration.newton3);
1055 if (traversalPtr->isApplicable()) {
1056 return std::optional{std::move(traversalPtr)};
1057 } else {
1058 return std::nullopt;
1059 }
1060 });
1061 } else {
1062 const auto [needsLiveInfo, needsHomogeneityAndMaxDensity] = _autoTuner.prepareIteration();
1063
1064 if (needsLiveInfo) {
1065 LiveInfo info{};
1066 info.gather(_containerSelector.getCurrentContainer(), functor, _neighborListRebuildFrequency);
1067 _autoTuner.getTuningStrategy().receiveLiveInfo(info);
1068 }
1069 if (needsHomogeneityAndMaxDensity) {
1070 utils::Timer timerCalculateHomogeneity;
1071 timerCalculateHomogeneity.start();
1072 const auto &container = _containerSelector.getCurrentContainer();
1073 const auto [homogeneity, maxDensity] =
1074 autopas::utils::calculateHomogeneityAndMaxDensity(container, container.getBoxMin(), container.getBoxMax());
1075 timerCalculateHomogeneity.stop();
1076 _autoTuner.addHomogeneityAndMaxDensity(homogeneity, maxDensity, timerCalculateHomogeneity.getTotalTime());
1077 }
1078
1079 std::tie(configuration, stillTuning) = _autoTuner.getNextConfig();
1080
1081 // loop as long as we don't get a valid configuration
1082 bool rejectIndefinitely = false;
1083 while (true) {
1084 // applicability check also sets the container
1085 std::tie(traversalPtrOpt, rejectIndefinitely) = isConfigurationApplicable(configuration, functor);
1086 if (traversalPtrOpt.has_value()) {
1087 break;
1088 }
1089 // if no config is left after rejecting this one an exception is thrown here.
1090 std::tie(configuration, stillTuning) = _autoTuner.rejectConfig(configuration, rejectIndefinitely);
1091 }
1092 }
1093
1094 return {configuration, std::move(traversalPtrOpt.value()), stillTuning};
1095}

1098template <class Functor>
1099bool LogicHandler<Particle>::iteratePairwisePipeline(Functor *functor) {
1101 utils::Timer tuningTimer;
1102 tuningTimer.start();
1103 const auto [configuration, traversalPtr, stillTuning] = selectConfiguration(*functor);
1104 tuningTimer.stop();
1105 AutoPasLog(DEBUG, "Selecting a configuration took {} ns.", tuningTimer.getTotalTime());
1106 _autoTuner.logIteration(configuration, stillTuning, tuningTimer.getTotalTime());
1107
1109 AutoPasLog(DEBUG, "Iterating with configuration: {} tuning: {}", configuration.toString(), stillTuning);
1110 const IterationMeasurements measurements = iteratePairwise(*functor, *traversalPtr);
1111
1113 auto bufferSizeListing = [](const auto &buffers) -> std::string {
1114 std::stringstream ss;
1115 size_t sum = 0;
1116 for (const auto &buffer : buffers) {
1117 ss << buffer.numParticles() << ", ";
1118 sum += buffer.numParticles();
1119 }
1120 ss << " Total: " << sum;
1121 return ss.str();
1122 };
1123 AutoPasLog(TRACE, "particleBuffer size : {}", bufferSizeListing(_particleBuffer));
1124 AutoPasLog(TRACE, "haloParticleBuffer size : {}", bufferSizeListing(_haloParticleBuffer));
1125 AutoPasLog(DEBUG, "Container::iteratePairwise took {} ns", measurements.timeIteratePairwise);
1126 AutoPasLog(DEBUG, "RemainderTraversal took {} ns", measurements.timeRemainderTraversal);
1127 AutoPasLog(DEBUG, "RebuildNeighborLists took {} ns", measurements.timeRebuild);
1128 AutoPasLog(DEBUG, "Container::iteratePairwise took {} ns", measurements.timeTotal);
1129 if (measurements.energyMeasurementsPossible) {
1130 AutoPasLog(DEBUG, "Energy Consumption: Psys: {} Joules Pkg: {} Joules Ram: {} Joules", measurements.energyPsys,
1131 measurements.energyPkg, measurements.energyRam);
1132 }
1133 _iterationLogger.logIteration(configuration, _iteration, stillTuning, measurements.timeIteratePairwise,
1134 measurements.timeRemainderTraversal, measurements.timeRebuild, measurements.timeTotal,
1135 tuningTimer.getTotalTime(), measurements.energyPsys, measurements.energyPkg,
1136 measurements.energyRam);
1137
1139 // if this was a major iteration add measurements and bump counters
1140 if (functor->isRelevantForTuning()) {
1141 if (stillTuning) {
1142 switch (_autoTuner.getTuningMetric()) {
1143 case TuningMetricOption::time:
1144 _autoTuner.addMeasurement(measurements.timeTotal, not neighborListsAreValid());
1145 break;
1146 case TuningMetricOption::energy:
1147 _autoTuner.addMeasurement(measurements.energyTotal, not neighborListsAreValid());
1148 break;
1149 }
1150 } else {
1151 AutoPasLog(TRACE, "Skipping adding of sample because functor is not marked relevant.");
1152 }
1153
1154 // this function depends on LogicHandler's and the AutoTuner's iteration counters,
1155 // that should not have been updated yet.
1156 if (not neighborListsAreValid() /*we have done a rebuild now*/) {
1157 // list is now valid
1158 _neighborListsAreValid.store(true, std::memory_order_relaxed);
1159 _stepsSinceLastListRebuild = 0;
1160 }
1161 ++_stepsSinceLastListRebuild;
1162
1163 _autoTuner.bumpIterationCounters();
1164 }
1165 return stillTuning;
1166}

1168template <typename Particle>
1169template <class PairwiseFunctor>
1170std::tuple<std::optional<std::unique_ptr<TraversalInterface>>, bool> LogicHandler<Particle>::isConfigurationApplicable(
1171 const Configuration &conf, PairwiseFunctor &pairwiseFunctor) {
1172 // Check if the container supports the traversal
1173 const auto allContainerTraversals = compatibleTraversals::allCompatibleTraversals(conf.container);
1174 if (allContainerTraversals.find(conf.traversal) == allContainerTraversals.end()) {
1175 AutoPasLog(DEBUG, "Configuration rejected: Container {} does not support the traversal {}.", conf.container,
1176 conf.traversal);
1177 return {std::nullopt, true};
1178 }
1179
1180 // Check if the required Newton 3 mode is supported by the functor
1181 if ((conf.newton3 == Newton3Option::enabled and not pairwiseFunctor.allowsNewton3()) or
1182 (conf.newton3 == Newton3Option::disabled and not pairwiseFunctor.allowsNonNewton3())) {
1183 AutoPasLog(DEBUG, "Configuration rejected: The functor doesn't support Newton 3 {}!", conf.newton3);
1184 return {std::nullopt, true};
1185 }
1186
1187 // Check if the traversal is applicable to the current state of the container
1188 _containerSelector.selectContainer(
1189 conf.container,
1190 ContainerSelectorInfo(conf.cellSizeFactor,
1191 _containerSelector.getCurrentContainer().getVerletSkin() / _neighborListRebuildFrequency,
1192 _neighborListRebuildFrequency, _verletClusterSize, conf.loadEstimator));
1193 const auto &container = _containerSelector.getCurrentContainer();
1194 const auto traversalInfo = container.getTraversalSelectorInfo();
1195
1196 auto traversalPtrOpt = autopas::utils::withStaticCellType<Particle>(
1197 container.getParticleCellTypeEnum(),
1198 [&](const auto &particleCellDummy) -> std::optional<std::unique_ptr<TraversalInterface>> {
1199 // Can't make this unique_ptr const otherwise we can't move it later.
1200 auto traversalPtr =
1201 TraversalSelector<std::decay_t<decltype(particleCellDummy)>>::template generateTraversal<PairwiseFunctor>(
1202 conf.traversal, pairwiseFunctor, traversalInfo, conf.dataLayout, conf.newton3);
1203 if (traversalPtr->isApplicable()) {
1204 return std::optional{std::move(traversalPtr)};
1205 } else {
1206 return std::nullopt;
1207 }
1208 });
1209 return {std::move(traversalPtrOpt), false};
1210}
