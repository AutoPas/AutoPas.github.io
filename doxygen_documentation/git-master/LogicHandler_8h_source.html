108 [[nodiscard]] std::vector<Particle> collectLeavingParticlesFromBuffer(bool insertOwnedParticlesToContainer) {
109 const auto &boxMin = _containerSelector.getCurrentContainer().getBoxMin();
110 const auto &boxMax = _containerSelector.getCurrentContainer().getBoxMax();
111 std::vector<Particle> leavingBufferParticles{};
112 for (auto &cell : _particleBuffer) {
113 auto &buffer = cell._particles;
114 if (insertOwnedParticlesToContainer) {
115 // Can't be const because we potentially modify ownership before re-adding
116 for (auto &p : buffer) {
117 if (p.isDummy()) {
118 continue;
119 }
120 if (utils::inBox(p.getR(), boxMin, boxMax)) {
121 p.setOwnershipState(OwnershipState::owned);
122 _containerSelector.getCurrentContainer().addParticle(p);
123 } else {
124 leavingBufferParticles.push_back(p);
125 }
126 }
127 buffer.clear();
128 } else {
129 for (auto iter = buffer.begin(); iter < buffer.end();) {
130 auto &p = *iter;
131
132 auto fastRemoveP = [&]() {
133 // Fast remove of particle, i.e., swap with last entry && pop.
134 std::swap(p, buffer.back());
135 buffer.pop_back();
136 // Do not increment the iter afterward!
137 };
138 if (p.isDummy()) {
139 // We remove dummies!
140 fastRemoveP();
141 // In case we swapped a dummy here, don't increment the iterator and do another iteration to check again.
142 continue;
143 }
144 // if p was a dummy a new particle might now be at the memory location of p so we need to check that.
145 // We also just might have deleted the last particle in the buffer in that case the inBox check is meaningless
146 if (not buffer.empty() and utils::notInBox(p.getR(), boxMin, boxMax)) {
147 leavingBufferParticles.push_back(p);
148 fastRemoveP();
149 } else {
150 ++iter;
151 }
152 }
153 }
154 }
155 return leavingBufferParticles;

207 std::vector<Particle> resizeBox(const std::array<double, 3> &boxMin, const std::array<double, 3> &boxMax) {
208 using namespace autopas::utils::ArrayMath::literals;
209 const auto &oldMin = _containerSelector.getCurrentContainer().getBoxMin();
210 const auto &oldMax = _containerSelector.getCurrentContainer().getBoxMax();
211
212 // if nothing changed do nothing
213 if (oldMin == boxMin and oldMax == boxMax) {
214 return {};
215 }
216
217 // sanity check that new size is actually positive
218 for (size_t i = 0; i < boxMin.size(); ++i) {
219 if (boxMin[i] >= boxMax[i]) {
220 utils::ExceptionHandler::exception(
221 "New box size in dimension {} is not positive!\nboxMin[{}] = {}\nboxMax[{}] = {}", i, i, boxMin[i], i,
222 boxMax[i]);
223 }
224 }
225
226 // warn if domain changes too drastically
227 const auto newLength = boxMax - boxMin;
228 const auto oldLength = oldMax - oldMin;
229 const auto relDiffLength = newLength / oldLength;
230 for (size_t i = 0; i < newLength.size(); ++i) {
231 // warning threshold is set arbitrary and up for change if needed
232 if (relDiffLength[i] > 1.3 or relDiffLength[i] < 0.7) {
233 AutoPasLog(WARN,
234 "LogicHandler.resize(): Domain size changed drastically in dimension {}! Gathered AutoTuning "
235 "information might not be applicable anymore!\n"
236 "Size old box : {}\n"
237 "Size new box : {}\n"
238 "Relative diff: {}",
239 i, utils::ArrayUtils::to_string(oldLength), utils::ArrayUtils::to_string(newLength),
240 utils::ArrayUtils::to_string(relDiffLength));
241 }
242 }
243
244 // check all particles
245 std::vector<Particle> particlesNowOutside;
246 for (auto pIter = _containerSelector.getCurrentContainer().begin(); pIter.isValid(); ++pIter) {
247 // make sure only owned ones are present
248 if (not pIter->isOwned()) {
249 utils::ExceptionHandler::exception(
250 "LogicHandler::resizeBox() encountered non owned particle. "
251 "When calling resizeBox() these should be already deleted. "
252 "This could be solved by calling updateContainer() before resizeBox().");
253 }
254 // owned particles that are now outside are removed from the container and returned
255 if (not utils::inBox(pIter->getR(), boxMin, boxMax)) {
256 particlesNowOutside.push_back(*pIter);
257 decreaseParticleCounter(*pIter);
258 internal::markParticleAsDeleted(*pIter);
259 }
260 }
261
262 // actually resize the container
263 _containerSelector.resizeBox(boxMin, boxMax);
264 // The container might have changed sufficiently enough so that we need more or less spacial locks
265 const auto boxLength = boxMax - boxMin;
266 const auto interactionLengthInv = 1. / (_containerSelector.getCurrentContainer().getInteractionLength());
267 initSpacialLocks(boxLength, interactionLengthInv);
268
269 // Set this flag, s.t., the container is rebuilt!
270 _neighborListsAreValid.store(false, std::memory_order_relaxed);
271
272 return particlesNowOutside;
273 }

385 // find the buffer the particle belongs to
386 auto &bufferCollection = particle.isOwned() ? _particleBuffer : _haloParticleBuffer;
387 for (auto &cell : bufferCollection) {
388 auto &buffer = cell._particles;
389 // if the address of the particle is between start and end of the buffer it is in this buffer
390 if (not buffer.empty() and &(buffer.front()) <= &particle and &particle <= &(buffer.back())) {
391 const bool isRearParticle = &particle == &buffer.back();
392 // swap-delete
393 particle = buffer.back();
394 buffer.pop_back();
395 return {true, not isRearParticle};
396 }

664 void initSpacialLocks(const std::array<double, 3> &boxLength, double interactionLengthInv) {
665 using namespace autopas::utils::ArrayMath::literals;
666 using autopas::utils::ArrayMath::ceil;
667 using autopas::utils::ArrayUtils::static_cast_copy_array;
668
669 // The maximum number of spatial locks is capped at 1e6.
670 // This limit is chosen more or less arbitrary. It is big enough so that our regular MD simulations
671 // fall well within it and small enough so that no memory issues arise.
672 // There were no rigorous tests for an optimal number of locks.
673 // Without this cap, very large domains (or tiny cutoffs) would generate an insane number of locks,
674 // that could blow up the memory.
675 constexpr size_t maxNumSpacialLocks{1000000};
676
677 // One lock per interaction length or less if this would generate too many.
678 const std::array<size_t, 3> locksPerDim = [&]() {
679 // First naively calculate the number of locks if we simply take the desired cell length.
680 // Ceil because both decisions are possible, and we are generous gods.
681 const std::array<size_t, 3> locksPerDimNaive =
682 static_cast_copy_array<size_t>(ceil(boxLength * interactionLengthInv));
683 const auto totalLocksNaive =
684 std::accumulate(locksPerDimNaive.begin(), locksPerDimNaive.end(), 1ul, std::multiplies<>());
685 // If the number of locks is within the limits everything is fine and we can return.
686 if (totalLocksNaive <= maxNumSpacialLocks) {
687 return locksPerDimNaive;
688 } else {
689 // If the number of locks grows too large, calculate the locks per dimension proportionally to the side lengths.
690 // Calculate side length relative to dimension 0.
691 const std::array<double, 3> boxSideProportions = {
692 1.,
693 boxLength[0] / boxLength[1],
694 boxLength[0] / boxLength[2],
695 };
696 // With this, calculate the number of locks the first dimension should receive.
697 const auto prodProportions =
698 std::accumulate(boxSideProportions.begin(), boxSideProportions.end(), 1., std::multiplies<>());
699 // Needs floor, otherwise we exceed the limit.
700 const auto locksInFirstDimFloat = std::floor(std::cbrt(maxNumSpacialLocks * prodProportions));
701 // From this and the proportions relative to the first dimension, we can calculate the remaining number of locks
702 const std::array<size_t, 3> locksPerDimLimited = {
703 static_cast<size_t>(locksInFirstDimFloat), // omitted div by 1
704 static_cast<size_t>(locksInFirstDimFloat / boxSideProportions[1]),
705 static_cast<size_t>(locksInFirstDimFloat / boxSideProportions[2]),
706 };
707 return locksPerDimLimited;
708 }
709 }();
710 _spacialLocks.resize(locksPerDim[0]);
711 for (auto &lockVecVec : _spacialLocks) {
712 lockVecVec.resize(locksPerDim[1]);
713 for (auto &lockVec : lockVecVec) {
714 lockVec.resize(locksPerDim[2]);
715 for (auto &lockPtr : lockVec) {
716 if (not lockPtr) {
717 lockPtr = std::make_unique<std::mutex>();
718 }
719 }
720 }
721 }

1117 const std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
1118 auto exchangeBuffer = [](const auto &newBuffers, auto &oldBuffers, auto &particleCounter) {
1119 // sanity check
1120 if (oldBuffers.size() < newBuffers.size()) {
1121 autopas::utils::ExceptionHandler::exception(
1122 "The number of new buffers ({}) is larger than number of existing buffers ({})!", newBuffers.size(),
1123 oldBuffers.size());
1124 }
1125
1126 // we will clear the old buffers so subtract the particles from the counters.
1127 const auto numParticlesInOldBuffers =
1128 std::transform_reduce(oldBuffers.begin(), std::next(oldBuffers.begin(), newBuffers.size()), 0, std::plus<>(),
1129 [](const auto &cell) { return cell.size(); });
1130 particleCounter.fetch_sub(numParticlesInOldBuffers, std::memory_order_relaxed);
1131
1132 // clear the old buffers and copy the content of the new buffers over.
1133 size_t numParticlesInNewBuffers = 0;
1134 for (size_t i = 0; i < newBuffers.size(); ++i) {
1135 oldBuffers[i].clear();
1136 for (const auto &p : newBuffers[i]) {
1137 ++numParticlesInNewBuffers;
1138 oldBuffers[i].addParticle(p);
1139 }
1140 }
1141 // update the counters.
1142 particleCounter.fetch_add(numParticlesInNewBuffers, std::memory_order_relaxed);
1143 };
1144
1145 exchangeBuffer(particleBuffers, _particleBuffer, _numParticlesOwned);
1146 exchangeBuffer(haloParticleBuffers, _haloParticleBuffer, _numParticlesHalo);

1156template <class Functor>
1157IterationMeasurements LogicHandler<Particle>::computeInteractions(Functor &functor, TraversalInterface &traversal) {
1158 // Helper to derive the Functor type at compile time
1159 constexpr auto interactionType = [] {
1160 if (utils::isPairwiseFunctor<Functor>()) {
1161 return InteractionTypeOption::pairwise;
1162 } else if (utils::isTriwiseFunctor<Functor>()) {
1163 return InteractionTypeOption::triwise;
1164 } else {
1165 utils::ExceptionHandler::exception(
1166 "LogicHandler::computeInteractions(): Functor is not valid. Only pairwise and triwise functors are "
1167 "supported. "
1168 "Please use a functor derived from "
1169 "PairwiseFunctor or TriwiseFunctor.");
1170 }
1171 }();
1172
1173 const bool doListRebuild = not _neighborListsAreValid.load(std::memory_order_relaxed);
1174 auto &autoTuner = *_autoTunerRefs[interactionType];
1175 const bool newton3 = autoTuner.getCurrentConfig().newton3;
1176 auto &container = _containerSelector.getCurrentContainer();
1177
1178 autopas::utils::Timer timerTotal;
1179 autopas::utils::Timer timerRebuild;
1180 autopas::utils::Timer timerComputeInteractions;
1181 autopas::utils::Timer timerComputeRemainder;
1182
1183 const bool energyMeasurementsPossible = autoTuner.resetEnergy();
1184
1185 timerTotal.start();
1186 functor.initTraversal();
1187
1188 if (doListRebuild) {
1189 timerRebuild.start();
1190#ifdef AUTOPAS_ENABLE_DYNAMIC_CONTAINERS
1191 this->updateRebuildPositions();
1192#endif
1193 container.rebuildNeighborLists(&traversal);
1194#ifdef AUTOPAS_ENABLE_DYNAMIC_CONTAINERS
1195 this->resetNeighborListsInvalidDoDynamicRebuild();
1196#endif
1197 timerRebuild.stop();
1198 _neighborListsAreValid.store(true, std::memory_order_relaxed);
1199 }
1200
1201 timerComputeInteractions.start();
1202 container.computeInteractions(&traversal);
1203 timerComputeInteractions.stop();
1204
1205 timerComputeRemainder.start();
1206 computeRemainderInteractions(functor, newton3);
1207 timerComputeRemainder.stop();
1208
1209 functor.endTraversal(newton3);
1210
1211 const auto [energyPsys, energyPkg, energyRam, energyTotal] = autoTuner.sampleEnergy();
1212 timerTotal.stop();
1213
1214 constexpr auto nanD = std::numeric_limits<double>::quiet_NaN();
1215 constexpr auto nanL = std::numeric_limits<long>::quiet_NaN();
1216 return {timerComputeInteractions.getTotalTime(),
1217 timerComputeRemainder.getTotalTime(),
1218 timerRebuild.getTotalTime(),
1219 timerTotal.getTotalTime(),
1220 energyMeasurementsPossible,
1221 energyMeasurementsPossible ? energyPsys : nanD,
1222 energyMeasurementsPossible ? energyPkg : nanD,
1223 energyMeasurementsPossible ? energyRam : nanD,
1224 energyMeasurementsPossible ? energyTotal : nanL};

1249template <class Particle>
1250template <bool newton3, class ContainerType, class PairwiseFunctor>
1251void LogicHandler<Particle>::computeRemainderInteractions2B(
1252 PairwiseFunctor *f, ContainerType &container, std::vector<FullParticleCell<Particle>> &particleBuffers,
1253 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
1254 // Sanity check. If this is violated feel free to add some logic here that adapts the number of locks.
1255 if (_bufferLocks.size() < particleBuffers.size()) {
1256 utils::ExceptionHandler::exception("Not enough locks for non-halo buffers! Num Locks: {}, Buffers: {}",
1257 _bufferLocks.size(), particleBuffers.size());
1258 }
1259
1260 // Balance buffers. This makes processing them with static scheduling quite efficient.
1261 // Also, if particles were not inserted in parallel, this enables us to process them in parallel now.
1262 // Cost is at max O(2N) worst O(N) per buffer collection and negligible compared to interacting them.
1263 auto cellToVec = [](auto &cell) -> std::vector<Particle> & { return cell._particles; };
1264 utils::ArrayUtils::balanceVectors(particleBuffers, cellToVec);
1265 utils::ArrayUtils::balanceVectors(haloParticleBuffers, cellToVec);
1266
1267 // The following part performs the main remainder traversal. The actual calculation is done in 4 steps carried out
1268 // in three helper functions.
1269
1270 // only activate time measurements if it will actually be logged
1271#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
1272 autopas::utils::Timer timerBufferContainer;
1273 autopas::utils::Timer timerPBufferPBuffer;
1274 autopas::utils::Timer timerPBufferHBuffer;
1275
1276 timerBufferContainer.start();
1277#endif
1278 // steps 1 & 2. particleBuffer with all close particles in container and haloParticleBuffer with owned, close
1279 // particles in container
1280 remainderHelperBufferContainer<newton3>(f, container, particleBuffers, haloParticleBuffers);
1281
1282#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
1283 timerBufferContainer.stop();
1284 timerPBufferPBuffer.start();
1285#endif
1286
1287 // step 3. particleBuffer with itself and all other buffers
1288 remainderHelperBufferBuffer<newton3>(f, particleBuffers, haloParticleBuffers);
1289
1290#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
1291 timerPBufferPBuffer.stop();
1292 timerPBufferHBuffer.start();
1293#endif
1294
1295 // step 4. particleBuffer with haloParticleBuffer
1296 remainderHelperBufferHaloBuffer<newton3>(f, particleBuffers, haloParticleBuffers);
1297
1298#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
1299 timerPBufferHBuffer.stop();
1300#endif
1301
1302 // unpack particle SoAs. Halo data is not interesting
1303 for (auto &buffer : particleBuffers) {
1304 f->SoAExtractor(buffer, buffer._particleSoABuffer, 0);
1305 }
1306
1307 AutoPasLog(TRACE, "Timer Buffers <-> Container (1+2): {}", timerBufferContainer.getTotalTime());
1308 AutoPasLog(TRACE, "Timer PBuffers<-> PBuffer ( 3): {}", timerPBufferPBuffer.getTotalTime());
1309 AutoPasLog(TRACE, "Timer PBuffers<-> HBuffer ( 4): {}", timerPBufferHBuffer.getTotalTime());
1310
1311 // Note: haloParticleBuffer with itself is NOT needed, as interactions between halo particles are unneeded!
1312}

1314template <class Particle>
1315template <bool newton3, class ContainerType, class PairwiseFunctor>
1316void LogicHandler<Particle>::remainderHelperBufferContainer(
1317 PairwiseFunctor *f, ContainerType &container, std::vector<FullParticleCell<Particle>> &particleBuffers,
1318 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
1319 using autopas::utils::ArrayUtils::static_cast_copy_array;
1320 using namespace autopas::utils::ArrayMath::literals;
1321
1322 // Bunch of shorthands
1323 const auto cutoff = container.getCutoff();
1324 const auto interactionLength = container.getInteractionLength();
1325 const auto interactionLengthInv = 1. / interactionLength;
1326 const auto haloBoxMin = container.getBoxMin() - interactionLength;
1327 const auto totalBoxLengthInv = 1. / (container.getBoxMax() + interactionLength - haloBoxMin);
1328 const std::array<size_t, 3> spacialLocksPerDim{
1329 _spacialLocks.size(),
1330 _spacialLocks[0].size(),
1331 _spacialLocks[0][0].size(),
1332 };
1333
1334 // Helper function to obtain the lock responsible for a given position.
1335 // Implemented as lambda because it can reuse a lot of information that is known in this context.
1336 const auto getSpacialLock = [&](const std::array<double, 3> &pos) -> std::mutex & {
1337 const auto posDistFromLowerCorner = pos - haloBoxMin;
1338 const auto relativePos = posDistFromLowerCorner * totalBoxLengthInv;
1339 // Lock coordinates are the position scaled by the number of locks
1340 const auto lockCoords =
1341 static_cast_copy_array<size_t>(static_cast_copy_array<double>(spacialLocksPerDim) * relativePos);
1342 return *_spacialLocks[lockCoords[0]][lockCoords[1]][lockCoords[2]];
1343 };
1344
1345 // one halo and particle buffer pair per thread
1346 AUTOPAS_OPENMP(parallel for schedule(static, 1) default(shared))
1347 for (int bufferId = 0; bufferId < particleBuffers.size(); ++bufferId) {
1348 auto &particleBuffer = particleBuffers[bufferId];
1349 auto &haloParticleBuffer = haloParticleBuffers[bufferId];
1350 // 1. particleBuffer with all close particles in container
1351 for (auto &&p1 : particleBuffer) {
1352 const auto pos = p1.getR();
1353 const auto min = pos - cutoff;
1354 const auto max = pos + cutoff;
1355 container.forEachInRegion(
1356 [&](auto &p2) {
1357 if constexpr (newton3) {
1358 const std::lock_guard<std::mutex> lock(getSpacialLock(p2.getR()));
1359 f->AoSFunctor(p1, p2, true);
1360 } else {
1361 f->AoSFunctor(p1, p2, false);
1362 // no need to calculate force enacted on a halo
1363 if (not p2.isHalo()) {
1364 const std::lock_guard<std::mutex> lock(getSpacialLock(p2.getR()));
1365 f->AoSFunctor(p2, p1, false);
1366 }
1367 }
1368 },
1369 min, max, IteratorBehavior::ownedOrHalo);
1370 }
1371
1372 // 2. haloParticleBuffer with owned, close particles in container
1373 for (auto &&p1halo : haloParticleBuffer) {
1374 const auto pos = p1halo.getR();
1375 const auto min = pos - cutoff;
1376 const auto max = pos + cutoff;
1377 container.forEachInRegion(
1378 [&](auto &p2) {
1379 // No need to apply anything to p1halo
1380 // -> AoSFunctor(p1, p2, false) not needed as it neither adds force nor Upot (potential energy)
1381 // -> newton3 argument needed for correct globals
1382 const std::lock_guard<std::mutex> lock(getSpacialLock(p2.getR()));
1383 f->AoSFunctor(p2, p1halo, newton3);
1384 },
1385 min, max, IteratorBehavior::owned);
1386 }
1387 }
1388}

1391template <bool newton3, class PairwiseFunctor>
1392void LogicHandler<Particle>::remainderHelperBufferBuffer(PairwiseFunctor *f,
1393 std::vector<FullParticleCell<Particle>> &particleBuffers,
1394 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
1395 // All (halo-)buffer interactions shall happen vectorized, hence, load all buffer data into SoAs
1396 for (auto &buffer : particleBuffers) {
1397 f->SoALoader(buffer, buffer._particleSoABuffer, 0, /*skipSoAResize*/ false);
1398 }
1399 for (auto &buffer : haloParticleBuffers) {
1400 f->SoALoader(buffer, buffer._particleSoABuffer, 0, /*skipSoAResize*/ false);
1401 }
1402
1403 AUTOPAS_OPENMP(parallel) {
1404 // For buffer interactions where bufferA == bufferB we can always enable newton3. For all interactions between
1405 // different buffers we turn newton3 always off, which ensures that only one thread at a time is writing to a
1406 // buffer. This saves expensive locks.
1407 // we can not use collapse here without locks, otherwise races would occur.
1408 AUTOPAS_OPENMP(for)
1409 for (size_t i = 0; i < particleBuffers.size(); ++i) {
1410 for (size_t jj = 0; jj < particleBuffers.size(); ++jj) {
1411 auto *particleBufferSoAA = &particleBuffers[i]._particleSoABuffer;
1412 const auto j = (i + jj) % particleBuffers.size();
1413 if (i == j) {
1414 f->SoAFunctorSingle(*particleBufferSoAA, true);
1415 } else {
1416 auto *particleBufferSoAB = &particleBuffers[j]._particleSoABuffer;
1417 f->SoAFunctorPair(*particleBufferSoAA, *particleBufferSoAB, false);
1418 }
1419 }
1420 }
1421 }

1490size_t LogicHandler<Particle>::collectBufferParticles(std::vector<Particle *> &bufferParticles,
1491 std::vector<FullParticleCell<Particle>> &particleBuffers,
1492 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {

1494 auto cellToVec = [](auto &cell) -> std::vector<Particle> & { return cell._particles; };
1495
1496 const size_t numOwnedBufferParticles =
1497 std::transform_reduce(particleBuffers.begin(), particleBuffers.end(), 0, std::plus<>(),
1498 [&](auto &vec) { return cellToVec(vec).size(); });
1499
1500 const size_t numHaloBufferParticles =
1501 std::transform_reduce(haloParticleBuffers.begin(), haloParticleBuffers.end(), 0, std::plus<>(),

1514 for (auto &buffer : haloParticleBuffers) {
1515 for (auto &particle : buffer._particles) {
1516 bufferParticles.push_back(&particle);

1632 Functor &functor, const InteractionTypeOption &interactionType) {

1634 Configuration configuration{};

1636 auto &autoTuner = *_autoTunerRefs[interactionType];
1637 // Todo: Make LiveInfo persistent between multiple functor calls in the same timestep (e.g. 2B + 3B)
1638 // https://github.com/AutoPas/AutoPas/issues/916
1639 LiveInfo info{};
1640
1641 // if this iteration is not relevant take the same algorithm config as before.
1642 if (not functor.isRelevantForTuning()) {
1643 stillTuning = false;
1644 configuration = autoTuner.getCurrentConfig();

1646 // change if necessary. (see https://github.com/AutoPas/AutoPas/issues/871)
1647 if (_containerSelector.getCurrentContainer().getContainerType() != configuration.container) {
1648 _containerSelector.selectContainer(
1649 configuration.container,
1650 ContainerSelectorInfo(configuration.cellSizeFactor, _containerSelector.getCurrentContainer().getVerletSkin(),
1651 _neighborListRebuildFrequency, _verletClusterSize, configuration.loadEstimator));
1652 }
1653 const auto &container = _containerSelector.getCurrentContainer();

1655 container.getParticleCellTypeEnum(), [&](const auto &particleCellDummy) -> decltype(traversalPtrOpt) {
1656 // Can't make this unique_ptr const otherwise we can't move it later.
1657 auto traversalPtr =
1658 TraversalSelector<std::decay_t<decltype(particleCellDummy)>>::template generateTraversal<Functor>(
1659 configuration.traversal, functor, container.getTraversalSelectorInfo(), configuration.dataLayout,
1660 configuration.newton3);
1661
1662 // set sortingThreshold of the traversal if it can be casted to a CellTraversal and uses the CellFunctor
1663 if (auto *cellTraversalPtr =
1664 dynamic_cast<autopas::CellTraversal<std::decay_t<decltype(particleCellDummy)>> *>(
1665 traversalPtr.get())) {
1666 cellTraversalPtr->setSortingThreshold(_sortingThreshold);
1667 }
1668 if (traversalPtr->isApplicable()) {
1669 return std::optional{std::move(traversalPtr)};
1670 } else {
1671 return std::nullopt;
1672 }
1673 });
1674 } else {
1675 if (autoTuner.needsHomogeneityAndMaxDensityBeforePrepare()) {
1676 utils::Timer timerCalculateHomogeneity;
1677 timerCalculateHomogeneity.start();
1678 const auto &container = _containerSelector.getCurrentContainer();
1679 const auto [homogeneity, maxDensity] = autopas::utils::calculateHomogeneityAndMaxDensity(container);
1680 timerCalculateHomogeneity.stop();
1681 autoTuner.addHomogeneityAndMaxDensity(homogeneity, maxDensity, timerCalculateHomogeneity.getTotalTime());
1682 }

1685
1686 if (needsLiveInfo) {
1687 info.gather(_containerSelector.getCurrentContainer(), functor, _neighborListRebuildFrequency);
1688 autoTuner.receiveLiveInfo(info);
1689 }
1690
1691 std::tie(configuration, stillTuning) = autoTuner.getNextConfig();
1692
1693 // loop as long as we don't get a valid configuration
1694 bool rejectIndefinitely = false;
1695 while (true) {
1696 // applicability check also sets the container
1697 std::tie(traversalPtrOpt, rejectIndefinitely) =
1698 isConfigurationApplicable(configuration, functor, interactionType);
1699 if (traversalPtrOpt.has_value()) {
1700 break;
1701 }
1702 // if no config is left after rejecting this one an exception is thrown here.
1703 std::tie(configuration, stillTuning) = autoTuner.rejectConfig(configuration, rejectIndefinitely);
1704 }
1705 }
1706
1707#ifdef AUTOPAS_LOG_LIVEINFO
1708 // if live info has not been gathered yet, gather it now and log it
1709 if (info.get().empty()) {
1710 info.gather(_containerSelector.getCurrentContainer(), functor, _neighborListRebuildFrequency);
1711 }
1712 _liveInfoLogger.logLiveInfo(info, _iteration);
1713#endif
1714
1715 return {configuration, std::move(traversalPtrOpt.value()), stillTuning};
1716}
