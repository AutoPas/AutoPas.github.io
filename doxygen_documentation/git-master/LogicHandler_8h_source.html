109 [[nodiscard]] std::vector<Particle_T> collectLeavingParticlesFromBuffer(bool insertOwnedParticlesToContainer) {
110 const auto &boxMin = _currentContainer->getBoxMin();
111 const auto &boxMax = _currentContainer->getBoxMax();
112 std::vector<Particle_T> leavingBufferParticles{};
113 for (auto &cell : _particleBuffer) {
114 auto &buffer = cell._particles;
115 if (insertOwnedParticlesToContainer) {
116 // Can't be const because we potentially modify ownership before re-adding
117 for (auto &p : buffer) {
118 if (p.isDummy()) {
119 continue;
120 }
121 if (utils::inBox(p.getR(), boxMin, boxMax)) {
122 p.setOwnershipState(OwnershipState::owned);
123 _currentContainer->addParticle(p);
124 } else {
125 leavingBufferParticles.push_back(p);
126 }
127 }
128 buffer.clear();
129 } else {
130 for (auto iter = buffer.begin(); iter < buffer.end();) {
131 auto &p = *iter;
132
133 auto fastRemoveP = [&]() {
134 // Fast remove of particle, i.e., swap with last entry && pop.
135 std::swap(p, buffer.back());
136 buffer.pop_back();
137 // Do not increment the iter afterward!
138 };
139 if (p.isDummy()) {
140 // We remove dummies!
141 fastRemoveP();
142 // In case we swapped a dummy here, don't increment the iterator and do another iteration to check again.
143 continue;
144 }
145 // if p was a dummy a new particle might now be at the memory location of p so we need to check that.
146 // We also just might have deleted the last particle in the buffer in that case the inBox check is meaningless
147 if (not buffer.empty() and utils::notInBox(p.getR(), boxMin, boxMax)) {
148 leavingBufferParticles.push_back(p);
149 fastRemoveP();
150 } else {
151 ++iter;
152 }
153 }
154 }
155 }
156 return leavingBufferParticles;

165#endif

167
168 if (_functorCalls > 0) {

172 // Called before bumpIterationCounters as it would return false after that.

175 }
176 autoTuner->bumpIterationCounters(needsToWait);

209 std::vector<Particle_T> resizeBox(const std::array<double, 3> &boxMin, const std::array<double, 3> &boxMax) {
210 using namespace autopas::utils::ArrayMath::literals;
211 const auto &oldMin = _currentContainer->getBoxMin();
212 const auto &oldMax = _currentContainer->getBoxMax();
213
214 // if nothing changed, do nothing
215 if (oldMin == boxMin and oldMax == boxMax) {
216 return {};
217 }
218
219 // sanity check that new size is actually positive
220 for (size_t i = 0; i < boxMin.size(); ++i) {
221 if (boxMin[i] >= boxMax[i]) {
222 utils::ExceptionHandler::exception(
223 "New box size in dimension {} is not positive!\nboxMin[{}] = {}\nboxMax[{}] = {}", i, i, boxMin[i], i,
224 boxMax[i]);
225 }
226 }
227
228 // warn if domain changes too drastically
229 const auto newLength = boxMax - boxMin;
230 const auto oldLength = oldMax - oldMin;
231 const auto relDiffLength = newLength / oldLength;
232 for (size_t i = 0; i < newLength.size(); ++i) {
233 // warning threshold is set arbitrary and up for change if needed
234 if (relDiffLength[i] > 1.3 or relDiffLength[i] < 0.7) {
235 AutoPasLog(WARN,
236 "LogicHandler.resize(): Domain size changed drastically in dimension {}! Gathered AutoTuning "
237 "information might not be applicable anymore!\n"
238 "Size old box : {}\n"
239 "Size new box : {}\n"
240 "Relative diff: {}",
241 i, utils::ArrayUtils::to_string(oldLength), utils::ArrayUtils::to_string(newLength),
242 utils::ArrayUtils::to_string(relDiffLength));
243 }
244 }
245
246 // The new box size is valid, so update the current container info.
247 _currentContainerSelectorInfo.boxMin = boxMin;
248 _currentContainerSelectorInfo.boxMax = boxMax;
249
250 // check all particles
251 std::vector<Particle_T> particlesNowOutside;
252 for (auto pIter = _currentContainer->begin(); pIter.isValid(); ++pIter) {
253 // make sure only owned ones are present
254 if (not pIter->isOwned()) {
255 utils::ExceptionHandler::exception(
256 "LogicHandler::resizeBox() encountered non owned particle. "
257 "When calling resizeBox() these should be already deleted. "
258 "This could be solved by calling updateContainer() before resizeBox().");
259 }
260 // owned particles that are now outside are removed from the container and returned
261 if (not utils::inBox(pIter->getR(), boxMin, boxMax)) {
262 particlesNowOutside.push_back(*pIter);
263 decreaseParticleCounter(*pIter);
264 internal::markParticleAsDeleted(*pIter);
265 }
266 }
267
268 // Resize by generating a new container with the new box size and moving all particles to it.
269 auto newContainer = ContainerSelector<Particle_T>::generateContainer(_currentContainer->getContainerType(),
270 _currentContainerSelectorInfo);
271 setCurrentContainer(std::move(newContainer));
272 // The container might have changed sufficiently that we would need a different number of spatial locks.
273 const auto boxLength = boxMax - boxMin;
274 const auto interactionLengthInv = 1. / _currentContainer->getInteractionLength();
275 initSpatialLocks(boxLength, interactionLengthInv);
276
277 // Set this flag, s.t., the container is rebuilt!
278 _neighborListsAreValid.store(false, std::memory_order_relaxed);
279
280 return particlesNowOutside;
281 }

696 void initSpatialLocks(const std::array<double, 3> &boxLength, double interactionLengthInv) {
697 using namespace autopas::utils::ArrayMath::literals;
698 using utils::ArrayMath::ceil;
699 using utils::ArrayUtils::static_cast_copy_array;
700
701 // The maximum number of spatial locks is capped at 1e6.
702 // This limit is chosen more or less arbitrary. It is big enough so that our regular MD simulations
703 // fall well within it and small enough so that no memory issues arise.
704 // There were no rigorous tests for an optimal number of locks.
705 // Without this cap, very large domains (or tiny cutoffs) would generate an insane number of locks,
706 // that could blow up the memory.
707 constexpr size_t maxNumSpacialLocks{1000000};
708
709 // One lock per interaction length or less if this would generate too many.
710 const std::array<size_t, 3> locksPerDim = [&]() {
711 // First naively calculate the number of locks if we simply take the desired cell length.
712 // Ceil because both decisions are possible, and we are generous gods.
713 const std::array<size_t, 3> locksPerDimNaive =
714 static_cast_copy_array<size_t>(ceil(boxLength * interactionLengthInv));
715 const auto totalLocksNaive =
716 std::accumulate(locksPerDimNaive.begin(), locksPerDimNaive.end(), 1ul, std::multiplies<>());
717 // If the number of locks is within the limits everything is fine and we can return.
718 if (totalLocksNaive <= maxNumSpacialLocks) {
719 return locksPerDimNaive;
720 } else {
721 // If the number of locks grows too large, calculate the locks per dimension proportionally to the side lengths.
722 // Calculate side length relative to dimension 0.
723 const std::array<double, 3> boxSideProportions = {
724 1.,
725 boxLength[0] / boxLength[1],
726 boxLength[0] / boxLength[2],
727 };
728 // With this, calculate the number of locks the first dimension should receive.
729 const auto prodProportions =
730 std::accumulate(boxSideProportions.begin(), boxSideProportions.end(), 1., std::multiplies<>());
731 // Needs floor, otherwise we exceed the limit.
732 const auto locksInFirstDimFloat = std::floor(std::cbrt(maxNumSpacialLocks * prodProportions));
733 // From this and the proportions relative to the first dimension, we can calculate the remaining number of locks
734 const std::array<size_t, 3> locksPerDimLimited = {
735 static_cast<size_t>(locksInFirstDimFloat), // omitted div by 1
736 static_cast<size_t>(locksInFirstDimFloat / boxSideProportions[1]),
737 static_cast<size_t>(locksInFirstDimFloat / boxSideProportions[2]),
738 };
739 return locksPerDimLimited;
740 }
741 }();
742 _spatialLocks.resize(locksPerDim[0]);
743 for (auto &lockVecVec : _spatialLocks) {
744 lockVecVec.resize(locksPerDim[1]);
745 for (auto &lockVec : lockVecVec) {
746 lockVec.resize(locksPerDim[2]);
747 for (auto &lockPtr : lockVec) {
748 if (not lockPtr) {
749 lockPtr = std::make_unique<std::mutex>();
750 }
751 }
752 }
753 }

1021template <typename Particle_T>
1022void LogicHandler<Particle_T>::setParticleBuffers(
1023 const std::vector<FullParticleCell<Particle_T>> &particleBuffers,
1024 const std::vector<FullParticleCell<Particle_T>> &haloParticleBuffers) {
1025 auto exchangeBuffer = [](const auto &newBuffers, auto &oldBuffers, auto &particleCounter) {
1026 // sanity check
1027 if (oldBuffers.size() < newBuffers.size()) {
1028 autopas::utils::ExceptionHandler::exception(
1029 "The number of new buffers ({}) is larger than number of existing buffers ({})!", newBuffers.size(),
1030 oldBuffers.size());
1031 }
1032
1033 // we will clear the old buffers so subtract the particles from the counters.
1034 const auto numParticlesInOldBuffers =
1035 std::transform_reduce(oldBuffers.begin(), std::next(oldBuffers.begin(), newBuffers.size()), 0, std::plus<>(),
1036 [](const auto &cell) { return cell.size(); });
1037 particleCounter.fetch_sub(numParticlesInOldBuffers, std::memory_order_relaxed);
1038
1039 // clear the old buffers and copy the content of the new buffers over.
1040 size_t numParticlesInNewBuffers = 0;
1041 for (size_t i = 0; i < newBuffers.size(); ++i) {
1042 oldBuffers[i].clear();
1043 for (const auto &p : newBuffers[i]) {
1044 ++numParticlesInNewBuffers;
1045 oldBuffers[i].addParticle(p);
1046 }
1047 }
1048 // update the counters.
1049 particleCounter.fetch_add(numParticlesInNewBuffers, std::memory_order_relaxed);
1050 };
1051
1052 exchangeBuffer(particleBuffers, _particleBuffer, _numParticlesOwned);
1053 exchangeBuffer(haloParticleBuffers, _haloParticleBuffer, _numParticlesHalo);
1054}

1063template <class Functor>
1064IterationMeasurements LogicHandler<Particle_T>::computeInteractions(Functor &functor, TraversalInterface &traversal) {
1065 // Helper to derive the Functor type at compile time
1066 constexpr auto interactionType = [] {
1067 if (utils::isPairwiseFunctor<Functor>()) {
1068 return InteractionTypeOption::pairwise;
1069 } else if (utils::isTriwiseFunctor<Functor>()) {
1070 return InteractionTypeOption::triwise;
1071 } else {
1072 utils::ExceptionHandler::exception(
1073 "LogicHandler::computeInteractions(): Functor is not valid. Only pairwise and triwise functors are "
1074 "supported. "
1075 "Please use a functor derived from "
1076 "PairwiseFunctor or TriwiseFunctor.");
1077 }
1078 }();
1079
1080 auto &autoTuner = *_autoTunerRefs[interactionType];
1081#ifdef AUTOPAS_ENABLE_DYNAMIC_CONTAINERS
1082 if (autoTuner.inFirstTuningIteration()) {
1083 _numRebuildsInNonTuningPhase = 0;
1084 }
1085
1086 // Rebuild frequency estimation should be triggered early in the tuning phase.
1087 // This is necessary because runtime prediction for each trial configuration
1088 // depends on the rebuild frequency.
1089 // To avoid the influence of poorly initialized velocities at the start of the simulation,
1090 // the rebuild frequency is estimated at iteration corresponding to the last sample of the first configuration.
1091 // The rebuild frequency estimated here is then reused for the remainder of the tuning phase.
1092 if (autoTuner.inFirstConfigurationLastSample()) {
1093 // Fetch the needed information for estimating the rebuild frequency from _logicHandlerInfo
1094 // and estimate the current rebuild frequency using the velocity method.
1095 double rebuildFrequencyEstimate =
1096 getVelocityMethodRFEstimate(_logicHandlerInfo.verletSkin, _logicHandlerInfo.deltaT);
1097 double userProvidedRF = static_cast<double>(_neighborListRebuildFrequency);
1098 // The user defined rebuild frequency is considered as the upper bound.
1099 // If velocity method estimate exceeds upper bound, set the rebuild frequency to the user defined value.
1100 // This is done because we currently use the user defined rebuild frequency as the upper bound to avoid expensive
1101 // buffer interactions.
1102 if (rebuildFrequencyEstimate > userProvidedRF) {
1103 autoTuner.setRebuildFrequency(userProvidedRF);
1104 } else {
1105 autoTuner.setRebuildFrequency(rebuildFrequencyEstimate);
1106 }
1107 }
1108#endif
1109 utils::Timer timerTotal;
1110 utils::Timer timerRebuild;
1111 utils::Timer timerComputeInteractions;
1112 utils::Timer timerComputeRemainder;
1113
1114 const bool energyMeasurementsPossible = autoTuner.resetEnergy();
1115
1116 timerTotal.start();
1117 functor.initTraversal();
1118
1119 // if lists are not valid -> rebuild;
1120 if (not _neighborListsAreValid.load(std::memory_order_relaxed)) {
1121 timerRebuild.start();
1122#ifdef AUTOPAS_ENABLE_DYNAMIC_CONTAINERS
1123 this->updateRebuildPositions();
1124#endif
1125 _currentContainer->rebuildNeighborLists(&traversal);
1126#ifdef AUTOPAS_ENABLE_DYNAMIC_CONTAINERS
1127 this->resetNeighborListsInvalidDoDynamicRebuild();
1128 _numRebuilds++;
1129 if (not autoTuner.inTuningPhase()) {
1130 _numRebuildsInNonTuningPhase++;
1131 }
1132#endif
1133 timerRebuild.stop();
1134 _neighborListsAreValid.store(true, std::memory_order_relaxed);
1135 }
1136
1137 timerComputeInteractions.start();
1138 _currentContainer->computeInteractions(&traversal);
1139 timerComputeInteractions.stop();
1140
1141 timerComputeRemainder.start();
1142 const bool newton3 = autoTuner.getCurrentConfig().newton3;
1143 const auto dataLayout = autoTuner.getCurrentConfig().dataLayout;
1144 computeRemainderInteractions(functor, newton3, dataLayout);
1145 timerComputeRemainder.stop();
1146
1147 functor.endTraversal(newton3);
1148
1149 const auto [energyWatts, energyJoules, energyDeltaT, energyTotal] = autoTuner.sampleEnergy();
1150 timerTotal.stop();
1151
1152 constexpr auto nanD = std::numeric_limits<double>::quiet_NaN();
1153 constexpr auto nanL = std::numeric_limits<long>::quiet_NaN();
1154 return {timerComputeInteractions.getTotalTime(),
1155 timerComputeRemainder.getTotalTime(),
1156 timerRebuild.getTotalTime(),
1157 timerTotal.getTotalTime(),
1158 energyMeasurementsPossible,
1159 energyMeasurementsPossible ? energyWatts : nanD,
1160 energyMeasurementsPossible ? energyJoules : nanD,
1161 energyMeasurementsPossible ? energyDeltaT : nanD,
1162 energyMeasurementsPossible ? energyTotal : nanL};
1163}

1191std::tuple<Configuration, std::unique_ptr<TraversalInterface>, bool> LogicHandler<Particle_T>::selectConfiguration(
1192 Functor &functor, const InteractionTypeOption &interactionType) {
1193 auto &autoTuner = *_autoTunerRefs[interactionType];
1194
1195 // Todo: Make LiveInfo persistent between multiple functor calls in the same timestep (e.g. 2B + 3B)
1196 // https://github.com/AutoPas/AutoPas/issues/916
1197 LiveInfo info{};
1198#ifdef AUTOPAS_LOG_LIVEINFO
1199 auto particleIter = this->begin(IteratorBehavior::ownedOrHalo);
1200 info.gather(particleIter, _neighborListRebuildFrequency, getNumberOfParticlesOwned(), _logicHandlerInfo.boxMin,
1201 _logicHandlerInfo.boxMax, _logicHandlerInfo.cutoff, _logicHandlerInfo.verletSkin);
1202 _liveInfoLogger.logLiveInfo(info, _iteration);
1203#endif
1204
1205 // if this iteration is not relevant, take the same algorithm config as before.
1206 if (not functor.isRelevantForTuning()) {
1207 auto configuration = autoTuner.getCurrentConfig();
1208 auto [traversalPtr, _] = isConfigurationApplicable(configuration, functor);
1209
1210 if (not traversalPtr) {
1211 // TODO: Can we handle this case gracefully?
1212 utils::ExceptionHandler::exception(
1213 "LogicHandler: Functor {} is not relevant for tuning but the given configuration is not applicable!",
1214 functor.getName());
1215 }
1216 return {configuration, std::move(traversalPtr), false};
1217 }
1218
1219 if (autoTuner.needsLiveInfo()) {
1220 // If live info has not been gathered yet, gather it now and send it to the tuner.
1221 if (info.get().empty()) {
1222 auto particleIter = this->begin(IteratorBehavior::ownedOrHalo);
1223 info.gather(particleIter, _neighborListRebuildFrequency, getNumberOfParticlesOwned(), _logicHandlerInfo.boxMin,
1224 _logicHandlerInfo.boxMax, _logicHandlerInfo.cutoff, _logicHandlerInfo.verletSkin);
1225 }
1226 autoTuner.receiveLiveInfo(info);
1227 }
1228
1229 size_t numRejectedConfigs = 0;
1230#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
1231 utils::Timer selectConfigurationTimer;
1232 selectConfigurationTimer.start();
1233#endif
1234 auto [configuration, stillTuning] = autoTuner.getNextConfig();
1235
1236 // loop as long as we don't get a valid configuration
1237 do {
1238 // applicability check also sets the container
1239 auto [traversalPtr, rejectIndefinitely] = isConfigurationApplicable(configuration, functor);
1240 if (traversalPtr) {
1241#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
1242 selectConfigurationTimer.stop();
1243 AutoPasLog(TRACE, "Select Configuration took {} ms. A total of {} configurations were rejected.",
1244 selectConfigurationTimer.getTotalTime(), numRejectedConfigs);
1245#endif
1246 return {configuration, std::move(traversalPtr), stillTuning};
1247 }
1248 // if no config is left after rejecting this one, an exception is thrown here.
1249 numRejectedConfigs++;
1250 std::tie(configuration, stillTuning) = autoTuner.rejectConfig(configuration, rejectIndefinitely);
1251 } while (true);
1252}

1296 autoTuner.logTuningResult(stillTuning, tuningTimer.getTotalTime());

1300
1302 AutoPasLog(DEBUG, "Iterating with configuration: {} tuning: {}", configuration.toString(), stillTuning);

1359std::tuple<std::unique_ptr<TraversalInterface>, bool> LogicHandler<Particle_T>::isConfigurationApplicable(
1360 const Configuration &config, Functor &functor) {
1361 // Check if the container supports the traversal
1362 const auto allContainerTraversals =
1363 compatibleTraversals::allCompatibleTraversals(config.container, config.interactionType);
1364 if (allContainerTraversals.find(config.traversal) == allContainerTraversals.end()) {
1365 AutoPasLog(WARN, "Configuration rejected: Container {} does not support the traversal {}.", config.container,
1366 config.traversal);
1367 return {nullptr, /*rejectIndefinitely*/ true};
1368 }
1369
1370 // Check if the functor supports the required Newton 3 mode
1371 if ((config.newton3 == Newton3Option::enabled and not functor.allowsNewton3()) or
1372 (config.newton3 == Newton3Option::disabled and not functor.allowsNonNewton3())) {
1373 AutoPasLog(DEBUG, "Configuration rejected: The functor doesn't support Newton 3 {}!", config.newton3);
1374 return {nullptr, /*rejectIndefinitely*/ true};
1375 }
1376
1377 std::unique_ptr<ParticleContainerInterface<Particle_T>> containerPtr{nullptr};
1378 auto containerInfo =
1379 ContainerSelectorInfo(_currentContainer->getBoxMin(), _currentContainer->getBoxMax(),
1380 _currentContainer->getCutoff(), config.cellSizeFactor, _currentContainer->getVerletSkin(),
1381 _verletClusterSize, _sortingThreshold, config.loadEstimator);
1382
1383 // If we have no current container or needs to be updated to the new config.container, we need to generate a new
1384 // container.
1385 const bool generateNewContainer = _currentContainer == nullptr or
1386 _currentContainer->getContainerType() != config.container or
1387 containerInfo != _currentContainerSelectorInfo;
1388
1389 if (generateNewContainer) {
1390 // For now, set the local containerPtr to the new container. We do not copy the particles over and set the member
1391 // _currentContainer until after we know that the traversal is applicable to the domain.
1392 containerPtr = ContainerSelector<Particle_T>::generateContainer(config.container, containerInfo);
1393 }
1394
1395 const auto traversalInfo =
1396 generateNewContainer ? containerPtr->getTraversalSelectorInfo() : _currentContainer->getTraversalSelectorInfo();
1397
1398 // Generates a traversal if applicable, otherwise returns a nullptr
1399 auto traversalPtr =
1400 TraversalSelector::generateTraversalFromConfig<Particle_T, Functor>(config, functor, traversalInfo);
1401
1402 // If the traversal is applicable to the domain, and the configuration requires generating a new container,
1403 // update the member _currentContainer with setCurrentContainer, copying the particle data over, and update
1404 // _currentContainerSelectorInfo.
1405 if (traversalPtr and generateNewContainer) {
1406 _currentContainerSelectorInfo = containerInfo;
1407 setCurrentContainer(std::move(containerPtr));
1408 }
1409
1410 return {std::move(traversalPtr), /*rejectIndefinitely*/ false};
1411}
