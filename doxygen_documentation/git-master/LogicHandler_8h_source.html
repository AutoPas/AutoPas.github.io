67  return {};

72  if (boxMin[i] >= boxMax[i]) {
73  utils::ExceptionHandler::exception(
74  "New box size in dimension {} is not positive!\nboxMin[{}] = {}\nboxMax[{}] = {}", i, i, boxMin[i], i,
75  boxMax[i]);
76  }

79  // warn if domain changes too drastically
80  const auto newLength = utils::ArrayMath::sub(boxMax, boxMin);
81  const auto oldLength = utils::ArrayMath::sub(oldMax, oldMin);

83  for (size_t i = 0; i < newLength.size(); ++i) {
84  // warning threshold is set arbitrary and up for change if needed

92  i, utils::ArrayUtils::to_string(oldLength), utils::ArrayUtils::to_string(newLength),
93  utils::ArrayUtils::to_string(relDiffLength));
94  }
