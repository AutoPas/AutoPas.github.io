41 [[nodiscard]] std::vector<Particle> collectLeavingParticlesFromBuffer(bool insertOwnedParticlesToContainer) {
42 std::vector<Particle> leavingBufferParticles;
43 if (insertOwnedParticlesToContainer) {
44 for (const auto &p : _particleBuffer) {
45 if (p.isDummy()) {
46 continue;
47 }
48 if (utils::inBox(p.getR(), _autoTuner.getContainer()->getBoxMin(), _autoTuner.getContainer()->getBoxMax())) {
49 _autoTuner.getContainer()->template addParticle(p);
50 } else {
51 leavingBufferParticles.push_back(p);
52 }
53 }
54 _particleBuffer.clear();
55 } else {
56 for (auto iter = _particleBuffer.begin(); iter != _particleBuffer.end();) {
57 auto &p = *iter;
58
59 auto fastRemoveP = [&]() {
60 // Fast remove of particle, i.e., swap with last entry && pop.
61 std::swap(p, _particleBuffer.back());
62 _particleBuffer.pop_back();
63 // Do not increment the iter afterwards!
64 };
65 if (p.isDummy()) {
66 // We remove dummies!
67 fastRemoveP();
68 }
69 if (utils::notInBox(p.getR(), _autoTuner.getContainer()->getBoxMin(), _autoTuner.getContainer()->getBoxMax())) {
70 leavingBufferParticles.push_back(p);
71 fastRemoveP();
72 } else {
73 ++iter;
74 }
75 }
76 }
77 return leavingBufferParticles;
78 }

83 [[nodiscard]] std::vector<Particle> updateContainer() {
84 bool doDataStructureUpdate = not neighborListsAreValid();
85
86 if (doDataStructureUpdate) {
87 _neighborListsAreValid = false;
88 }
89 // The next call also adds particles to the container if doDataStructureUpdate is true.
90 auto leavingBufferParticles = collectLeavingParticlesFromBuffer(doDataStructureUpdate);
91
92 AutoPasLog(debug, "Initiating container update.");
93 auto leavingParticles = _autoTuner.getContainer()->updateContainer(not doDataStructureUpdate);
94 leavingParticles.insert(leavingParticles.end(), leavingBufferParticles.begin(), leavingBufferParticles.end());
95
96 // Substract the amount of leaving particles from the number of owned particles.
97 _numParticlesOwned.fetch_sub(leavingParticles.size(), std::memory_order_relaxed);
98 // updateContainer deletes all halo particles.
99 _haloParticleBuffer.clear();
100 _numParticlesHalo.exchange(0, std::memory_order_relaxed);
101 return leavingParticles;
102 }

110 std::vector<Particle> resizeBox(const std::array<double, 3> &boxMin, const std::array<double, 3> &boxMax) {
111 const auto &oldMin = _autoTuner.getContainer()->getBoxMin();
112 const auto &oldMax = _autoTuner.getContainer()->getBoxMax();
113
114 // if nothing changed do nothing
115 if (oldMin == boxMin and oldMax == boxMax) {
116 return {};
117 }
118
119 // sanity check that new size is actually positive
120 for (size_t i = 0; i < boxMin.size(); ++i) {
121 if (boxMin[i] >= boxMax[i]) {
122 utils::ExceptionHandler::exception(
123 "New box size in dimension {} is not positive!\nboxMin[{}] = {}\nboxMax[{}] = {}", i, i, boxMin[i], i,
124 boxMax[i]);
125 }
126 }
127
128 // warn if domain changes too drastically
129 const auto newLength = utils::ArrayMath::sub(boxMax, boxMin);
130 const auto oldLength = utils::ArrayMath::sub(oldMax, oldMin);
131 const auto relDiffLength = utils::ArrayMath::div(newLength, oldLength);
132 for (size_t i = 0; i < newLength.size(); ++i) {
133 // warning threshold is set arbitrary and up for change if needed
134 if (relDiffLength[i] > 1.3 or relDiffLength[i] < 0.7) {
135 AutoPasLog(warn,
136 "LogicHandler.resize(): Domain size changed drastically in dimension {}! Gathered AutoTuning "
137 "information might not be applicable anymore!\n"
138 "Size old box : {}\n"
139 "Size new box : {}\n"
140 "Relative diff: {}",
141 i, utils::ArrayUtils::to_string(oldLength), utils::ArrayUtils::to_string(newLength),
142 utils::ArrayUtils::to_string(relDiffLength));
143 }
144 }
145
146 // check all particles
147 std::vector<Particle> particlesNowOutside;
148 for (auto pIter = _autoTuner.getContainer()->begin(); pIter.isValid(); ++pIter) {
149 // make sure only owned ones are present
150 if (not pIter->isOwned()) {
151 utils::ExceptionHandler::exception(
152 "LogicHandler::resizeBox() encountered non owned particle. "
153 "When calling resizeBox() these should be already deleted. "
154 "This could be solved by calling updateContainer() before resizeBox().");
155 }
156 // owned particles that are now outside are removed from the container and returned
157 if (not utils::inBox(pIter->getR(), boxMin, boxMax)) {
158 particlesNowOutside.push_back(*pIter);
159 deleteParticle(pIter);
160 }
161 }
162
163 // actually resize the container
164 _autoTuner.resizeBox(boxMin, boxMax);
165 // Set this flag, s.t., the container is rebuilt!
166 _neighborListsAreValid = false;
167
168 return particlesNowOutside;
169 }
