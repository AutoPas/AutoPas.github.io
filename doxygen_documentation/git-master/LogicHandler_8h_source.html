38 [[nodiscard]] std::pair<std::vector<Particle>, bool> updateContainer(bool forced) {

61 std::vector<Particle> resizeBox(const std::array<double, 3> &boxMin, const std::array<double, 3> &boxMax) {
62 const auto &oldMin = _autoTuner.getContainer()->getBoxMin();
63 const auto &oldMax = _autoTuner.getContainer()->getBoxMax();
64
65 // if nothing changed do nothing
66 if (oldMin == boxMin and oldMax == boxMax) {
67 return {};
68 }
69
70 // sanity check that new size is actually positive
71 for (size_t i = 0; i < boxMin.size(); ++i) {
72 if (boxMin[i] >= boxMax[i]) {
73 utils::ExceptionHandler::exception(
74 "New box size in dimension {} is not positive!\nboxMin[{}] = {}\nboxMax[{}] = {}", i, i, boxMin[i], i,
75 boxMax[i]);
76 }
77 }
78
79 // warn if domain changes too drastically
80 const auto newLength = utils::ArrayMath::sub(boxMax, boxMin);
81 const auto oldLength = utils::ArrayMath::sub(oldMax, oldMin);
82 const auto relDiffLength = utils::ArrayMath::div(newLength, oldLength);
83 for (size_t i = 0; i < newLength.size(); ++i) {
84 // warning threshold is set arbitrary and up for change if needed
85 if (relDiffLength[i] > 1.3 or relDiffLength[i] < 0.7) {
86 AutoPasLog(warn,
87 "LogicHandler.resize(): Domain size changed drastically in dimension {}! Gathered AutoTuning "
88 "information might not be applicable anymore!\n"
89 "Size old box : {}\n"
90 "Size new box : {}\n"
91 "Relative diff: {}",
92 i, utils::ArrayUtils::to_string(oldLength), utils::ArrayUtils::to_string(newLength),
93 utils::ArrayUtils::to_string(relDiffLength));
94 }
95 }
96
97 // check all particles
98 std::vector<Particle> particlesNowOutside;
99 for (auto pIter = _autoTuner.getContainer()->begin(); pIter.isValid(); ++pIter) {
100 // make sure only owned ones are present
101 if (not pIter->isOwned()) {
102 utils::ExceptionHandler::exception(
103 "LogicHandler::resizeBox() encountered non owned particle. "
104 "When calling resizeBox() these should be already deleted. "
105 "This could be solved by calling updateContainer() before resizeBox().");
106 }
107 // owned particles that are now outside are removed from the container and returned
108 if (not utils::inBox(pIter->getR(), boxMin, boxMax)) {
109 particlesNowOutside.push_back(*pIter);
110 deleteParticle(pIter);
111 }
112 }
113
114 // actually resize the container
115 _autoTuner.resizeBox(boxMin, boxMax);
116
117 return particlesNowOutside;
118 }

311 container->getBoxMin()[dim], dim, container->getBoxMax()[dim], container->getCutoff(), container->getSkin(),
