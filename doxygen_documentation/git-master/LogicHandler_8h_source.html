45 [[nodiscard]] std::vector<Particle> collectLeavingParticlesFromBuffer(bool insertOwnedParticlesToContainer) {
46 const auto &boxMin = _autoTuner.getContainer().getBoxMin();
47 const auto &boxMax = _autoTuner.getContainer().getBoxMax();
48 std::vector<Particle> leavingBufferParticles{};
49 for (auto &cell : _particleBuffer) {
50 auto &buffer = cell._particles;
51 if (insertOwnedParticlesToContainer) {
52 for (const auto &p : buffer) {
53 if (p.isDummy()) {
54 continue;
55 }
56 if (utils::inBox(p.getR(), boxMin, boxMax)) {
57 _autoTuner.getContainer().addParticle(p);
58 } else {
59 leavingBufferParticles.push_back(p);
60 }
61 }
62 buffer.clear();
63 } else {
64 for (auto iter = buffer.begin(); iter != buffer.end();) {
65 auto &p = *iter;
66
67 auto fastRemoveP = [&]() {
68 // Fast remove of particle, i.e., swap with last entry && pop.
69 std::swap(p, buffer.back());
70 buffer.pop_back();
71 // Do not increment the iter afterwards!
72 };
73 if (p.isDummy()) {
74 // We remove dummies!
75 fastRemoveP();
76 }
77 if (utils::notInBox(p.getR(), boxMin, boxMax)) {
78 leavingBufferParticles.push_back(p);
79 fastRemoveP();
80 } else {
81 ++iter;
82 }
83 }
84 }
85 }
86 return leavingBufferParticles;
87 }

92 [[nodiscard]] std::vector<Particle> updateContainer() {
93 bool doDataStructureUpdate = not neighborListsAreValid();
94
95 if (doDataStructureUpdate) {
96 _neighborListsAreValid.store(false, std::memory_order_relaxed);
97 }
98 // The next call also adds particles to the container if doDataStructureUpdate is true.
99 auto leavingBufferParticles = collectLeavingParticlesFromBuffer(doDataStructureUpdate);
100
101 AutoPasLog(DEBUG, "Initiating container update.");
102 auto leavingParticles = _autoTuner.getContainer().updateContainer(not doDataStructureUpdate);
103 leavingParticles.insert(leavingParticles.end(), leavingBufferParticles.begin(), leavingBufferParticles.end());
104
105 // Substract the amount of leaving particles from the number of owned particles.
106 _numParticlesOwned.fetch_sub(leavingParticles.size(), std::memory_order_relaxed);
107 // updateContainer deletes all halo particles.
108 std::for_each(_haloParticleBuffer.begin(), _haloParticleBuffer.end(), [](auto &buffer) { buffer.clear(); });
109 _numParticlesHalo.store(0, std::memory_order_relaxed);
110 return leavingParticles;
111 }

119 std::vector<Particle> resizeBox(const std::array<double, 3> &boxMin, const std::array<double, 3> &boxMax) {
120 using namespace autopas::utils::ArrayMath::literals;
121 const auto &oldMin = _autoTuner.getContainer().getBoxMin();
122 const auto &oldMax = _autoTuner.getContainer().getBoxMax();
123
124 // if nothing changed do nothing
125 if (oldMin == boxMin and oldMax == boxMax) {
126 return {};
127 }
128
129 // sanity check that new size is actually positive
130 for (size_t i = 0; i < boxMin.size(); ++i) {
131 if (boxMin[i] >= boxMax[i]) {
132 utils::ExceptionHandler::exception(
133 "New box size in dimension {} is not positive!\nboxMin[{}] = {}\nboxMax[{}] = {}", i, i, boxMin[i], i,
134 boxMax[i]);
135 }
136 }
137
138 // warn if domain changes too drastically
139 const auto newLength = boxMax - boxMin;
140 const auto oldLength = oldMax - oldMin;
141 const auto relDiffLength = newLength / oldLength;
142 for (size_t i = 0; i < newLength.size(); ++i) {
143 // warning threshold is set arbitrary and up for change if needed
144 if (relDiffLength[i] > 1.3 or relDiffLength[i] < 0.7) {
145 AutoPasLog(WARN,
146 "LogicHandler.resize(): Domain size changed drastically in dimension {}! Gathered AutoTuning "
147 "information might not be applicable anymore!\n"
148 "Size old box : {}\n"
149 "Size new box : {}\n"
150 "Relative diff: {}",
151 i, utils::ArrayUtils::to_string(oldLength), utils::ArrayUtils::to_string(newLength),
152 utils::ArrayUtils::to_string(relDiffLength));
153 }
154 }
155
156 // check all particles
157 std::vector<Particle> particlesNowOutside;
158 for (auto pIter = _autoTuner.getContainer().begin(); pIter.isValid(); ++pIter) {
159 // make sure only owned ones are present
160 if (not pIter->isOwned()) {
161 utils::ExceptionHandler::exception(
162 "LogicHandler::resizeBox() encountered non owned particle. "
163 "When calling resizeBox() these should be already deleted. "
164 "This could be solved by calling updateContainer() before resizeBox().");
165 }
166 // owned particles that are now outside are removed from the container and returned
167 if (not utils::inBox(pIter->getR(), boxMin, boxMax)) {
168 particlesNowOutside.push_back(*pIter);
169 decreaseParticleCounter(*pIter);
170 internal::markParticleAsDeleted(*pIter);
171 }
172 }
173
174 // actually resize the container
175 _autoTuner.resizeBox(boxMin, boxMax);
176 // Set this flag, s.t., the container is rebuilt!
177 _neighborListsAreValid.store(false, std::memory_order_relaxed);
178
179 return particlesNowOutside;
180 }

288 std::tuple<bool, bool> deleteParticleFromBuffers(Particle &particle) {
289 // find the buffer the particle belongs to
290 auto &bufferCollection = particle.isOwned() ? _particleBuffer : _haloParticleBuffer;
291 for (auto &cell : bufferCollection) {
292 auto &buffer = cell._particles;
293 // if the address of the particle is between start and end of the buffer it is in this buffer
294 if (not buffer.empty() and &(buffer.front()) <= &particle and &particle <= &(buffer.back())) {
295 const bool isRearParticle = &particle == &buffer.back();
296 // swap-delete
297 particle = buffer.back();
298 buffer.pop_back();
299 return {true, not isRearParticle};
300 }
301 }
302 return {false, true};
303 }

343 template <class Iterator>
344 typename Iterator::ParticleVecType gatherAdditionalVectors(IteratorBehavior behavior) {
345 typename Iterator::ParticleVecType additionalVectors;
346 additionalVectors.reserve(static_cast<bool>(behavior & IteratorBehavior::owned) * _particleBuffer.size() +
347 static_cast<bool>(behavior & IteratorBehavior::halo) * _haloParticleBuffer.size());
348 if (behavior & IteratorBehavior::owned) {
349 for (auto &buffer : _particleBuffer) {
350 // Don't insert empty buffers. This also means that we won't pick up particles added during iterating if they
351 // go to the buffers. But since we wouldn't pick them up if they go into the container to a cell that the
352 // iterators already passed this is unsupported anyways.
353 if (not buffer.isEmpty()) {
354 additionalVectors.push_back(&(buffer._particles));
355 }
356 }
357 }
358 if (behavior & IteratorBehavior::halo) {
359 for (auto &buffer : _haloParticleBuffer) {
360 if (not buffer.isEmpty()) {
361 additionalVectors.push_back(&(buffer._particles));
362 }
363 }
364 }
365 return additionalVectors;
