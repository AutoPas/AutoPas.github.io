107 [[nodiscard]] std::vector<Particle> collectLeavingParticlesFromBuffer(bool insertOwnedParticlesToContainer) {
108 const auto &boxMin = _containerSelector.getCurrentContainer().getBoxMin();
109 const auto &boxMax = _containerSelector.getCurrentContainer().getBoxMax();
110 std::vector<Particle> leavingBufferParticles{};
111 for (auto &cell : _particleBuffer) {
112 auto &buffer = cell._particles;
113 if (insertOwnedParticlesToContainer) {
114 // Can't be const because we potentially modify ownership before re-adding
115 for (auto &p : buffer) {
116 if (p.isDummy()) {
117 continue;
118 }
119 if (utils::inBox(p.getR(), boxMin, boxMax)) {
120 p.setOwnershipState(OwnershipState::owned);
121 _containerSelector.getCurrentContainer().addParticle(p);
122 } else {
123 leavingBufferParticles.push_back(p);
124 }
125 }
126 buffer.clear();
127 } else {
128 for (auto iter = buffer.begin(); iter < buffer.end();) {
129 auto &p = *iter;
130
131 auto fastRemoveP = [&]() {
132 // Fast remove of particle, i.e., swap with last entry && pop.
133 std::swap(p, buffer.back());
134 buffer.pop_back();
135 // Do not increment the iter afterward!
136 };
137 if (p.isDummy()) {
138 // We remove dummies!
139 fastRemoveP();
140 // In case we swapped a dummy here, don't increment the iterator and do another iteration to check again.
141 continue;
142 }
143 // if p was a dummy a new particle might now be at the memory location of p so we need to check that.
144 // We also just might have deleted the last particle in the buffer in that case the inBox check is meaningless
145 if (not buffer.empty() and utils::notInBox(p.getR(), boxMin, boxMax)) {
146 leavingBufferParticles.push_back(p);
147 fastRemoveP();
148 } else {
149 ++iter;
150 }
151 }
152 }
153 }
154 return leavingBufferParticles;

200 std::vector<Particle> resizeBox(const std::array<double, 3> &boxMin, const std::array<double, 3> &boxMax) {
201 using namespace autopas::utils::ArrayMath::literals;
202 const auto &oldMin = _containerSelector.getCurrentContainer().getBoxMin();
203 const auto &oldMax = _containerSelector.getCurrentContainer().getBoxMax();
204
205 // if nothing changed do nothing
206 if (oldMin == boxMin and oldMax == boxMax) {
207 return {};
208 }
209
210 // sanity check that new size is actually positive
211 for (size_t i = 0; i < boxMin.size(); ++i) {
212 if (boxMin[i] >= boxMax[i]) {
213 utils::ExceptionHandler::exception(
214 "New box size in dimension {} is not positive!\nboxMin[{}] = {}\nboxMax[{}] = {}", i, i, boxMin[i], i,
215 boxMax[i]);
216 }
217 }
218
219 // warn if domain changes too drastically
220 const auto newLength = boxMax - boxMin;
221 const auto oldLength = oldMax - oldMin;
222 const auto relDiffLength = newLength / oldLength;
223 for (size_t i = 0; i < newLength.size(); ++i) {
224 // warning threshold is set arbitrary and up for change if needed
225 if (relDiffLength[i] > 1.3 or relDiffLength[i] < 0.7) {
226 AutoPasLog(WARN,
227 "LogicHandler.resize(): Domain size changed drastically in dimension {}! Gathered AutoTuning "
228 "information might not be applicable anymore!\n"
229 "Size old box : {}\n"
230 "Size new box : {}\n"
231 "Relative diff: {}",
232 i, utils::ArrayUtils::to_string(oldLength), utils::ArrayUtils::to_string(newLength),
233 utils::ArrayUtils::to_string(relDiffLength));
234 }
235 }
236
237 // check all particles
238 std::vector<Particle> particlesNowOutside;
239 for (auto pIter = _containerSelector.getCurrentContainer().begin(); pIter.isValid(); ++pIter) {
240 // make sure only owned ones are present
241 if (not pIter->isOwned()) {
242 utils::ExceptionHandler::exception(
243 "LogicHandler::resizeBox() encountered non owned particle. "
244 "When calling resizeBox() these should be already deleted. "
245 "This could be solved by calling updateContainer() before resizeBox().");
246 }
247 // owned particles that are now outside are removed from the container and returned
248 if (not utils::inBox(pIter->getR(), boxMin, boxMax)) {
249 particlesNowOutside.push_back(*pIter);
250 decreaseParticleCounter(*pIter);
251 internal::markParticleAsDeleted(*pIter);
252 }
253 }
254
255 // actually resize the container
256 _containerSelector.resizeBox(boxMin, boxMax);
257 // The container might have changed sufficiently enough so that we need more or less spacial locks
258 const auto boxLength = boxMax - boxMin;
259 const auto interactionLengthInv = 1. / (_containerSelector.getCurrentContainer().getInteractionLength());
260 initSpacialLocks(boxLength, interactionLengthInv);
261
262 // Set this flag, s.t., the container is rebuilt!
263 _neighborListsAreValid.store(false, std::memory_order_relaxed);
264
265 return particlesNowOutside;
266 }

377 std::tuple<bool, bool> deleteParticleFromBuffers(Particle &particle) {
378 // find the buffer the particle belongs to
379 auto &bufferCollection = particle.isOwned() ? _particleBuffer : _haloParticleBuffer;
380 for (auto &cell : bufferCollection) {
381 auto &buffer = cell._particles;
382 // if the address of the particle is between start and end of the buffer it is in this buffer
383 if (not buffer.empty() and &(buffer.front()) <= &particle and &particle <= &(buffer.back())) {
384 const bool isRearParticle = &particle == &buffer.back();
385 // swap-delete
386 particle = buffer.back();
387 buffer.pop_back();
388 return {true, not isRearParticle};
389 }
390 }
391 return {false, true};

603 void initSpacialLocks(const std::array<double, 3> &boxLength, double interactionLengthInv) {
604 using namespace autopas::utils::ArrayMath::literals;
605 using autopas::utils::ArrayMath::ceil;
606 using autopas::utils::ArrayUtils::static_cast_copy_array;
607
608 // The maximum number of spatial locks is capped at 1e6.
609 // This limit is chosen more or less arbitrary. It is big enough so that our regular MD simulations
610 // fall well within it and small enough so that no memory issues arise.
611 // There were no rigorous tests for an optimal number of locks.
612 // Without this cap, very large domains (or tiny cutoffs) would generate an insane number of locks,
613 // that could blow up the memory.
614 constexpr size_t maxNumSpacialLocks{1000000};
615
616 // One lock per interaction length or less if this would generate too many.
617 const std::array<size_t, 3> locksPerDim = [&]() {
618 // First naively calculate the number of locks if we simply take the desired cell length.
619 // Ceil because both decisions are possible, and we are generous gods.
620 const std::array<size_t, 3> locksPerDimNaive =
621 static_cast_copy_array<size_t>(ceil(boxLength * interactionLengthInv));
622 const auto totalLocksNaive =
623 std::accumulate(locksPerDimNaive.begin(), locksPerDimNaive.end(), 1ul, std::multiplies<>());
624 // If the number of locks is within the limits everything is fine and we can return.
625 if (totalLocksNaive <= maxNumSpacialLocks) {
626 return locksPerDimNaive;
627 } else {
628 // If the number of locks grows too large, calculate the locks per dimension proportionally to the side lengths.
629 // Calculate side length relative to dimension 0.
630 const std::array<double, 3> boxSideProportions = {
631 1.,
632 boxLength[0] / boxLength[1],
633 boxLength[0] / boxLength[2],
634 };
635 // With this, calculate the number of locks the first dimension should receive.
636 const auto prodProportions =
637 std::accumulate(boxSideProportions.begin(), boxSideProportions.end(), 1., std::multiplies<>());
638 // Needs floor, otherwise we exceed the limit.
639 const auto locksInFirstDimFloat = std::floor(std::cbrt(maxNumSpacialLocks * prodProportions));
640 // From this and the proportions relative to the first dimension, we can calculate the remaining number of locks
641 const std::array<size_t, 3> locksPerDimLimited = {
642 static_cast<size_t>(locksInFirstDimFloat), // omitted div by 1
643 static_cast<size_t>(locksInFirstDimFloat / boxSideProportions[1]),
644 static_cast<size_t>(locksInFirstDimFloat / boxSideProportions[2]),
645 };
646 return locksPerDimLimited;
647 }
648 }();
649 _spacialLocks.resize(locksPerDim[0]);
650 for (auto &lockVecVec : _spacialLocks) {
651 lockVecVec.resize(locksPerDim[1]);
652 for (auto &lockVec : lockVecVec) {
653 lockVec.resize(locksPerDim[2]);
654 for (auto &lockPtr : lockVec) {
655 if (not lockPtr) {
656 lockPtr = std::make_unique<std::mutex>();
657 }
658 }
659 }
660 }

999template <typename Particle>
1000void LogicHandler<Particle>::setParticleBuffers(const std::vector<FullParticleCell<Particle>> &particleBuffers,
1001 const std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
1002 auto exchangeBuffer = [](const auto &newBuffers, auto &oldBuffers, auto &particleCounter) {
1003 // sanity check
1004 if (oldBuffers.size() < newBuffers.size()) {
1005 autopas::utils::ExceptionHandler::exception(
1006 "The number of new buffers ({}) is larger than number of existing buffers ({})!", newBuffers.size(),
1007 oldBuffers.size());
1008 }
1009
1010 // we will clear the old buffers so subtract the particles from the counters.
1011 const auto numParticlesInOldBuffers =
1012 std::transform_reduce(oldBuffers.begin(), std::next(oldBuffers.begin(), newBuffers.size()), 0, std::plus<>(),
1013 [](const auto &cell) { return cell.size(); });
1014 particleCounter.fetch_sub(numParticlesInOldBuffers, std::memory_order_relaxed);
1015
1016 // clear the old buffers and copy the content of the new buffers over.
1017 size_t numParticlesInNewBuffers = 0;
1018 for (size_t i = 0; i < newBuffers.size(); ++i) {
1019 oldBuffers[i].clear();
1020 for (const auto &p : newBuffers[i]) {
1021 ++numParticlesInNewBuffers;
1022 oldBuffers[i].addParticle(p);
1023 }
1024 }
1025 // update the counters.
1026 particleCounter.fetch_add(numParticlesInNewBuffers, std::memory_order_relaxed);
1027 };
1028
1029 exchangeBuffer(particleBuffers, _particleBuffer, _numParticlesOwned);
1030 exchangeBuffer(haloParticleBuffers, _haloParticleBuffer, _numParticlesHalo);
1031}

1053 "PairwiseFunctor or TriwiseFunctor.");

1055 }();

1126template <bool newton3, class ContainerType, class PairwiseFunctor>
1127void LogicHandler<Particle>::computeRemainderInteractions2B(
1128 PairwiseFunctor *f, ContainerType &container, std::vector<FullParticleCell<Particle>> &particleBuffers,
1129 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
1130 // Sanity check. If this is violated feel free to add some logic here that adapts the number of locks.
1131 if (_bufferLocks.size() < particleBuffers.size()) {
1132 utils::ExceptionHandler::exception("Not enough locks for non-halo buffers! Num Locks: {}, Buffers: {}",
1133 _bufferLocks.size(), particleBuffers.size());
1134 }
1135
1136 // Balance buffers. This makes processing them with static scheduling quite efficient.
1137 // Also, if particles were not inserted in parallel, this enables us to process them in parallel now.
1138 // Cost is at max O(2N) worst O(N) per buffer collection and negligible compared to interacting them.
1139 auto cellToVec = [](auto &cell) -> std::vector<Particle> & { return cell._particles; };
1140 utils::ArrayUtils::balanceVectors(particleBuffers, cellToVec);
1141 utils::ArrayUtils::balanceVectors(haloParticleBuffers, cellToVec);
1142
1143 // The following part performs the main remainder traversal. The actual calculation is done in 4 steps carried out
1144 // in three helper functions.
1145
1146 // only activate time measurements if it will actually be logged
1147#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
1148 autopas::utils::Timer timerBufferContainer;
1149 autopas::utils::Timer timerPBufferPBuffer;
1150 autopas::utils::Timer timerPBufferHBuffer;
1151
1152 timerBufferContainer.start();
1153#endif
1154 // steps 1 & 2. particleBuffer with all close particles in container and haloParticleBuffer with owned, close
1155 // particles in container
1156 remainderHelperBufferContainer<newton3>(f, container, particleBuffers, haloParticleBuffers);
1157
1158#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
1159 timerBufferContainer.stop();
1160 timerPBufferPBuffer.start();
1161#endif
1162
1163 // step 3. particleBuffer with itself and all other buffers
1164 remainderHelperBufferBuffer<newton3>(f, particleBuffers, haloParticleBuffers);
1165
1166#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
1167 timerPBufferPBuffer.stop();
1168 timerPBufferHBuffer.start();
1169#endif
1170
1171 // step 4. particleBuffer with haloParticleBuffer
1172 remainderHelperBufferHaloBuffer<newton3>(f, particleBuffers, haloParticleBuffers);
1173
1174#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
1175 timerPBufferHBuffer.stop();
1176#endif

1178 // unpack particle SoAs. Halo data is not interesting
1179 for (auto &buffer : particleBuffers) {
1180 f->SoAExtractor(buffer, buffer._particleSoABuffer, 0);
1181 }
1182
1183 AutoPasLog(TRACE, "Timer Buffers <-> Container (1+2): {}", timerBufferContainer.getTotalTime());

1188}

1190template <class Particle>
1191template <bool newton3, class ContainerType, class PairwiseFunctor>
1192void LogicHandler<Particle>::remainderHelperBufferContainer(
1193 PairwiseFunctor *f, ContainerType &container, std::vector<FullParticleCell<Particle>> &particleBuffers,
1194 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
1195 using autopas::utils::ArrayUtils::static_cast_copy_array;
1196 using namespace autopas::utils::ArrayMath::literals;
1197
1198 // Bunch of shorthands
1199 const auto cutoff = container.getCutoff();
1200 const auto interactionLength = container.getInteractionLength();
1201 const auto interactionLengthInv = 1. / interactionLength;
1202 const auto haloBoxMin = container.getBoxMin() - interactionLength;
1203 const auto totalBoxLengthInv = 1. / (container.getBoxMax() + interactionLength - haloBoxMin);
1204 const std::array<size_t, 3> spacialLocksPerDim{
1205 _spacialLocks.size(),
1206 _spacialLocks[0].size(),
1207 _spacialLocks[0][0].size(),
1208 };
1209
1210 // Helper function to obtain the lock responsible for a given position.
1211 // Implemented as lambda because it can reuse a lot of information that is known in this context.
1212 const auto getSpacialLock = [&](const std::array<double, 3> &pos) -> std::mutex & {
1213 const auto posDistFromLowerCorner = pos - haloBoxMin;
1214 const auto relativePos = posDistFromLowerCorner * totalBoxLengthInv;
1215 // Lock coordinates are the position scaled by the number of locks
1216 const auto lockCoords =
1217 static_cast_copy_array<size_t>(static_cast_copy_array<double>(spacialLocksPerDim) * relativePos);
1218 return *_spacialLocks[lockCoords[0]][lockCoords[1]][lockCoords[2]];
1219 };
1220
1221 // one halo and particle buffer pair per thread
1222 AUTOPAS_OPENMP(parallel for schedule(static, 1) default(shared))
1223 for (int bufferId = 0; bufferId < particleBuffers.size(); ++bufferId) {
1224 auto &particleBuffer = particleBuffers[bufferId];
1225 auto &haloParticleBuffer = haloParticleBuffers[bufferId];
1226 // 1. particleBuffer with all close particles in container
1227 for (auto &&p1 : particleBuffer) {
1228 const auto pos = p1.getR();
1229 const auto min = pos - cutoff;
1230 const auto max = pos + cutoff;
1231 container.forEachInRegion(
1232 [&](auto &p2) {
1233 if constexpr (newton3) {
1234 const std::lock_guard<std::mutex> lock(getSpacialLock(p2.getR()));
1235 f->AoSFunctor(p1, p2, true);
1236 } else {
1237 f->AoSFunctor(p1, p2, false);
1238 // no need to calculate force enacted on a halo
1239 if (not p2.isHalo()) {
1240 const std::lock_guard<std::mutex> lock(getSpacialLock(p2.getR()));
1241 f->AoSFunctor(p2, p1, false);
1242 }
1243 }
1244 },
1245 min, max, IteratorBehavior::ownedOrHalo);
1246 }
1247
1248 // 2. haloParticleBuffer with owned, close particles in container
1249 for (auto &&p1halo : haloParticleBuffer) {
1250 const auto pos = p1halo.getR();
1251 const auto min = pos - cutoff;
1252 const auto max = pos + cutoff;
1253 container.forEachInRegion(
1254 [&](auto &p2) {
1255 // No need to apply anything to p1halo
1256 // -> AoSFunctor(p1, p2, false) not needed as it neither adds force nor Upot (potential energy)
1257 // -> newton3 argument needed for correct globals
1258 const std::lock_guard<std::mutex> lock(getSpacialLock(p2.getR()));
1259 f->AoSFunctor(p2, p1halo, newton3);
1260 },
1261 min, max, IteratorBehavior::owned);
1262 }
1263 }
1264}

1267template <bool newton3, class PairwiseFunctor>
1268void LogicHandler<Particle>::remainderHelperBufferBuffer(PairwiseFunctor *f,
1269 std::vector<FullParticleCell<Particle>> &particleBuffers,
1270 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
1271 // All (halo-)buffer interactions shall happen vectorized, hence, load all buffer data into SoAs
1272 for (auto &buffer : particleBuffers) {
1273 f->SoALoader(buffer, buffer._particleSoABuffer, 0, /*skipSoAResize*/ false);
1274 }
1275 for (auto &buffer : haloParticleBuffers) {

1279 AUTOPAS_OPENMP(parallel) {
1280 // For buffer interactions where bufferA == bufferB we can always enable newton3. For all interactions between
1281 // different buffers we turn newton3 always off, which ensures that only one thread at a time is writing to a
1282 // buffer. This saves expensive locks.
1283 // we can not use collapse here without locks, otherwise races would occur.
1284 AUTOPAS_OPENMP(for)
1285 for (size_t i = 0; i < particleBuffers.size(); ++i) {
1286 for (size_t jj = 0; jj < particleBuffers.size(); ++jj) {
1287 auto *particleBufferSoAA = &particleBuffers[i]._particleSoABuffer;
1288 const auto j = (i + jj) % particleBuffers.size();
1289 if (i == j) {
1290 f->SoAFunctorSingle(*particleBufferSoAA, true);
1291 } else {
1292 auto *particleBufferSoAB = &particleBuffers[j]._particleSoABuffer;
1293 f->SoAFunctorPair(*particleBufferSoAA, *particleBufferSoAB, false);
1294 }
1295 }
1296 }
1297 }

1512 auto &autoTuner = *_autoTunerRefs[interactionType];

1514 // https://github.com/AutoPas/AutoPas/issues/916
1515 LiveInfo info{};

1517 // if this iteration is not relevant take the same algorithm config as before.
1518 if (not functor.isRelevantForTuning()) {
1519 stillTuning = false;
1520 configuration = autoTuner.getCurrentConfig();
1521 // The currently selected container might not be compatible with the configuration for this functor. Check and
1522 // change if necessary. (see https://github.com/AutoPas/AutoPas/issues/871)
1523 if (_containerSelector.getCurrentContainer().getContainerType() != configuration.container) {
1524 _containerSelector.selectContainer(
1525 configuration.container,
1526 ContainerSelectorInfo(
1527 configuration.cellSizeFactor,
1528 _containerSelector.getCurrentContainer().getVerletSkin() / _neighborListRebuildFrequency,
1529 _neighborListRebuildFrequency, _verletClusterSize, configuration.loadEstimator));
1530 }
1531 const auto &container = _containerSelector.getCurrentContainer();
1532 traversalPtrOpt = autopas::utils::withStaticCellType<Particle>(
1533 container.getParticleCellTypeEnum(), [&](const auto &particleCellDummy) -> decltype(traversalPtrOpt) {
1534 // Can't make this unique_ptr const otherwise we can't move it later.
1535 auto traversalPtr =
1536 TraversalSelector<std::decay_t<decltype(particleCellDummy)>>::template generateTraversal<Functor>(
1537 configuration.traversal, functor, container.getTraversalSelectorInfo(), configuration.dataLayout,
1538 configuration.newton3);
1539
1540 // set sortingThreshold of the traversal if it can be casted to a CellTraversal and uses the CellFunctor
1541 if (auto *cellTraversalPtr =
1542 dynamic_cast<autopas::CellTraversal<std::decay_t<decltype(particleCellDummy)>> *>(
1543 traversalPtr.get())) {
1544 cellTraversalPtr->setSortingThreshold(_sortingThreshold);
1545 }
1546 if (traversalPtr->isApplicable()) {
1547 return std::optional{std::move(traversalPtr)};
1548 } else {
1549 return std::nullopt;
1550 }
1551 });
1552 } else {
1553 if (autoTuner.needsHomogeneityAndMaxDensityBeforePrepare()) {
1554 utils::Timer timerCalculateHomogeneity;
1555 timerCalculateHomogeneity.start();
1556 const auto &container = _containerSelector.getCurrentContainer();
1557 const auto [homogeneity, maxDensity] = autopas::utils::calculateHomogeneityAndMaxDensity(container);
1558 timerCalculateHomogeneity.stop();
1559 autoTuner.addHomogeneityAndMaxDensity(homogeneity, maxDensity, timerCalculateHomogeneity.getTotalTime());
1560 }
1561
1562 const auto needsLiveInfo = autoTuner.prepareIteration();
1563
1564 if (needsLiveInfo) {
1565 info.gather(_containerSelector.getCurrentContainer(), functor, _neighborListRebuildFrequency);
1566 autoTuner.receiveLiveInfo(info);
1567 }
1568
1569 std::tie(configuration, stillTuning) = autoTuner.getNextConfig();
1570
1571 // loop as long as we don't get a valid configuration
1572 bool rejectIndefinitely = false;
1573 while (true) {
1574 // applicability check also sets the container
1575 std::tie(traversalPtrOpt, rejectIndefinitely) =
1576 isConfigurationApplicable(configuration, functor, interactionType);
1577 if (traversalPtrOpt.has_value()) {
1578 break;
1579 }
1580 // if no config is left after rejecting this one an exception is thrown here.
1581 std::tie(configuration, stillTuning) = autoTuner.rejectConfig(configuration, rejectIndefinitely);
1582 }
1583 }
1584
1585#ifdef AUTOPAS_LOG_LIVEINFO
1586 // if live info has not been gathered yet, gather it now and log it
1587 if (info.get().empty()) {
1588 info.gather(_containerSelector.getCurrentContainer(), functor, _neighborListRebuildFrequency);
1589 }
1590 _liveInfoLogger.logLiveInfo(info, _iteration);
1591#endif
1592
1593 return {configuration, std::move(traversalPtrOpt.value()), stillTuning};
1594}
