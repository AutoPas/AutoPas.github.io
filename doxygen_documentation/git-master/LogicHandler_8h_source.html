44 [[nodiscard]] std::vector<Particle> collectLeavingParticlesFromBuffer(bool insertOwnedParticlesToContainer) {
45 const auto &boxMin = _autoTuner.getContainer()->getBoxMin();
46 const auto &boxMax = _autoTuner.getContainer()->getBoxMax();
47 std::vector<Particle> leavingBufferParticles{};
48 for (auto &buffer : _particleBuffer) {
49 if (insertOwnedParticlesToContainer) {
50 for (const auto &p : buffer) {
51 if (p.isDummy()) {
52 continue;
53 }
54 if (utils::inBox(p.getR(), boxMin, boxMax)) {
55 _autoTuner.getContainer()->template addParticle(p);
56 } else {
57 leavingBufferParticles.push_back(p);
58 }
59 }
60 buffer.clear();
61 } else {
62 for (auto iter = buffer.begin(); iter != buffer.end();) {
63 auto &p = *iter;
64
65 auto fastRemoveP = [&]() {
66 // Fast remove of particle, i.e., swap with last entry && pop.
67 std::swap(p, buffer.back());
68 buffer.pop_back();
69 // Do not increment the iter afterwards!
70 };
71 if (p.isDummy()) {
72 // We remove dummies!
73 fastRemoveP();
74 }
75 if (utils::notInBox(p.getR(), boxMin, boxMax)) {
76 leavingBufferParticles.push_back(p);
77 fastRemoveP();
78 } else {
79 ++iter;
80 }
81 }
82 }
83 }
84 return leavingBufferParticles;
85 }

90 [[nodiscard]] std::vector<Particle> updateContainer() {
91 bool doDataStructureUpdate = not neighborListsAreValid();
92
93 if (doDataStructureUpdate) {
94 _neighborListsAreValid = false;
95 }
96 // The next call also adds particles to the container if doDataStructureUpdate is true.
97 auto leavingBufferParticles = collectLeavingParticlesFromBuffer(doDataStructureUpdate);
98
99 AutoPasLog(debug, "Initiating container update.");
100 auto leavingParticles = _autoTuner.getContainer()->updateContainer(not doDataStructureUpdate);
101 leavingParticles.insert(leavingParticles.end(), leavingBufferParticles.begin(), leavingBufferParticles.end());
102
103 // Substract the amount of leaving particles from the number of owned particles.
104 _numParticlesOwned.fetch_sub(leavingParticles.size(), std::memory_order_relaxed);
105 // updateContainer deletes all halo particles.
106 std::for_each(_haloParticleBuffer.begin(), _haloParticleBuffer.end(), [](auto &buffer) { buffer.clear(); });
107 _numParticlesHalo.exchange(0, std::memory_order_relaxed);
108 return leavingParticles;
109 }

117 std::vector<Particle> resizeBox(const std::array<double, 3> &boxMin, const std::array<double, 3> &boxMax) {
118 const auto &oldMin = _autoTuner.getContainer()->getBoxMin();
119 const auto &oldMax = _autoTuner.getContainer()->getBoxMax();
120
121 // if nothing changed do nothing
122 if (oldMin == boxMin and oldMax == boxMax) {
123 return {};
124 }
125
126 // sanity check that new size is actually positive
127 for (size_t i = 0; i < boxMin.size(); ++i) {
128 if (boxMin[i] >= boxMax[i]) {
129 utils::ExceptionHandler::exception(
130 "New box size in dimension {} is not positive!\nboxMin[{}] = {}\nboxMax[{}] = {}", i, i, boxMin[i], i,
131 boxMax[i]);
132 }
133 }
134
135 // warn if domain changes too drastically
136 const auto newLength = utils::ArrayMath::sub(boxMax, boxMin);
137 const auto oldLength = utils::ArrayMath::sub(oldMax, oldMin);
138 const auto relDiffLength = utils::ArrayMath::div(newLength, oldLength);
139 for (size_t i = 0; i < newLength.size(); ++i) {
140 // warning threshold is set arbitrary and up for change if needed
141 if (relDiffLength[i] > 1.3 or relDiffLength[i] < 0.7) {
142 AutoPasLog(warn,
143 "LogicHandler.resize(): Domain size changed drastically in dimension {}! Gathered AutoTuning "
144 "information might not be applicable anymore!\n"
145 "Size old box : {}\n"
146 "Size new box : {}\n"
147 "Relative diff: {}",
148 i, utils::ArrayUtils::to_string(oldLength), utils::ArrayUtils::to_string(newLength),
149 utils::ArrayUtils::to_string(relDiffLength));
150 }
151 }
152
153 // check all particles
154 std::vector<Particle> particlesNowOutside;
155 for (auto pIter = _autoTuner.getContainer()->begin(); pIter.isValid(); ++pIter) {
156 // make sure only owned ones are present
157 if (not pIter->isOwned()) {
158 utils::ExceptionHandler::exception(
159 "LogicHandler::resizeBox() encountered non owned particle. "
160 "When calling resizeBox() these should be already deleted. "
161 "This could be solved by calling updateContainer() before resizeBox().");
162 }
163 // owned particles that are now outside are removed from the container and returned
164 if (not utils::inBox(pIter->getR(), boxMin, boxMax)) {
165 particlesNowOutside.push_back(*pIter);
166 deleteParticle(pIter);
167 }
168 }
169
170 // actually resize the container
171 _autoTuner.resizeBox(boxMin, boxMax);
172 // Set this flag, s.t., the container is rebuilt!
173 _neighborListsAreValid = false;
174
175 return particlesNowOutside;
176 }

181 void addParticle(const Particle &p) {
182 if (not neighborListsAreValid()) {
183 // Container has to be invalid to be able to add Particles!
184 _autoTuner.getContainer()->addParticle(p);
185 } else {
186 const auto &boxMin = _autoTuner.getContainer()->getBoxMin();
187 const auto &boxMax = _autoTuner.getContainer()->getBoxMax();
188 if (utils::notInBox(p.getR(), boxMin, boxMax)) {
189 autopas::utils::ExceptionHandler::exception(
190 "LogicHandler: Trying to add a particle that is not in the bounding box.\n"
191 "Box Min {}\n"
192 "Box Max {}\n"
193 "{}",
194 utils::ArrayUtils::to_string(boxMin), utils::ArrayUtils::to_string(boxMax), p.toString());
195 }
196 // If the container is valid, we add it to the particle buffer.
197 _particleBuffer[autopas_get_thread_num()].push_back(p);
198 }
199 _numParticlesOwned.fetch_add(1, std::memory_order_relaxed);

399 container->getVerletSkin(), container->getCutoff() + container->getVerletSkin());
