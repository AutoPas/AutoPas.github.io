102 [[nodiscard]] std::vector<Particle> collectLeavingParticlesFromBuffer(bool insertOwnedParticlesToContainer) {
103 const auto &boxMin = _containerSelector.getCurrentContainer().getBoxMin();
104 const auto &boxMax = _containerSelector.getCurrentContainer().getBoxMax();
105 std::vector<Particle> leavingBufferParticles{};
106 for (auto &cell : _particleBuffer) {
107 auto &buffer = cell._particles;
108 if (insertOwnedParticlesToContainer) {
109 for (const auto &p : buffer) {
110 if (p.isDummy()) {
111 continue;
112 }
113 if (utils::inBox(p.getR(), boxMin, boxMax)) {
114 _containerSelector.getCurrentContainer().addParticle(p);
115 } else {
116 leavingBufferParticles.push_back(p);
117 }
118 }
119 buffer.clear();
120 } else {
121 for (auto iter = buffer.begin(); iter < buffer.end();) {
122 auto &p = *iter;
123
124 auto fastRemoveP = [&]() {
125 // Fast remove of particle, i.e., swap with last entry && pop.
126 std::swap(p, buffer.back());
127 buffer.pop_back();
128 // Do not increment the iter afterwards!
129 };
130 if (p.isDummy()) {
131 // We remove dummies!
132 fastRemoveP();
133 // In case we swapped a dummy here, don't increment the iterator and do another iteration to check again.
134 continue;
135 }
136 // if p was a dummy a new particle might now be at the memory location of p so we need to check that.
137 // We also just might have deleted the last particle in the buffer in that case the inBox check is meaningless
138 if (not buffer.empty() and utils::notInBox(p.getR(), boxMin, boxMax)) {
139 leavingBufferParticles.push_back(p);
140 fastRemoveP();
141 } else {
142 ++iter;
143 }
144 }
145 }
146 }
147 return leavingBufferParticles;
148 }

153 [[nodiscard]] std::vector<Particle> updateContainer() {
154 bool doDataStructureUpdate = not neighborListsAreValid();
155
156 if (doDataStructureUpdate) {
157 _neighborListsAreValid.store(false, std::memory_order_relaxed);
158 }
159 // The next call also adds particles to the container if doDataStructureUpdate is true.
160 auto leavingBufferParticles = collectLeavingParticlesFromBuffer(doDataStructureUpdate);
161
162 AutoPasLog(DEBUG, "Initiating container update.");
163 auto leavingParticles = _containerSelector.getCurrentContainer().updateContainer(not doDataStructureUpdate);
164 leavingParticles.insert(leavingParticles.end(), leavingBufferParticles.begin(), leavingBufferParticles.end());
165
166 // Substract the amount of leaving particles from the number of owned particles.
167 _numParticlesOwned.fetch_sub(leavingParticles.size(), std::memory_order_relaxed);
168 // updateContainer deletes all halo particles.
169 std::for_each(_haloParticleBuffer.begin(), _haloParticleBuffer.end(), [](auto &buffer) { buffer.clear(); });
170 _numParticlesHalo.store(0, std::memory_order_relaxed);

180 std::vector<Particle> resizeBox(const std::array<double, 3> &boxMin, const std::array<double, 3> &boxMax) {
181 using namespace autopas::utils::ArrayMath::literals;
182 const auto &oldMin = _containerSelector.getCurrentContainer().getBoxMin();
183 const auto &oldMax = _containerSelector.getCurrentContainer().getBoxMax();
184
185 // if nothing changed do nothing
186 if (oldMin == boxMin and oldMax == boxMax) {
187 return {};
188 }
189
190 // sanity check that new size is actually positive
191 for (size_t i = 0; i < boxMin.size(); ++i) {
192 if (boxMin[i] >= boxMax[i]) {
193 utils::ExceptionHandler::exception(
194 "New box size in dimension {} is not positive!\nboxMin[{}] = {}\nboxMax[{}] = {}", i, i, boxMin[i], i,
195 boxMax[i]);
196 }
197 }
198
199 // warn if domain changes too drastically
200 const auto newLength = boxMax - boxMin;
201 const auto oldLength = oldMax - oldMin;
202 const auto relDiffLength = newLength / oldLength;
203 for (size_t i = 0; i < newLength.size(); ++i) {
204 // warning threshold is set arbitrary and up for change if needed
205 if (relDiffLength[i] > 1.3 or relDiffLength[i] < 0.7) {
206 AutoPasLog(WARN,
207 "LogicHandler.resize(): Domain size changed drastically in dimension {}! Gathered AutoTuning "
208 "information might not be applicable anymore!\n"
209 "Size old box : {}\n"
210 "Size new box : {}\n"
211 "Relative diff: {}",
212 i, utils::ArrayUtils::to_string(oldLength), utils::ArrayUtils::to_string(newLength),
213 utils::ArrayUtils::to_string(relDiffLength));
214 }
215 }
216
217 // check all particles
218 std::vector<Particle> particlesNowOutside;
219 for (auto pIter = _containerSelector.getCurrentContainer().begin(); pIter.isValid(); ++pIter) {
220 // make sure only owned ones are present
221 if (not pIter->isOwned()) {
222 utils::ExceptionHandler::exception(
223 "LogicHandler::resizeBox() encountered non owned particle. "
224 "When calling resizeBox() these should be already deleted. "
225 "This could be solved by calling updateContainer() before resizeBox().");
226 }
227 // owned particles that are now outside are removed from the container and returned
228 if (not utils::inBox(pIter->getR(), boxMin, boxMax)) {
229 particlesNowOutside.push_back(*pIter);
230 decreaseParticleCounter(*pIter);
231 internal::markParticleAsDeleted(*pIter);
232 }
233 }
234
235 // actually resize the container
236 _containerSelector.resizeBox(boxMin, boxMax);
237 // The container might have changed sufficiently enough so that we need more or less spacial locks
238 const auto boxLength = boxMax - boxMin;
239 const auto interactionLengthInv = 1. / (_containerSelector.getCurrentContainer().getInteractionLength());
240 initSpacialLocks(boxLength, interactionLengthInv);
241
242 // Set this flag, s.t., the container is rebuilt!
243 _neighborListsAreValid.store(false, std::memory_order_relaxed);
244
245 return particlesNowOutside;
246 }

354 std::tuple<bool, bool> deleteParticleFromBuffers(Particle &particle) {
355 // find the buffer the particle belongs to
356 auto &bufferCollection = particle.isOwned() ? _particleBuffer : _haloParticleBuffer;
357 for (auto &cell : bufferCollection) {
358 auto &buffer = cell._particles;
359 // if the address of the particle is between start and end of the buffer it is in this buffer
360 if (not buffer.empty() and &(buffer.front()) <= &particle and &particle <= &(buffer.back())) {
361 const bool isRearParticle = &particle == &buffer.back();
362 // swap-delete
363 particle = buffer.back();
364 buffer.pop_back();
365 return {true, not isRearParticle};
366 }
367 }
368 return {false, true};
369 }

412 template <class Iterator>
413 typename Iterator::ParticleVecType gatherAdditionalVectors(IteratorBehavior behavior) {
414 typename Iterator::ParticleVecType additionalVectors;
415 additionalVectors.reserve(static_cast<bool>(behavior & IteratorBehavior::owned) * _particleBuffer.size() +
416 static_cast<bool>(behavior & IteratorBehavior::halo) * _haloParticleBuffer.size());
417 if (behavior & IteratorBehavior::owned) {
418 for (auto &buffer : _particleBuffer) {
419 // Don't insert empty buffers. This also means that we won't pick up particles added during iterating if they
420 // go to the buffers. But since we wouldn't pick them up if they go into the container to a cell that the
421 // iterators already passed this is unsupported anyways.
422 if (not buffer.isEmpty()) {
423 additionalVectors.push_back(&(buffer._particles));
424 }
425 }
426 }
427 if (behavior & IteratorBehavior::halo) {
428 for (auto &buffer : _haloParticleBuffer) {
429 if (not buffer.isEmpty()) {
430 additionalVectors.push_back(&(buffer._particles));
431 }
432 }
433 }
434 return additionalVectors;

797template <typename Particle>
798void LogicHandler<Particle>::setParticleBuffers(const std::vector<FullParticleCell<Particle>> &particleBuffers,
799 const std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
800 auto exchangeBuffer = [](const auto &newBuffers, auto &oldBuffers, auto &particleCounter) {
801 // sanity check
802 if (oldBuffers.size() < newBuffers.size()) {
803 autopas::utils::ExceptionHandler::exception(
804 "The number of new buffers ({}) is larger than number of existing buffers ({})!", newBuffers.size(),
805 oldBuffers.size());
806 }
807
808 // we will clear the old buffers so subtract the particles from the counters.
809 const auto numParticlesInOldBuffers =
810 std::transform_reduce(oldBuffers.begin(), std::next(oldBuffers.begin(), newBuffers.size()), 0, std::plus<>(),
811 [](const auto &cell) { return cell.size(); });
812 particleCounter.fetch_sub(numParticlesInOldBuffers, std::memory_order_relaxed);
813
814 // clear the old buffers and copy the content of the new buffers over.
815 size_t numParticlesInNewBuffers = 0;
816 for (size_t i = 0; i < newBuffers.size(); ++i) {
817 oldBuffers[i].clear();
818 for (const auto &p : newBuffers[i]) {
819 ++numParticlesInNewBuffers;
820 oldBuffers[i].addParticle(p);
821 }
822 }
823 // update the counters.
824 particleCounter.fetch_add(numParticlesInNewBuffers, std::memory_order_relaxed);
825 };
826
827 exchangeBuffer(particleBuffers, _particleBuffer, _numParticlesOwned);
828 exchangeBuffer(haloParticleBuffers, _haloParticleBuffer, _numParticlesHalo);
829}

863 withStaticContainerType(container, [&](auto &actualContainerType) {
864 if (configuration.newton3) {
865 doRemainderTraversal<true>(&functor, actualContainerType, _particleBuffer, _haloParticleBuffer);
866 } else {
867 doRemainderTraversal<false>(&functor, actualContainerType, _particleBuffer, _haloParticleBuffer);
868 }
869 });
870 timerRemainderTraversal.stop();
871 functor.endTraversal(configuration.newton3);
872
873 const auto [energyPsys, energyPkg, energyRam, energyTotal] = _autoTuner.sampleEnergy();
874
875 timerTotal.stop();
876
877 constexpr auto nanD = std::numeric_limits<double>::quiet_NaN();
878 constexpr auto nanL = std::numeric_limits<long>::quiet_NaN();
879 return {timerIteratePairwise.getTotalTime(),
880 timerRemainderTraversal.getTotalTime(),
881 timerRebuild.getTotalTime(),
882 timerTotal.getTotalTime(),
883 energyMeasurementsPossible,
884 energyMeasurementsPossible ? energyPsys : nanD,
885 energyMeasurementsPossible ? energyPkg : nanD,

890template <class Particle>
891template <bool newton3, class ContainerType, class PairwiseFunctor>
892void LogicHandler<Particle>::doRemainderTraversal(PairwiseFunctor *f, ContainerType &container,
893 std::vector<FullParticleCell<Particle>> &particleBuffers,
894 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
895 // Sanity check. If this is violated feel free to add some logic here that adapts the number of locks.
896 if (_bufferLocks.size() < particleBuffers.size()) {
897 utils::ExceptionHandler::exception("Not enough locks for non-halo buffers! Num Locks: {}, Buffers: {}",
898 _bufferLocks.size(), particleBuffers.size());
899 }
900
901 // Balance buffers. This makes processing them with static scheduling quite efficient.
902 // Also, if particles were not inserted in parallel, this enables us to process them in parallel now.
903 // Cost is at max O(2N) worst O(N) per buffer collection and negligible compared to interacting them.
904 auto cellToVec = [](auto &cell) -> std::vector<Particle> & { return cell._particles; };
905 utils::ArrayUtils::balanceVectors(particleBuffers, cellToVec);
906 utils::ArrayUtils::balanceVectors(haloParticleBuffers, cellToVec);
907
908 // The following part performs the main remainder traversal. The actual calculation is done in 4 steps carried out
909 // in three helper functions.
910
911 // only activate time measurements if it will actually be logged
912#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
913 autopas::utils::Timer timerBufferContainer;
914 autopas::utils::Timer timerPBufferPBuffer;
915 autopas::utils::Timer timerPBufferHBuffer;
916
917 timerBufferContainer.start();
918#endif
919 // steps 1 & 2. particleBuffer with all close particles in container and haloParticleBuffer with owned, close
920 // particles in container
921 remainderHelperBufferContainer<newton3>(f, container, particleBuffers, haloParticleBuffers);
922
923#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
924 timerBufferContainer.stop();
925 timerPBufferPBuffer.start();
926#endif
927
928 // step 3. particleBuffer with itself and all other buffers
929 remainderHelperBufferBuffer<newton3>(f, particleBuffers, haloParticleBuffers);
930
931#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
932 timerPBufferPBuffer.stop();
933 timerPBufferHBuffer.start();
934#endif
935
936 // step 4. particleBuffer with haloParticleBuffer
937 remainderHelperBufferHaloBuffer<newton3>(f, particleBuffers, haloParticleBuffers);
938
939#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
940 timerPBufferHBuffer.stop();
941#endif
942
943 // unpack particle SoAs. Halo data is not interesting
944 for (auto &buffer : particleBuffers) {
945 f->SoAExtractor(buffer, buffer._particleSoABuffer, 0);
946 }
947
948 AutoPasLog(TRACE, "Timer Buffers <-> Container (1+2): {}", timerBufferContainer.getTotalTime());
949 AutoPasLog(TRACE, "Timer PBuffers<-> PBuffer ( 3): {}", timerPBufferPBuffer.getTotalTime());
950 AutoPasLog(TRACE, "Timer PBuffers<-> HBuffer ( 4): {}", timerPBufferHBuffer.getTotalTime());
951
952 // Note: haloParticleBuffer with itself is NOT needed, as interactions between halo particles are unneeded!
953}

955template <class Particle>
956template <bool newton3, class ContainerType, class PairwiseFunctor>
957void LogicHandler<Particle>::remainderHelperBufferContainer(
958 PairwiseFunctor *f, ContainerType &container, std::vector<FullParticleCell<Particle>> &particleBuffers,
959 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
960 using autopas::utils::ArrayUtils::static_cast_copy_array;
961 using namespace autopas::utils::ArrayMath::literals;
962
963 const auto haloBoxMin = container.getBoxMin() - container.getInteractionLength();
964 const auto interactionLengthInv = 1. / container.getInteractionLength();
965
966 const double cutoff = container.getCutoff();
967 // one halo and particle buffer pair per thread
968 AUTOPAS_OPENMP(parallel for schedule(static, 1) shared(f, _spacialLocks, haloBoxMin, interactionLengthInv))
969 for (int bufferId = 0; bufferId < particleBuffers.size(); ++bufferId) {
970 auto &particleBuffer = particleBuffers[bufferId];
971 auto &haloParticleBuffer = haloParticleBuffers[bufferId];
972 // 1. particleBuffer with all close particles in container
973 for (auto &&p1 : particleBuffer) {
974 const auto pos = p1.getR();
975 const auto min = pos - cutoff;
976 const auto max = pos + cutoff;
977 container.forEachInRegion(
978 [&](auto &p2) {
979 const auto lockCoords = static_cast_copy_array<size_t>((p2.getR() - haloBoxMin) * interactionLengthInv);
980 if constexpr (newton3) {
981 const std::lock_guard<std::mutex> lock(*_spacialLocks[lockCoords[0]][lockCoords[1]][lockCoords[2]]);
982 f->AoSFunctor(p1, p2, true);
983 } else {
984 f->AoSFunctor(p1, p2, false);
985 // no need to calculate force enacted on a halo
986 if (not p2.isHalo()) {
987 const std::lock_guard<std::mutex> lock(*_spacialLocks[lockCoords[0]][lockCoords[1]][lockCoords[2]]);
988 f->AoSFunctor(p2, p1, false);
989 }
990 }
991 },
992 min, max, IteratorBehavior::ownedOrHalo);
993 }
994
995 // 2. haloParticleBuffer with owned, close particles in container
996 for (auto &&p1halo : haloParticleBuffer) {
997 const auto pos = p1halo.getR();
998 const auto min = pos - cutoff;
999 const auto max = pos + cutoff;
1000 container.forEachInRegion(
1001 [&](auto &p2) {
1002 const auto lockCoords = static_cast_copy_array<size_t>((p2.getR() - haloBoxMin) * interactionLengthInv);
1003 // No need to apply anything to p1halo
1004 // -> AoSFunctor(p1, p2, false) not needed as it neither adds force nor Upot (potential energy)
1005 // -> newton3 argument needed for correct globals
1006 const std::lock_guard<std::mutex> lock(*_spacialLocks[lockCoords[0]][lockCoords[1]][lockCoords[2]]);
1007 f->AoSFunctor(p2, p1halo, newton3);
1008 },
1009 min, max, IteratorBehavior::owned);
1010 }
1011 }
1012}

1014template <class Particle>
1015template <bool newton3, class PairwiseFunctor>
1016void LogicHandler<Particle>::remainderHelperBufferBuffer(PairwiseFunctor *f,
1017 std::vector<FullParticleCell<Particle>> &particleBuffers,
1018 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
1019 // All (halo-)buffer interactions shall happen vectorized, hence, load all buffer data into SoAs
1020 for (auto &buffer : particleBuffers) {
1021 f->SoALoader(buffer, buffer._particleSoABuffer, 0, /*skipSoAResize*/ false);
1022 }
1023 for (auto &buffer : haloParticleBuffers) {
1024 f->SoALoader(buffer, buffer._particleSoABuffer, 0, /*skipSoAResize*/ false);
1025 }
1026
1027 AUTOPAS_OPENMP(parallel) {
1028 // For buffer interactions where bufferA == bufferB we can always enable newton3. For all interactions between
1029 // different buffers we turn newton3 always off, which ensures that only one thread at a time is writing to a
1030 // buffer. This saves expensive locks.
1031 // we can not use collapse here without locks, otherwise races would occur.
1032 AUTOPAS_OPENMP(for)
1033 for (size_t i = 0; i < particleBuffers.size(); ++i) {
1034 for (size_t jj = 0; jj < particleBuffers.size(); ++jj) {
1035 auto *particleBufferSoAA = &particleBuffers[i]._particleSoABuffer;
1036 const auto j = (i + jj) % particleBuffers.size();
1037 if (i == j) {
1038 f->SoAFunctorSingle(*particleBufferSoAA, true);
1039 } else {
1040 auto *particleBufferSoAB = &particleBuffers[j]._particleSoABuffer;
1041 f->SoAFunctorPair(*particleBufferSoAA, *particleBufferSoAB, false);
1042 }
1043 }
1044 }
1045 }
1046}

1049template <bool newton3, class PairwiseFunctor>
1050void LogicHandler<Particle>::remainderHelperBufferHaloBuffer(
1051 PairwiseFunctor *f, std::vector<FullParticleCell<Particle>> &particleBuffers,
1052 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
1053 // Here, phase / color based parallelism turned out to be more efficient than tasks
1054 AUTOPAS_OPENMP(parallel)
1055 for (int interactionOffset = 0; interactionOffset < haloParticleBuffers.size(); ++interactionOffset) {
1056 AUTOPAS_OPENMP(for)
1057 for (size_t i = 0; i < particleBuffers.size(); ++i) {
1058 auto &particleBufferSoA = particleBuffers[i]._particleSoABuffer;
1059 auto &haloBufferSoA =
1060 haloParticleBuffers[(i + interactionOffset) % haloParticleBuffers.size()]._particleSoABuffer;
1061 f->SoAFunctorPair(particleBufferSoA, haloBufferSoA, false);
1062 }
1063 }

1066template <typename Particle>
1067template <class Functor>
1068std::tuple<Configuration, std::unique_ptr<TraversalInterface>, bool> LogicHandler<Particle>::selectConfiguration(
1069 Functor &functor) {
1070 bool stillTuning = false;
1071 Configuration configuration{};
1072 std::optional<std::unique_ptr<TraversalInterface>> traversalPtrOpt{};
1073 LiveInfo info{};
1074 // if this iteration is not relevant take the same algorithm config as before.
1075 if (not functor.isRelevantForTuning()) {
1076 stillTuning = false;
1077 configuration = _autoTuner.getCurrentConfig();
1078 const auto &container = _containerSelector.getCurrentContainer();
1079 traversalPtrOpt = autopas::utils::withStaticCellType<Particle>(
1080 container.getParticleCellTypeEnum(), [&](const auto &particleCellDummy) -> decltype(traversalPtrOpt) {
1081 // Can't make this unique_ptr const otherwise we can't move it later.
1082 auto traversalPtr =
1083 TraversalSelector<std::decay_t<decltype(particleCellDummy)>>::template generateTraversal<Functor>(
1084 configuration.traversal, functor, container.getTraversalSelectorInfo(), configuration.dataLayout,
1085 configuration.newton3);
1086
1087 // set sortingThreshold of the traversal if it can be casted to a CellPairTraversal and uses the CellFunctor
1088 if (auto *cellPairTraversalPtr =
1089 dynamic_cast<autopas::CellPairTraversal<std::decay_t<decltype(particleCellDummy)>> *>(
1090 traversalPtr.get())) {
1091 cellPairTraversalPtr->setSortingThreshold(_sortingThreshold);
1092 }
1093
1094 if (traversalPtr->isApplicable()) {
1095 return std::optional{std::move(traversalPtr)};
1096 } else {
1097 return std::nullopt;
1098 }
1099 });
1100 } else {
1101 if (_autoTuner.needsHomogeneityAndMaxDensityBeforePrepare()) {
1102 utils::Timer timerCalculateHomogeneity;
1103 timerCalculateHomogeneity.start();
1104 const auto &container = _containerSelector.getCurrentContainer();
1105 const auto [homogeneity, maxDensity] = autopas::utils::calculateHomogeneityAndMaxDensity(container);
1106 timerCalculateHomogeneity.stop();
1107 _autoTuner.addHomogeneityAndMaxDensity(homogeneity, maxDensity, timerCalculateHomogeneity.getTotalTime());
1108 }
1109
1110 const auto needsLiveInfo = _autoTuner.prepareIteration();
1111
1112 if (needsLiveInfo) {
1113 info.gather(_containerSelector.getCurrentContainer(), functor, _neighborListRebuildFrequency);
1114 _autoTuner.receiveLiveInfo(info);
1115 }
1116
1117 std::tie(configuration, stillTuning) = _autoTuner.getNextConfig();
1118
1119 // loop as long as we don't get a valid configuration
1120 bool rejectIndefinitely = false;
1121 while (true) {
1122 // applicability check also sets the container
1123 std::tie(traversalPtrOpt, rejectIndefinitely) = isConfigurationApplicable(configuration, functor);
1124 if (traversalPtrOpt.has_value()) {
1125 break;
1126 }
1127 // if no config is left after rejecting this one an exception is thrown here.
1128 std::tie(configuration, stillTuning) = _autoTuner.rejectConfig(configuration, rejectIndefinitely);
1129 }
1130 }
1131
1132#ifdef AUTOPAS_LOG_LIVEINFO
1133 // if live info has not been gathered yet, gather it now and log it
1134 if (info.get().empty()) {
1135 info.gather(_containerSelector.getCurrentContainer(), functor, _neighborListRebuildFrequency);
1136 }
1137 _liveInfoLogger.logLiveInfo(info, _iteration);
1138#endif
1139
1140 return {configuration, std::move(traversalPtrOpt.value()), stillTuning};
1141}

1150 tuningTimer.stop();

1161 for (const auto &buffer : buffers) {
1162 ss << buffer.size() << ", ";
1163 sum += buffer.size();
1164 }
1165 ss << " Total: " << sum;

1168 AutoPasLog(TRACE, "particleBuffer size : {}", bufferSizeListing(_particleBuffer));

1171 AutoPasLog(DEBUG, "RemainderTraversal took {} ns", measurements.timeRemainderTraversal);
1172 AutoPasLog(DEBUG, "RebuildNeighborLists took {} ns", measurements.timeRebuild);
1173 AutoPasLog(DEBUG, "Container::iteratePairwise took {} ns", measurements.timeTotal);

1183 // if this was a major iteration add measurements and bump counters
1184 if (functor->isRelevantForTuning()) {

1186 // choose the metric of interest
1187 const auto measurement = [&]() {
1188 switch (_autoTuner.getTuningMetric()) {
1189 case TuningMetricOption::time:
1190 return measurements.timeTotal;
1191 case TuningMetricOption::energy:
1192 return measurements.energyTotal;
1193 default:
1194 autopas::utils::ExceptionHandler::exception(
1195 "LogicHandler::iteratePairwisePipeline(): Unknown tuning metric.");

1197 }
1198 }();
1199 _autoTuner.addMeasurement(measurement, not neighborListsAreValid());
1200 } else {
1201 AutoPasLog(TRACE, "Skipping adding of sample because functor is not marked relevant.");
1202 }
1203
1204 // this function depends on LogicHandler's and the AutoTuner's iteration counters,
1205 // that should not have been updated yet.
1206 if (not neighborListsAreValid() /*we have done a rebuild now*/) {
1207 // list is now valid
1208 _neighborListsAreValid.store(true, std::memory_order_relaxed);

1213 _containerSelector.getCurrentContainer().setStepsSinceLastRebuild(_stepsSinceLastListRebuild);
1214
1215 _autoTuner.bumpIterationCounters();

1217 }
1218 return stillTuning;

1221template <typename Particle>
1222template <class PairwiseFunctor>
1223std::tuple<std::optional<std::unique_ptr<TraversalInterface>>, bool> LogicHandler<Particle>::isConfigurationApplicable(
1224 const Configuration &conf, PairwiseFunctor &pairwiseFunctor) {
1225 // Check if the container supports the traversal
1226 const auto allContainerTraversals = compatibleTraversals::allCompatibleTraversals(conf.container);
1227 if (allContainerTraversals.find(conf.traversal) == allContainerTraversals.end()) {
1228 AutoPasLog(DEBUG, "Configuration rejected: Container {} does not support the traversal {}.", conf.container,
1229 conf.traversal);
1230 return {std::nullopt, true};
1231 }
1232
1233 // Check if the required Newton 3 mode is supported by the functor
1234 if ((conf.newton3 == Newton3Option::enabled and not pairwiseFunctor.allowsNewton3()) or
1235 (conf.newton3 == Newton3Option::disabled and not pairwiseFunctor.allowsNonNewton3())) {
1236 AutoPasLog(DEBUG, "Configuration rejected: The functor doesn't support Newton 3 {}!", conf.newton3);
1237 return {std::nullopt, true};
1238 }
1239
1240 // Check if the traversal is applicable to the current state of the container
1241 _containerSelector.selectContainer(
1242 conf.container,
1243 ContainerSelectorInfo(conf.cellSizeFactor,
1244 _containerSelector.getCurrentContainer().getVerletSkin() / _neighborListRebuildFrequency,
1245 _neighborListRebuildFrequency, _verletClusterSize, conf.loadEstimator));
1246 const auto &container = _containerSelector.getCurrentContainer();
1247 const auto traversalInfo = container.getTraversalSelectorInfo();
1248
1249 auto traversalPtrOpt = autopas::utils::withStaticCellType<Particle>(
1250 container.getParticleCellTypeEnum(),
1251 [&](const auto &particleCellDummy) -> std::optional<std::unique_ptr<TraversalInterface>> {
1252 // Can't make this unique_ptr const otherwise we can't move it later.
1253 auto traversalPtr =
1254 TraversalSelector<std::decay_t<decltype(particleCellDummy)>>::template generateTraversal<PairwiseFunctor>(
1255 conf.traversal, pairwiseFunctor, traversalInfo, conf.dataLayout, conf.newton3);
1256
1257 // set sortingThreshold of the traversal if it can be casted to a CellPairTraversal and uses the CellFunctor
1258 if (auto *cellPairTraversalPtr =
1259 dynamic_cast<autopas::CellPairTraversal<std::decay_t<decltype(particleCellDummy)>> *>(
1260 traversalPtr.get())) {
1261 cellPairTraversalPtr->setSortingThreshold(_sortingThreshold);
1262 }
1263
1264 if (traversalPtr->isApplicable()) {
1265 return std::optional{std::move(traversalPtr)};
1266 } else {
1267 return std::nullopt;
1268 }
1269 });
1270 return {std::move(traversalPtrOpt), false};
1271}
