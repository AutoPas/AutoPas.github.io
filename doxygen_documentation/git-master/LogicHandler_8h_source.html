97 [[nodiscard]] std::vector<Particle> collectLeavingParticlesFromBuffer(bool insertOwnedParticlesToContainer) {
98 const auto &boxMin = _containerSelector.getCurrentContainer().getBoxMin();
99 const auto &boxMax = _containerSelector.getCurrentContainer().getBoxMax();
100 std::vector<Particle> leavingBufferParticles{};
101 for (auto &cell : _particleBuffer) {
102 auto &buffer = cell._particles;
103 if (insertOwnedParticlesToContainer) {
104 for (const auto &p : buffer) {
105 if (p.isDummy()) {
106 continue;
107 }
108 if (utils::inBox(p.getR(), boxMin, boxMax)) {
109 _containerSelector.getCurrentContainer().addParticle(p);
110 } else {
111 leavingBufferParticles.push_back(p);
112 }
113 }
114 buffer.clear();
115 } else {
116 for (auto iter = buffer.begin(); iter < buffer.end();) {
117 auto &p = *iter;
118
119 auto fastRemoveP = [&]() {
120 // Fast remove of particle, i.e., swap with last entry && pop.
121 std::swap(p, buffer.back());
122 buffer.pop_back();
123 // Do not increment the iter afterwards!
124 };
125 if (p.isDummy()) {
126 // We remove dummies!
127 fastRemoveP();
128 // In case we swapped a dummy here, don't increment the iterator and do another iteration to check again.
129 continue;
130 }
131 // if p was a dummy a new particle might now be at the memory location of p so we need to check that.
132 // We also just might have deleted the last particle in the buffer in that case the inBox check is meaningless
133 if (not buffer.empty() and utils::notInBox(p.getR(), boxMin, boxMax)) {
134 leavingBufferParticles.push_back(p);
135 fastRemoveP();
136 } else {
137 ++iter;
138 }
139 }
140 }
141 }
142 return leavingBufferParticles;
143 }

148 [[nodiscard]] std::vector<Particle> updateContainer() {
149 bool doDataStructureUpdate = not neighborListsAreValid();
150
151 if (doDataStructureUpdate) {
152 _neighborListsAreValid.store(false, std::memory_order_relaxed);
153 }
154 // The next call also adds particles to the container if doDataStructureUpdate is true.
155 auto leavingBufferParticles = collectLeavingParticlesFromBuffer(doDataStructureUpdate);
156
157 AutoPasLog(DEBUG, "Initiating container update.");
158 auto leavingParticles = _containerSelector.getCurrentContainer().updateContainer(not doDataStructureUpdate);
159 leavingParticles.insert(leavingParticles.end(), leavingBufferParticles.begin(), leavingBufferParticles.end());
160
161 // Substract the amount of leaving particles from the number of owned particles.
162 _numParticlesOwned.fetch_sub(leavingParticles.size(), std::memory_order_relaxed);
163 // updateContainer deletes all halo particles.
164 std::for_each(_haloParticleBuffer.begin(), _haloParticleBuffer.end(), [](auto &buffer) { buffer.clear(); });
165 _numParticlesHalo.store(0, std::memory_order_relaxed);
166 return leavingParticles;

175 std::vector<Particle> resizeBox(const std::array<double, 3> &boxMin, const std::array<double, 3> &boxMax) {
176 using namespace autopas::utils::ArrayMath::literals;
177 const auto &oldMin = _containerSelector.getCurrentContainer().getBoxMin();
178 const auto &oldMax = _containerSelector.getCurrentContainer().getBoxMax();
179
180 // if nothing changed do nothing
181 if (oldMin == boxMin and oldMax == boxMax) {
182 return {};
183 }
184
185 // sanity check that new size is actually positive
186 for (size_t i = 0; i < boxMin.size(); ++i) {
187 if (boxMin[i] >= boxMax[i]) {
188 utils::ExceptionHandler::exception(
189 "New box size in dimension {} is not positive!\nboxMin[{}] = {}\nboxMax[{}] = {}", i, i, boxMin[i], i,
190 boxMax[i]);
191 }
192 }
193
194 // warn if domain changes too drastically
195 const auto newLength = boxMax - boxMin;
196 const auto oldLength = oldMax - oldMin;
197 const auto relDiffLength = newLength / oldLength;
198 for (size_t i = 0; i < newLength.size(); ++i) {
199 // warning threshold is set arbitrary and up for change if needed
200 if (relDiffLength[i] > 1.3 or relDiffLength[i] < 0.7) {
201 AutoPasLog(WARN,
202 "LogicHandler.resize(): Domain size changed drastically in dimension {}! Gathered AutoTuning "
203 "information might not be applicable anymore!\n"
204 "Size old box : {}\n"
205 "Size new box : {}\n"
206 "Relative diff: {}",
207 i, utils::ArrayUtils::to_string(oldLength), utils::ArrayUtils::to_string(newLength),
208 utils::ArrayUtils::to_string(relDiffLength));
209 }
210 }
211
212 // check all particles
213 std::vector<Particle> particlesNowOutside;
214 for (auto pIter = _containerSelector.getCurrentContainer().begin(); pIter.isValid(); ++pIter) {
215 // make sure only owned ones are present
216 if (not pIter->isOwned()) {
217 utils::ExceptionHandler::exception(
218 "LogicHandler::resizeBox() encountered non owned particle. "
219 "When calling resizeBox() these should be already deleted. "
220 "This could be solved by calling updateContainer() before resizeBox().");
221 }
222 // owned particles that are now outside are removed from the container and returned
223 if (not utils::inBox(pIter->getR(), boxMin, boxMax)) {
224 particlesNowOutside.push_back(*pIter);
225 decreaseParticleCounter(*pIter);
226 internal::markParticleAsDeleted(*pIter);
227 }
228 }
229
230 // actually resize the container
231 _containerSelector.resizeBox(boxMin, boxMax);
232 // The container might have changed sufficiently enough so that we need more or less spacial locks
233 const auto boxLength = boxMax - boxMin;
234 const auto interactionLengthInv = 1. / (_containerSelector.getCurrentContainer().getInteractionLength());
235 initSpacialLocks(boxLength, interactionLengthInv);
236
237 // Set this flag, s.t., the container is rebuilt!
238 _neighborListsAreValid.store(false, std::memory_order_relaxed);
239
240 return particlesNowOutside;
241 }

349 std::tuple<bool, bool> deleteParticleFromBuffers(Particle &particle) {
350 // find the buffer the particle belongs to
351 auto &bufferCollection = particle.isOwned() ? _particleBuffer : _haloParticleBuffer;
352 for (auto &cell : bufferCollection) {
353 auto &buffer = cell._particles;
354 // if the address of the particle is between start and end of the buffer it is in this buffer
355 if (not buffer.empty() and &(buffer.front()) <= &particle and &particle <= &(buffer.back())) {
356 const bool isRearParticle = &particle == &buffer.back();
357 // swap-delete
358 particle = buffer.back();
359 buffer.pop_back();
360 return {true, not isRearParticle};
361 }
362 }
363 return {false, true};
364 }

407 template <class Iterator>
408 typename Iterator::ParticleVecType gatherAdditionalVectors(IteratorBehavior behavior) {
409 typename Iterator::ParticleVecType additionalVectors;
410 additionalVectors.reserve(static_cast<bool>(behavior & IteratorBehavior::owned) * _particleBuffer.size() +
411 static_cast<bool>(behavior & IteratorBehavior::halo) * _haloParticleBuffer.size());
412 if (behavior & IteratorBehavior::owned) {
413 for (auto &buffer : _particleBuffer) {
414 // Don't insert empty buffers. This also means that we won't pick up particles added during iterating if they
415 // go to the buffers. But since we wouldn't pick them up if they go into the container to a cell that the
416 // iterators already passed this is unsupported anyways.
417 if (not buffer.isEmpty()) {
418 additionalVectors.push_back(&(buffer._particles));
419 }
420 }
421 }
422 if (behavior & IteratorBehavior::halo) {
423 for (auto &buffer : _haloParticleBuffer) {
424 if (not buffer.isEmpty()) {
425 additionalVectors.push_back(&(buffer._particles));
426 }
427 }
428 }
429 return additionalVectors;

799template <typename Particle>
800void LogicHandler<Particle>::setParticleBuffers(const std::vector<FullParticleCell<Particle>> &particleBuffers,
801 const std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
802 auto exchangeBuffer = [](const auto &newBuffers, auto &oldBuffers, auto &particleCounter) {
803 // sanity check
804 if (oldBuffers.size() < newBuffers.size()) {
805 autopas::utils::ExceptionHandler::exception(
806 "The number of new buffers ({}) is larger than number of existing buffers ({})!", newBuffers.size(),
807 oldBuffers.size());
808 }
809
810 // we will clear the old buffers so subtract the particles from the counters.
811 const auto numParticlesInOldBuffers =
812 std::transform_reduce(oldBuffers.begin(), std::next(oldBuffers.begin(), newBuffers.size()), 0, std::plus<>(),
813 [](const auto &cell) { return cell.size(); });
814 particleCounter.fetch_sub(numParticlesInOldBuffers, std::memory_order_relaxed);
815
816 // clear the old buffers and copy the content of the new buffers over.
817 size_t numParticlesInNewBuffers = 0;
818 for (size_t i = 0; i < newBuffers.size(); ++i) {
819 oldBuffers[i].clear();
820 for (const auto &p : newBuffers[i]) {
821 ++numParticlesInNewBuffers;
822 oldBuffers[i].addParticle(p);
823 }
824 }
825 // update the counters.
826 particleCounter.fetch_add(numParticlesInNewBuffers, std::memory_order_relaxed);
827 };
828
829 exchangeBuffer(particleBuffers, _particleBuffer, _numParticlesOwned);
830 exchangeBuffer(haloParticleBuffers, _haloParticleBuffer, _numParticlesHalo);
831}

867 if (configuration.newton3) {
868 doRemainderTraversal<true>(&functor, actualContainerType, _particleBuffer, _haloParticleBuffer);
869 } else {
870 doRemainderTraversal<false>(&functor, actualContainerType, _particleBuffer, _haloParticleBuffer);
871 }
872 });
873 timerRemainderTraversal.stop();
874 functor.endTraversal(configuration.newton3);
875
876 const auto [energyPsys, energyPkg, energyRam, energyTotal] = _autoTuner.sampleEnergy();
877
878 timerTotal.stop();
879
880 constexpr auto nanD = std::numeric_limits<double>::quiet_NaN();
881 constexpr auto nanL = std::numeric_limits<long>::quiet_NaN();
882 return {timerIteratePairwise.getTotalTime(),
883 timerRemainderTraversal.getTotalTime(),
884 timerRebuild.getTotalTime(),
885 timerTotal.getTotalTime(),
886 energyMeasurementsPossible,
887 energyMeasurementsPossible ? energyPsys : nanD,
888 energyMeasurementsPossible ? energyPkg : nanD,
889 energyMeasurementsPossible ? energyRam : nanD,
890 energyMeasurementsPossible ? energyTotal : nanL};

893template <class Particle>
894template <bool newton3, class ContainerType, class PairwiseFunctor>
895void LogicHandler<Particle>::doRemainderTraversal(PairwiseFunctor *f, ContainerType &container,
896 std::vector<FullParticleCell<Particle>> &particleBuffers,
897 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
898 // Sanity check. If this is violated feel free to add some logic here that adapts the number of locks.
899 if (_bufferLocks.size() < particleBuffers.size()) {
900 utils::ExceptionHandler::exception("Not enough locks for non-halo buffers! Num Locks: {}, Buffers: {}",
901 _bufferLocks.size(), particleBuffers.size());
902 }
903
904 // Balance buffers. This makes processing them with static scheduling quite efficient.
905 // Also, if particles were not inserted in parallel, this enables us to process them in parallel now.
906 // Cost is at max O(2N) worst O(N) per buffer collection and negligible compared to interacting them.
907 auto cellToVec = [](auto &cell) -> std::vector<Particle> & { return cell._particles; };
908 utils::ArrayUtils::balanceVectors(particleBuffers, cellToVec);
909 utils::ArrayUtils::balanceVectors(haloParticleBuffers, cellToVec);
910
911 // The following part performs the main remainder traversal. The actual calculation is done in 4 steps carried out
912 // in three helper functions.
913
914 // only activate time measurements if it will actually be logged
915#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
916 autopas::utils::Timer timerBufferContainer;
917 autopas::utils::Timer timerPBufferPBuffer;
918 autopas::utils::Timer timerPBufferHBuffer;
919
920 timerBufferContainer.start();
921#endif
922 // steps 1 & 2. particleBuffer with all close particles in container and haloParticleBuffer with owned, close
923 // particles in container
924 remainderHelperBufferContainer<newton3>(f, container, particleBuffers, haloParticleBuffers);
925
926#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
927 timerBufferContainer.stop();
928 timerPBufferPBuffer.start();
929#endif
930
931 // step 3. particleBuffer with itself and all other buffers
932 remainderHelperBufferBuffer<newton3>(f, particleBuffers, haloParticleBuffers);
933
934#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
935 timerPBufferPBuffer.stop();
936 timerPBufferHBuffer.start();
937#endif
938
939 // step 4. particleBuffer with haloParticleBuffer
940 remainderHelperBufferHaloBuffer<newton3>(f, particleBuffers, haloParticleBuffers);
941
942#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
943 timerPBufferHBuffer.stop();
944#endif
945
946 // unpack particle SoAs. Halo data is not interesting
947 for (auto &buffer : particleBuffers) {
948 f->SoAExtractor(buffer, buffer._particleSoABuffer, 0);
949 }
950
951 AutoPasLog(TRACE, "Timer Buffers <-> Container (1+2): {}", timerBufferContainer.getTotalTime());
952 AutoPasLog(TRACE, "Timer PBuffers<-> PBuffer ( 3): {}", timerPBufferPBuffer.getTotalTime());
953 AutoPasLog(TRACE, "Timer PBuffers<-> HBuffer ( 4): {}", timerPBufferHBuffer.getTotalTime());
954
955 // Note: haloParticleBuffer with itself is NOT needed, as interactions between halo particles are unneeded!
956}

958template <class Particle>
959template <bool newton3, class ContainerType, class PairwiseFunctor>
960void LogicHandler<Particle>::remainderHelperBufferContainer(
961 PairwiseFunctor *f, ContainerType &container, std::vector<FullParticleCell<Particle>> &particleBuffers,
962 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
963 using autopas::utils::ArrayUtils::static_cast_copy_array;
964 using namespace autopas::utils::ArrayMath::literals;
965
966 const auto haloBoxMin = container.getBoxMin() - container.getInteractionLength();
967 const auto interactionLengthInv = 1. / container.getInteractionLength();
968
969 const double cutoff = container.getCutoff();
970 // one halo and particle buffer pair per thread
971 AUTOPAS_OPENMP(parallel for schedule(static, 1) shared(f, _spacialLocks, haloBoxMin, interactionLengthInv))
972 for (int bufferId = 0; bufferId < particleBuffers.size(); ++bufferId) {
973 auto &particleBuffer = particleBuffers[bufferId];
974 auto &haloParticleBuffer = haloParticleBuffers[bufferId];
975 // 1. particleBuffer with all close particles in container
976 for (auto &&p1 : particleBuffer) {
977 const auto pos = p1.getR();
978 const auto min = pos - cutoff;
979 const auto max = pos + cutoff;
980 container.forEachInRegion(
981 [&](auto &p2) {
982 const auto lockCoords = static_cast_copy_array<size_t>((p2.getR() - haloBoxMin) * interactionLengthInv);
983 if constexpr (newton3) {
984 const std::lock_guard<std::mutex> lock(*_spacialLocks[lockCoords[0]][lockCoords[1]][lockCoords[2]]);
985 f->AoSFunctor(p1, p2, true);
986 } else {
987 f->AoSFunctor(p1, p2, false);
988 // no need to calculate force enacted on a halo
989 if (not p2.isHalo()) {
990 const std::lock_guard<std::mutex> lock(*_spacialLocks[lockCoords[0]][lockCoords[1]][lockCoords[2]]);
991 f->AoSFunctor(p2, p1, false);
992 }
993 }
994 },
995 min, max, IteratorBehavior::ownedOrHalo);
996 }
997
998 // 2. haloParticleBuffer with owned, close particles in container
999 for (auto &&p1halo : haloParticleBuffer) {
1000 const auto pos = p1halo.getR();
1001 const auto min = pos - cutoff;
1002 const auto max = pos + cutoff;
1003 container.forEachInRegion(
1004 [&](auto &p2) {
1005 const auto lockCoords = static_cast_copy_array<size_t>((p2.getR() - haloBoxMin) * interactionLengthInv);
1006 // No need to apply anything to p1halo
1007 // -> AoSFunctor(p1, p2, false) not needed as it neither adds force nor Upot (potential energy)
1008 // -> newton3 argument needed for correct globals
1009 const std::lock_guard<std::mutex> lock(*_spacialLocks[lockCoords[0]][lockCoords[1]][lockCoords[2]]);
1010 f->AoSFunctor(p2, p1halo, newton3);
1011 },
1012 min, max, IteratorBehavior::owned);
1013 }
1014 }
1015}

1017template <class Particle>
1018template <bool newton3, class PairwiseFunctor>
1019void LogicHandler<Particle>::remainderHelperBufferBuffer(PairwiseFunctor *f,
1020 std::vector<FullParticleCell<Particle>> &particleBuffers,
1021 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
1022 // All (halo-)buffer interactions shall happen vectorized, hence, load all buffer data into SoAs
1023 for (auto &buffer : particleBuffers) {
1024 f->SoALoader(buffer, buffer._particleSoABuffer, 0, /*skipSoAResize*/ false);
1025 }
1026 for (auto &buffer : haloParticleBuffers) {
1027 f->SoALoader(buffer, buffer._particleSoABuffer, 0, /*skipSoAResize*/ false);
1028 }
1029
1030 AUTOPAS_OPENMP(parallel) {
1031 // For buffer interactions where bufferA == bufferB we can always enable newton3. For all interactions between
1032 // different buffers we turn newton3 always off, which ensures that only one thread at a time is writing to a
1033 // buffer. This saves expensive locks.
1034 // we can not use collapse here without locks, otherwise races would occur.
1035 AUTOPAS_OPENMP(for)
1036 for (size_t i = 0; i < particleBuffers.size(); ++i) {
1037 for (size_t jj = 0; jj < particleBuffers.size(); ++jj) {
1038 auto *particleBufferSoAA = &particleBuffers[i]._particleSoABuffer;
1039 const auto j = (i + jj) % particleBuffers.size();
1040 if (i == j) {
1041 f->SoAFunctorSingle(*particleBufferSoAA, true);
1042 } else {
1043 auto *particleBufferSoAB = &particleBuffers[j]._particleSoABuffer;
1044 f->SoAFunctorPair(*particleBufferSoAA, *particleBufferSoAB, false);
1045 }
1046 }
1047 }
1048 }
1049}

1052template <bool newton3, class PairwiseFunctor>
1053void LogicHandler<Particle>::remainderHelperBufferHaloBuffer(
1054 PairwiseFunctor *f, std::vector<FullParticleCell<Particle>> &particleBuffers,
1055 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
1056 // Here, phase / color based parallelism turned out to be more efficient than tasks
1057 AUTOPAS_OPENMP(parallel)
1058 for (int interactionOffset = 0; interactionOffset < haloParticleBuffers.size(); ++interactionOffset) {
1059 AUTOPAS_OPENMP(for)
1060 for (size_t i = 0; i < particleBuffers.size(); ++i) {
1061 auto &particleBufferSoA = particleBuffers[i]._particleSoABuffer;
1062 auto &haloBufferSoA =
1063 haloParticleBuffers[(i + interactionOffset) % haloParticleBuffers.size()]._particleSoABuffer;
1064 f->SoAFunctorPair(particleBufferSoA, haloBufferSoA, false);
1065 }
1066 }

1069template <typename Particle>
1070template <class Functor>
1071std::tuple<Configuration, std::unique_ptr<TraversalInterface>, bool> LogicHandler<Particle>::selectConfiguration(
1072 Functor &functor) {
1073 bool stillTuning = false;
1074 Configuration configuration{};
1075 std::optional<std::unique_ptr<TraversalInterface>> traversalPtrOpt{};
1076 // if this iteration is not relevant take the same algorithm config as before.
1077 if (not functor.isRelevantForTuning()) {
1078 stillTuning = false;
1079 configuration = _autoTuner.getCurrentConfig();
1080 const auto &container = _containerSelector.getCurrentContainer();
1081 traversalPtrOpt = autopas::utils::withStaticCellType<Particle>(
1082 container.getParticleCellTypeEnum(), [&](const auto &particleCellDummy) -> decltype(traversalPtrOpt) {
1083 // Can't make this unique_ptr const otherwise we can't move it later.
1084 auto traversalPtr =
1085 TraversalSelector<std::decay_t<decltype(particleCellDummy)>>::template generateTraversal<Functor>(
1086 configuration.traversal, functor, container.getTraversalSelectorInfo(), configuration.dataLayout,
1087 configuration.newton3);
1088
1089 // set sortingThreshold of the traversal if it can be casted to a CellPairTraversal and uses the CellFunctor
1090 if (auto *cellPairTraversalPtr =
1091 dynamic_cast<autopas::CellPairTraversal<std::decay_t<decltype(particleCellDummy)>> *>(
1092 traversalPtr.get())) {
1093 cellPairTraversalPtr->setSortingThreshold(_sortingThreshold);
1094 }
1095
1096 if (traversalPtr->isApplicable()) {
1097 return std::optional{std::move(traversalPtr)};
1098 } else {
1099 return std::nullopt;
1100 }
1101 });
1102 } else {
1103 if (_autoTuner.needsHomogeneityAndMaxDensityBeforePrepare()) {
1104 utils::Timer timerCalculateHomogeneity;
1105 timerCalculateHomogeneity.start();
1106 const auto &container = _containerSelector.getCurrentContainer();
1107 const auto [homogeneity, maxDensity] =
1108 autopas::utils::calculateHomogeneityAndMaxDensity(container, container.getBoxMin(), container.getBoxMax());
1109 timerCalculateHomogeneity.stop();
1110 _autoTuner.addHomogeneityAndMaxDensity(homogeneity, maxDensity, timerCalculateHomogeneity.getTotalTime());
1111 }
1112
1113 const auto needsLiveInfo = _autoTuner.prepareIteration();
1114
1115 if (needsLiveInfo) {
1116 LiveInfo info{};
1117 info.gather(_containerSelector.getCurrentContainer(), functor, _neighborListRebuildFrequency);
1118 _autoTuner.receiveLiveInfo(info);
1119 }
1120
1121 std::tie(configuration, stillTuning) = _autoTuner.getNextConfig();
1122
1123 // loop as long as we don't get a valid configuration
1124 bool rejectIndefinitely = false;
1125 while (true) {
1126 // applicability check also sets the container
1127 std::tie(traversalPtrOpt, rejectIndefinitely) = isConfigurationApplicable(configuration, functor);
1128 if (traversalPtrOpt.has_value()) {
1129 break;
1130 }
1131 // if no config is left after rejecting this one an exception is thrown here.
1132 std::tie(configuration, stillTuning) = _autoTuner.rejectConfig(configuration, rejectIndefinitely);
1133 }
1134 }
1135
1136 return {configuration, std::move(traversalPtrOpt.value()), stillTuning};
1137}

1143 utils::Timer tuningTimer;
1144 tuningTimer.start();
1145 const auto [configuration, traversalPtr, stillTuning] = selectConfiguration(*functor);
1146 tuningTimer.stop();
1147 _autoTuner.logIteration(configuration, stillTuning, tuningTimer.getTotalTime());
1148
1150 AutoPasLog(DEBUG, "Iterating with configuration: {} tuning: {}", configuration.toString(), stillTuning);
1151 const IterationMeasurements measurements = iteratePairwise(*functor, *traversalPtr);

1155 std::stringstream ss;
1156 size_t sum = 0;
1157 for (const auto &buffer : buffers) {
1158 ss << buffer.size() << ", ";
1159 sum += buffer.size();
1160 }
1161 ss << " Total: " << sum;

1167 AutoPasLog(DEBUG, "RemainderTraversal took {} ns", measurements.timeRemainderTraversal);
1168 AutoPasLog(DEBUG, "RebuildNeighborLists took {} ns", measurements.timeRebuild);
1169 AutoPasLog(DEBUG, "Container::iteratePairwise took {} ns", measurements.timeTotal);
1170 if (measurements.energyMeasurementsPossible) {
1171 AutoPasLog(DEBUG, "Energy Consumption: Psys: {} Joules Pkg: {} Joules Ram: {} Joules", measurements.energyPsys,
1172 measurements.energyPkg, measurements.energyRam);

1175 measurements.timeRemainderTraversal, measurements.timeRebuild, measurements.timeTotal,
1176 tuningTimer.getTotalTime(), measurements.energyPsys, measurements.energyPkg,
1177 measurements.energyRam);
1178
1180 // if this was a major iteration add measurements and bump counters
1181 if (functor->isRelevantForTuning()) {
1182 if (stillTuning) {
1183 switch (_autoTuner.getTuningMetric()) {
1184 case TuningMetricOption::time:
1185 _autoTuner.addMeasurement(measurements.timeTotal, not neighborListsAreValid());
1186 break;
1187 case TuningMetricOption::energy:
1188 _autoTuner.addMeasurement(measurements.energyTotal, not neighborListsAreValid());
1189 break;
1190 }
1191 } else {
1192 AutoPasLog(TRACE, "Skipping adding of sample because functor is not marked relevant.");
1193 }
1194

1196 // that should not have been updated yet.
1197 if (not neighborListsAreValid() /*we have done a rebuild now*/) {
1198 // list is now valid
1199 _neighborListsAreValid.store(true, std::memory_order_relaxed);
1200 _stepsSinceLastListRebuild = 0;
1201 }
1202 ++_stepsSinceLastListRebuild;
1203
1204 _containerSelector.getCurrentContainer().setStepsSinceLastRebuild(_stepsSinceLastListRebuild);
1205
1206 _autoTuner.bumpIterationCounters();

1208 }
1209 return stillTuning;
1210}

1212template <typename Particle>
1213template <class PairwiseFunctor>
1214std::tuple<std::optional<std::unique_ptr<TraversalInterface>>, bool> LogicHandler<Particle>::isConfigurationApplicable(
1215 const Configuration &conf, PairwiseFunctor &pairwiseFunctor) {
1216 // Check if the container supports the traversal
1217 const auto allContainerTraversals = compatibleTraversals::allCompatibleTraversals(conf.container);
1218 if (allContainerTraversals.find(conf.traversal) == allContainerTraversals.end()) {
1219 AutoPasLog(DEBUG, "Configuration rejected: Container {} does not support the traversal {}.", conf.container,
1220 conf.traversal);
1221 return {std::nullopt, true};
1222 }
1223
1224 // Check if the required Newton 3 mode is supported by the functor
1225 if ((conf.newton3 == Newton3Option::enabled and not pairwiseFunctor.allowsNewton3()) or
1226 (conf.newton3 == Newton3Option::disabled and not pairwiseFunctor.allowsNonNewton3())) {
1227 AutoPasLog(DEBUG, "Configuration rejected: The functor doesn't support Newton 3 {}!", conf.newton3);
1228 return {std::nullopt, true};
1229 }
1230
1231 // Check if the traversal is applicable to the current state of the container
1232 _containerSelector.selectContainer(
1233 conf.container,
1234 ContainerSelectorInfo(conf.cellSizeFactor,
1235 _containerSelector.getCurrentContainer().getVerletSkin() / _neighborListRebuildFrequency,
1236 _neighborListRebuildFrequency, _verletClusterSize, conf.loadEstimator));
1237 const auto &container = _containerSelector.getCurrentContainer();
1238 const auto traversalInfo = container.getTraversalSelectorInfo();
1239
1240 auto traversalPtrOpt = autopas::utils::withStaticCellType<Particle>(
1241 container.getParticleCellTypeEnum(),
1242 [&](const auto &particleCellDummy) -> std::optional<std::unique_ptr<TraversalInterface>> {
1243 // Can't make this unique_ptr const otherwise we can't move it later.
1244 auto traversalPtr =
1245 TraversalSelector<std::decay_t<decltype(particleCellDummy)>>::template generateTraversal<PairwiseFunctor>(
1246 conf.traversal, pairwiseFunctor, traversalInfo, conf.dataLayout, conf.newton3);
1247
1248 // set sortingThreshold of the traversal if it can be casted to a CellPairTraversal and uses the CellFunctor
1249 if (auto *cellPairTraversalPtr =
1250 dynamic_cast<autopas::CellPairTraversal<std::decay_t<decltype(particleCellDummy)>> *>(
1251 traversalPtr.get())) {
1252 cellPairTraversalPtr->setSortingThreshold(_sortingThreshold);
1253 }
1254
1255 if (traversalPtr->isApplicable()) {
1256 return std::optional{std::move(traversalPtr)};
1257 } else {
1258 return std::nullopt;
1259 }
1260 });
1261 return {std::move(traversalPtrOpt), false};
1262}
