267 bool AutoTuner<Particle>::iteratePairwise(PairwiseFunctor *f, bool doListRebuild) {
268  bool isTuning = false;
269  // tune if :
270  // - more than one config exists
271  // - currently in tuning phase
272  // - functor is relevant
273  if ((not _tuningStrategy->searchSpaceIsTrivial()) and _iterationsSinceTuning >= _tuningInterval and
274  f->isRelevantForTuning()) {
275  isTuning = tune<PairwiseFunctor>(*f);
276  if (not isTuning) {
277  _iterationsSinceTuning = 0;
278  }
279  }
280 
281  // large case differentiation for data layout and newton 3
282  switch (_tuningStrategy->getCurrentConfiguration().dataLayout) {
283  case DataLayoutOption::aos: {
284  if (_tuningStrategy->getCurrentConfiguration().newton3 == Newton3Option::enabled) {
285  if (isTuning) {
286  iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::aos, /*Newton3*/ true,
287  /*tuning*/ true>(f, doListRebuild);
288  } else {
289  iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::aos, /*Newton3*/ true,
290  /*tuning*/ false>(f, doListRebuild);
291  }
292  } else {
293  if (isTuning) {
294  iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::aos, /*Newton3*/ false,
295  /*tuning*/ true>(f, doListRebuild);
296  } else {
297  iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::aos, /*Newton3*/ false,
298  /*tuning*/ false>(f, doListRebuild);
299  }
300  }
301  break;
302  }
303  case DataLayoutOption::soa: {
304  if (_tuningStrategy->getCurrentConfiguration().newton3 == Newton3Option::enabled) {
305  if (isTuning) {
306  iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::soa, /*Newton3*/ true,
307  /*tuning*/ true>(f, doListRebuild);
308  } else {
309  iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::soa, /*Newton3*/ true,
310  /*tuning*/ false>(f, doListRebuild);
311  }
312  } else {
313  if (isTuning) {
314  iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::soa, /*Newton3*/ false,
315  /*tuning*/ true>(f, doListRebuild);
316  } else {
317  iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::soa, /*Newton3*/ false,
318  /*tuning*/ false>(f, doListRebuild);
319  }
320  }
321  break;
322  }
323 #if defined(AUTOPAS_CUDA)
324  case DataLayoutOption::cuda: {
325  if (_tuningStrategy->getCurrentConfiguration().newton3 == Newton3Option::enabled) {
326  if (isTuning) {
327  iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::cuda, /*Newton3*/ true,

329  } else {

332  }
333  } else {
334  if (isTuning) {
335  iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::cuda, /*Newton3*/ false,
336  /*tuning*/ true>(f, doListRebuild);
337  } else {
338  iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::cuda, /*Newton3*/ false,
339  /*tuning*/ false>(f, doListRebuild);
340  }
341  }
342  break;
343  }
344 #endif
345  default:
346  utils::ExceptionHandler::exception("AutoTuner: Unknown data layout : {}",
347  _tuningStrategy->getCurrentConfiguration().dataLayout);
348  }
349

351  ++_iterationsSinceTuning;
352  ++_iteration;
353  }
354  return isTuning;

358 template <class PairwiseFunctor, DataLayoutOption::Value dataLayout, bool useNewton3, bool inTuningPhase>
359 void AutoTuner<Particle>::iteratePairwiseTemplateHelper(PairwiseFunctor *f, bool doListRebuild) {
360  auto containerPtr = getContainer();
361  AutoPasLog(debug, "Iterating with configuration: {} tuning: {}",
362  _tuningStrategy->getCurrentConfiguration().toString(), inTuningPhase ? "true" : "false");
363 
364  auto traversal = autopas::utils::withStaticCellType<Particle>(
365  containerPtr->getParticleCellTypeEnum(), [&](auto particleCellDummy) {
366  return TraversalSelector<decltype(particleCellDummy)>::template generateTraversal<PairwiseFunctor, dataLayout,
367  useNewton3>(
368  _tuningStrategy->getCurrentConfiguration().traversal, *f, containerPtr->getTraversalSelectorInfo());
369  });
370 
371  if (not traversal->isApplicable()) {
372  autopas::utils::ExceptionHandler::exception(
373  "Error: Trying to execute a traversal that is not applicable. Two common reasons:\n"
374  "1. The search space is trivial, but no traversals are applicable. \n"
375  "2. You are using multiple functors and one of the not first is not supporting all configuration options of "
376  "the first.\n"
377  "Config: {}\n"
378  "Current functor: {}",
379  _tuningStrategy->getCurrentConfiguration().toString(), typeid(*f).name());
380  }
381 
382  autopas::utils::Timer timerTotal;
383  autopas::utils::Timer timerRebuild;
384  timerTotal.start();
385 
386  f->initTraversal();
387  if (doListRebuild) {
388  timerRebuild.start();
389  containerPtr->rebuildNeighborLists(traversal.get());
390  timerRebuild.stop();
391  }
392  containerPtr->iteratePairwise(traversal.get());
393  f->endTraversal(useNewton3);
394 
395  if (doListRebuild) {
396  AutoPasLog(debug, "rebuildNeighborLists took {} nanoseconds", timerRebuild.getTotalTime());
397  }
398 
399  auto runtimeTotal = timerTotal.stop();
400  AutoPasLog(debug, "IteratePairwise took {} nanoseconds", runtimeTotal);
401 
402  // if tuning execute with time measurements
403  if (inTuningPhase) {
404  addTimeMeasurement(*f, runtimeTotal);
405  }
406 }

410 bool AutoTuner<Particle>::tune(PairwiseFunctor &pairwiseFunctor) {
411  bool stillTuning = true;
412 
413  // need more samples; keep current config
414  if (_samples.size() < _maxSamples) {
415  return stillTuning;
416  }
417  utils::Timer tuningTimer;
418  tuningTimer.start();
419  // first tuning iteration -> reset to first config
420  if (_iterationsSinceTuning == _tuningInterval) {
421  _tuningStrategy->reset(_iteration);
422  } else { // enough samples -> next config
423  stillTuning = _tuningStrategy->tune();
424  }
425 
426  // repeat as long as traversals are not applicable or we run out of configs
427  while (true) {
428  // check if newton3 works with this functor and remove config if not
429  if ((_tuningStrategy->getCurrentConfiguration().newton3 == Newton3Option::enabled and
430  not pairwiseFunctor.allowsNewton3()) or
431  (_tuningStrategy->getCurrentConfiguration().newton3 == Newton3Option::disabled and
432  not pairwiseFunctor.allowsNonNewton3())) {
433  AutoPasLog(warn, "Configuration with newton 3 {} called with a functor that does not support this!",
434  _tuningStrategy->getCurrentConfiguration().newton3.to_string());
435 
436  _tuningStrategy->removeN3Option(_tuningStrategy->getCurrentConfiguration().newton3);
437  } else {
438  if (configApplicable(_tuningStrategy->getCurrentConfiguration(), pairwiseFunctor)) {
439  // we found a valid config!
440  break;
441  } else {
442  AutoPasLog(debug, "Skip not applicable configuration {}",
443  _tuningStrategy->getCurrentConfiguration().toString());
444  stillTuning = _tuningStrategy->tune(true);
445  }
446  }
447  }
448  // samples should only be cleared if we are still tuning, see `if (_samples.size() < _maxSamples)` from before.
449  if (stillTuning) {
450  // samples are no longer needed. Delete them here so willRebuild() works as expected.
451  _samples.clear();
452  }
453  tuningTimer.stop();
454  AutoPasLog(debug, "Tuning took {} ns.", tuningTimer.getTotalTime());
455 
456  selectCurrentContainer();
457  return stillTuning;
458 }
