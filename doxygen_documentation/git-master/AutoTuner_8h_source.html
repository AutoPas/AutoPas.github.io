92 std::tuple<bool, bool> prepareIteration() {
93 const bool isFirstTuningIteration = _iterationsSinceTuning == _tuningInterval;
94
95 // first tuning iteration -> reset everything
96 if (isFirstTuningIteration) {
97 // call the appropriate version of reset
98 if (auto *mpiStrategy = dynamic_cast<MPIParallelizedStrategy *>(_tuningStrategy.get())) {
99 const std::pair<double, double> smoothedHomogeneityAndMaxDensity{
100 autopas::OptimumSelector::medianValue(_homogeneitiesOfLastTenIterations),
101 autopas::OptimumSelector::medianValue(_maxDensitiesOfLastTenIterations)};
102 mpiStrategy->reset(_iteration, smoothedHomogeneityAndMaxDensity, _mpiTuningMaxDifferenceForBucket,
103 _mpiTuningWeightForMaxDensity);
104 } else {
105 _tuningStrategy->reset(_iteration);
106 }
107
108 // Homogeneity was calculated directly before the tuning phase so reset it now.
109 if (_tuningStrategy->smoothedHomogeneityAndMaxDensityNeeded()) {
110 AutoPasLog(DEBUG, "Calculating homogeneities over 10 iterations took in total {} ns on rank {}.",
111 _timerCalculateHomogeneity.getTotalTime(), []() {
112 int rank{0};
113 AutoPas_MPI_Comm_rank(AUTOPAS_MPI_COMM_WORLD, &rank);
114 return rank;
115 });
116 _homogeneitiesOfLastTenIterations.clear();
117 _maxDensitiesOfLastTenIterations.clear();
118 }
119 }
120
121 // if necessary, we need to collect live info in the first tuning iteration
122 const bool needsLiveInfo = isFirstTuningIteration and _tuningStrategy->needsLiveInfo();
123
124 // calc homogeneity if needed, and we are within 10 iterations of the next tuning phase
125 const size_t numIterationsForHomogeneity = 10;
126 const bool needsHomogeneityAndMaxDensity =
127 _tuningStrategy->smoothedHomogeneityAndMaxDensityNeeded() and
128 _iterationsSinceTuning > _tuningInterval - numIterationsForHomogeneity and
129 _iterationsSinceTuning <= _tuningInterval;
130
131 return {needsLiveInfo, needsHomogeneityAndMaxDensity};
132 }
