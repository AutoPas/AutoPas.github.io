223 using namespace autopas::utils::ArrayMath::literals;
224 using autopas::utils::ArrayMath::ceil;
225 using autopas::utils::ArrayUtils::static_cast_copy_array;

269 [[nodiscard]] long estimateRuntimeFromSamples() const {
270 // reduce samples for rebuild and non-rebuild iterations with the given selector strategy
271 const auto reducedValueBuilding =
272 autopas::OptimumSelector::optimumValue(_samplesRebuildingNeighborLists, _selectorStrategy);
273 // if there is no data for the non rebuild iterations we have to assume them taking the same time as rebuilding ones
274 // this might neither be a good estimate nor fair but the best we can do
275 const auto reducedValueNotBuilding =
276 _samplesNotRebuildingNeighborLists.empty()
277 ? reducedValueBuilding
278 : autopas::OptimumSelector::optimumValue(_samplesNotRebuildingNeighborLists, _selectorStrategy);
279
280 const auto numIterationsNotBuilding =
281 std::max(0, static_cast<int>(_rebuildFrequency) - static_cast<int>(_samplesRebuildingNeighborLists.size()));
282 const auto numIterationsBuilding = _rebuildFrequency - numIterationsNotBuilding;
283
284 // calculate weighted estimate for one iteration
285 return (numIterationsBuilding * reducedValueBuilding + numIterationsNotBuilding * reducedValueNotBuilding) /
286 _rebuildFrequency;
287 }

504template <class Particle>
505template <class PairwiseFunctor>
506bool AutoTuner<Particle>::iteratePairwise(PairwiseFunctor *f, bool doListRebuild,
507 std::vector<FullParticleCell<Particle>> &particleBuffers,
508 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
509 bool isTuning = false;
510 // tune if :
511 // - more than one config exists
512 // - currently in tuning phase
513 // - functor is relevant
514 if (_tuningStrategy->smoothedHomogeneityAndMaxDensityNeeded() and _iterationsSinceTuning >= _tuningInterval - 9 and
515 _iterationsSinceTuning <= _tuningInterval) {
516 _timerCalculateHomogeneity.start();
517 const auto &container = getContainer();
518 const auto [homogeneity, maxDensity] =
519 autopas::utils::calculateHomogeneityAndMaxDensity(*container, container->getBoxMin(), container->getBoxMax());
520 _homogeneitiesOfLastTenIterations.push_back(homogeneity);
521 _maxDensitiesOfLastTenIterations.push_back(maxDensity);
522 _timerCalculateHomogeneity.stop();
523 }
524 if ((not _tuningStrategy->searchSpaceIsTrivial()) and _iterationsSinceTuning >= _tuningInterval and
525 f->isRelevantForTuning()) {
526 isTuning = tune<PairwiseFunctor>(*f);
527 if (not isTuning) {
528 _iterationsSinceTuning = 0;
529 }
530 }
531
532 // large case differentiation for data layout and newton 3
533 switch (_tuningStrategy->getCurrentConfiguration().dataLayout) {
534 case DataLayoutOption::aos: {
535 if (_tuningStrategy->getCurrentConfiguration().newton3 == Newton3Option::enabled) {
536 if (isTuning) {
537 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::aos, /*Newton3*/ true,
538 /*tuning*/ true>(f, doListRebuild, particleBuffers, haloParticleBuffers);
539 } else {
540 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::aos, /*Newton3*/ true,
541 /*tuning*/ false>(f, doListRebuild, particleBuffers, haloParticleBuffers);
542 }
543 } else {
544 if (isTuning) {
545 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::aos, /*Newton3*/ false,
546 /*tuning*/ true>(f, doListRebuild, particleBuffers, haloParticleBuffers);
547 } else {
548 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::aos, /*Newton3*/ false,
549 /*tuning*/ false>(f, doListRebuild, particleBuffers, haloParticleBuffers);
550 }
551 }
552 break;
553 }
554 case DataLayoutOption::soa: {
555 if (_tuningStrategy->getCurrentConfiguration().newton3 == Newton3Option::enabled) {
556 if (isTuning) {
557 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::soa, /*Newton3*/ true,
558 /*tuning*/ true>(f, doListRebuild, particleBuffers, haloParticleBuffers);
559 } else {
560 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::soa, /*Newton3*/ true,
561 /*tuning*/ false>(f, doListRebuild, particleBuffers, haloParticleBuffers);
562 }
563 } else {
564 if (isTuning) {
565 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::soa, /*Newton3*/ false,
566 /*tuning*/ true>(f, doListRebuild, particleBuffers, haloParticleBuffers);
567 } else {
568 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::soa, /*Newton3*/ false,
569 /*tuning*/ false>(f, doListRebuild, particleBuffers, haloParticleBuffers);
570 }
571 }
572 break;
573 }
574 default:
575 utils::ExceptionHandler::exception("AutoTuner: Unknown data layout : {}",
576 _tuningStrategy->getCurrentConfiguration().dataLayout);
577 }
578
579 if (f->isRelevantForTuning()) {
580 ++_iterationsSinceTuning;
581 ++_iteration;
582 }
583 return isTuning;
584}

586template <class Particle>
587template <bool newton3, class T, class PairwiseFunctor>
588void AutoTuner<Particle>::doRemainderTraversal(PairwiseFunctor *f, T containerPtr,
589 std::vector<FullParticleCell<Particle>> &particleBuffers,
590 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
591 // Sanity check. If this is violated feel free to add some logic here that adapts the number of locks.
592 if (_bufferLocks.size() < particleBuffers.size()) {
593 utils::ExceptionHandler::exception("Not enough locks for non-halo buffers! Num Locks: {}, Buffers: {}",
594 _bufferLocks.size(), particleBuffers.size());
595 }
596
597 // Balance buffers. This makes processing them with static scheduling quite efficient.
598 // Also, if particles were not inserted in parallel, this enables us to process them in parallel now.
599 // Cost is at max O(2N) worst O(N) per buffer collection and negligible compared to interacting them.
600 auto cellToVec = [](auto &cell) -> std::vector<Particle> & { return cell._particles; };
601 utils::ArrayUtils::balanceVectors(particleBuffers, cellToVec);
602 utils::ArrayUtils::balanceVectors(haloParticleBuffers, cellToVec);
603
604 // The following part performes the main remainder traversal. The actual calculation is done in 4 steps carried out in
605 // three helper functions.
606
607 // only activate time measurements if it will actually be logged
608#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
609 autopas::utils::Timer timerBufferContainer;
610 autopas::utils::Timer timerPBufferPBuffer;
611 autopas::utils::Timer timerPBufferHBuffer;
612
613 timerBufferContainer.start();
614#endif
615 // steps 1 & 2. particleBuffer with all close particles in container and haloParticleBuffer with owned, close
616 // particles in container
617 remainderHelperBufferContainer<newton3>(f, containerPtr, particleBuffers, haloParticleBuffers);
618
619#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
620 timerBufferContainer.stop();
621 timerPBufferPBuffer.start();
622#endif
623
624 // step 3. particleBuffer with itself and all other buffers
625 remainderHelperBufferBuffer<newton3>(f, containerPtr, particleBuffers, haloParticleBuffers);
626
627#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
628 timerPBufferPBuffer.stop();
629 timerPBufferHBuffer.start();
630#endif
631
632 // step 4. particleBuffer with haloParticleBuffer
633 remainderHelperBufferHaloBuffer<newton3>(f, containerPtr, particleBuffers, haloParticleBuffers);
634
635#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
636 timerPBufferHBuffer.stop();
637#endif
638
639 // unpack particle SoAs. Halo data is not interesting
640 for (auto &buffer : particleBuffers) {
641 f->SoAExtractor(buffer, buffer._particleSoABuffer, 0);
642 }
643
644 AutoPasLog(TRACE, "Timer Buffers <-> Container (1+2): {}", timerBufferContainer.getTotalTime());
645 AutoPasLog(TRACE, "Timer PBuffers<-> PBuffer ( 3): {}", timerPBufferPBuffer.getTotalTime());
646 AutoPasLog(TRACE, "Timer PBuffers<-> HBuffer ( 4): {}", timerPBufferHBuffer.getTotalTime());
647
648 // Note: haloParticleBuffer with itself is NOT needed, as interactions between halo particles are unneeded!
649}

651template <class Particle>
652template <bool newton3, class T, class PairwiseFunctor>
653void AutoTuner<Particle>::remainderHelperBufferContainer(PairwiseFunctor *f, T containerPtr,
654 std::vector<FullParticleCell<Particle>> &particleBuffers,
655 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
656 using autopas::utils::ArrayUtils::static_cast_copy_array;
657 using namespace autopas::utils::ArrayMath::literals;
658
659 const auto haloBoxMin = containerPtr->getBoxMin() - containerPtr->getInteractionLength();
660 const auto interactionLengthInv = 1. / containerPtr->getInteractionLength();
661
662 withStaticContainerType(containerPtr, [&](auto staticTypedContainerPtr) {
663 const double cutoff = staticTypedContainerPtr->getCutoff();
664#ifdef AUTOPAS_OPENMP
665// one halo and particle buffer pair per thread
666#pragma omp parallel for schedule(static, 1), shared(f, _spacialLocks, haloBoxMin, interactionLengthInv)
667#endif
668 for (int bufferId = 0; bufferId < particleBuffers.size(); ++bufferId) {
669 auto &particleBuffer = particleBuffers[bufferId];
670 auto &haloParticleBuffer = haloParticleBuffers[bufferId];
671 // 1. particleBuffer with all close particles in container
672 for (auto &&p1 : particleBuffer) {
673 const auto pos = p1.getR();
674 const auto min = pos - cutoff;
675 const auto max = pos + cutoff;
676 staticTypedContainerPtr->forEachInRegion(
677 [&](auto &p2) {
678 const auto lockCoords = static_cast_copy_array<size_t>((p2.getR() - haloBoxMin) * interactionLengthInv);
679 if constexpr (newton3) {
680 const std::lock_guard<std::mutex> lock(*_spacialLocks[lockCoords[0]][lockCoords[1]][lockCoords[2]]);
681 f->AoSFunctor(p1, p2, true);
682 } else {
683 f->AoSFunctor(p1, p2, false);
684 // no need to calculate force enacted on a halo
685 if (not p2.isHalo()) {
686 const std::lock_guard<std::mutex> lock(*_spacialLocks[lockCoords[0]][lockCoords[1]][lockCoords[2]]);
687 f->AoSFunctor(p2, p1, false);
688 }
689 }
690 },
691 min, max, IteratorBehavior::ownedOrHalo);
692 }
693
694 // 2. haloParticleBuffer with owned, close particles in container
695 for (auto &&p1halo : haloParticleBuffer) {
696 const auto pos = p1halo.getR();
697 const auto min = pos - cutoff;
698 const auto max = pos + cutoff;
699 staticTypedContainerPtr->forEachInRegion(
700 [&](auto &p2) {
701 const auto lockCoords = static_cast_copy_array<size_t>((p2.getR() - haloBoxMin) * interactionLengthInv);
702 // No need to apply anything to p1halo
703 // -> AoSFunctor(p1, p2, false) not needed as it neither adds force nor Upot (potential energy)
704 // -> newton3 argument needed for correct globals
705 const std::lock_guard<std::mutex> lock(*_spacialLocks[lockCoords[0]][lockCoords[1]][lockCoords[2]]);
706 f->AoSFunctor(p2, p1halo, newton3);
707 },
708 min, max, IteratorBehavior::owned);
709 }
710 }
711 });
712}

714template <class Particle>
715template <bool newton3, class T, class PairwiseFunctor>
716void AutoTuner<Particle>::remainderHelperBufferBuffer(PairwiseFunctor *f, T containerPtr,
717 std::vector<FullParticleCell<Particle>> &particleBuffers,
718 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
719 // All (halo-)buffer interactions shall happen vectorized, hence, load all buffer data into SoAs
720 for (auto &buffer : particleBuffers) {
721 f->SoALoader(buffer, buffer._particleSoABuffer, 0);
722 }
723 for (auto &buffer : haloParticleBuffers) {
724 f->SoALoader(buffer, buffer._particleSoABuffer, 0);
725 }
726
727#ifdef AUTOPAS_OPENMP
728#pragma omp parallel
729#endif
730 {
731 // For buffer interactions where bufferA == bufferB we can always enable newton3. For all interactions between
732 // different buffers we turn newton3 always off, which ensures that only one thread at a time is writing to a
733 // buffer. This saves expensive locks.
734#ifdef AUTOPAS_OPENMP
735 // we can not use collapse here without locks, otherwise races would occur.
736#pragma omp for
737#endif
738 for (size_t i = 0; i < particleBuffers.size(); ++i) {
739 for (size_t jj = 0; jj < particleBuffers.size(); ++jj) {
740 auto *particleBufferSoAA = &particleBuffers[i]._particleSoABuffer;
741 const auto j = (i + jj) % particleBuffers.size();
742 if (i == j) {
743 f->SoAFunctorSingle(*particleBufferSoAA, true);
744 } else {
745 auto *particleBufferSoAB = &particleBuffers[j]._particleSoABuffer;
746 f->SoAFunctorPair(*particleBufferSoAA, *particleBufferSoAB, false);
747 }
748 }
749 }
750 }
751}

777void AutoTuner<Particle>::iteratePairwiseTemplateHelper(PairwiseFunctor *f, bool doListRebuild,
778 std::vector<FullParticleCell<Particle>> &particleBuffer,
779 std::vector<FullParticleCell<Particle>> &haloParticleBuffer) {
780 auto containerPtr = getContainer();
781 AutoPasLog(DEBUG, "Iterating with configuration: {} tuning: {}",
782 _tuningStrategy->getCurrentConfiguration().toString(), inTuningPhase ? "true" : "false");
783
784 auto traversal = autopas::utils::withStaticCellType<Particle>(
785 containerPtr->getParticleCellTypeEnum(), [&](auto particleCellDummy) {
786 return TraversalSelector<decltype(particleCellDummy)>::template generateTraversal<PairwiseFunctor, dataLayout,
787 useNewton3>(
788 _tuningStrategy->getCurrentConfiguration().traversal, *f, containerPtr->getTraversalSelectorInfo());
789 });
790
791 if (not traversal->isApplicable()) {
792 autopas::utils::ExceptionHandler::exception(
793 "Error: Trying to execute a traversal that is not applicable. Two common reasons:\n"
794 "1. The search space is trivial, but no traversals are applicable. \n"
795 "2. You are using multiple functors and one of the not first is not supporting all configuration options of "
796 "the first.\n"
797 "Config: {}\n"
798 "Current functor: {}",
799 _tuningStrategy->getCurrentConfiguration().toString(), typeid(*f).name());
800 }
801
802 autopas::utils::Timer timerTotal;
803 autopas::utils::Timer timerRebuild;
804 autopas::utils::Timer timerIteratePairwise;
805 autopas::utils::Timer timerRemainderTraversal;
806 timerTotal.start();
807
808 f->initTraversal();
809 if (doListRebuild) {
810 timerRebuild.start();
811 containerPtr->rebuildNeighborLists(traversal.get());
812 timerRebuild.stop();
813 }
814 timerIteratePairwise.start();
815 containerPtr->iteratePairwise(traversal.get());
816 timerIteratePairwise.stop();
817
818 timerRemainderTraversal.start();
819 doRemainderTraversal<useNewton3>(f, containerPtr, particleBuffer, haloParticleBuffer);
820 timerRemainderTraversal.stop();
821 f->endTraversal(useNewton3);
822
823 timerTotal.stop();
824
825 auto bufferSizeListing = [](const auto &buffers) -> std::string {
826 std::stringstream ss;
827 size_t sum = 0;
828 for (const auto &buffer : buffers) {
829 ss << buffer.numParticles() << ", ";
830 sum += buffer.numParticles();
831 }
832 ss << " Total: " << sum;

843 timerRemainderTraversal.getTotalTime(), timerRebuild.getTotalTime(),
844 timerTotal.getTotalTime());

846 // if tuning execute with time measurements
847 if (inTuningPhase) {
848 if (f->isRelevantForTuning()) {

856template <class Particle>
857template <class PairwiseFunctor>
858bool AutoTuner<Particle>::tune(PairwiseFunctor &pairwiseFunctor) {
859 bool stillTuning = true;
860
861 // need more samples; keep current config
862 if (getCurrentNumSamples() < _maxSamples) {
863 return stillTuning;
864 }
865 utils::Timer tuningTimer;
866 tuningTimer.start();
867
868 // first tuning iteration -> reset to first config
869 if (_iterationsSinceTuning == _tuningInterval) {
870 if (auto *mpiStrategy = dynamic_cast<MPIParallelizedStrategy *>(_tuningStrategy.get())) {
871 const std::pair<double, double> smoothedHomogeneityAndMaxDensity{
872 autopas::OptimumSelector::medianValue(_homogeneitiesOfLastTenIterations),
873 autopas::OptimumSelector::medianValue(_maxDensitiesOfLastTenIterations)};
874 mpiStrategy->reset<Particle>(_iteration, getContainer(), smoothedHomogeneityAndMaxDensity,
875 _mpiTuningMaxDifferenceForBucket, _mpiTuningWeightForMaxDensity);
876 } else {
877 _tuningStrategy->reset(_iteration);
878 }
879 // only used temporarily of evaluation of smoothing
880 if (_tuningStrategy->smoothedHomogeneityAndMaxDensityNeeded()) {
881 int rank{0};
882#ifdef AUTOPAS_INTERNODE_TUNING
883 MPI_Comm_rank(MPI_COMM_WORLD, &rank);
884#endif
885 AutoPasLog(DEBUG, "Calculating homogeneities took added up {} ns on rank {}.",
886 _timerCalculateHomogeneity.getTotalTime(), rank);
887 _homogeneitiesOfLastTenIterations.erase(_homogeneitiesOfLastTenIterations.begin(),
888 _homogeneitiesOfLastTenIterations.end());
889 _maxDensitiesOfLastTenIterations.erase(_maxDensitiesOfLastTenIterations.begin(),
890 _maxDensitiesOfLastTenIterations.end());
891 }
892 } else { // enough samples -> next config
893 stillTuning = _tuningStrategy->tune();
894 }
895
896 // repeat as long as traversals are not applicable or we run out of configs
897 while (true) {
898 auto &currentConfig = getCurrentConfig();
899 // check if newton3 works with this functor and remove config if not
900 if ((currentConfig.newton3 == Newton3Option::enabled and not pairwiseFunctor.allowsNewton3()) or
901 (currentConfig.newton3 == Newton3Option::disabled and not pairwiseFunctor.allowsNonNewton3())) {
902 AutoPasLog(WARN, "Configuration with newton 3 {} called with a functor that does not support this!",
903 currentConfig.newton3.to_string());
904
905 _tuningStrategy->removeN3Option(currentConfig.newton3);
906 } else {
907 if (configApplicable(currentConfig, pairwiseFunctor)) {
908 // we found a valid config!
909 break;
910 } else {
911 AutoPasLog(DEBUG, "Skip not applicable configuration {}", currentConfig.toString());
912 stillTuning = _tuningStrategy->tune(true);
913 }
914 }
915 }
916 // samples should only be cleared if we are still tuning, see `if (_samples.size() < _maxSamples)` from before.
917 if (stillTuning) {
918 // samples are no longer needed. Delete them here so willRebuild() works as expected.
919 _samplesNotRebuildingNeighborLists.clear();
920 _samplesRebuildingNeighborLists.clear();
921 }
922 tuningTimer.stop();
923 // when a tuning result is found log it
924 if (not stillTuning) {
925 AutoPasLog(DEBUG, "Selected Configuration {}", getCurrentConfig().toString());
926 _tuningResultLogger.logTuningResult(getCurrentConfig(), _iteration, tuningTimer.getTotalTime());
927 }
928 AutoPasLog(DEBUG, "Tuning took {} ns.", tuningTimer.getTotalTime());
929 _iterationLogger.logTimeTuning(tuningTimer.getTotalTime());
930
931 selectCurrentContainer();
932 return stillTuning;
933}

964template <class Particle>
965void AutoTuner<Particle>::addTimeMeasurement(long time, bool neighborListRebuilt) {
966 const auto &currentConfig = _tuningStrategy->getCurrentConfiguration();
967 if (getCurrentNumSamples() < _maxSamples) {
968 AutoPasLog(TRACE, "Adding sample.");
969 if (neighborListRebuilt) {
970 _samplesRebuildingNeighborLists.push_back(time);
971 } else {
972 _samplesNotRebuildingNeighborLists.push_back(time);
973 }
974 // if this was the last sample:
975 if (getCurrentNumSamples() == _maxSamples) {
976 auto &evidenceCurrentConfig = _evidence[currentConfig];
977
978 const long reducedValue = estimateRuntimeFromSamples();
979
980 evidenceCurrentConfig.emplace_back(_iteration, reducedValue);
981
982 // smooth evidence to remove high outliers. If smoothing results in a higher value use the original value.
983 const auto smoothedValue = std::min(reducedValue, smoothing::smoothLastPoint(evidenceCurrentConfig, 5));
984
985 // replace collected evidence with smoothed value to improve next smoothing
986 evidenceCurrentConfig.back().second = smoothedValue;
987
988 _tuningStrategy->addEvidence(smoothedValue, _iteration);
989
990 // print config, times and reduced value
991 if (autopas::Logger::get()->level() <= autopas::Logger::LogLevel::debug) {
992 std::ostringstream ss;
993 // print config
994 ss << currentConfig.toString() << " : ";
995 // print all timings
996 ss << utils::ArrayUtils::to_string(_samplesRebuildingNeighborLists, " ",
997 {"With rebuilding neighbor lists [ ", " ]"});
998 ss << utils::ArrayUtils::to_string(_samplesNotRebuildingNeighborLists, " ",
999 {"Without rebuilding neighbor lists [ ", " ]"});
1000 ss << " Smoothed value: " << smoothedValue;
1001 AutoPasLog(DEBUG, "Collected times for {}", ss.str());
1002 }
1003 _tuningDataLogger.logTuningData(currentConfig, _samplesRebuildingNeighborLists,
1004 _samplesNotRebuildingNeighborLists, _iteration, reducedValue, smoothedValue);
1005 }
1006 }
