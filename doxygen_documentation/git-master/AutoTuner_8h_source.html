253 void initSpacialLocks(const std::array<double, 3> &boxLength, double interactionLengthInv) {
254 using namespace autopas::utils::ArrayMath::literals;
255 using autopas::utils::ArrayMath::ceil;
256 using autopas::utils::ArrayUtils::static_cast_copy_array;
257
258 // one lock per interaction length + one for each halo region
259 const auto locksPerDim = static_cast_copy_array<size_t>(ceil(boxLength * interactionLengthInv) + 2.);
260 _spacialLocks.resize(locksPerDim[0]);
261 for (auto &lockVecVec : _spacialLocks) {
262 lockVecVec.resize(locksPerDim[1]);
263 for (auto &lockVec : lockVecVec) {
264 lockVec.resize(locksPerDim[2]);
265 for (auto &lockPtr : lockVec) {
266 if (not lockPtr) {
267 lockPtr = std::make_unique<std::mutex>();
268 }
269 }
270 }
271 }

300 [[nodiscard]] long estimateRuntimeFromSamples() const {
301 // reduce samples for rebuild and non-rebuild iterations with the given selector strategy
302 const auto reducedValueBuilding =
303 autopas::OptimumSelector::optimumValue(_samplesRebuildingNeighborLists, _selectorStrategy);
304 // if there is no data for the non rebuild iterations we have to assume them taking the same time as rebuilding ones
305 // this might neither be a good estimate nor fair but the best we can do
306 const auto reducedValueNotBuilding =
307 _samplesNotRebuildingNeighborLists.empty()
308 ? reducedValueBuilding
309 : autopas::OptimumSelector::optimumValue(_samplesNotRebuildingNeighborLists, _selectorStrategy);
310
311 const auto numIterationsNotBuilding =
312 std::max(0, static_cast<int>(_rebuildFrequency) - static_cast<int>(_samplesRebuildingNeighborLists.size()));
313 const auto numIterationsBuilding = _rebuildFrequency - numIterationsNotBuilding;
314
315 // calculate weighted estimate for one iteration
316 return (numIterationsBuilding * reducedValueBuilding + numIterationsNotBuilding * reducedValueNotBuilding) /
317 _rebuildFrequency;
318 }

539template <class Particle>
540template <class PairwiseFunctor>
541bool AutoTuner<Particle>::iteratePairwise(PairwiseFunctor *f, bool doListRebuild,
542 std::vector<FullParticleCell<Particle>> &particleBuffers,
543 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
544 // calc homogeneity if needed, and we are within 10 iterations of the next tuning phase
545 if (_tuningStrategy->smoothedHomogeneityAndMaxDensityNeeded() and _iterationsSinceTuning >= _tuningInterval - 9 and
546 _iterationsSinceTuning <= _tuningInterval) {
547 _timerCalculateHomogeneity.start();
548 const auto &container = getContainer();
549 const auto [homogeneity, maxDensity] =
550 autopas::utils::calculateHomogeneityAndMaxDensity(container, container.getBoxMin(), container.getBoxMax());
551 _homogeneitiesOfLastTenIterations.push_back(homogeneity);
552 _maxDensitiesOfLastTenIterations.push_back(maxDensity);
553 _timerCalculateHomogeneity.stop();
554 }
555 bool isTuning = false;
556 // tune if :
557 // - more than one config exists
558 // - currently in tuning phase
559 // - functor is relevant
560 if ((not _tuningStrategy->searchSpaceIsTrivial()) and _iterationsSinceTuning >= _tuningInterval and
561 f->isRelevantForTuning()) {
562 isTuning = tune<PairwiseFunctor>(*f);
563 if (not isTuning) {
564 _iterationsSinceTuning = 0;
565 }
566 }
567
568 // large case differentiation for data layout and newton 3
569 switch (_tuningStrategy->getCurrentConfiguration().dataLayout) {
570 case DataLayoutOption::aos: {
571 if (_tuningStrategy->getCurrentConfiguration().newton3 == Newton3Option::enabled) {
572 if (isTuning) {
573 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::aos, /*Newton3*/ true,
574 /*tuning*/ true>(f, doListRebuild, particleBuffers, haloParticleBuffers);
575 } else {
576 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::aos, /*Newton3*/ true,
577 /*tuning*/ false>(f, doListRebuild, particleBuffers, haloParticleBuffers);
578 }
579 } else {
580 if (isTuning) {
581 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::aos, /*Newton3*/ false,
582 /*tuning*/ true>(f, doListRebuild, particleBuffers, haloParticleBuffers);
583 } else {
584 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::aos, /*Newton3*/ false,
585 /*tuning*/ false>(f, doListRebuild, particleBuffers, haloParticleBuffers);
586 }
587 }
588 break;
589 }
590 case DataLayoutOption::soa: {
591 if (_tuningStrategy->getCurrentConfiguration().newton3 == Newton3Option::enabled) {
592 if (isTuning) {
593 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::soa, /*Newton3*/ true,
594 /*tuning*/ true>(f, doListRebuild, particleBuffers, haloParticleBuffers);
595 } else {
596 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::soa, /*Newton3*/ true,
597 /*tuning*/ false>(f, doListRebuild, particleBuffers, haloParticleBuffers);
598 }
599 } else {
600 if (isTuning) {
601 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::soa, /*Newton3*/ false,
602 /*tuning*/ true>(f, doListRebuild, particleBuffers, haloParticleBuffers);
603 } else {
604 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::soa, /*Newton3*/ false,
605 /*tuning*/ false>(f, doListRebuild, particleBuffers, haloParticleBuffers);
606 }
607 }
608 break;
609 }
610 default:
611 utils::ExceptionHandler::exception("AutoTuner: Unknown data layout : {}",
612 _tuningStrategy->getCurrentConfiguration().dataLayout);
613 }
614
615 if (f->isRelevantForTuning()) {
616 ++_iterationsSinceTuning;
617 ++_iteration;
618 }
619 return isTuning;
620}

623template <bool newton3, class ContainerType, class PairwiseFunctor>
624void AutoTuner<Particle>::doRemainderTraversal(PairwiseFunctor *f, ContainerType &container,
625 std::vector<FullParticleCell<Particle>> &particleBuffers,
626 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
627 // Sanity check. If this is violated feel free to add some logic here that adapts the number of locks.
628 if (_bufferLocks.size() < particleBuffers.size()) {
629 utils::ExceptionHandler::exception("Not enough locks for non-halo buffers! Num Locks: {}, Buffers: {}",
630 _bufferLocks.size(), particleBuffers.size());
631 }
632
633 // Balance buffers. This makes processing them with static scheduling quite efficient.
634 // Also, if particles were not inserted in parallel, this enables us to process them in parallel now.
635 // Cost is at max O(2N) worst O(N) per buffer collection and negligible compared to interacting them.
636 auto cellToVec = [](auto &cell) -> std::vector<Particle> & { return cell._particles; };
637 utils::ArrayUtils::balanceVectors(particleBuffers, cellToVec);
638 utils::ArrayUtils::balanceVectors(haloParticleBuffers, cellToVec);
639
640 // The following part performes the main remainder traversal. The actual calculation is done in 4 steps carried out in
641 // three helper functions.
642
643 // only activate time measurements if it will actually be logged
644#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
645 autopas::utils::Timer timerBufferContainer;
646 autopas::utils::Timer timerPBufferPBuffer;
647 autopas::utils::Timer timerPBufferHBuffer;
648
649 timerBufferContainer.start();
650#endif
651 // steps 1 & 2. particleBuffer with all close particles in container and haloParticleBuffer with owned, close
652 // particles in container
653 remainderHelperBufferContainer<newton3>(f, container, particleBuffers, haloParticleBuffers);
654
655#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
656 timerBufferContainer.stop();
657 timerPBufferPBuffer.start();
658#endif
659
660 // step 3. particleBuffer with itself and all other buffers
661 remainderHelperBufferBuffer<newton3>(f, particleBuffers, haloParticleBuffers);
662
663#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
664 timerPBufferPBuffer.stop();
665 timerPBufferHBuffer.start();
666#endif
667
668 // step 4. particleBuffer with haloParticleBuffer
669 remainderHelperBufferHaloBuffer<newton3>(f, particleBuffers, haloParticleBuffers);
670
671#if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_TRACE
672 timerPBufferHBuffer.stop();
673#endif
674
675 // unpack particle SoAs. Halo data is not interesting
676 for (auto &buffer : particleBuffers) {
677 f->SoAExtractor(buffer, buffer._particleSoABuffer, 0);
678 }
679
680 AutoPasLog(TRACE, "Timer Buffers <-> Container (1+2): {}", timerBufferContainer.getTotalTime());
681 AutoPasLog(TRACE, "Timer PBuffers<-> PBuffer ( 3): {}", timerPBufferPBuffer.getTotalTime());
682 AutoPasLog(TRACE, "Timer PBuffers<-> HBuffer ( 4): {}", timerPBufferHBuffer.getTotalTime());
683
684 // Note: haloParticleBuffer with itself is NOT needed, as interactions between halo particles are unneeded!
685}

687template <class Particle>
688template <bool newton3, class ContainerType, class PairwiseFunctor>
689void AutoTuner<Particle>::remainderHelperBufferContainer(PairwiseFunctor *f, ContainerType &container,
690 std::vector<FullParticleCell<Particle>> &particleBuffers,
691 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
692 using autopas::utils::ArrayUtils::static_cast_copy_array;
693 using namespace autopas::utils::ArrayMath::literals;
694
695 const auto haloBoxMin = container.getBoxMin() - container.getInteractionLength();
696 const auto interactionLengthInv = 1. / container.getInteractionLength();
697
698 const double cutoff = container.getCutoff();
699#ifdef AUTOPAS_OPENMP
700// one halo and particle buffer pair per thread
701#pragma omp parallel for schedule(static, 1), shared(f, _spacialLocks, haloBoxMin, interactionLengthInv)
702#endif
703 for (int bufferId = 0; bufferId < particleBuffers.size(); ++bufferId) {
704 auto &particleBuffer = particleBuffers[bufferId];
705 auto &haloParticleBuffer = haloParticleBuffers[bufferId];
706 // 1. particleBuffer with all close particles in container
707 for (auto &&p1 : particleBuffer) {
708 const auto pos = p1.getR();
709 const auto min = pos - cutoff;
710 const auto max = pos + cutoff;
711 container.forEachInRegion(
712 [&](auto &p2) {
713 const auto lockCoords = static_cast_copy_array<size_t>((p2.getR() - haloBoxMin) * interactionLengthInv);
714 if constexpr (newton3) {
715 const std::lock_guard<std::mutex> lock(*_spacialLocks[lockCoords[0]][lockCoords[1]][lockCoords[2]]);
716 f->AoSFunctor(p1, p2, true);
717 } else {
718 f->AoSFunctor(p1, p2, false);
719 // no need to calculate force enacted on a halo
720 if (not p2.isHalo()) {
721 const std::lock_guard<std::mutex> lock(*_spacialLocks[lockCoords[0]][lockCoords[1]][lockCoords[2]]);
722 f->AoSFunctor(p2, p1, false);
723 }
724 }
725 },
726 min, max, IteratorBehavior::ownedOrHalo);
727 }
728
729 // 2. haloParticleBuffer with owned, close particles in container
730 for (auto &&p1halo : haloParticleBuffer) {
731 const auto pos = p1halo.getR();
732 const auto min = pos - cutoff;
733 const auto max = pos + cutoff;
734 container.forEachInRegion(
735 [&](auto &p2) {
736 const auto lockCoords = static_cast_copy_array<size_t>((p2.getR() - haloBoxMin) * interactionLengthInv);
737 // No need to apply anything to p1halo
738 // -> AoSFunctor(p1, p2, false) not needed as it neither adds force nor Upot (potential energy)
739 // -> newton3 argument needed for correct globals
740 const std::lock_guard<std::mutex> lock(*_spacialLocks[lockCoords[0]][lockCoords[1]][lockCoords[2]]);
741 f->AoSFunctor(p2, p1halo, newton3);
742 },
743 min, max, IteratorBehavior::owned);
744 }
745 }
746}

748template <class Particle>
749template <bool newton3, class PairwiseFunctor>
750void AutoTuner<Particle>::remainderHelperBufferBuffer(PairwiseFunctor *f,
751 std::vector<FullParticleCell<Particle>> &particleBuffers,
752 std::vector<FullParticleCell<Particle>> &haloParticleBuffers) {
753 // All (halo-)buffer interactions shall happen vectorized, hence, load all buffer data into SoAs
754 for (auto &buffer : particleBuffers) {
755 f->SoALoader(buffer, buffer._particleSoABuffer, 0);
756 }
757 for (auto &buffer : haloParticleBuffers) {
758 f->SoALoader(buffer, buffer._particleSoABuffer, 0);
759 }
760
761#ifdef AUTOPAS_OPENMP
762#pragma omp parallel
763#endif
764 {
765 // For buffer interactions where bufferA == bufferB we can always enable newton3. For all interactions between
766 // different buffers we turn newton3 always off, which ensures that only one thread at a time is writing to a
767 // buffer. This saves expensive locks.
768#ifdef AUTOPAS_OPENMP
769 // we can not use collapse here without locks, otherwise races would occur.
770#pragma omp for
771#endif
772 for (size_t i = 0; i < particleBuffers.size(); ++i) {
773 for (size_t jj = 0; jj < particleBuffers.size(); ++jj) {
774 auto *particleBufferSoAA = &particleBuffers[i]._particleSoABuffer;
775 const auto j = (i + jj) % particleBuffers.size();
776 if (i == j) {
777 f->SoAFunctorSingle(*particleBufferSoAA, true);
778 } else {
779 auto *particleBufferSoAB = &particleBuffers[j]._particleSoABuffer;
780 f->SoAFunctorPair(*particleBufferSoAA, *particleBufferSoAB, false);
781 }
782 }
783 }
784 }

809template <class Particle>
810template <class PairwiseFunctor, DataLayoutOption::Value dataLayout, bool useNewton3, bool inTuningPhase>
811void AutoTuner<Particle>::iteratePairwiseTemplateHelper(PairwiseFunctor *f, bool doListRebuild,
812 std::vector<FullParticleCell<Particle>> &particleBuffer,
813 std::vector<FullParticleCell<Particle>> &haloParticleBuffer) {
814 auto &container = getContainer();
815 AutoPasLog(DEBUG, "Iterating with configuration: {} tuning: {}",
816 _tuningStrategy->getCurrentConfiguration().toString(), inTuningPhase ? "true" : "false");
817
818 auto traversal =
819 autopas::utils::withStaticCellType<Particle>(container.getParticleCellTypeEnum(), [&](auto particleCellDummy) {
820 return TraversalSelector<decltype(particleCellDummy)>::template generateTraversal<PairwiseFunctor, dataLayout,
821 useNewton3>(
822 _tuningStrategy->getCurrentConfiguration().traversal, *f, container.getTraversalSelectorInfo());
823 });
824
825 if (not traversal->isApplicable()) {
826 autopas::utils::ExceptionHandler::exception(
827 "Error: Trying to execute a traversal that is not applicable. Two common reasons:\n"
828 "1. The search space is trivial, but no traversals are applicable. \n"
829 "2. You are using multiple functors and one of the not first is not supporting all configuration options of "
830 "the first.\n"
831 "Config: {}\n"
832 "Current functor: {}",
833 _tuningStrategy->getCurrentConfiguration().toString(), typeid(*f).name());
834 }
835

839 autopas::utils::Timer timerRemainderTraversal;
840 if (_energy_measurement_available) {
841 try {
842 _raplMeter.reset();
843 } catch (const utils::ExceptionHandler::AutoPasException &e) {
848 AutoPasLog(WARN, "Energy Measurement no longer possible:\n\t{}", e.what());
849 _energy_measurement_available = false;
850 if (_tuningMetric == TuningMetricOption::energy) {
851 throw e;
852 }
853 }
854 }
855 timerTotal.start();
856
857 f->initTraversal();
858 if (doListRebuild) {
859 timerRebuild.start();
860 container.rebuildNeighborLists(traversal.get());
861 timerRebuild.stop();
862 }
863 timerIteratePairwise.start();
864 container.iteratePairwise(traversal.get());
865 timerIteratePairwise.stop();
866
867 timerRemainderTraversal.start();
868 withStaticContainerType(container, [&](auto &actualContainerType) {
869 doRemainderTraversal<useNewton3>(f, actualContainerType, particleBuffer, haloParticleBuffer);
870 });
871 timerRemainderTraversal.stop();
872 f->endTraversal(useNewton3);
873
874 if (_energy_measurement_available) {
875 try {
876 _raplMeter.sample();
877 } catch (const utils::ExceptionHandler::AutoPasException &e) {
878 AutoPasLog(WARN, "Energy Measurement no longer possible:\n\t{}", e.what());
879 _energy_measurement_available = false;
880 if (_tuningMetric == TuningMetricOption::energy) {
881 throw e;
882 }
883 }
884 }
885
886 timerTotal.stop();
887

889 std::stringstream ss;

891 for (const auto &buffer : buffers) {
892 ss << buffer.numParticles() << ", ";
893 sum += buffer.numParticles();
894 }
895 ss << " Total: " << sum;
896 return ss.str();
897 };
898 AutoPasLog(TRACE, "particleBuffer size : {}", bufferSizeListing(particleBuffer));
899 AutoPasLog(TRACE, "haloParticleBuffer size : {}", bufferSizeListing(haloParticleBuffer));
900 AutoPasLog(DEBUG, "Container::iteratePairwise took {} ns", timerIteratePairwise.getTotalTime());
901 AutoPasLog(DEBUG, "RemainderTraversal took {} ns", timerRemainderTraversal.getTotalTime());
902 AutoPasLog(DEBUG, "RebuildNeighborLists took {} ns", timerRebuild.getTotalTime());
903 AutoPasLog(DEBUG, "AutoTuner::iteratePairwise took {} ns", timerTotal.getTotalTime());

908 timerRemainderTraversal.getTotalTime(), timerRebuild.getTotalTime(),
909 timerTotal.getTotalTime(), _raplMeter.get_psys_energy(), _raplMeter.get_pkg_energy(),
910 _raplMeter.get_ram_energy());
911 } else {
912 constexpr auto nan = std::numeric_limits<double>::quiet_NaN();
913 _iterationLogger.logIteration(getCurrentConfig(), _iteration, inTuningPhase, timerIteratePairwise.getTotalTime(),
914 timerRemainderTraversal.getTotalTime(), timerRebuild.getTotalTime(),
915 timerTotal.getTotalTime(), nan, nan, nan);
916 }
917 // if tuning execute with time measurements
918 if (inTuningPhase) {
919 if (f->isRelevantForTuning()) {
920 switch (this->_tuningMetric) {
921 case TuningMetricOption::time:
922 addMeasurement(timerTotal.getTotalTime(), doListRebuild);
923 break;
924 case TuningMetricOption::energy:
925 addMeasurement(_raplMeter.get_total_energy(), doListRebuild);
926 break;
927 }
928 } else {
929 AutoPasLog(TRACE, "Skipping adding of sample because functor is not marked relevant.");
930 }
931 }

934template <class Particle>
935template <class PairwiseFunctor>
936bool AutoTuner<Particle>::tune(PairwiseFunctor &pairwiseFunctor) {
937 bool stillTuning = true;
938
939 // need more samples; keep current config
940 if (getCurrentNumSamples() < _maxSamples) {
941 return stillTuning;
942 }
943 utils::Timer tuningTimer;
944 tuningTimer.start();
945
946 // first tuning iteration -> reset to first config
947 if (_iterationsSinceTuning == _tuningInterval) {
948 // if necessary gather current live info and pass it to the actual strategy
949 if (_tuningStrategy->needsLiveInfo()) {
950 LiveInfo info{};
951 info.gather(_containerSelector.getCurrentContainer(), pairwiseFunctor, _rebuildFrequency);
952 _tuningStrategy->receiveLiveInfo(info);
953 }
954 // call the appropriate version of reset
955 if (auto *mpiStrategy = dynamic_cast<MPIParallelizedStrategy *>(_tuningStrategy.get())) {
956 const std::pair<double, double> smoothedHomogeneityAndMaxDensity{
957 autopas::OptimumSelector::medianValue(_homogeneitiesOfLastTenIterations),
958 autopas::OptimumSelector::medianValue(_maxDensitiesOfLastTenIterations)};
959 mpiStrategy->reset<Particle>(_iteration, getContainer(), smoothedHomogeneityAndMaxDensity,
960 _mpiTuningMaxDifferenceForBucket, _mpiTuningWeightForMaxDensity);
961 } else {
962 _tuningStrategy->reset(_iteration);
963 }
964 // Homogeneity was calculated directly before the tuning phase so reset it now.
965 if (_tuningStrategy->smoothedHomogeneityAndMaxDensityNeeded()) {
966 AutoPasLog(DEBUG, "Calculating homogeneities took added up {} ns on rank {}.",
967 _timerCalculateHomogeneity.getTotalTime(), []() {
968 int rank{0};
969 AutoPas_MPI_Comm_rank(AUTOPAS_MPI_COMM_WORLD, &rank);
970 return rank;
971 });
972 _homogeneitiesOfLastTenIterations.clear();
973 _maxDensitiesOfLastTenIterations.clear();
974 }
975 } else { // enough samples -> next config
976 stillTuning = _tuningStrategy->tune();
977 }
978
979 // repeat as long as traversals are not applicable, or we run out of configs
980 while (true) {
981 auto &currentConfig = getCurrentConfig();
982 // check if newton3 works with this functor and remove config if not
983 if ((currentConfig.newton3 == Newton3Option::enabled and not pairwiseFunctor.allowsNewton3()) or
984 (currentConfig.newton3 == Newton3Option::disabled and not pairwiseFunctor.allowsNonNewton3())) {
985 AutoPasLog(WARN, "Configuration with newton 3 {} called with a functor that does not support this!",
986 currentConfig.newton3.to_string());
987
988 _tuningStrategy->removeN3Option(currentConfig.newton3);
989 } else {
990 if (configApplicable(currentConfig, pairwiseFunctor)) {
991 // we found a valid config!
992 break;
993 } else {
994 AutoPasLog(DEBUG, "Skip not applicable configuration {}", currentConfig.toString());
995 stillTuning = _tuningStrategy->tune(true);
996 }
997 }
998 }
999 // samples should only be cleared if we are still tuning, see `if (_samples.size() < _maxSamples)` from before.
1000 if (stillTuning) {
1001 // samples are no longer needed. Delete them here so willRebuild() works as expected.
1002 _samplesNotRebuildingNeighborLists.clear();
1003 _samplesRebuildingNeighborLists.clear();
1004 }
1005 tuningTimer.stop();
1006 // when a tuning result is found log it
1007 if (not stillTuning) {
1008 AutoPasLog(DEBUG, "Selected Configuration {}", getCurrentConfig().toString());
1009 _tuningResultLogger.logTuningResult(getCurrentConfig(), _iteration, tuningTimer.getTotalTime());
1010 }
1011 AutoPasLog(DEBUG, "Tuning took {} ns.", tuningTimer.getTotalTime());
1012 _iterationLogger.logTimeTuning(tuningTimer.getTotalTime());
1013
1014 selectCurrentContainer();
1015 return stillTuning;
1016}

1046template <class Particle>
1047void AutoTuner<Particle>::addMeasurement(long sample, bool neighborListRebuilt) {
1048 const auto &currentConfig = _tuningStrategy->getCurrentConfiguration();
1049 if (getCurrentNumSamples() < _maxSamples) {
1050 AutoPasLog(TRACE, "Adding sample.");
1051 if (neighborListRebuilt) {
1052 _samplesRebuildingNeighborLists.push_back(sample);
1053 } else {
1054 _samplesNotRebuildingNeighborLists.push_back(sample);
1055 }
1056 // if this was the last sample:
1057 if (getCurrentNumSamples() == _maxSamples) {
1058 auto &evidenceCurrentConfig = _evidence[currentConfig];
1059
1060 const long reducedValue = estimateRuntimeFromSamples();
1061
1062 evidenceCurrentConfig.emplace_back(_iteration, reducedValue);
1063
1064 // smooth evidence to remove high outliers. If smoothing results in a higher value use the original value.
1065 const auto smoothedValue = std::min(reducedValue, smoothing::smoothLastPoint(evidenceCurrentConfig, 5));
1066
1067 // replace collected evidence with smoothed value to improve next smoothing
1068 evidenceCurrentConfig.back().second = smoothedValue;
1069
1070 _tuningStrategy->addEvidence(smoothedValue, _iteration);
1071
1072 // print config, times and reduced value
1073 if (autopas::Logger::get()->level() <= autopas::Logger::LogLevel::debug) {
1074 std::ostringstream ss;
1075 // print config
1076 ss << currentConfig.toString() << " : ";
1077 // print all timings
1078 ss << utils::ArrayUtils::to_string(_samplesRebuildingNeighborLists, " ",
1079 {"With rebuilding neighbor lists [ ", " ]"});
1080 ss << utils::ArrayUtils::to_string(_samplesNotRebuildingNeighborLists, " ",
1081 {"Without rebuilding neighbor lists [ ", " ]"});
1082 ss << " Smoothed value: " << smoothedValue;
1083 switch (this->_tuningMetric) {
1084 case TuningMetricOption::time:
1085 AutoPasLog(DEBUG, "Collected times for {}", ss.str());
1086 break;
1087 case TuningMetricOption::energy:
1088 AutoPasLog(DEBUG, "Collected energy consumption for {}", ss.str());
1089 break;
1090 }
1091 }
1092 _tuningDataLogger.logTuningData(currentConfig, _samplesRebuildingNeighborLists,
1093 _samplesNotRebuildingNeighborLists, _iteration, reducedValue, smoothedValue);
1094 }
1095 }
