217 [[nodiscard]] long estimateRuntimeFromSamples() const {
218 // reduce samples for rebuild and non-rebuild iterations with the given selector strategy
219 const auto reducedValueBuilding =
220 autopas::OptimumSelector::optimumValue(_samplesRebuildingNeighborLists, _selectorStrategy);
221 // if there is no data for the non rebuild iterations we have to assume them taking the same time as rebuilding ones
222 // this might neither be a good estimate nor fair but the best we can do
223 const auto reducedValueNotBuilding =
224 _samplesNotRebuildingNeighborLists.empty()
225 ? reducedValueBuilding
226 : autopas::OptimumSelector::optimumValue(_samplesNotRebuildingNeighborLists, _selectorStrategy);
227
228 const auto numIterationsNotBuilding =
229 std::max(0, static_cast<int>(_rebuildFrequency) - static_cast<int>(_samplesRebuildingNeighborLists.size()));
230 const auto numIterationsBuilding = _rebuildFrequency - numIterationsNotBuilding;
231
232 // calculate weighted estimate for one iteration
233 return (numIterationsBuilding * reducedValueBuilding + numIterationsNotBuilding * reducedValueNotBuilding) /
234 _rebuildFrequency;
235 }

372template <class Particle>
373template <class PairwiseFunctor>
374bool AutoTuner<Particle>::iteratePairwise(PairwiseFunctor *f, bool doListRebuild,
375 std::vector<std::vector<Particle>> &particleBuffers,
376 std::vector<std::vector<Particle>> &haloParticleBuffers) {
377 bool isTuning = false;
378 // tune if :
379 // - more than one config exists
380 // - currently in tuning phase
381 // - functor is relevant
382 if (_tuningStrategy->smoothedHomogeneityAndMaxDensityNeeded() and _iterationsSinceTuning >= _tuningInterval - 9 and
383 _iterationsSinceTuning <= _tuningInterval) {
384 _timerCalculateHomogeneity.start();
385 const auto &container = getContainer();
386 const auto [homogeneity, maxDensity] =
387 autopas::utils::calculateHomogeneityAndMaxDensity(*container, container->getBoxMin(), container->getBoxMax());
388 _homogeneitiesOfLastTenIterations.push_back(homogeneity);
389 _maxDensitiesOfLastTenIterations.push_back(maxDensity);
390 _timerCalculateHomogeneity.stop();
391 }
392 if ((not _tuningStrategy->searchSpaceIsTrivial()) and _iterationsSinceTuning >= _tuningInterval and
393 f->isRelevantForTuning()) {
394 isTuning = tune<PairwiseFunctor>(*f);
395 if (not isTuning) {
396 _iterationsSinceTuning = 0;
397 }
398 }
399
400 // large case differentiation for data layout and newton 3
401 switch (_tuningStrategy->getCurrentConfiguration().dataLayout) {
402 case DataLayoutOption::aos: {
403 if (_tuningStrategy->getCurrentConfiguration().newton3 == Newton3Option::enabled) {
404 if (isTuning) {
405 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::aos, /*Newton3*/ true,
406 /*tuning*/ true>(f, doListRebuild, particleBuffers, haloParticleBuffers);
407 } else {
408 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::aos, /*Newton3*/ true,
409 /*tuning*/ false>(f, doListRebuild, particleBuffers, haloParticleBuffers);
410 }
411 } else {
412 if (isTuning) {
413 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::aos, /*Newton3*/ false,
414 /*tuning*/ true>(f, doListRebuild, particleBuffers, haloParticleBuffers);
415 } else {
416 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::aos, /*Newton3*/ false,
417 /*tuning*/ false>(f, doListRebuild, particleBuffers, haloParticleBuffers);
418 }
419 }
420 break;
421 }
422 case DataLayoutOption::soa: {
423 if (_tuningStrategy->getCurrentConfiguration().newton3 == Newton3Option::enabled) {
424 if (isTuning) {
425 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::soa, /*Newton3*/ true,
426 /*tuning*/ true>(f, doListRebuild, particleBuffers, haloParticleBuffers);
427 } else {
428 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::soa, /*Newton3*/ true,
429 /*tuning*/ false>(f, doListRebuild, particleBuffers, haloParticleBuffers);
430 }
431 } else {
432 if (isTuning) {
433 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::soa, /*Newton3*/ false,
434 /*tuning*/ true>(f, doListRebuild, particleBuffers, haloParticleBuffers);
435 } else {
436 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::soa, /*Newton3*/ false,
437 /*tuning*/ false>(f, doListRebuild, particleBuffers, haloParticleBuffers);
438 }
439 }
440 break;
441 }
442 default:
443 utils::ExceptionHandler::exception("AutoTuner: Unknown data layout : {}",
444 _tuningStrategy->getCurrentConfiguration().dataLayout);
445 }
446
447 if (f->isRelevantForTuning()) {
448 ++_iterationsSinceTuning;
449 ++_iteration;
450 }
451 return isTuning;
452}

475template <bool newton3, class Particle, class T, class PairwiseFunctor>
476void doRemainderTraversal(PairwiseFunctor *f, T containerPtr, std::vector<std::vector<Particle>> &particleBuffers,
477 std::vector<std::vector<Particle>> &haloParticleBuffers) {
478 const double cutoff = containerPtr->getCutoff();
479 for (int bufferId = 0; bufferId < particleBuffers.size(); ++bufferId) {
480 auto &particleBuffer = particleBuffers[bufferId];
481 auto &haloParticleBuffer = haloParticleBuffers[bufferId];
482 // 1. particleBuffer with all close particles in container
483 for (auto &&p : particleBuffer) {
484 auto pos = p.getR();
485 auto min = autopas::utils::ArrayMath::subScalar(pos, cutoff);
486 auto max = autopas::utils::ArrayMath::addScalar(pos, cutoff);
487 for (auto iter2 = containerPtr->getRegionIterator(min, max, IteratorBehavior::ownedOrHalo); iter2.isValid();
488 ++iter2) {
489 if (newton3) {
490 f->AoSFunctor(p, *iter2, true);
491 } else {
492 f->AoSFunctor(p, *iter2, false);
493 f->AoSFunctor(*iter2, p, false);
494 }
495 }
496 }
497
498 // 2. haloParticleBuffer with owned, close particles in container
499 for (auto &&p : haloParticleBuffer) {
500 auto pos = p.getR();
501 auto min = autopas::utils::ArrayMath::subScalar(pos, cutoff);
502 auto max = autopas::utils::ArrayMath::addScalar(pos, cutoff);
503 for (auto iter2 = containerPtr->getRegionIterator(min, max, IteratorBehavior::owned); iter2.isValid(); ++iter2) {
504 if (newton3) {
505 f->AoSFunctor(p, *iter2, true);
506 } else {
507 // Here, we do not need to interact p with *iter2, because p is a halo particle and an AoSFunctor call with an
508 // halo particle as first argument has no effect if newton3 == false.
509 f->AoSFunctor(*iter2, p, false);
510 }
511 }
512 }
513 }
514
515 // 3. particleBuffer with itself and all other buffers
516 for (size_t bufferIdOuter = 0; bufferIdOuter < particleBuffers.size(); ++bufferIdOuter) {
517 auto &particleBufferOuter = particleBuffers[bufferIdOuter];
518 for (size_t i = 0; i < particleBufferOuter.size(); ++i) {
519 auto &&p1 = particleBufferOuter[i];
520 for (size_t bufferIdInner = bufferIdOuter; bufferIdInner < particleBuffers.size(); ++bufferIdInner) {
521 auto &particleBufferInner = particleBuffers[bufferIdInner];
522 for (size_t j = bufferIdOuter == bufferIdInner ? i + 1 : 0; j < particleBufferInner.size(); ++j) {
523 auto &&p2 = particleBufferInner[j];
524 if (newton3) {
525 f->AoSFunctor(p1, p2, true);
526 } else {
527 f->AoSFunctor(p1, p2, false);
528 f->AoSFunctor(p2, p1, false);
529 }
530 }
531 }
532 }
533 }
534 // 4. particleBuffer with haloParticleBuffer
535 for (size_t bufferIdOuter = 0; bufferIdOuter < particleBuffers.size(); ++bufferIdOuter) {
536 auto &particleBufferOuter = particleBuffers[bufferIdOuter];
537 for (size_t i = 0; i < particleBufferOuter.size(); ++i) {
538 auto &&p1 = particleBufferOuter[i];
539 for (size_t bufferIdInner = 0; bufferIdInner < haloParticleBuffers.size(); ++bufferIdInner) {
540 auto &haloBufferInner = haloParticleBuffers[bufferIdInner];
541 for (size_t j = 0; j < haloBufferInner.size(); ++j) {
542 auto &&p2 = haloBufferInner[j];
543 if (newton3) {
544 f->AoSFunctor(p1, p2, true);
545 } else {
546 // Here, we do not need to interact p2 with p1, because p2 is a halo particle and an AoSFunctor call with an
547 // halo particle as first argument has no effect if newton3 == false.
548 f->AoSFunctor(p1, p2, false);
549 }
550 }
551 }
552 }
553 }
554
555 // Note: haloParticleBuffer with itself is NOT needed, as interactions between halo particles are unneeded!
556}

558template <class Particle>
559template <class PairwiseFunctor, DataLayoutOption::Value dataLayout, bool useNewton3, bool inTuningPhase>
560void AutoTuner<Particle>::iteratePairwiseTemplateHelper(PairwiseFunctor *f, bool doListRebuild,
561 std::vector<std::vector<Particle>> &particleBuffer,
562 std::vector<std::vector<Particle>> &haloParticleBuffer) {
563 auto containerPtr = getContainer();
564 AutoPasLog(debug, "Iterating with configuration: {} tuning: {}",
565 _tuningStrategy->getCurrentConfiguration().toString(), inTuningPhase ? "true" : "false");
566
567 auto traversal = autopas::utils::withStaticCellType<Particle>(
568 containerPtr->getParticleCellTypeEnum(), [&](auto particleCellDummy) {
569 return TraversalSelector<decltype(particleCellDummy)>::template generateTraversal<PairwiseFunctor, dataLayout,
570 useNewton3>(
571 _tuningStrategy->getCurrentConfiguration().traversal, *f, containerPtr->getTraversalSelectorInfo());
572 });
573
574 if (not traversal->isApplicable()) {
575 autopas::utils::ExceptionHandler::exception(
576 "Error: Trying to execute a traversal that is not applicable. Two common reasons:\n"
577 "1. The search space is trivial, but no traversals are applicable. \n"
578 "2. You are using multiple functors and one of the not first is not supporting all configuration options of "
579 "the first.\n"
580 "Config: {}\n"
581 "Current functor: {}",
582 _tuningStrategy->getCurrentConfiguration().toString(), typeid(*f).name());
583 }
584
585 autopas::utils::Timer timerTotal;
586 autopas::utils::Timer timerRebuild;
587 autopas::utils::Timer timerIteratePairwise;
588 timerTotal.start();
589
590 f->initTraversal();
591 if (doListRebuild) {
592 timerRebuild.start();
593 containerPtr->rebuildNeighborLists(traversal.get());
594 timerRebuild.stop();
595 }
596 timerIteratePairwise.start();
597 containerPtr->iteratePairwise(traversal.get());
598
599 doRemainderTraversal<useNewton3>(f, containerPtr, particleBuffer, haloParticleBuffer);
600 timerIteratePairwise.stop();
601 f->endTraversal(useNewton3);
602
603 timerTotal.stop();
604
605 if (doListRebuild) {
606 AutoPasLog(debug, "rebuildNeighborLists took {} nanoseconds", timerRebuild.getTotalTime());
607 }
608 // TODO: Fix misleading output
609 // actually this is the time for iteratePairwise + init/endTraversal + rebuildNeighborLists
610 // this containing all of this has legacy reasons so that old plot scripts work
611 AutoPasLog(debug, "IteratePairwise took {} nanoseconds", timerTotal.getTotalTime());
612
613 _iterationLogger.logIteration(getCurrentConfig(), _iteration, inTuningPhase, timerIteratePairwise.getTotalTime(),
614 timerRebuild.getTotalTime(), timerTotal.getTotalTime());
615
616 // if tuning execute with time measurements
617 if (inTuningPhase) {
618 if (f->isRelevantForTuning()) {
619 addTimeMeasurement(timerTotal.getTotalTime(), doListRebuild);
620 } else {
621 AutoPasLog(trace, "Skipping adding of time measurement because functor is not marked relevant.");
622 }
623 }

626template <class Particle>
627template <class PairwiseFunctor>
628bool AutoTuner<Particle>::tune(PairwiseFunctor &pairwiseFunctor) {
629 bool stillTuning = true;
630
631 // need more samples; keep current config
632 if (getCurrentNumSamples() < _maxSamples) {
633 return stillTuning;
634 }
635 utils::Timer tuningTimer;
636 tuningTimer.start();
637
638 // first tuning iteration -> reset to first config
639 if (_iterationsSinceTuning == _tuningInterval) {
640 if (auto *mpiStrategy = dynamic_cast<MPIParallelizedStrategy *>(_tuningStrategy.get())) {
641 const std::pair<double, double> smoothedHomogeneityAndMaxDensity{
642 autopas::OptimumSelector::medianValue(_homogeneitiesOfLastTenIterations),
643 autopas::OptimumSelector::medianValue(_maxDensitiesOfLastTenIterations)};
644 mpiStrategy->reset<Particle>(_iteration, getContainer(), smoothedHomogeneityAndMaxDensity,
645 _mpiTuningMaxDifferenceForBucket, _mpiTuningWeightForMaxDensity);
646 } else {
647 _tuningStrategy->reset(_iteration);
648 }
649 // only used temporarily of evaluation of smoothing
650 if (_tuningStrategy->smoothedHomogeneityAndMaxDensityNeeded()) {
651 int rank{0};
652#ifdef AUTOPAS_INTERNODE_TUNING
653 MPI_Comm_rank(MPI_COMM_WORLD, &rank);
654#endif
655 AutoPasLog(debug, "Calculating homogeneities took added up {} ns on rank {}.",
656 _timerCalculateHomogeneity.getTotalTime(), rank);
657 _homogeneitiesOfLastTenIterations.erase(_homogeneitiesOfLastTenIterations.begin(),
658 _homogeneitiesOfLastTenIterations.end());
659 _maxDensitiesOfLastTenIterations.erase(_maxDensitiesOfLastTenIterations.begin(),
660 _maxDensitiesOfLastTenIterations.end());
661 }
662 } else { // enough samples -> next config
663 stillTuning = _tuningStrategy->tune();
664 }
665
666 // repeat as long as traversals are not applicable or we run out of configs
667 while (true) {
668 auto &currentConfig = getCurrentConfig();
669 // check if newton3 works with this functor and remove config if not
670 if ((currentConfig.newton3 == Newton3Option::enabled and not pairwiseFunctor.allowsNewton3()) or
671 (currentConfig.newton3 == Newton3Option::disabled and not pairwiseFunctor.allowsNonNewton3())) {
672 AutoPasLog(warn, "Configuration with newton 3 {} called with a functor that does not support this!",
673 currentConfig.newton3.to_string());
674
675 _tuningStrategy->removeN3Option(currentConfig.newton3);
676 } else {
677 if (configApplicable(currentConfig, pairwiseFunctor)) {
678 // we found a valid config!
679 break;
680 } else {
681 AutoPasLog(debug, "Skip not applicable configuration {}", currentConfig.toString());
682 stillTuning = _tuningStrategy->tune(true);
683 }
684 }
685 }
686 // samples should only be cleared if we are still tuning, see `if (_samples.size() < _maxSamples)` from before.
687 if (stillTuning) {
688 // samples are no longer needed. Delete them here so willRebuild() works as expected.
689 _samplesNotRebuildingNeighborLists.clear();
690 _samplesRebuildingNeighborLists.clear();
691 }
692 tuningTimer.stop();
693 // when a tuning result is found log it
694 if (not stillTuning) {
695 AutoPasLog(debug, "Selected Configuration {}", getCurrentConfig().toString());
696 _tuningResultLogger.logTuningResult(getCurrentConfig(), _iteration, tuningTimer.getTotalTime());
697 }
698 AutoPasLog(debug, "Tuning took {} ns.", tuningTimer.getTotalTime());
699 _iterationLogger.logTimeTuning(tuningTimer.getTotalTime());
700
701 selectCurrentContainer();
702 return stillTuning;
703}

734template <class Particle>
735void AutoTuner<Particle>::addTimeMeasurement(long time, bool neighborListRebuilt) {
736 const auto &currentConfig = _tuningStrategy->getCurrentConfiguration();
737 if (getCurrentNumSamples() < _maxSamples) {
738 AutoPasLog(trace, "Adding sample.");
739 if (neighborListRebuilt) {
740 _samplesRebuildingNeighborLists.push_back(time);
741 } else {
742 _samplesNotRebuildingNeighborLists.push_back(time);
743 }
744 // if this was the last sample:
745 if (getCurrentNumSamples() == _maxSamples) {
746 auto &evidenceCurrentConfig = _evidence[currentConfig];
747
748 const long reducedValue = estimateRuntimeFromSamples();
749
750 evidenceCurrentConfig.emplace_back(_iteration, reducedValue);
751
752 // smooth evidence to remove high outliers. If smoothing results in a higher value use the original value.
753 const auto smoothedValue = std::min(reducedValue, smoothing::smoothLastPoint(evidenceCurrentConfig, 5));
754
755 // replace collected evidence with smoothed value to improve next smoothing
756 evidenceCurrentConfig.back().second = smoothedValue;
757
758 _tuningStrategy->addEvidence(smoothedValue, _iteration);
759
760 // print config, times and reduced value
761 if (autopas::Logger::get()->level() <= autopas::Logger::LogLevel::debug) {
762 std::ostringstream ss;
763 // print config
764 ss << currentConfig.toString() << " : ";
765 // print all timings
766 ss << utils::ArrayUtils::to_string(_samplesRebuildingNeighborLists, " ",
767 {"With rebuilding neighbor lists [ ", " ]"});
768 ss << utils::ArrayUtils::to_string(_samplesNotRebuildingNeighborLists, " ",
769 {"Without rebuilding neighbor lists [ ", " ]"});
770 ss << " Smoothed value: " << smoothedValue;
771 AutoPasLog(debug, "Collected times for {}", ss.str());
772 }
773 _tuningDataLogger.logTuningData(currentConfig, _samplesRebuildingNeighborLists,
774 _samplesNotRebuildingNeighborLists, _iteration, reducedValue, smoothedValue);
775 }
776 }
