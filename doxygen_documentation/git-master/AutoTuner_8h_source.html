273template <class PairwiseFunctor>
274bool AutoTuner<Particle>::iteratePairwise(PairwiseFunctor *f, bool doListRebuild, std::vector<Particle> &particleBuffer,
275 std::vector<Particle> &haloParticleBuffer) {
276 bool isTuning = false;
277 // tune if :
278 // - more than one config exists
279 // - currently in tuning phase
280 // - functor is relevant
281 if ((not _tuningStrategy->searchSpaceIsTrivial()) and _iterationsSinceTuning >= _tuningInterval and
282 f->isRelevantForTuning()) {
283 isTuning = tune<PairwiseFunctor>(*f);
284 if (not isTuning) {
285 _iterationsSinceTuning = 0;
286 }
287 }
288
289 // large case differentiation for data layout and newton 3
290 switch (_tuningStrategy->getCurrentConfiguration().dataLayout) {
291 case DataLayoutOption::aos: {
292 if (_tuningStrategy->getCurrentConfiguration().newton3 == Newton3Option::enabled) {
293 if (isTuning) {
294 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::aos, /*Newton3*/ true,
295 /*tuning*/ true>(f, doListRebuild, particleBuffer, haloParticleBuffer);
296 } else {
297 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::aos, /*Newton3*/ true,
298 /*tuning*/ false>(f, doListRebuild, particleBuffer, haloParticleBuffer);
299 }
300 } else {
301 if (isTuning) {
302 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::aos, /*Newton3*/ false,
303 /*tuning*/ true>(f, doListRebuild, particleBuffer, haloParticleBuffer);
304 } else {
305 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::aos, /*Newton3*/ false,
306 /*tuning*/ false>(f, doListRebuild, particleBuffer, haloParticleBuffer);
307 }
308 }
309 break;
310 }
311 case DataLayoutOption::soa: {
312 if (_tuningStrategy->getCurrentConfiguration().newton3 == Newton3Option::enabled) {
313 if (isTuning) {
314 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::soa, /*Newton3*/ true,
315 /*tuning*/ true>(f, doListRebuild, particleBuffer, haloParticleBuffer);
316 } else {
317 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::soa, /*Newton3*/ true,
318 /*tuning*/ false>(f, doListRebuild, particleBuffer, haloParticleBuffer);
319 }
320 } else {
321 if (isTuning) {
322 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::soa, /*Newton3*/ false,
323 /*tuning*/ true>(f, doListRebuild, particleBuffer, haloParticleBuffer);
324 } else {
325 iteratePairwiseTemplateHelper<PairwiseFunctor, DataLayoutOption::soa, /*Newton3*/ false,
326 /*tuning*/ false>(f, doListRebuild, particleBuffer, haloParticleBuffer);
327 }
328 }
329 break;
330 }
331 default:
332 utils::ExceptionHandler::exception("AutoTuner: Unknown data layout : {}",
333 _tuningStrategy->getCurrentConfiguration().dataLayout);
334 }
335
336 if (f->isRelevantForTuning()) {
337 ++_iterationsSinceTuning;
338 ++_iteration;
339 }
340 return isTuning;
341}

356template <bool newton3, class Particle, class T, class PairwiseFunctor>
357void doRemainderTraversal(PairwiseFunctor *f, T containerPtr, std::vector<Particle> &particleBuffer,
358 std::vector<Particle> &haloParticleBuffer) {
359 // 1. particleBuffer with all close particles in container
360 for (auto &&p : particleBuffer) {
361 auto pos = p.getR();
362 auto min = autopas::utils::ArrayMath::subScalar(pos, containerPtr->getCutoff());
363 auto max = autopas::utils::ArrayMath::addScalar(pos, containerPtr->getCutoff());
364 for (auto iter2 = containerPtr->getRegionIterator(min, max, IteratorBehavior::ownedOrHalo); iter2.isValid();
365 ++iter2) {
366 if (newton3) {
367 f->AoSFunctor(p, *iter2, true);
368 } else {
369 f->AoSFunctor(p, *iter2, false);
370 f->AoSFunctor(*iter2, p, false);
371 }
372 }
373 }
374
375 // 2. haloParticleBuffer with owned, close particles in container
376 for (auto &&p : haloParticleBuffer) {
377 auto pos = p.getR();
378 auto min = autopas::utils::ArrayMath::subScalar(pos, containerPtr->getCutoff());
379 auto max = autopas::utils::ArrayMath::addScalar(pos, containerPtr->getCutoff());
380 for (auto iter2 = containerPtr->getRegionIterator(min, max, IteratorBehavior::owned); iter2.isValid(); ++iter2) {
381 if (newton3) {
382 f->AoSFunctor(p, *iter2, true);
383 } else {
384 // Here, we do not need to interact p with *iter2, because p is a halo particle and an AoSFunctor call with an
385 // halo particle as first argument has no effect if newton3 == false.
386 f->AoSFunctor(*iter2, p, false);
387 }
388 }
389 }
390
391 // 3. particleBuffer with itself
392 for (size_t i = 0; i < particleBuffer.size(); ++i) {
393 auto &&p1 = particleBuffer[i];
394 for (size_t j = i + 1; j < particleBuffer.size(); ++j) {
395 if (i == j) {
396 continue;
397 }
398 auto &&p2 = particleBuffer[j];
399 if (newton3) {
400 f->AoSFunctor(p1, p2, true);
401 } else {
402 f->AoSFunctor(p1, p2, false);
403 f->AoSFunctor(p2, p1, false);
404 }
405 }
406 }
407
408 // 4. particleBuffer with haloParticleBuffer
409 for (size_t i = 0; i < particleBuffer.size(); ++i) {
410 auto &&p1 = particleBuffer[i];
411 for (size_t j = 0; j < haloParticleBuffer.size(); ++j) {
412 auto &&p2 = haloParticleBuffer[j];
413 if (newton3) {
414 f->AoSFunctor(p1, p2, true);
415 } else {
416 // Here, we do not need to interact p2 with p1, because p2 is a halo particle and an AoSFunctor call with an
417 // halo particle as first argument has no effect if newton3 == false.
418 f->AoSFunctor(p1, p2, false);
419 }
420 }
421 }
422
423 // Note: haloParticleBuffer with itself is NOT needed, as interactions between halo particles are unneeded!
424}

427template <class PairwiseFunctor, DataLayoutOption::Value dataLayout, bool useNewton3, bool inTuningPhase>
428void AutoTuner<Particle>::iteratePairwiseTemplateHelper(PairwiseFunctor *f, bool doListRebuild,
429 std::vector<Particle> &particleBuffer,
430 std::vector<Particle> &haloParticleBuffer) {
431 auto containerPtr = getContainer();
432 AutoPasLog(debug, "Iterating with configuration: {} tuning: {}",
433 _tuningStrategy->getCurrentConfiguration().toString(), inTuningPhase ? "true" : "false");
434
435 auto traversal = autopas::utils::withStaticCellType<Particle>(
436 containerPtr->getParticleCellTypeEnum(), [&](auto particleCellDummy) {
437 return TraversalSelector<decltype(particleCellDummy)>::template generateTraversal<PairwiseFunctor, dataLayout,
438 useNewton3>(
439 _tuningStrategy->getCurrentConfiguration().traversal, *f, containerPtr->getTraversalSelectorInfo());
440 });
441
442 if (not traversal->isApplicable()) {
443 autopas::utils::ExceptionHandler::exception(
444 "Error: Trying to execute a traversal that is not applicable. Two common reasons:\n"
445 "1. The search space is trivial, but no traversals are applicable. \n"
446 "2. You are using multiple functors and one of the not first is not supporting all configuration options of "
447 "the first.\n"
448 "Config: {}\n"
449 "Current functor: {}",
450 _tuningStrategy->getCurrentConfiguration().toString(), typeid(*f).name());
451 }
452
453 autopas::utils::Timer timerTotal;
454 autopas::utils::Timer timerRebuild;
455 autopas::utils::Timer timerIteratePairwise;
456 timerTotal.start();
457
458 f->initTraversal();
459 if (doListRebuild) {
460 timerRebuild.start();
461 containerPtr->rebuildNeighborLists(traversal.get());
462 timerRebuild.stop();
463 }
464 timerIteratePairwise.start();
465 containerPtr->iteratePairwise(traversal.get());
466
467 doRemainderTraversal<useNewton3>(f, containerPtr, particleBuffer, haloParticleBuffer);
468 timerIteratePairwise.stop();
469 f->endTraversal(useNewton3);
470
471 timerTotal.stop();
472
473 if (doListRebuild) {
474 AutoPasLog(debug, "rebuildNeighborLists took {} nanoseconds", timerRebuild.getTotalTime());
475 }
476 // TODO: Fix misleading output
477 // actually this is the time for iteratePairwise + init/endTraversal + rebuildNeighborLists
478 // this containing all of this has legacy reasons so that old plot scripts work
479 AutoPasLog(debug, "IteratePairwise took {} nanoseconds", timerTotal.getTotalTime());
480
481 _iterationLogger.logIteration(getCurrentConfig(), _iteration, inTuningPhase, timerIteratePairwise.getTotalTime(),
482 timerRebuild.getTotalTime(), timerTotal.getTotalTime());
483
484 // if tuning execute with time measurements
485 if (inTuningPhase) {
486 if (f->isRelevantForTuning()) {
487 addTimeMeasurement(timerTotal.getTotalTime());
488 } else {
489 AutoPasLog(trace, "Skipping adding of time measurement because functor is not marked relevant.");
490 }
491 }

494template <class Particle>
495template <class PairwiseFunctor>
496bool AutoTuner<Particle>::tune(PairwiseFunctor &pairwiseFunctor) {
497 bool stillTuning = true;
498
499 // need more samples; keep current config
500 if (_samples.size() < _maxSamples) {
501 return stillTuning;
502 }
503 utils::Timer tuningTimer;
504 tuningTimer.start();
505 // first tuning iteration -> reset to first config
506 if (_iterationsSinceTuning == _tuningInterval) {
507 _tuningStrategy->reset(_iteration);
508 } else { // enough samples -> next config
509 stillTuning = _tuningStrategy->tune();
510 }
511
512 // repeat as long as traversals are not applicable or we run out of configs
513 while (true) {
514 auto &currentConfig = getCurrentConfig();
515 // check if newton3 works with this functor and remove config if not
516 if ((currentConfig.newton3 == Newton3Option::enabled and not pairwiseFunctor.allowsNewton3()) or
517 (currentConfig.newton3 == Newton3Option::disabled and not pairwiseFunctor.allowsNonNewton3())) {
518 AutoPasLog(warn, "Configuration with newton 3 {} called with a functor that does not support this!",
519 currentConfig.newton3.to_string());
520
521 _tuningStrategy->removeN3Option(currentConfig.newton3);
522 } else {
523 if (configApplicable(currentConfig, pairwiseFunctor)) {
524 // we found a valid config!
525 break;
526 } else {
527 AutoPasLog(debug, "Skip not applicable configuration {}", currentConfig.toString());
528 stillTuning = _tuningStrategy->tune(true);
529 }
530 }
531 }
532 // samples should only be cleared if we are still tuning, see `if (_samples.size() < _maxSamples)` from before.
533 if (stillTuning) {
534 // samples are no longer needed. Delete them here so willRebuild() works as expected.
535 _samples.clear();
536 }
537 tuningTimer.stop();
538 // when a tuning result is found log it
539 if (not stillTuning) {
540 AutoPasLog(debug, "Selected Configuration {}", getCurrentConfig().toString());
541 _tuningResultLogger.logTuningResult(getCurrentConfig(), _iteration, tuningTimer.getTotalTime());
542 }
543 AutoPasLog(debug, "Tuning took {} ns.", tuningTimer.getTotalTime());
544 _iterationLogger.logTimeTuning(tuningTimer.getTotalTime());
545
546 selectCurrentContainer();
547 return stillTuning;
548}

578template <class Particle>
579void AutoTuner<Particle>::addTimeMeasurement(long time) {
580 const auto &currentConfig = _tuningStrategy->getCurrentConfiguration();
581
582 if (_samples.size() < _maxSamples) {
583 AutoPasLog(trace, "Adding sample.");
584 _samples.push_back(time);
585 // if this was the last sample:
586 if (_samples.size() == _maxSamples) {
587 auto &evidenceCurrentConfig = _evidence[currentConfig];
588
589 const auto reducedValue = OptimumSelector::optimumValue(_samples, _selectorStrategy);
590 evidenceCurrentConfig.emplace_back(_iteration, reducedValue);
591
592 // smooth evidence to remove high outliers. If smoothing results in a higher value use the original value.
593 const auto smoothedValue = std::min(reducedValue, smoothing::smoothLastPoint(evidenceCurrentConfig, 5));
594
595 // replace collected evidence with smoothed value to improve next smoothing
596 evidenceCurrentConfig.back().second = smoothedValue;
597
598 _tuningStrategy->addEvidence(smoothedValue, _iteration);
599
600 // print config, times and reduced value
601 if (autopas::Logger::get()->level() <= autopas::Logger::LogLevel::debug) {
602 std::ostringstream ss;
603 // print config
604 ss << currentConfig.toString() << " : ";
605 // print all timings
606 ss << utils::ArrayUtils::to_string(_samples, " ", {"[ ", " ]"});
607 ss << " Smoothed value: " << smoothedValue;
608 AutoPasLog(debug, "Collected times for {}", ss.str());
609 }
610 _tuningDataLogger.logTuningData(currentConfig, _samples, _iteration, reducedValue, smoothedValue);
611 }
612 }
