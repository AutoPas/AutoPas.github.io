187 bool updateHaloParticle(const Particle &particle) override {
188 Particle pCopy = particle;
189 pCopy.setOwnershipState(OwnershipState::halo);
190 auto cells = _linkedCells.getCellBlock().getNearbyHaloCells(pCopy.getR(), this->getVerletSkin());
191 for (auto cellptr : cells) {
192 bool updated = internal::checkParticleInCellAndUpdateByID(*cellptr, pCopy);
193 if (updated) {
194 return true;
195 }
196 }
197 AutoPasLog(TRACE,
198 "updateHaloParticle was not able to update particle at "
199 "[{}, {}, {}]",
200 pCopy.getR()[0], pCopy.getR()[1], pCopy.getR()[2]);
201 return false;
202 }
