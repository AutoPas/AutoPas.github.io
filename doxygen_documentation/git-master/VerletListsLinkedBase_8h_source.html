169 bool updateHaloParticle(const Particle &particle) override {
170 Particle pCopy = particle;
171 pCopy.setOwnershipState(OwnershipState::halo);
172 auto cells = _linkedCells.getCellBlock().getNearbyHaloCells(pCopy.getR(), this->getVerletSkin());
173 for (auto cellptr : cells) {
174 bool updated = internal::checkParticleInCellAndUpdateByID(*cellptr, pCopy);
175 if (updated) {
176 return true;
177 }
178 }
179 AutoPasLog(TRACE,
180 "updateHaloParticle was not able to update particle at "
181 "[{}, {}, {}]",
182 pCopy.getR()[0], pCopy.getR()[1], pCopy.getR()[2]);
183 return false;
184 }
