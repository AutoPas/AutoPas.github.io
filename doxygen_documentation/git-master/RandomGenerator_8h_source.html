119void RandomGenerator::fillWithHaloParticles(Container &container, const Particle &defaultParticle, double haloWidth,
120 unsigned long numParticles, const HaloAddFunction &haloAddFunction,
121 unsigned int seed) {
122 srand(seed); // fixed seedpoint
123
124 auto haloBoxMin = container.getBoxMin();
125 auto haloBoxMax = container.getBoxMax();
126
127 // increase the box size not exactly by the width to make it exclusive
128 for (unsigned int i = 0; i < 3; ++i) {
129 haloBoxMin[i] -= haloWidth * .99;
130 haloBoxMax[i] += haloWidth * .99;
131 }
132
133 for (unsigned long i = defaultParticle.getID(); i < defaultParticle.getID() + numParticles; ++i) {
134 auto pos = randomPosition(haloBoxMin, haloBoxMax);
135 // we only want to add particles not in the actual box
136 while (autopas::utils::inBox(pos, container.getBoxMin(), container.getBoxMax())) {
137 pos = randomPosition(haloBoxMin, haloBoxMax);
138 }
139 Particle particle(defaultParticle);
140 particle.setR(pos);
141 particle.setID(i);
142 haloAddFunction(container, particle);
143 }
