83void FullSearch::selectOptimalConfiguration() {
84 if (_searchSpace.size() == 1) {
85 _currentConfig = _searchSpace.begin();
86 return;
87 }
88
89 // Time measure strategy
90 if (_traversalTimes.empty()) {
91 utils::ExceptionHandler::exception(
92 "FullSearch: Trying to determine fastest configuration without any measurements! "
93 "Either selectOptimalConfiguration was called too early or no applicable configurations were found");
94 }
95
96 const auto optimum = std::min_element(_traversalTimes.begin(), _traversalTimes.end(),
97 [](std::pair<Configuration, size_t> a,
98 std::pair<Configuration, size_t> b) -> bool { return a.second < b.second; });
99
100 _currentConfig = _searchSpace.find(optimum->first);
101 // sanity check
102 if (_currentConfig == _searchSpace.end()) {
103 autopas::utils::ExceptionHandler::exception(
104 "FullSearch: Optimal configuration not found in list of configurations!");
105 }
106}
