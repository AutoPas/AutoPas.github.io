172 bool BayesianSearch::tune(bool currentInvalid) {
173  if (currentInvalid) {
174  _invalidConfigs.insert(_currentConfig);
175  }
176 
177  if (searchSpaceIsEmpty()) {
178  // no valid configuration
179  _currentConfig = FeatureVector();
180  return false;
181  }
182 
183  if (_gaussianProcess.numEvidence() >= _maxEvidence) {
184  // select best config
185  _currentConfig = _encoder.oneHotDecode(_gaussianProcess.getEvidenceMax());
186  AutoPasLog(debug, "Selected Configuration {}", _currentConfig.toString());
187  return false;
188  }
189 
190  // try to sample a valid vector which is expected to yield a good acquisition
191  for (size_t i = 0; i < maxAttempts; ++i) {
192  if (_currentSamples.empty()) {
193  sampleAcquisitions(_predNumLHSamples, _predAcqFunction);
194  }
195 
196  // test best vectors until empty
197  while (not _currentSamples.empty()) {
198  if (_invalidConfigs.find(_currentSamples.back()) == _invalidConfigs.end()) {
199  // valid config found!
200  _currentConfig = _currentSamples.back();
201  return true;
202  } else {
203  // invalid: dispose
204  _currentSamples.pop_back();
205  }
206  }
207 
208  // No valid configuration. This should rarely happen.
209  AutoPasLog(debug, "Tuning could not generate a valid configuration.");
210  }
211 
212  utils::ExceptionHandler::exception("BayesianSearch: Failed to sample an valid FeatureVector");
213  _currentConfig = FeatureVector();
214  return false;
215 }

217 void BayesianSearch::sampleAcquisitions(size_t n, AcquisitionFunctionOption af) {
218  // create n lhs samples
219  _currentSamples = _encoder.lhsSampleFeatures(n, _rng);
220 
221  // map container and calculate all acquisition function values
222  std::map<FeatureVector, double> acquisitions;
223  for (auto &sample : _currentSamples) {
224  acquisitions[sample] = _gaussianProcess.calcAcquisition(af, _encoder.oneHotEncode(sample));
225  }
226 
227  // sort by acquisition
228  std::sort(_currentSamples.begin(), _currentSamples.end(),
229  [&acquisitions](const FeatureVector &f1, const FeatureVector &f2) {
230  return acquisitions[f1] < acquisitions[f2];
231  });
232 }
