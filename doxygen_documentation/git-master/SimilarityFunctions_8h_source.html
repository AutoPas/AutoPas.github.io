27template <class Container>
28std::pair<double, double> calculateHomogeneityAndMaxDensity(const Container &container,
29 const std::array<double, 3> &startCorner,
30 const std::array<double, 3> &endCorner) {
31 unsigned int numberOfParticles = static_cast<unsigned int>(container.getNumberOfParticles());
32 autopas::AutoPas_MPI_Allreduce(AUTOPAS_MPI_IN_PLACE, &numberOfParticles, 1, AUTOPAS_MPI_UNSIGNED_INT, AUTOPAS_MPI_SUM,
33 AUTOPAS_MPI_COMM_WORLD);
34 // approximately the resolution we want to get.
35 size_t numberOfCells = ceil(numberOfParticles / 10.);
36 std::array<double, 3> domainSizePerDimension = {};
37 for (int i = 0; i < 3; ++i) {
38 domainSizePerDimension[i] = endCorner[i] - startCorner[i];
39 }
40
41 // get cellLength which is equal in each direction, derived from the domainsize and the requested number of cells
42 const double volume = domainSizePerDimension[0] * domainSizePerDimension[1] * domainSizePerDimension[2];
43 const double cellVolume = volume / numberOfCells;
44 const double cellLength = cbrt(cellVolume);
45
46 // calculate the size of the boundary cells, which might be smaller then the other cells
47 std::array<size_t, 3> cellsPerDimension = {};
48 // size of the last cell layer per dimension. This cell might get truncated to fit in the domain.
49 std::array<double, 3> outerCellSizePerDimension = {};
50 for (int i = 0; i < 3; ++i) {
51 outerCellSizePerDimension[i] =
52 domainSizePerDimension[i] - (floor(domainSizePerDimension[i] / cellLength) * cellLength);
53 cellsPerDimension[i] = ceil(domainSizePerDimension[i] / cellLength);
54 }
55 // Actual number of cells we end up with
56 numberOfCells = cellsPerDimension[0] * cellsPerDimension[1] * cellsPerDimension[2];
57
58 std::vector<size_t> particlesPerCell(numberOfCells, 0);
59 std::vector<double> allVolumes(numberOfCells, 0);
60
61 // add particles accordingly to their cell to get the amount of particles in each cell
62 for (auto particleItr = container.begin(autopas::IteratorBehavior::owned); particleItr.isValid(); ++particleItr) {
63 const auto &particleLocation = particleItr->getR();
64 std::array<size_t, 3> index = {};
65 for (size_t i = 0; i < particleLocation.size(); i++) {
66 index[i] = particleLocation[i] / cellLength;
67 }
68 const size_t cellIndex = autopas::utils::ThreeDimensionalMapping::threeToOneD(index, cellsPerDimension);
69 particlesPerCell[cellIndex] += 1;
70 // calculate the size of the current cell
71 allVolumes[cellIndex] = 1;
72 for (size_t i = 0; i < cellsPerDimension.size(); ++i) {
73 // the last cell layer has a special size
74 if (index[i] == cellsPerDimension[i] - 1) {
75 allVolumes[cellIndex] *= outerCellSizePerDimension[i];
76 } else {
77 allVolumes[cellIndex] *= cellLength;
78 }
79 }
80 }
81
82 // calculate density for each cell
83 double maxDensity{0.};
84 std::vector<double> densityPerCell(numberOfCells, 0.0);
85 for (size_t i = 0; i < particlesPerCell.size(); i++) {
86 densityPerCell[i] =
87 (allVolumes[i] == 0) ? 0 : (particlesPerCell[i] / allVolumes[i]); // make sure there is no division of zero
88 if (densityPerCell[i] > maxDensity) {
89 maxDensity = densityPerCell[i];
90 }
91 }
92
93 if (maxDensity < 0.0)
94 throw std::runtime_error("maxDensity can never be smaller than 0.0, but is:" + std::to_string(maxDensity));
95
96 // get mean and reserve variable for densityVariance
97 const double densityMean = numberOfParticles / volume;
98 double densityVariance = 0.0;
99
100 // calculate densityVariance
101 for (size_t r = 0; r < densityPerCell.size(); ++r) {
102 double distance = densityPerCell[r] - densityMean;
103 densityVariance += (distance * distance / densityPerCell.size());
104 }
105
106 // finally calculate standard deviation
107 const double homogeneity = sqrt(densityVariance);
108 // normally between 0.0 and 1.5
109 if (homogeneity < 0.0)
110 throw std::runtime_error("homogeneity can never be smaller than 0.0, but is:" + std::to_string(homogeneity));
111 return {homogeneity, maxDensity};
112}
