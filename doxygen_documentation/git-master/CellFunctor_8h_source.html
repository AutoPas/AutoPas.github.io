163void CellFunctor<Particle, ParticleCell, ParticleFunctor, DataLayout, useNewton3, bidirectional>::processCellAoS(
164 ParticleCell &cell) {
165 if (cell.numParticles() > _startSorting) {
166 SortedCellView<Particle, ParticleCell> cellSorted(
167 cell, utils::ArrayMath::normalize(std::array<double, 3>{1.0, 1.0, 1.0}));
168
169 auto outer = cellSorted._particles.begin();
170 for (; outer != cellSorted._particles.end(); ++outer) {
171 Particle &p1 = *outer->second;
172
173 auto inner = outer;
174 ++inner;
175 for (; inner != cellSorted._particles.end(); ++inner) {
176 if (std::abs(outer->first - inner->first) > _sortingCutoff) {
177 break;
178 }
179 Particle &p2 = *inner->second;
180 if constexpr (newton3) {
181 _functor->AoSFunctor(p1, p2, true);
182 } else {
183 _functor->AoSFunctor(p1, p2, false);
184 _functor->AoSFunctor(p2, p1, false);
185 }
186 }
187 }
188 } else {
189 auto outer = getStaticCellIter(cell);
190 for (; outer.isValid(); ++outer) {
191 Particle &p1 = *outer;
192
193 auto inner = outer;
194 ++inner;
195 for (; inner.isValid(); ++inner) {
196 Particle &p2 = *inner;
197
198 if constexpr (newton3) {
199 _functor->AoSFunctor(p1, p2, true);
200 } else {
201 _functor->AoSFunctor(p1, p2, false);
202 _functor->AoSFunctor(p2, p1, false);
203 }
204 }
205 }

220

223 break;

225 Particle &p2 = *inner.second;
226 _functor->AoSFunctor(p1, p2, true);

228 }

232
233 for (; outer.isValid(); ++outer) {
234 Particle &p1 = *outer;
235
236 for (auto inner = innerStart; inner.isValid(); ++inner) {
