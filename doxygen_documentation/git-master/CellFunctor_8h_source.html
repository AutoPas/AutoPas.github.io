191 
192  auto outer = cellSorted._particles.begin();
193  for (; outer != cellSorted._particles.end(); ++outer) {
194  Particle &p1 = *outer->second;
195 
196  auto inner = outer;
197  ++inner;

199  if (std::abs(outer->first - inner->first) > _sortingCutoff) {

203  if constexpr (newton3) {
