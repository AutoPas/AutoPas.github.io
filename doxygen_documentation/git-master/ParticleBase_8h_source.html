167 bool setRDistanceCheck(const std::array<double, 3> &r, double maxDistSquared) {
168 using namespace autopas::utils::ArrayMath::literals;
169 const auto distanceVec = r - _r;
170 const double distanceSquared = utils::ArrayMath::dot(distanceVec, distanceVec);
171 setR(r);
172 const bool distanceIsFine =
173 distanceSquared < maxDistSquared or autopas::utils::Math::isNearAbs(maxDistSquared, 0., 1e-12);
174 if (not distanceIsFine) {
175 AutoPasLog(WARN, "Particle {}: Distance between old and new position is larger than expected: {} > {}", _id,
176 distanceSquared, maxDistSquared);
177 }
