158 void generateSoAFromAoS() override {
159 // Generate a map from pointer to particle index in the SoA. This works, because during loadSoA"()" the particles
160 // are loaded in the same order.
161 std::unordered_map<Particle *, size_t> _aos2soaMap;
162 _aos2soaMap.reserve(_baseLinkedCells->getNumberOfParticles());
163 size_t i = 0;
164 // needs to iterate also over dummies!
165 for (auto iter = _baseLinkedCells->begin(IteratorBehavior::ownedOrHaloOrDummy); iter.isValid(); ++iter, ++i) {
166 _aos2soaMap[&(*iter)] = i;
167 }
168
169 for (int color = 0; color < _numColors; color++) {
170 unsigned int numThreads = _aosNeighborList[color].size();
171 _soaNeighborList[color].resize(numThreads);
172#if defined(AUTOPAS_OPENMP)
173#pragma omp parallel num_threads(numThreads)
174#endif
175#if defined(AUTOPAS_OPENMP)
176#pragma omp for schedule(static)
177#endif
178 for (unsigned int thread = 0; thread < numThreads; thread++) {
179 auto &currentThreadList = _soaNeighborList[color][thread];
180 currentThreadList.clear();
181 for (const auto &pair : _aosNeighborList[color][thread]) {
182 size_t indexFirst = _aos2soaMap[pair.first];
183 std::vector<size_t, AlignedAllocator<size_t>> neighbors;
184 neighbors.reserve(pair.second.size());
185 for (const auto &second : pair.second) {
186 size_t indexSecond = _aos2soaMap[second];
187 neighbors.push_back(indexSecond);
188 }
189 currentThreadList.push_back({indexFirst, std::move(neighbors)});
190 }
191 }
192 }

255 // Check all neighbor lists for the pair, but the one that the pair would be in if it was not moved first.
256 auto &oldThreadNeighborList = _aosNeighborList[_currentColor][currentThreadIndex];
257 if (isPairInList(oldThreadNeighborList, first, second)) {
258 for (int color = 0; color < _numColors; color++) {
259 for (unsigned int thread = 0; thread < _aosNeighborList[color].size(); thread++) {
260 if (not isPairInList(_aosNeighborList[_currentColor][currentThreadIndex], first, second)) {
