167 void generateSoAFromAoS() override {
168 // Generate a map from pointer to particle index in the SoA. This works, because during loadSoA"()" the particles
169 // are loaded in the same order.
170 std::unordered_map<Particle *, size_t> _aos2soaMap;
171 _aos2soaMap.reserve(_baseLinkedCells->getNumParticles());
172 size_t i = 0;
173 // needs to iterate also over dummies!
174 for (auto iter = _baseLinkedCells->begin(IteratorBehavior::ownedOrHaloOrDummy); iter.isValid(); ++iter, ++i) {
175 _aos2soaMap[&(*iter)] = i;
176 }
177
178 for (int color = 0; color < _numColors; color++) {
179 unsigned int numThreads = _aosNeighborList[color].size();
180 _soaNeighborList[color].resize(numThreads);
181#if defined(AUTOPAS_OPENMP)
182#pragma omp parallel num_threads(numThreads)
183#endif
184#if defined(AUTOPAS_OPENMP)
185#pragma omp for schedule(static)
186#endif
187 for (unsigned int thread = 0; thread < numThreads; thread++) {
188 auto &currentThreadList = _soaNeighborList[color][thread];
189 currentThreadList.clear();
190 for (const auto &pair : _aosNeighborList[color][thread]) {
191 size_t indexFirst = _aos2soaMap[pair.first];
192 std::vector<size_t, AlignedAllocator<size_t>> neighbors;
193 neighbors.reserve(pair.second.size());
194 for (const auto &second : pair.second) {
195 size_t indexSecond = _aos2soaMap[second];
196 neighbors.push_back(indexSecond);
197 }
198 currentThreadList.push_back({indexFirst, std::move(neighbors)});
199 }
200 }
201 }

203 _soaListIsValid = true;
