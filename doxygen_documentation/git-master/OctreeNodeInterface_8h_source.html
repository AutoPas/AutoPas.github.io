191 static double getEnclosedVolumeWith(const std::array<double, 3> &aMin, const std::array<double, 3> &aMax,
192 const std::array<double, 3> &bMin, const std::array<double, 3> &bMax) {
193 auto product = 1.0;
194 int count = 0;
195 for (auto d = 0; d < 3; ++d) {
196 auto minOnAxis = std::max(aMin[d], bMin[d]);
197 auto maxOnAxis = std::min(aMax[d], bMax[d]);
198 auto dim = (maxOnAxis - minOnAxis);
199 if (dim > 0) {
200 ++count;
201 } else {
202 break;
203 }
204 product *= dim;
205 }
206 auto result = count == 3 ? product : 0;
207 return result;
208 }

327 OctreeNodeInterface<Particle> *neighbor = GTEQ_FACE_NEIGHBOR(face);

455template <class Particle>
456static octree::Octant SONTYPE(const OctreeNodeInterface<Particle> *node) {
457 octree::Octant result = octree::OOO;
458 if (FATHER(node)) {
459 for (octree::Vertex test : octree::Tables::vertices) {
460 if (FATHER(node)->SON(test) == node) {
461 result = test;
462 break;
463 }
464 }
465 if (result == octree::OOO) {
466 throw std::runtime_error("[OctreeNodeInterface::SONTYPE()] Unable to determine SONTYPE");
467 }
468 }
469 return result;

473OctreeNodeInterface<Particle> *OctreeNodeInterface<Particle>::GTEQ_FACE_NEIGHBOR(const octree::Face I) {
474 // Check precondition
475 if (not isFace(I)) {
476 throw std::runtime_error("[OctreeNodeInterface::GTEQ_FACE_NEIGHBOR()] Received invalid face.");
477 }
478
479 auto null = [](OctreeNodeInterface<Particle> *T) { return T == nullptr; };
480
481 // Find a common ancestor
482 OctreeNodeInterface<Particle> *Q, *P = this;
483 if ((not null(FATHER(P))) and ADJ(I, SONTYPE(P))) {
484 Q = FATHER(P)->GTEQ_FACE_NEIGHBOR(I);
485 } else {
486 Q = FATHER(P);
487 }
488
489 if ((not null(Q)) and GRAY(Q)) {
490 // Follow the reflected path to locate the neighbor
491 return Q->SON(REFLECT(I, SONTYPE(P)));
492 } else {
493 return Q;
494 }

504 auto null = [](OctreeNodeInterface<Particle> *T) { return T == nullptr; };

506 // Find a common ancestor
507 OctreeNodeInterface<Particle> *Q, *P = this;
508 if (null(FATHER(P))) {
509 Q = nullptr;

511 Q = FATHER(P)->GTEQ_EDGE_NEIGHBOR(I);
512 } else if (octree::Face common = COMMON_FACE(I, SONTYPE(P)); common != octree::O) {
513 Q = FATHER(P)->GTEQ_FACE_NEIGHBOR(common);
514 } else {
515 Q = FATHER(P);

519 // Follow opposite path to locate the neighbor
