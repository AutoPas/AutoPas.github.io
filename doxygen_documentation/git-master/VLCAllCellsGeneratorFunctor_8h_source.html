57  void AoSFunctor(Particle &i, Particle &j, bool newton3) override {
58  if (i.isDummy() or j.isDummy()) {
59  return;
60  }
61  auto dist = utils::ArrayMath::sub(i.getR(), j.getR());
62  double distsquare = utils::ArrayMath::dot(dist, dist);
63  if (distsquare < _cutoffskinsquared) {
64  // this is thread safe, only if particle i is accessed by only one
65  // thread at a time. which is ensured, as particle i resides in a

70  auto &[cellIndex, particleIndex] = _particleToCellMap[&i];
71  _neighborLists[cellIndex][particleIndex].second.push_back(&j);

84  double *const __restrict__ zptr = soa.template begin<Particle::AttributeNames::posZ>();

86  // index of cell1 is particleToCellMap of ptr1ptr, same for 2
87  auto cell1 = _particleToCellMap.at(ptrptr[0]).first;
88 
89  auto &currentList = _neighborLists[cell1];

91  // iterating over particle indices and accessing currentList at index i works
92  // because the particles are iterated in the same order they are loaded in
93  // which is the same order they were initialized when building the aosNeighborList
94  size_t numPart = soa.getNumParticles();
95  for (unsigned int i = 0; i < numPart; ++i) {
96  for (unsigned int j = i + 1; j < numPart; ++j) {
97  const double drx = xptr[i] - xptr[j];
98  const double dry = yptr[i] - yptr[j];
99  const double drz = zptr[i] - zptr[j];
100 
101  const double drx2 = drx * drx;
102  const double dry2 = dry * dry;
103  const double drz2 = drz * drz;
104 
105  const double dr2 = drx2 + dry2 + drz2;
106 
107  if (dr2 < _cutoffskinsquared) {
108  currentList[i].second.push_back(ptrptr[j]);
109  if (not newton3) {
