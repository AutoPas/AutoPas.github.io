53 void AoSFunctor(Particle &i, Particle &j, bool newton3) override {
54 if (i.isDummy() or j.isDummy()) {
55 return;
56 }
57 auto dist = utils::ArrayMath::sub(i.getR(), j.getR());
58 double distsquare = utils::ArrayMath::dot(dist, dist);
59 if (distsquare < _cutoffskinsquared) {
60 // this is thread safe, only if particle i is accessed by only one
61 // thread at a time. which is ensured, as particle i resides in a
62 // specific cell and each cell is only accessed by one thread at a time
63 // (ensured by traversals)
64 // also the list is not allowed to be resized!
65
66 auto &[cellIndex, particleIndex] = _particleToCellMap[&i];
67 _neighborLists[cellIndex][particleIndex].second.push_back(&j);
68 }

74 void SoAFunctorSingle(SoAView<SoAArraysType> soa, bool newton3) override {

77 auto **const __restrict__ ptrptr = soa.template begin<Particle::AttributeNames::ptr>();
78 double *const __restrict__ xptr = soa.template begin<Particle::AttributeNames::posX>();
79 double *const __restrict__ yptr = soa.template begin<Particle::AttributeNames::posY>();
80 double *const __restrict__ zptr = soa.template begin<Particle::AttributeNames::posZ>();
81
82 // index of cell1 is particleToCellMap of ptr1ptr, same for 2
83 auto cell1 = _particleToCellMap.at(ptrptr[0]).first;
84
85 auto &currentList = _neighborLists[cell1];
86
87 // iterating over particle indices and accessing currentList at index i works
88 // because the particles are iterated in the same order they are loaded in
89 // which is the same order they were initialized when building the aosNeighborList
90 size_t numPart = soa.getNumParticles();
91 for (unsigned int i = 0; i < numPart; ++i) {
92 for (unsigned int j = i + 1; j < numPart; ++j) {
93 const double drx = xptr[i] - xptr[j];
94 const double dry = yptr[i] - yptr[j];
95 const double drz = zptr[i] - zptr[j];
96
97 const double drx2 = drx * drx;
98 const double dry2 = dry * dry;
99 const double drz2 = drz * drz;
100
101 const double dr2 = drx2 + dry2 + drz2;
102
103 if (dr2 < _cutoffskinsquared) {
104 currentList[i].second.push_back(ptrptr[j]);
105 if (not newton3) {
106 // we need this here, as SoAFunctorSingle will only be called once for both newton3=true and false.
107 currentList[j].second.push_back(ptrptr[i]);
108 }
109 }
110 }
