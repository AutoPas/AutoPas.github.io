658template <class T, std::size_t SIZE>
659double boxDistanceSquared(const std::array<T, SIZE> &aMin, const std::array<T, SIZE> &aMax,
660 const std::array<T, SIZE> &bMin, const std::array<T, SIZE> &bMax) {
661 using namespace autopas::utils::ArrayMath::literals;
662 using autopas::utils::ArrayMath::dot;
663 using autopas::utils::ArrayMath::max;
664
665 const auto aToB = max(std::array<T, SIZE>{}, aMin - bMax);
666 const auto bToA = max(std::array<T, SIZE>{}, bMin - aMax);
667
668 return dot(aToB, aToB) + dot(bToA, bToA);
669}
