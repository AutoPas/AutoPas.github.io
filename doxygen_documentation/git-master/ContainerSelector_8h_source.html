97template <class Particle>
98std::unique_ptr<autopas::ParticleContainerInterface<Particle>> ContainerSelector<Particle>::generateContainer(
99 ContainerOption containerChoice, ContainerSelectorInfo containerInfo) {
100 std::unique_ptr<autopas::ParticleContainerInterface<Particle>> container;
101 switch (containerChoice) {
102 case ContainerOption::directSum: {
103 container = std::make_unique<DirectSum<Particle>>(_boxMin, _boxMax, _cutoff, containerInfo.verletSkinPerTimestep,
104 containerInfo.verletRebuildFrequency);
105 break;
106 }
107
108 case ContainerOption::linkedCells: {
109 container = std::make_unique<LinkedCells<Particle>>(
110 _boxMin, _boxMax, _cutoff, containerInfo.verletSkinPerTimestep, containerInfo.verletRebuildFrequency,
111 containerInfo.cellSizeFactor, containerInfo.loadEstimator);
112 break;
113 }
114 case ContainerOption::linkedCellsReferences: {
115 container = std::make_unique<LinkedCellsReferences<Particle>>(
116 _boxMin, _boxMax, _cutoff, containerInfo.verletSkinPerTimestep, containerInfo.verletRebuildFrequency,
117 containerInfo.cellSizeFactor);
118 break;
119 }
120 case ContainerOption::verletLists: {
121 container = std::make_unique<VerletLists<Particle>>(
122 _boxMin, _boxMax, _cutoff, containerInfo.verletSkinPerTimestep, containerInfo.verletRebuildFrequency,
123 VerletLists<Particle>::BuildVerletListType::VerletSoA, containerInfo.cellSizeFactor);
124 break;
125 }
126 case ContainerOption::verletListsCells: {
127 container = std::make_unique<VerletListsCells<Particle, VLCAllCellsNeighborList<Particle>>>(
128 _boxMin, _boxMax, _cutoff, containerInfo.verletSkinPerTimestep, containerInfo.verletRebuildFrequency,
129 containerInfo.cellSizeFactor, containerInfo.loadEstimator,
130 VerletListsCellsHelpers<Particle>::VLCBuildType::Value::soaBuild);
131 break;
132 }
133 case ContainerOption::verletClusterLists: {
134 container = std::make_unique<VerletClusterLists<Particle>>(
135 _boxMin, _boxMax, _cutoff, containerInfo.verletSkinPerTimestep, containerInfo.verletRebuildFrequency,
136 containerInfo.verletClusterSize, containerInfo.loadEstimator);
137 break;
138 }
139 case ContainerOption::varVerletListsAsBuild: {
140 container = std::make_unique<VarVerletLists<Particle, VerletNeighborListAsBuild<Particle>>>(
141 _boxMin, _boxMax, _cutoff, containerInfo.verletSkinPerTimestep, containerInfo.verletRebuildFrequency,
142 containerInfo.cellSizeFactor);
143 break;
144 }
145
146 case ContainerOption::pairwiseVerletLists: {
147 container = std::make_unique<VerletListsCells<Particle, VLCCellPairNeighborList<Particle>>>(
148 _boxMin, _boxMax, _cutoff, containerInfo.verletSkinPerTimestep, containerInfo.verletRebuildFrequency,
149 containerInfo.cellSizeFactor, containerInfo.loadEstimator,
150 VerletListsCellsHelpers<Particle>::VLCBuildType::Value::soaBuild);
151 break;
152 }
153 case ContainerOption::octree: {
154 container =
155 std::make_unique<Octree<Particle>>(_boxMin, _boxMax, _cutoff, containerInfo.verletSkinPerTimestep,
156 containerInfo.verletRebuildFrequency, containerInfo.cellSizeFactor);
157 break;
158 }
159 default: {
160 utils::ExceptionHandler::exception("ContainerSelector: Container type {} is not a known type!",
161 containerChoice.to_string());
162 }
163 }
164
165 // copy particles so they do not get lost when container is switched
166 if (_currentContainer != nullptr) {
167 for (auto particleIter = _currentContainer->begin(IteratorBehavior::ownedOrHalo); particleIter.isValid();
168 ++particleIter) {
169 // add particle as inner if it is owned
170 if (particleIter->isOwned()) {
171 container->addParticle(*particleIter);
172 } else {
173 container->addHaloParticle(*particleIter);
174 }
175 }
176 }
177
178 return container;
179}

200template <class Particle>
201void ContainerSelector<Particle>::selectContainer(ContainerOption containerOption,
202 ContainerSelectorInfo containerInfo) {
203 // Only do something if we have no container, a new type is required, or the info changed
204 if (_currentContainer == nullptr or _currentContainer->getContainerType() != containerOption or
205 _currentInfo != containerInfo) {
206 _currentContainer = std::move(generateContainer(containerOption, containerInfo));
207 _currentInfo = containerInfo;
208 }
