161 void verifyCurrentConfigTime() const {
162 for (const auto &order : _lastApplicableConfigurationOrders) {
163 bool shouldBeBetter;
164 if (order.smaller.matches(*_currentConfig)) {
165 shouldBeBetter = true;
166 } else if (order.greater.matches(*_currentConfig)) {
167 shouldBeBetter = false;
168 } else {
169 continue;
170 }
171
172 const auto currentConfigTime = _traversalTimes.at(*_currentConfig);
173 const auto &comparePattern = shouldBeBetter ? order.greater : order.smaller;
174 for (const auto &[otherConfig, time] : _traversalTimes) {
175 bool error = false;
176 if (comparePattern.matches(otherConfig) and order.haveEqualSameProperties(*_currentConfig, otherConfig)) {
177 if ((shouldBeBetter and time > currentConfigTime) or (not shouldBeBetter and time < currentConfigTime)) {
178 error = true;
179 }
180 }
181 if (error) {
182 if (shouldBeBetter) {
183 _tuningErrorPrinter(order, *_currentConfig, currentConfigTime, otherConfig, time, _currentLiveInfo);
184 } else {
185 _tuningErrorPrinter(order, otherConfig, time, *_currentConfig, currentConfigTime, _currentLiveInfo);
186 }
187 }
188 }

196 std::vector<rule_syntax::ConfigurationOrder> applyRules() {
197 AutoPasLog(DEBUG, _currentLiveInfo.toString());
198
199 std::vector<RuleVM::MemoryCell> initialStack;
200 std::vector<std::pair<std::string, rule_syntax::Define>> defines{};
201 for (const auto &[name, value] : _currentLiveInfo.get()) {
202 initialStack.emplace_back(value);
203 defines.push_back({name, {name, std::make_shared<rule_syntax::Literal>(value)}});
204 }
205
206 std::ifstream ifs{_ruleFileName};
207 const std::string content((std::istreambuf_iterator<char>(ifs)), (std::istreambuf_iterator<char>()));
208
209 rule_syntax::RuleBasedProgramParser parser{defines};
210 auto [programTree, context] = parser.parse(content);
211
212 const auto generatedProgram = programTree.generateCode(context);
213
214 RuleVM vm{};
215 const auto removePatterns = vm.execute(generatedProgram, initialStack);
216
217 AutoPasLog(DEBUG, "Remove patterns (Count {}):", removePatterns.size());
218 std::vector<ConfigurationPattern> toRemovePatterns{};
219 std::vector<rule_syntax::ConfigurationOrder> applicableConfigurationOrders{};
220 for (const auto &patternIdx : removePatterns) {
221 const auto pattern = context.smallerConfigurationPatternByIndex(patternIdx);
222 toRemovePatterns.push_back(pattern);
223 AutoPasLog(DEBUG, "Remove {}", pattern.toString());
224
225 applicableConfigurationOrders.push_back(context.getConfigurationOrders().at(patternIdx));
226 }
227
228 auto newSearchSpace = _originalSearchSpace;
229 _removedConfigurations.clear();
230 auto &removedConfigsLocal = _removedConfigurations;
231 newSearchSpace.remove_if([&toRemovePatterns, &removedConfigsLocal](const Configuration &configuration) {
232 const bool remove = std::any_of(toRemovePatterns.begin(), toRemovePatterns.end(),
233 [&configuration](const auto &pattern) { return pattern.matches(configuration); });
234 if (remove) {
235 removedConfigsLocal.insert(configuration);
236 }
237 return remove;
238 });
239 AutoPasLog(DEBUG, "Rules remove {} out of {} configurations", _originalSearchSpace.size() - newSearchSpace.size(),
240 _originalSearchSpace.size());
241 if (not _verifyModeEnabled) {
242 this->_searchSpace = {newSearchSpace.begin(), newSearchSpace.end()};
243 }
244
245 return applicableConfigurationOrders;
246 }
