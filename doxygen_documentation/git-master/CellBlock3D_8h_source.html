210 std::vector<ParticleCell *> getNearbyHaloCells(const std::array<double, 3> &position, double allowedDistance) const {
211 using namespace autopas::utils::ArrayMath::literals;
212
213 auto index3d = get3DIndexOfPosition(position);
214
215 std::vector<ParticleCell *> closeHaloCells;
216
217 auto isHaloCell = [&](const auto &index) {
218 for (size_t i = 0; i < 3; ++i) {
219 if (index[i] < _cellsPerInteractionLength or
220 index[i] >= _cellsPerDimensionWithHalo[i] - _cellsPerInteractionLength) {
221 return true;
222 }
223 }
224 return false;
225 };
226
227 // always add the cell the particle is currently in first, for that we test if it is a halo cell.
228 if (isHaloCell(index3d)) {
229 closeHaloCells.push_back(&getCell(index3d));
230 }
231
232 const auto lowIndex3D = get3DIndexOfPosition(position - allowedDistance);
233 const auto highIndex3D = get3DIndexOfPosition(position + allowedDistance);
234 // these indices are (already) at least 0 and at most _cellsPerDimensionWithHalo[i]-1
235
236 auto currentIndex = lowIndex3D;
237
238 for (currentIndex[0] = lowIndex3D[0]; currentIndex[0] <= highIndex3D[0]; ++currentIndex[0]) {
239 for (currentIndex[1] = lowIndex3D[1]; currentIndex[1] <= highIndex3D[1]; ++currentIndex[1]) {
240 for (currentIndex[2] = lowIndex3D[2]; currentIndex[2] <= highIndex3D[2]; ++currentIndex[2]) {
241 // we have already added the cell which normally would belong to the particle, so skip here:
242 if (currentIndex == index3d) {
243 continue;
244 }
245 // we need to return the cell is it is a halo cell.
246 if (isHaloCell(currentIndex)) {
247 closeHaloCells.push_back(&getCell(currentIndex));
248 }
249 }
250 }
251 }
252
253 return closeHaloCells;
254 }

332template <class ParticleCell>
333inline std::array<typename CellBlock3D<ParticleCell>::index_t, 3> CellBlock3D<ParticleCell>::get3DIndexOfPosition(
334 const std::array<double, 3> &pos) const {
335 std::array<typename CellBlock3D<ParticleCell>::index_t, 3> cellIndex{};
336
337 for (size_t dim = 0; dim < 3; dim++) {
338 // 0 <= cellIndex < cellsPerDimWithHalo
339 const long int value = (static_cast<long int>(std::floor((pos[dim] - _boxMin[dim]) * _cellLengthReciprocal[dim]))) +
340 _cellsPerInteractionLength;
341 const index_t nonNegativeValue = static_cast<index_t>(std::max(value, 0l));
342 const index_t nonLargerValue = std::min(nonNegativeValue, _cellsPerDimensionWithHalo[dim] - 1);
343 cellIndex[dim] = nonLargerValue;
344
345 // sanity checks and precautions for numerical instabilities around the edge of the container box,
346 // which would lead to doubling of particles
347 if (pos[dim] >= _boxMax[dim]) {
348 // pos is located outside the box. Make sure that cellIndex is at least in the positive halo range.
349 cellIndex[dim] = std::max(cellIndex[dim], _cellsPerDimensionWithHalo[dim] - _cellsPerInteractionLength);
350 } else if (pos[dim] < _boxMin[dim] and cellIndex[dim] == _cellsPerInteractionLength) {
351 // pos is located outside but the index resolved to be the last IN the box. Fix by decrementing the index.
352 --cellIndex[dim];
353 } else if (pos[dim] < _boxMax[dim] and
354 cellIndex[dim] == _cellsPerDimensionWithHalo[dim] - _cellsPerInteractionLength) {
355 // pos is located inside the box, but cellIndex resolves to be the first halo cell. Fix by decrementing the index.
356 --cellIndex[dim];
357 }
358 }
359
360 return cellIndex;
361}

371template <class ParticleCell>
372inline void CellBlock3D<ParticleCell>::rebuild(std::vector<ParticleCell> &vec, const std::array<double, 3> &bMin,
373 const std::array<double, 3> &bMax, double interactionLength,
374 double cellSizeFactor) {
375 using namespace autopas::utils::ArrayMath::literals;
376
377 _cells = &vec;
378 _boxMin = bMin;
379 _boxMax = bMax;
380 _interactionLength = interactionLength;
381
382 if (cellSizeFactor >= 1.0) {
383 _cellsPerInteractionLength = 1;
384 } else {
385 _cellsPerInteractionLength = ceil(1.0 / cellSizeFactor);
386 }
387 // compute cell length and number of cells
388 index_t numCells = 1;
389 for (int d = 0; d < 3; ++d) {
390 const double diff = _boxMax[d] - _boxMin[d];
391 auto cellsPerDim = static_cast<index_t>(std::floor(diff / (_interactionLength * cellSizeFactor)));
392 // at least one central cell
393 cellsPerDim = std::max(cellsPerDim, 1ul);
394
395 _cellsPerDimensionWithHalo[d] = cellsPerDim + 2 * _cellsPerInteractionLength;
396
397 _cellLength[d] = diff / cellsPerDim;
398
399 _cellLengthReciprocal[d] = cellsPerDim / diff; // compute with least rounding possible
400
401 _haloBoxMin[d] = _boxMin[d] - _cellsPerInteractionLength * _cellLength[d];
402 _haloBoxMax[d] = _boxMax[d] + _cellsPerInteractionLength * _cellLength[d];
403
404 numCells *= _cellsPerDimensionWithHalo[d];
405 }
406 AutoPasLog(TRACE, "Box Length incl Halo : {}", autopas::utils::ArrayUtils::to_string(_haloBoxMax - _haloBoxMin));
407 AutoPasLog(TRACE, "Cells/Dim incl Halo : {}", autopas::utils::ArrayUtils::to_string(_cellsPerDimensionWithHalo));
408 AutoPasLog(TRACE, "Cell Length : {}", autopas::utils::ArrayUtils::to_string(_cellLength));
409 AutoPasLog(TRACE, "Interaction Length : {}", interactionLength);
410 AutoPasLog(TRACE, "Cell Size Factor : {}", cellSizeFactor);
411
412 _firstOwnedCellIndex = _cellsPerDimensionWithHalo[0] * _cellsPerDimensionWithHalo[1] * _cellsPerInteractionLength +
413 _cellsPerDimensionWithHalo[0] * _cellsPerInteractionLength + _cellsPerInteractionLength;
414 _lastOwnedCellIndex = numCells - 1 - _firstOwnedCellIndex;
415
416 // initialize cells
417 _cells->resize(numCells);
418
419 for (auto &cell : *_cells) {
420 cell.setCellLength(_cellLength);
421 }
422}

445template <class ParticleCell>
446inline std::pair<std::array<double, 3>, std::array<double, 3>> CellBlock3D<ParticleCell>::getCellBoundingBox(
447 const std::array<index_t, 3> &index3d) const {
448 std::array<double, 3> boxmin{}, boxmax{};
449 for (int d = 0; d < 3; d++) {
450 // defaults
451 boxmin[d] = index3d[d] * this->_cellLength[d] + _haloBoxMin[d];
452 boxmax[d] = (index3d[d] + 1) * this->_cellLength[d] + _haloBoxMin[d];
453
454 // stupid rounding errors. Make sure that the lower corner is set correctly!
455 if (index3d[d] == 0) {
456 boxmin[d] = _haloBoxMin[d];
457 boxmax[d] = this->_cellLength[d];
458 } else if (index3d[d] == _cellsPerInteractionLength) {
459 boxmin[d] = _boxMin[d];
460 }
461 // no else, as this might ALSO be 1
462 if (index3d[d] == this->_cellsPerDimensionWithHalo[d] - _cellsPerInteractionLength - 1) {
463 boxmax[d] = _boxMax[d];
464 } else if (index3d[d] == this->_cellsPerDimensionWithHalo[d] - 1) {
465 boxmin[d] = _haloBoxMax[d] - this->_cellLength[d];
466 boxmax[d] = _haloBoxMax[d];
467 }
468 }
469 return {boxmin, boxmax};
470}

506template <class ParticleCell>
507void CellBlock3D<ParticleCell>::clearHaloCells() {
508 std::vector<index_t> haloSlices(2 * _cellsPerInteractionLength);
509 auto mid = haloSlices.begin() + _cellsPerInteractionLength;
510 std::iota(haloSlices.begin(), mid, 0);
511
512 // x: min and max of x
513 std::iota(mid, haloSlices.end(), _cellsPerDimensionWithHalo[0] - _cellsPerInteractionLength);
514 for (index_t i : haloSlices) {
515 for (index_t j = 0; j < _cellsPerDimensionWithHalo[1]; j++) {
516 for (index_t k = 0; k < _cellsPerDimensionWithHalo[2]; k++) {
517 index_t index = index1D({i, j, k});
518 (*_cells)[index].clear();
519 }
520 }
521 }
522 // y: min and max of y
523 std::iota(mid, haloSlices.end(), _cellsPerDimensionWithHalo[1] - _cellsPerInteractionLength);
524 for (index_t i = 1; i < _cellsPerDimensionWithHalo[0] - 1; i++) { // 0 and cells-1 already done in previous loop
525 for (index_t j : haloSlices) {
526 for (index_t k = 0; k < _cellsPerDimensionWithHalo[2]; k++) {
527 index_t index = index1D({i, j, k});
528 (*_cells)[index].clear();
529 }
530 }
531 }
532 // z: min and max of z
533 std::iota(mid, haloSlices.end(), _cellsPerDimensionWithHalo[2] - _cellsPerInteractionLength);
534 for (index_t i = 1; i < _cellsPerDimensionWithHalo[0] - 1; i++) { // 0 and cells-1 already done in previous loop
535 for (index_t j = 1; j < _cellsPerDimensionWithHalo[1] - 1; j++) { // 0 and cells-1 already done in previous loop
536 for (index_t k : haloSlices) {
537 index_t index = index1D({i, j, k});
538 (*_cells)[index].clear();
539 }
540 }
541 }
