197 std::vector<ParticleCell *> getNearbyHaloCells(const std::array<double, 3> &position, double allowedDistance) const {
198 auto index3d = get3DIndexOfPosition(position);
199
200 std::vector<ParticleCell *> closeHaloCells;
201
202 auto isHaloCell = [&](const auto &index) {
203 for (size_t i = 0; i < 3; ++i) {
204 if (index[i] < _cellsPerInteractionLength or
205 index[i] >= _cellsPerDimensionWithHalo[i] - _cellsPerInteractionLength) {
206 return true;
207 }
208 }
209 return false;
210 };
211
212 // always add the cell the particle is currently in first, for that we test if it is a halo cell.
213 if (isHaloCell(index3d)) {
214 closeHaloCells.push_back(&getCell(index3d));
215 }
216
217 auto lowIndex3D = get3DIndexOfPosition(utils::ArrayMath::subScalar(position, allowedDistance));
218 auto highIndex3D = get3DIndexOfPosition(utils::ArrayMath::addScalar(position, allowedDistance));
219 // these indices are (already) at least 0 and at most _cellsPerDimensionWithHalo[i]-1
220
221 auto currentIndex = lowIndex3D;
222
223 for (currentIndex[0] = lowIndex3D[0]; currentIndex[0] <= highIndex3D[0]; ++currentIndex[0]) {
224 for (currentIndex[1] = lowIndex3D[1]; currentIndex[1] <= highIndex3D[1]; ++currentIndex[1]) {
225 for (currentIndex[2] = lowIndex3D[2]; currentIndex[2] <= highIndex3D[2]; ++currentIndex[2]) {
226 // we have already added the cell which normally would belong to the particle, so skip here:
227 if (currentIndex == index3d) {
228 continue;
229 }
230 // we need to return the cell is it is a halo cell.
231 if (isHaloCell(currentIndex)) {
232 closeHaloCells.push_back(&getCell(currentIndex));
233 }
234 }
235 }
236 }
237
238 return closeHaloCells;

295inline std::array<typename CellBlock3D<ParticleCell>::index_t, 3> CellBlock3D<ParticleCell>::get3DIndexOfPosition(
296 const std::array<double, 3> &pos) const {
297 std::array<typename CellBlock3D<ParticleCell>::index_t, 3> cellIndex{};
298
299 for (size_t dim = 0; dim < 3; dim++) {
300 const long int value = (static_cast<long int>(std::floor((pos[dim] - _boxMin[dim]) * _cellLengthReciprocal[dim]))) +
301 _cellsPerInteractionLength;
302 const index_t nonNegativeValue = static_cast<index_t>(std::max(value, 0l));
303 const index_t nonLargerValue = std::min(nonNegativeValue, _cellsPerDimensionWithHalo[dim] - 1);
304 cellIndex[dim] = nonLargerValue;
305
306 // this is a sanity check to prevent doubling of particles
307 if (pos[dim] >= _boxMax[dim]) {
308 // pos[dim] is located outside of the box. Make sure that cellIndex also references cell outside of the box.
309 cellIndex[dim] = std::max(cellIndex[dim], _cellsPerDimensionWithHalo[dim] - _cellsPerInteractionLength);
310 } else if (pos[dim] < _boxMin[dim] && cellIndex[dim] == _cellsPerInteractionLength) {
311 // pos[dim] is located below the box (outside), but cellIndex references cell inside of the box. Correct cellIndex
312 // by subtracting 1.
313 --cellIndex[dim];
314 } else if (pos[dim] < _boxMax[dim] &&
315 cellIndex[dim] == _cellsPerDimensionWithHalo[dim] - _cellsPerInteractionLength) {
316 // pos[dim] is located inside of the box, but cellIndex references cell outside of the box. Correct cellIndex to
317 // last cell inside of the box.
318 cellIndex[dim] = _cellsPerDimensionWithHalo[dim] - _cellsPerInteractionLength - 1;
319 }
320 }
321
322 return cellIndex;
323 // in very rare cases rounding is stupid, thus we need a check...
325}

327template <class ParticleCell>
328inline void CellBlock3D<ParticleCell>::rebuild(std::vector<ParticleCell> &vec, const std::array<double, 3> &bMin,
329 const std::array<double, 3> &bMax, double interactionLength,
330 double cellSizeFactor) {
331 _cells = &vec;
332 _boxMin = bMin;
333 _boxMax = bMax;
334 _interactionLength = interactionLength;
335
336 if (cellSizeFactor >= 1.0) {
337 _cellsPerInteractionLength = 1;
338 } else {
339 _cellsPerInteractionLength = ceil(1.0 / cellSizeFactor);
340 }
341 // compute cell length
342 _numCells = 1;
343 for (int d = 0; d < 3; ++d) {
344 const double diff = _boxMax[d] - _boxMin[d];
345 auto cellsPerDim = static_cast<index_t>(std::floor(diff / (_interactionLength * cellSizeFactor)));
346 // at least one central cell
347 cellsPerDim = std::max(cellsPerDim, 1ul);
348
349 _cellsPerDimensionWithHalo[d] = cellsPerDim + 2 * _cellsPerInteractionLength;
350
351 _cellLength[d] = diff / cellsPerDim;
352
353 _cellLengthReciprocal[d] = cellsPerDim / diff; // compute with least rounding possible
354
355 _haloBoxMin[d] = _boxMin[d] - _cellsPerInteractionLength * _cellLength[d];
356 _haloBoxMax[d] = _boxMax[d] + _cellsPerInteractionLength * _cellLength[d];
357
358 _numCells *= _cellsPerDimensionWithHalo[d];
359 }
360 AutoPasLog(trace, "Box Length incl Halo : {}",
361 autopas::utils::ArrayUtils::to_string(autopas::utils::ArrayMath::sub(_haloBoxMax, _haloBoxMin)));
362 AutoPasLog(trace, "Cells/Dim incl Halo : {}", autopas::utils::ArrayUtils::to_string(_cellsPerDimensionWithHalo));
363 AutoPasLog(trace, "Cell Length : {}", autopas::utils::ArrayUtils::to_string(_cellLength));
364 AutoPasLog(trace, "Interaction Length : {}", interactionLength);
365 AutoPasLog(trace, "Cell Size Factor : {}", cellSizeFactor);
366
367 _cells->resize(_numCells);
368 for (auto &cell : *_cells) {
369 cell.setCellLength(_cellLength);
370 }
371}

379template <class ParticleCell>
380inline std::pair<std::array<double, 3>, std::array<double, 3>> CellBlock3D<ParticleCell>::getCellBoundingBox(
381 const std::array<index_t, 3> &index3d) const {
382 std::array<double, 3> boxmin{}, boxmax{};
383 for (int d = 0; d < 3; d++) {
384 // defaults
385 boxmin[d] = index3d[d] * this->_cellLength[d] + _haloBoxMin[d];
386 boxmax[d] = (index3d[d] + 1) * this->_cellLength[d] + _haloBoxMin[d];
387
388 // stupid rounding errors (makes sure that the lower corner is set
389 // correctly!
390 if (index3d[d] == 0) {
391 boxmin[d] = _haloBoxMin[d];
392 boxmax[d] = this->_cellLength[d];
393 } else if (index3d[d] == _cellsPerInteractionLength) {
394 boxmin[d] = _boxMin[d];
395 }
396 // no else!, as this might ALSO be 1
397 if (index3d[d] == this->_cellsPerDimensionWithHalo[d] - _cellsPerInteractionLength - 1) {
398 boxmax[d] = _boxMax[d];
399 } else if (index3d[d] == this->_cellsPerDimensionWithHalo[d] - 1) {
400 boxmin[d] = _haloBoxMax[d] - this->_cellLength[d];
401 boxmax[d] = _haloBoxMax[d];
402 }
403 }
404 return {boxmin, boxmax};
405}

441template <class ParticleCell>
442void CellBlock3D<ParticleCell>::clearHaloCells() {
443 std::vector<index_t> haloSlices(2 * _cellsPerInteractionLength);
444 auto mid = haloSlices.begin() + _cellsPerInteractionLength;
445 std::iota(haloSlices.begin(), mid, 0);
446
447 // x: min and max of x
448 std::iota(mid, haloSlices.end(), _cellsPerDimensionWithHalo[0] - _cellsPerInteractionLength);
449 for (index_t i : haloSlices) {
450 for (index_t j = 0; j < _cellsPerDimensionWithHalo[1]; j++) {
451 for (index_t k = 0; k < _cellsPerDimensionWithHalo[2]; k++) {
452 index_t index = index1D({i, j, k});
453 (*_cells)[index].clear();
454 }
455 }
456 }
457 // y: min and max of y
458 std::iota(mid, haloSlices.end(), _cellsPerDimensionWithHalo[1] - _cellsPerInteractionLength);
459 for (index_t i = 1; i < _cellsPerDimensionWithHalo[0] - 1; i++) { // 0 and cells-1 already done in previous loop
460 for (index_t j : haloSlices) {
461 for (index_t k = 0; k < _cellsPerDimensionWithHalo[2]; k++) {
462 index_t index = index1D({i, j, k});
463 (*_cells)[index].clear();
464 }
465 }
466 }
467 // z: min and max of z
468 std::iota(mid, haloSlices.end(), _cellsPerDimensionWithHalo[2] - _cellsPerInteractionLength);
469 for (index_t i = 1; i < _cellsPerDimensionWithHalo[0] - 1; i++) { // 0 and cells-1 already done in previous loop
470 for (index_t j = 1; j < _cellsPerDimensionWithHalo[1] - 1; j++) { // 0 and cells-1 already done in previous loop
471 for (index_t k : haloSlices) {
472 index_t index = index1D({i, j, k});
473 (*_cells)[index].clear();
474 }
475 }
476 }
