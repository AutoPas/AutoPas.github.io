61 const double w_ij = (dvdr < 0) ? dvdr / utils::ArrayMath::L2Norm(dr) : 0;
62 // const PS::F64 w_ij = (dv * dr < 0) ? dv * dr / sqrt(dr * dr) : 0;
63
64 const double v_sig = i.getSoundSpeed() + j.getSoundSpeed() - 3.0 * w_ij;

67 i.checkAndSetVSigMax(v_sig);
68 if (newton3) {
69 j.checkAndSetVSigMax(v_sig); // Newton 3
70 // v_sig_max = std::max(v_sig_max, v_sig);
71 }
72 const double AV = -0.5 * v_sig * w_ij / (0.5 * (i.getDensity() + j.getDensity()));
73 // const PS::F64 AV = - 0.5 * v_sig * w_ij / (0.5 * (ep_i[i].dens +
74 // ep_j[j].dens));

77 utils::ArrayMath::mulScalar(utils::ArrayMath::add(SPHKernels::gradW(dr, i.getSmoothingLength()),

88 // gradW_ij;

91 // Newton3, gradW_ij = -gradW_ji
92 }
93 double scale2i = j.getMass() * (i.getPressure() / (i.getDensity() * i.getDensity()) + 0.5 * AV);
94 i.addEngDot(utils::ArrayMath::dot(gradW_ij, dv) * scale2i);
95 // hydro[i].eng_dot += ep_j[j].mass * (ep_i[i].pres / (ep_i[i].dens *
96 // ep_i[i].dens) + 0.5 * AV) * dv * gradW_ij;
