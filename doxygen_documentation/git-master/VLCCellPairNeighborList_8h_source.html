142 void generateSoAFromAoS(LinkedCells<Particle> &linkedCells) override {
143 _soaNeighborList.clear();
144
145 // particle pointer to global index of particle
146 std::unordered_map<Particle *, size_t> particleToIndex;
147 particleToIndex.reserve(linkedCells.getNumParticles());
148 size_t i = 0;
149 for (auto iter = linkedCells.begin(IteratorBehavior::ownedOrHaloOrDummy); iter.isValid(); ++iter, ++i) {
150 particleToIndex[&(*iter)] = i;
151 }
152
153 _soaNeighborList.resize(linkedCells.getCells().size());
154
155 // iterate over cells
156 for (size_t firstCellIndex = 0; firstCellIndex < _aosNeighborList.size(); ++firstCellIndex) {
157 const auto &aosCurrentCell = _aosNeighborList[firstCellIndex];
158 auto &soaCurrentCell = _soaNeighborList[firstCellIndex];
159 soaCurrentCell.resize(aosCurrentCell.size());
160
161 // iterate over each cell's neighboring cells
162 for (size_t secondCellIndex = 0; secondCellIndex < aosCurrentCell.size(); ++secondCellIndex) {
163 const auto &aosCellPair = aosCurrentCell[secondCellIndex];
164 auto &soaCellPair = soaCurrentCell[secondCellIndex];
165 soaCellPair.reserve(aosCellPair.capacity());
166
167 // iterate over pairs of particle and neighbor list
168 for (const auto &aosParticleAndParticleList : aosCellPair) {
169 Particle *currentParticle = aosParticleAndParticleList.first;
170 // global index of current particle
171 size_t currentParticleGlobalIndex = particleToIndex.at(currentParticle);
172
173 // create SoA neighbor list for current particle
174 std::vector<size_t, autopas::AlignedAllocator<size_t>> currentSoANeighborList{};
175
176 // fill the SoA neighbor list with the indices of the particles from the corresponding AoS neighbor list
177 for (const auto &neighborOfCurrentParticle : aosParticleAndParticleList.second) {
178 currentSoANeighborList.emplace_back(particleToIndex.at(neighborOfCurrentParticle));
179 }
180
181 // add newly constructed pair of particle index and SoA particle neighbor list to cell pair
182 soaCellPair.emplace_back(std::make_pair(currentParticleGlobalIndex, currentSoANeighborList));
183 }
184 }
