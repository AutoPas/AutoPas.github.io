289bool BayesianClusterSearch::tune(bool currentInvalid) {
290 if (currentInvalid) {
291 _invalidConfigs.insert(_currentConfig);
292 }
293
294 // in the first tuning phase do a full search
295 if (_firstTuningPhase) {
296 if (_fullSearch.tune(currentInvalid)) {
297 // continue with full-search
298 _currentConfig = _fullSearch.getCurrentConfiguration();
299 return true;
300 } else {
301 // continue with GaussianCluster
302 _firstTuningPhase = false;
303 _currentNumEvidence = 0;
304 }
305 }
306
307 if (searchSpaceIsEmpty()) {
308 // no valid configuration
309 _currentConfig = FeatureVector();
310 return false;
311 }
312
313 // no more tunings steps
314 if (_currentNumEvidence >= _maxEvidence) {
315 // select best config of current tuning phase
316 _currentConfig = _currentOptimalConfig;
317
318 return false;
319 }
320
321 // try to sample a valid vector which is expected to yield a good acquisition
322 for (size_t i = 0; i < maxAttempts; ++i) {
323 if (_currentAcquisitions.empty()) {
324 sampleAcquisitions(_predNumLHSamples, _predAcqFunction);
325 }
326
327 // test best vectors until empty
328 while (not _currentAcquisitions.empty()) {
329 auto best = _encoder.convertFromCluster(_currentAcquisitions.back());
330
331 if (_invalidConfigs.find(best) == _invalidConfigs.end()) {
332 // valid config found!
333 _currentConfig = best;
334 return true;
335 } else {
336 // invalid: dispose
337 _currentAcquisitions.pop_back();
338 }
339 }
340
341 // No valid configuration. This should rarely happen.
342 AutoPasLog(debug, "Tuning could not generate a valid configuration.");
343 }
344
345 utils::ExceptionHandler::exception("BayesianClusterSearch: Failed to sample an valid FeatureVector");
346 _currentConfig = FeatureVector();
347 return false;
348}
