288 bool BayesianClusterSearch::tune(bool currentInvalid) {
289  if (currentInvalid) {
290  _invalidConfigs.insert(_currentConfig);
291  }
292 
293  // in the first tuning phase do a full search
294  if (_firstTuningPhase) {
295  if (_fullSearch.tune(currentInvalid)) {
296  // continue with full-search
297  _currentConfig = _fullSearch.getCurrentConfiguration();
298  return true;
299  } else {
300  // continue with GaussianCluster
301  _firstTuningPhase = false;
302  _currentNumEvidence = 0;
303  }
304  }
305 
306  if (searchSpaceIsEmpty()) {
307  // no valid configuration
308  _currentConfig = FeatureVector();
309  return false;
310  }
311 
312  // no more tunings steps
313  if (_currentNumEvidence >= _maxEvidence) {
314  // select best config of current tuning phase
315  _currentConfig = _currentOptimalConfig;
316  AutoPasLog(debug, "Selected Configuration {}", _currentConfig.toString());
317 
318  return false;
319  }
320 
321  // try to sample a valid vector which is expected to yield a good acquisition
322  for (size_t i = 0; i < maxAttempts; ++i) {
323  if (_currentAcquisitions.empty()) {
324  sampleAcquisitions(_predNumLHSamples, _predAcqFunction);
325  }
326 
327  // test best vectors until empty
328  while (not _currentAcquisitions.empty()) {
329  auto best = _encoder.convertFromCluster(_currentAcquisitions.back());
330 
331  if (_invalidConfigs.find(best) == _invalidConfigs.end()) {
332  // valid config found!
333  _currentConfig = best;
334  return true;
335  } else {
336  // invalid: dispose
337  _currentAcquisitions.pop_back();
338  }
339  }
340 
341  // No valid configuration. This should rarely happen.
342  AutoPasLog(debug, "Tuning could not generate a valid configuration.");
343  }
344 
345  utils::ExceptionHandler::exception("BayesianClusterSearch: Failed to sample an valid FeatureVector");
346  _currentConfig = FeatureVector();
347  return false;
348 }
