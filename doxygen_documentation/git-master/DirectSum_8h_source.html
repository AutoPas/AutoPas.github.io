116 [[nodiscard]] std::vector<ParticleType> updateContainer(bool keepNeighborListsValid) override {
117 if (keepNeighborListsValid) {
118 return autopas::LeavingParticleCollector::collectParticlesAndMarkNonOwnedAsDummy(*this);
119 }
120 // first we delete halo particles, as we don't want them here.
121 deleteHaloParticles();
122 getCell().deleteDummyParticles();
123
124 std::vector<ParticleType> invalidParticles{};
125 for (auto iter = getCell().begin(); iter.isValid(); ++iter) {
126 if (utils::notInBox(iter->getR(), this->getBoxMin(), this->getBoxMax())) {
127 invalidParticles.push_back(*iter);
128 internal::deleteParticle(iter);
129 }
130 }
131 return invalidParticles;
132 }
