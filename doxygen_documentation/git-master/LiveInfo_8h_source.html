79 template <class Particle, class PairwiseFunctor>
80 void gather(const autopas::ParticleContainerInterface<Particle> &container, const PairwiseFunctor &functor,
81 unsigned int rebuildFrequency) {
82 using namespace autopas::utils::ArrayMath::literals;
83 using autopas::utils::ArrayMath::ceilToInt;
84 using autopas::utils::ArrayMath::floorToInt;
85
86 // Some aliases for quicker access
87 const auto &boxMin = container.getBoxMin();
88 const auto &boxMax = container.getBoxMax();
89 const auto cutoff = container.getCutoff();
90 const auto cutoffInv = 1.0 / cutoff;
91 const auto numParticles = container.getNumberOfParticles();
92
93 infos["numParticles"] = numParticles;
94 infos["cutoff"] = cutoff;
95 infos["skin"] = container.getVerletSkin();
96 infos["rebuildFrequency"] = static_cast<size_t>(rebuildFrequency);
97 const auto domainSize = boxMax - boxMin;
98
99 infos["domainSizeX"] = domainSize[0];
100 infos["domainSizeY"] = domainSize[1];
101 infos["domainSizeZ"] = domainSize[2];
102
103 infos["particleSize"] = sizeof(Particle);
104
105 // Calculate number of cells for a linked cells container, assuming cell size factor == 1
106 const auto cellsPerDim = ceilToInt(domainSize * cutoffInv);
107 const auto numCells = static_cast<size_t>(cellsPerDim[0] * cellsPerDim[1] * cellsPerDim[2]);
108 infos["numCells"] = numCells;
109
110 // Count how many particles are in each cell via bin counting
111 std::vector<size_t> particleBins;
112 // +1 because we count all halo particles in the last bin
113 particleBins.resize(numCells + 1);
114 // Blurred cells divide the domain into 3x3x3 equivalent boxes.
115 std::vector<size_t> particleBinsBlurred;
116 particleBinsBlurred.resize(27);
117 const auto blurredCellDimsReciproc = std::array<double, 3>{3.0, 3.0, 3.0} / domainSize;
118 for (const Particle &particle : container) {
119 if (utils::inBox(particle.getR(), boxMin, boxMax)) {
120 // find the actual cell
121 const auto offsetIntoBox = particle.getR() - boxMin;
122 const auto cell = floorToInt(offsetIntoBox * cutoffInv);
123 const auto binIndex = (cell[2] * cellsPerDim[1] + cell[1]) * cellsPerDim[0] + cell[0];
124 particleBins[binIndex]++;
125
126 // find the blurred cell
127 const auto cellBlurred = floorToInt(offsetIntoBox * blurredCellDimsReciproc);
128 const auto binIndexBlurred = (cellBlurred[2] * 3 + cellBlurred[1]) * 3 + cellBlurred[0];
129 particleBinsBlurred[binIndexBlurred]++;
130 } else {
131 // found a halo particle
132 particleBins.back()++;
133 }
134 }
135
136 infos["numHaloParticles"] = particleBins.back();
137
138 // calculate statistics about particle distributions per cell
139 const auto avgParticlesPerCell = numParticles == 0
140 ? 0.
141 : static_cast<double>(container.getNumberOfParticles() - particleBins.back()) /
142 static_cast<double>(numCells);
143 const auto avgParticlesPerBlurredCell =
144 numParticles == 0 ? 0.
145 : static_cast<double>(container.getNumberOfParticles() - particleBins.back()) /
146 static_cast<double>(particleBinsBlurred.size());
147
148 const auto [estimatedNumNeighborInteractions, maxDiff, sumStddev, numEmptyCells, maxParticlesPerCell,
149 minParticlesPerCell] = [&]() {
150 double estimatedNumNeighborInteractionsLambda = 0.;
151 double maxDiffLambda = 0.;
152 double sumStddevLambda = 0.;
153 size_t numEmptyCellsLambda = 0;
154 size_t maxParticlesPerCellLambda = std::numeric_limits<size_t>::min();
155 size_t minParticlesPerCellLambda = std::numeric_limits<size_t>::max();
156 // go over all bins and calculate statistics
157 for (size_t i = 0; i < particleBins.size() - 1; i++) {
158 const auto particlesInBin = particleBins[i];
159 if (particlesInBin == 0) {
160 ++numEmptyCellsLambda;
161 } else {
162 // FIXME: Tobias explain this calculation (:
163 estimatedNumNeighborInteractionsLambda +=
164 static_cast<double>(particlesInBin * (particlesInBin * 27 - 1)) * 0.155;
165 }
166 maxParticlesPerCellLambda = std::max(particlesInBin, maxParticlesPerCellLambda);
167 minParticlesPerCellLambda = std::min(particlesInBin, minParticlesPerCellLambda);
168 const auto diffFromAvg = avgParticlesPerCell - static_cast<int>(particlesInBin);
169 maxDiffLambda = std::max(diffFromAvg, maxDiffLambda);
170 sumStddevLambda += diffFromAvg * diffFromAvg;
171 }
172 estimatedNumNeighborInteractionsLambda /= 2;
173
174 return std::tuple{estimatedNumNeighborInteractionsLambda,
175 maxDiffLambda,
176 sumStddevLambda,
177 numEmptyCellsLambda,
178 maxParticlesPerCellLambda,
179 minParticlesPerCellLambda};
180 }();
181
182 infos["numEmptyCells"] = numEmptyCells;
183 infos["maxParticlesPerCell"] = maxParticlesPerCell;
184 infos["minParticlesPerCell"] = minParticlesPerCell;
185 infos["particlesPerCellStdDev"] =
186 numParticles == 0 ? 0.
187 : std::sqrt(sumStddev) / static_cast<double>(particleBins.size() - 1) / avgParticlesPerCell;
188 infos["avgParticlesPerCell"] = avgParticlesPerCell;
189 infos["estimatedNumNeighborInteractions"] = static_cast<unsigned long>(estimatedNumNeighborInteractions);
190
191 const double sumStddevBlurred = [&]() {
192 double res = 0.;
193 for (auto numParticlesInBin : particleBinsBlurred) {
194 auto diff = avgParticlesPerBlurredCell - static_cast<int>(numParticlesInBin);
195 res += diff * diff;
196 }
197 return res;
198 }();
199 infos["particlesPerBlurredCellStdDev"] = numParticles == 0 ? 0.
200 : std::sqrt(sumStddevBlurred) /
201 static_cast<double>(particleBinsBlurred.size()) /
202 avgParticlesPerBlurredCell;
203
204 infos["threadCount"] = static_cast<size_t>(autopas::autopas_get_max_threads());
205
206 constexpr size_t particleSizeNeededByFunctor = calculateParticleSizeNeededByFunctor<Particle, PairwiseFunctor>(
207 std::make_index_sequence<PairwiseFunctor::getNeededAttr().size()>());
208 infos["particleSizeNeededByFunctor"] = particleSizeNeededByFunctor;
209 }

221 [[nodiscard]] std::string toString() const {
222 auto typeToString = [](auto type) {
223 if constexpr (std::is_same_v<decltype(type), bool> or std::is_same_v<decltype(type), double> or
224 std::is_same_v<decltype(type), size_t>) {
225 return std::to_string(type);
226 } else if constexpr (std::is_base_of_v<Option<decltype(type)>, decltype(type)>) {
227 return type.to_string();
228 }
229 return std::string{"fail"};
230 };
231 auto pairToString = [&](const auto &pair) { return pair.first + "=" + std::visit(typeToString, pair.second); };
232 std::string res{"Live Info: "};
233 if (not infos.empty()) {
234 // We initialize the accumulation with the first element. Hence, the accumulation starts at next(begin).
235 res += std::accumulate(std::next(infos.begin()), infos.end(), pairToString(*infos.begin()),
236 [&](std::string s, const auto &elem) { return std::move(s) + " " + pairToString(elem); });
237 }
238 return res;
239 }
