195 inline void LCC01Traversal<ParticleCell, PairwiseFunctor, dataLayout, useNewton3, combineSoA>::computeOffsets() {
196  _cellOffsets.resize(2 * this->_overlap[0] + 1);
197 
198  const auto interactionLengthSquare(this->_interactionLength * this->_interactionLength);
199 
200  for (long x = -this->_overlap[0]; x <= 0l; ++x) {
201  for (long y = -this->_overlap[1]; y <= static_cast<long>(this->_overlap[1]); ++y) {
202  for (long z = -this->_overlap[2]; z <= static_cast<long>(this->_overlap[2]); ++z) {
203  std::array<double, 3> pos = {};
204  pos[0] = std::max(0l, (std::abs(x) - 1l)) * this->_cellLength[0];
205  pos[1] = std::max(0l, (std::abs(y) - 1l)) * this->_cellLength[1];
206  pos[2] = std::max(0l, (std::abs(z) - 1l)) * this->_cellLength[2];
207  const double distSquare = utils::ArrayMath::dot(pos, pos);
208  if (distSquare <= interactionLengthSquare) {
209  const long currentOffset = utils::ThreeDimensionalMapping::threeToOneD(
210  x, y, z, utils::ArrayUtils::static_cast_array<long>(this->_cellsPerDimension));
211  const bool containCurrentOffset =
212  std::any_of(_cellOffsets[x + this->_overlap[0]].cbegin(), _cellOffsets[x + this->_overlap[0]].cend(),
213  [currentOffset](const auto &e) { return e.first == currentOffset; });
214  if (containCurrentOffset) {
215  continue;
216  }
217  for (long ix = x; ix <= std::abs(x); ++ix) {
218  const long offset = utils::ThreeDimensionalMapping::threeToOneD(
219  ix, y, z, utils::ArrayUtils::static_cast_array<long>(this->_cellsPerDimension));
220  const size_t index = ix + this->_overlap[0];
221  if (y == 0l and z == 0l) {
222  // make sure center of slice is always at the beginning
223  _cellOffsets[index].insert(_cellOffsets[index].cbegin(),
224  std::make_pair(offset, utils::ArrayMath::normalize(pos)));
225  } else {
226  _cellOffsets[index].push_back(std::make_pair(offset, utils::ArrayMath::normalize(pos)));
227  }
228  }
229  }
230  }
231  }
232  }
233 }

238  std::vector<ParticleCell> &cells, unsigned long x, unsigned long y, unsigned long z) {
239  unsigned long baseIndex = utils::ThreeDimensionalMapping::threeToOneD(x, y, z, this->_cellsPerDimension);
240  ParticleCell &baseCell = cells[baseIndex];
241  const size_t cOffSize = _cellOffsets.size();
242 
243  if constexpr (combineSoA) {
244  // Iteration along x
245 
246  const auto threadID = static_cast<size_t>(autopas_get_thread_num());
247  auto &currentSlice = _currentSlices[threadID * _cacheOffset];
248  auto &combinationSlice = _combinationSlices[threadID];
249 
250  // First cell needs to initialize whole buffer
251  if (x == this->_overlap[0]) {
252  currentSlice = 0;
253  for (unsigned int offsetSlice = 0; offsetSlice < cOffSize; offsetSlice++) {
254  combinationSlice[offsetSlice]._particleSoABuffer.clear();
255  for (const auto &offset : _cellOffsets[offsetSlice]) {
256  const unsigned long otherIndex = baseIndex + offset.first;
257  ParticleCell &otherCell = cells[otherIndex];
258  appendNeeded(combinationSlice[offsetSlice], otherCell,
259  std::make_index_sequence<PairwiseFunctor::getNeededAttr(std::false_type()).size()>{});
260  }
261  }
262  } else {
263  // reduce size
264  size_t i = 0;
265  const size_t midSlice = (currentSlice + this->_overlap[0] + 1) % cOffSize;
266  for (size_t slice = (currentSlice + 1) % cOffSize; slice != midSlice; ++slice %= cOffSize, ++i) {
267  size_t newSize = 0;
268  for (const auto &offset : _cellOffsets[i]) {
269  const unsigned long otherIndex = baseIndex + offset.first;
270  ParticleCell &otherCell = cells[otherIndex];
271  newSize += otherCell.numParticles();
272  }
273  combinationSlice[slice]._particleSoABuffer.resizeArrays(newSize);
274  }
275  // append buffers
276  for (size_t slice = midSlice; slice != currentSlice; ++slice %= cOffSize, ++i) {
277  for (auto offsetIndex = _cellOffsets[(i + 1) % cOffSize].size(); offsetIndex < _cellOffsets[i].size();
278  ++offsetIndex) {
279  const unsigned long otherIndex = baseIndex + _cellOffsets[i][offsetIndex].first;
280  ParticleCell &otherCell = cells[otherIndex];
281  appendNeeded(combinationSlice[slice], otherCell,
282  std::make_index_sequence<PairwiseFunctor::getNeededAttr(std::false_type()).size()>{});
283  }
284  }
285 
286  combinationSlice[currentSlice]._particleSoABuffer.clear();
287 
288  for (const auto &offset : _cellOffsets.back()) {
289  const unsigned long otherIndex = baseIndex + offset.first;
290  ParticleCell &otherCell = cells[otherIndex];
291  appendNeeded(combinationSlice[currentSlice], otherCell,
292  std::make_index_sequence<PairwiseFunctor::getNeededAttr(std::false_type()).size()>{});
293  }
294 
295  ++currentSlice %= cOffSize;
296  }
297 
298  // calculate all interactions
299  for (unsigned int slice = 0; slice < cOffSize; slice++) {
300  if (slice == (currentSlice + this->_overlap[0]) % cOffSize) {
301  // slice contains base cell -> skip particles of base cell. This is not supported by CellFunctor, so call
302  // pairwise functor directly.
303  auto startIndex = baseCell.numParticles();
304  auto endIndex = combinationSlice[slice]._particleSoABuffer.getNumParticles();
305  _pairwiseFunctor->SoAFunctorPair(baseCell._particleSoABuffer,
306  {&(combinationSlice[slice]._particleSoABuffer), startIndex, endIndex}, false);
307  // compute base cell
308  this->_cellFunctor.processCell(baseCell);
309  } else {
310  this->_cellFunctor.processCellPair(baseCell, combinationSlice[slice]);
311  }
312  }
313  } else {
314  for (const auto &slice : _cellOffsets) {
315  for (auto const &[offset, r] : slice) {
316  const unsigned long otherIndex = baseIndex + offset;
317  ParticleCell &otherCell = cells[otherIndex];
318 
319  if (baseIndex == otherIndex) {
320  this->_cellFunctor.processCell(baseCell);
321  } else {
322  this->_cellFunctor.processCellPair(baseCell, otherCell, r);
323  }
324  }
325  }
326  }
