86 size_t generateClusters() {
87 if (getNumActualParticles() > 0) {
88 _particlesStorage.sortByDim(2);
89
90 // if the number of particles is divisible by the cluster size this is 0
91 const auto sizeLastCluster = _particlesStorage.numParticles() % _clusterSize;
92 _numDummyParticles = sizeLastCluster == 0 ? 0 : _clusterSize - sizeLastCluster;
93
94 // fill the last cluster with dummy copies of the last particle
95 auto lastParticle = _particlesStorage[_particlesStorage.numParticles() - 1];
96 markParticleAsDeleted(lastParticle);
97 for (size_t i = 0; i < _numDummyParticles; i++) {
98 _particlesStorage.addParticle(lastParticle);
99 }
100
101 // Mark start of the different clusters by adding pointers to _particlesStorage
102 size_t numClusters = _particlesStorage.numParticles() / _clusterSize;
103 _clusters.reserve(numClusters);
104 for (size_t index = 0; index < numClusters; index++) {
105 _clusters.emplace_back(&(_particlesStorage[_clusterSize * index]), _clusterSize);
106 }
107 }
108
109 return getNumClusters();
110 }

119 void setDummyValues(double dummyStartX, double dummyDistZ) {
120 auto &lastCluster = getCluster(getNumClusters() - 1);
121 for (size_t index = 1; index <= _numDummyParticles; index++) {
122 auto &dummy = lastCluster[_clusterSize - index];
123 dummy = lastCluster[0]; // use first Particle in last cluster as dummy particle!
124 dummy.setOwnershipState(OwnershipState::dummy);
125 dummy.setR({dummyStartX, 0, dummyDistZ * static_cast<double>(index)});
126 dummy.setID(std::numeric_limits<size_t>::max());
127 }
128 }

134 void setDummyParticlesToLastActualParticle() {
135 if (_numDummyParticles > 0) {
136 auto &lastCluster = getCluster(getNumClusters() - 1);
137 auto lastActualParticle = lastCluster[_clusterSize - _numDummyParticles - 1];
138 for (size_t index = 1; index <= _numDummyParticles; index++) {
139 lastCluster[_clusterSize - index] = lastActualParticle;
140 }
141 }

330 void deleteByIndex(size_t index) override {
334
335 // swap particle that should be deleted to end of actual particles.
336 std::swap(_particlesStorage._particles[index], _particlesStorage._particles[getNumActualParticles() - 1]);
337 if (getNumTailDummyParticles() != 0) {
338 // swap particle that should be deleted (now at end of actual particles) with last dummy particle.
339 std::swap(_particlesStorage._particles[getNumActualParticles() - 1],
340 _particlesStorage._particles[_particlesStorage._particles.size() - 1]);
341 }
342 _particlesStorage._particles.pop_back();
343
344 if (_particleDeletionObserver) {
345 _particleDeletionObserver->notifyParticleDeleted();
346 }
