81 size_t generateClusters() {
82 if (getNumActualParticles() == 0) {
83 _clusters.resize(0, Cluster<Particle>(nullptr, _clusterSize));
84 return 0;
85 }
86 this->sortByDim(2);
87
88 // if the number of particles is divisible by the cluster size this is 0
89 const auto sizeLastCluster = this->_particles.size() % _clusterSize;
90 _numDummyParticles = sizeLastCluster == 0 ? 0 : _clusterSize - sizeLastCluster;
91
92 // fill the last cluster with dummy copies of the last particle
93 auto lastParticle = this->_particles[this->_particles.size() - 1];
94 markParticleAsDeleted(lastParticle);
95 for (size_t i = 0; i < _numDummyParticles; i++) {
96 this->addParticle(lastParticle);
97 }
98
99 // Mark start of the different clusters by adding pointers to _particles
100 const size_t numClusters = this->_particles.size() / _clusterSize;
101 _clusters.resize(numClusters, Cluster<Particle>(nullptr, _clusterSize));
102 for (size_t index = 0; index < numClusters; index++) {
103 _clusters[index].reset(&(this->_particles[_clusterSize * index]));
104 }
105 return getNumClusters();
106 }

115 void setDummyValues(double dummyStartX, double dummyDistZ) {
116 auto &lastCluster = getCluster(getNumClusters() - 1);
117 for (size_t index = 1; index <= _numDummyParticles; index++) {
118 auto &dummy = lastCluster[_clusterSize - index];
119 dummy = lastCluster[0]; // use first Particle in last cluster as dummy particle!
120 dummy.setOwnershipState(OwnershipState::dummy);
121 dummy.setR({dummyStartX, 0, dummyDistZ * static_cast<double>(index)});
122 dummy.setID(std::numeric_limits<size_t>::max());
123 }

189 std::vector<Particle> collectOutOfBoundsParticles(const std::array<double, 3> &boxMin,
190 const std::array<double, 3> &boxMax) {
191 // make sure to get rid of all dummies
192 deleteDummyParticles();
193 // move all particles that are not in the tower to the back of the storage
194 const auto firstOutOfBoundsParticleIter =
195 std::partition(this->_particles.begin(), this->_particles.end(),
196 [&](const auto &p) { return utils::inBox(p.getR(), boxMin, boxMax); });
197
198 // copy out of bounds particles
199 std::vector<Particle> outOfBoundsParticles(firstOutOfBoundsParticleIter, this->_particles.end());
200 // shrink the particle storage so all out of bounds particles are cut away
201 this->_particles.resize(std::distance(this->_particles.begin(), firstOutOfBoundsParticleIter),
202 *this->_particles.begin());
203 return outOfBoundsParticles;
204 }
