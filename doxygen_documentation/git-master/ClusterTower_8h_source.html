79  size_t generateClusters() {
80  if (getNumActualParticles() > 0) {
81  _particlesStorage.sortByDim(2);
82 
83  auto sizeLastCluster = (_particlesStorage.numParticles() % _clusterSize);
84  _numDummyParticles = sizeLastCluster != 0 ? _clusterSize - sizeLastCluster : 0;
85 
86  auto lastParticle = _particlesStorage[_particlesStorage.numParticles() - 1];
87  markParticleAsDeleted(lastParticle);
88  for (size_t i = 0; i < _numDummyParticles; i++) {
89  _particlesStorage.addParticle(lastParticle);
90  }

92  // Mark start of the different clusters by adding pointers to _particlesStorage
93  size_t numClusters = _particlesStorage.numParticles() / _clusterSize;
94  _clusters.reserve(numClusters);
95  for (size_t index = 0; index < numClusters; index++) {
96  _clusters.emplace_back(&(_particlesStorage[_clusterSize * index]), _clusterSize);
97  }
98  }

109  void fillUpWithDummyParticles(double dummyStartX, double dummyDistZ) {
