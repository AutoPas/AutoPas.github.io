81 size_t generateClusters() {
82 if (getNumActualParticles() > 0) {
83 _particlesStorage.sortByDim(2);
84
85 auto sizeLastCluster = (_particlesStorage.numParticles() % _clusterSize);
86 _numDummyParticles = sizeLastCluster != 0 ? _clusterSize - sizeLastCluster : 0;
87
88 auto lastParticle = _particlesStorage[_particlesStorage.numParticles() - 1];
89 markParticleAsDeleted(lastParticle);
90 for (size_t i = 0; i < _numDummyParticles; i++) {
91 _particlesStorage.addParticle(lastParticle);
92 }
93
94 // Mark start of the different clusters by adding pointers to _particlesStorage
95 size_t numClusters = _particlesStorage.numParticles() / _clusterSize;
96 _clusters.reserve(numClusters);
97 for (size_t index = 0; index < numClusters; index++) {
98 _clusters.emplace_back(&(_particlesStorage[_clusterSize * index]), _clusterSize);
99 }
100 }
101
102 return getNumClusters();
103 }

111 void fillUpWithDummyParticles(double dummyStartX, double dummyDistZ) {
112 auto &lastCluster = getCluster(getNumClusters() - 1);
113 for (size_t index = 1; index <= _numDummyParticles; index++) {
114 lastCluster[_clusterSize - index] = lastCluster[0]; // use first Particle in last cluster as dummy particle!
115 lastCluster[_clusterSize - index].setOwnershipState(OwnershipState::dummy);
116 lastCluster[_clusterSize - index].setR({dummyStartX, 0, dummyDistZ * index});
117 lastCluster[_clusterSize - index].setID(std::numeric_limits<size_t>::max());
118 }
119 }

125 void setDummyParticlesToLastActualParticle() {
126 if (_numDummyParticles > 0) {
127 auto &lastCluster = getCluster(getNumClusters() - 1);
128 auto lastActualParticle = lastCluster[_clusterSize - _numDummyParticles - 1];
129 for (size_t index = 1; index <= _numDummyParticles; index++) {
130 lastCluster[_clusterSize - index] = lastActualParticle;
131 }

292 void deleteByIndex(size_t index) override {
296
297 // swap particle that should be deleted to end of actual particles.
298 std::swap(_particlesStorage._particles[index], _particlesStorage._particles[getNumActualParticles() - 1]);
299 if (getNumDummyParticles() != 0) {

302 _particlesStorage._particles[_particlesStorage._particles.size() - 1]);
