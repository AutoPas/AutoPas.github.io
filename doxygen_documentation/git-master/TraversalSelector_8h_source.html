247template <class ParticleCell>
248template <class PairwiseFunctor>
249std::unique_ptr<TraversalInterface> TraversalSelector<ParticleCell>::generateTraversal(
250 TraversalOption traversalType, PairwiseFunctor &pairwiseFunctor, const TraversalSelectorInfo &traversalInfo,
251 DataLayoutOption dataLayout, Newton3Option newton3) {
252 switch (dataLayout) {
253 case DataLayoutOption::aos: {
254 if (newton3 == Newton3Option::enabled) {
255 return TraversalSelector<ParticleCell>::template generateTraversal<PairwiseFunctor, DataLayoutOption::aos,
256 true>(traversalType, pairwiseFunctor,
257 traversalInfo);
258 } else {
259 return TraversalSelector<ParticleCell>::template generateTraversal<PairwiseFunctor, DataLayoutOption::aos,
260 false>(traversalType, pairwiseFunctor,
261 traversalInfo);
262 }
263 }
264 case DataLayoutOption::soa: {
265 if (newton3 == Newton3Option::enabled) {
266 return TraversalSelector<ParticleCell>::template generateTraversal<PairwiseFunctor, DataLayoutOption::soa,
267 true>(traversalType, pairwiseFunctor,
268 traversalInfo);
269 } else {
270 return TraversalSelector<ParticleCell>::template generateTraversal<PairwiseFunctor, DataLayoutOption::soa,
271 false>(traversalType, pairwiseFunctor,
272 traversalInfo);
273 }
274 }
275 }
276
277 autopas::utils::ExceptionHandler::exception("Traversal type {} is not a known type!", traversalType.to_string());
278 return std::unique_ptr<TraversalInterface>(nullptr);
279}
