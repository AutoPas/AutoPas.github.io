193  for (size_t firstIndex = 0ul; firstIndex < _numRegisteredTypes; ++firstIndex) {
194  for (size_t secondIndex = 0ul; secondIndex < _numRegisteredTypes; ++secondIndex) {
195  auto globalIndex = _numRegisteredTypes * firstIndex + secondIndex;
196 
197  // epsilon

199  _computedMixingData[globalIndex].epsilon24 = epsilon24;

204  _computedMixingData[globalIndex].sigmaSquare = sigmaSquare;
