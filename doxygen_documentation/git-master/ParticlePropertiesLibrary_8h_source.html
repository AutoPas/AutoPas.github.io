276void ParticlePropertiesLibrary<floatType, intType>::addSiteType(intType siteID, floatType epsilon, floatType sigma,
277 floatType mass) {
278 if (_numRegisteredSiteTypes != siteID) {
279 autopas::utils::ExceptionHandler::exception(

282 siteID, _numRegisteredSiteTypes);

290template <typename floatType, typename intType>
291void ParticlePropertiesLibrary<floatType, intType>::addMolType(const intType molId, const std::vector<intType> siteIds,
292 const std::vector<std::array<floatType, 3>> relPos,
293 const std::array<floatType, 3> momentOfInertia) {
294 // Error handling
295#if MD_FLEXIBLE_MODE == SINGLESITE
296 AutoPasLog(WARN,
297 "ParticlePropertiesLibrary::addMolType(): trying to register a multi-site molecule type when md-flexible "
298 "has been compiled without support for multi-site molecules. Please compile with the CMake argument "
299 "'-DMD_FLEXIBLE_MODE=MULTISITE'.");
300#endif
301 if (_numRegisteredMolTypes != molId) {
302 autopas::utils::ExceptionHandler::exception(
303 "ParticlePropertiesLibrary::addMolType(): trying to register a molecule type with id {}. Please register types "
304 "consecutively, starting at id 0. Currently there are {} registered types.",
305 molId, _numRegisteredSiteTypes);
306 }
307 if (std::any_of(siteIds.cbegin(), siteIds.cend(), [this](intType i) { return i >= this->_numRegisteredSiteTypes; })) {
308 autopas::utils::ExceptionHandler::exception(
309 "ParticlePropertiesLibrary::addMolType(): trying to register a molecule type with an unregistered site type "
310 "Id.");
311 }
312 if (siteIds.size() != relPos.size()) {
313 autopas::utils::ExceptionHandler::exception(
314 "ParticlePropertiesLibrary::addMolType(): trying to register a molecule type with vectors of site IDs and site"
315 "positions that do not match in size.");
316 }
317
318 // Add molecule type if there are no errors.
319 ++_numRegisteredMolTypes;
320 _siteIds.emplace_back(siteIds);
321 _relativeSitePositions.emplace_back(relPos);
322 _numSites.emplace_back(siteIds.size());
323 floatType molMass{0.};
324 for (intType site = 0; site < siteIds.size(); ++site) {
325 molMass += _siteMasses[siteIds[site]];
326 }
327 _molMasses.emplace_back(molMass);
328 _momentOfInertias.emplace_back(momentOfInertia);
329
330 floatType molLargestSigma{0.};
331 for (size_t site = 0; site < siteIds.size(); site++) {
332 molLargestSigma = std::max(molLargestSigma, _sigmas[siteIds[site]]);
333 }
334 _moleculesLargestSigma.emplace_back(molLargestSigma);
335}

337template <typename floatType, typename intType>
338void ParticlePropertiesLibrary<floatType, intType>::calculateMixingCoefficients() {
339 if (_numRegisteredSiteTypes == 0) {
340 autopas::utils::ExceptionHandler::AutoPasException(
341 "ParticlePropertiesLibrary::calculateMixingCoefficients was called without any site types being registered!");
342 }
343
344 _computedMixingData.resize(_numRegisteredSiteTypes * _numRegisteredSiteTypes);
345
346 const auto cutoffSquared = _cutoff * _cutoff;
347
348 for (size_t firstIndex = 0ul; firstIndex < _numRegisteredSiteTypes; ++firstIndex) {
349 for (size_t secondIndex = 0ul; secondIndex < _numRegisteredSiteTypes; ++secondIndex) {
350 auto globalIndex = _numRegisteredSiteTypes * firstIndex + secondIndex;
351
352 // epsilon
353 const floatType epsilon24 = 24 * sqrt(_epsilons[firstIndex] * _epsilons[secondIndex]);
354 _computedMixingData[globalIndex].epsilon24 = epsilon24;
355
356 // sigma
357 const floatType sigma = (_sigmas[firstIndex] + _sigmas[secondIndex]) / 2.0;
358 const floatType sigmaSquared = sigma * sigma;
359 _computedMixingData[globalIndex].sigmaSquared = sigmaSquared;
360
361 // shift6
362 const floatType shift6 = calcShift6(epsilon24, sigmaSquared, cutoffSquared);
363 _computedMixingData[globalIndex].shift6 = shift6;
364 }
365 }

447template <typename floatType, typename intType>
448double ParticlePropertiesLibrary<floatType, intType>::calcShift6(double epsilon24, double sigmaSquared,
449 double cutoffSquared) {
450 const auto sigmaDivCutoffPow2 = sigmaSquared / cutoffSquared;
451 const auto sigmaDivCutoffPow6 = sigmaDivCutoffPow2 * sigmaDivCutoffPow2 * sigmaDivCutoffPow2;
452 const auto shift6 = epsilon24 * (sigmaDivCutoffPow6 - sigmaDivCutoffPow6 * sigmaDivCutoffPow6);
453 return shift6;
454}
