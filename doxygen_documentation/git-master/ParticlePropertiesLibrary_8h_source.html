187template <typename floatType, typename intType>
188void ParticlePropertiesLibrary<floatType, intType>::calculateMixingCoefficients() {
189 _computedMixingData.resize(_numRegisteredTypes * _numRegisteredTypes);
190
191 auto cutoffSquare = _cutoff * _cutoff;
192
193 for (size_t firstIndex = 0ul; firstIndex < _numRegisteredTypes; ++firstIndex) {
194 for (size_t secondIndex = 0ul; secondIndex < _numRegisteredTypes; ++secondIndex) {
195 auto globalIndex = _numRegisteredTypes * firstIndex + secondIndex;
196
197 // epsilon
198 floatType epsilon24 = 24 * sqrt(_epsilons[firstIndex] * _epsilons[secondIndex]);
199 _computedMixingData[globalIndex].epsilon24 = epsilon24;
200
201 // sigma
202 floatType sigma = (_sigmas[firstIndex] + _sigmas[secondIndex]) / 2.0;
203 floatType sigmaSquare = sigma * sigma;
204 _computedMixingData[globalIndex].sigmaSquare = sigmaSquare;
205
206 // shift6
207 floatType shift6 = calcShift6(epsilon24, sigmaSquare, cutoffSquare);
208 _computedMixingData[globalIndex].shift6 = shift6;
209 }
210 }

228template <typename floatType, typename intType>
229double ParticlePropertiesLibrary<floatType, intType>::calcShift6(double epsilon24, double sigmaSquare,
230 double cutoffSquare) {
231 auto sigmaPow2DivCutoff = sigmaSquare / (cutoffSquare);
232 auto sigmaPow6DivCutoff = sigmaPow2DivCutoff * sigmaPow2DivCutoff * sigmaPow2DivCutoff;
233 auto shift6 = epsilon24 * (sigmaPow6DivCutoff - sigmaPow6DivCutoff * sigmaPow6DivCutoff);
234 return shift6;
235}
