120 void AoSFunctor(Particle &i, Particle &j, bool newton3) final {
121 using namespace autopas::utils::ArrayMath::literals;
122
123 if (i.isDummy() or j.isDummy()) {
124 return;
125 }
126 auto sigmasquare = _sigmaSquareAoS;
127 auto epsilon24 = _epsilon24AoS;
128 auto shift6 = _shift6AoS;
129 if constexpr (useMixing) {
130 sigmasquare = _PPLibrary->mixingSigmaSquare(i.getTypeId(), j.getTypeId());
131 epsilon24 = _PPLibrary->mixing24Epsilon(i.getTypeId(), j.getTypeId());
132 if constexpr (applyShift) {
133 shift6 = _PPLibrary->mixingShift6(i.getTypeId(), j.getTypeId());
134 }
135 }
136 auto dr = i.getR() - j.getR();
137 double dr2 = utils::ArrayMath::dot(dr, dr);
138
139 if (dr2 > _cutoffsquareAoS) {
140 return;
141 }
142
143 double invdr2 = 1. / dr2;
144 double lj6 = sigmasquare * invdr2;
145 lj6 = lj6 * lj6 * lj6;
146 double lj12 = lj6 * lj6;
147 double lj12m6 = lj12 - lj6;
148 double fac = epsilon24 * (lj12 + lj12m6) * invdr2;
149 auto f = dr * fac;
150 i.addF(f);
151 if (newton3) {
152 // only if we use newton 3 here, we want to
153 j.subF(f);
154 }
155 if (calculateGlobals) {
156 auto virial = dr * f;
157 double upot = epsilon24 * lj12m6 + shift6;
158
159 const int threadnum = autopas_get_thread_num();
160 // for non-newton3 the division is in the post-processing step.
161 if (newton3) {
162 upot *= 0.5;
163 virial *= (double)0.5;
164 }
165 if (i.isOwned()) {
166 if (newton3) {
167 _aosThreadData[threadnum].upotSumN3 += upot * 0.5;
168 _aosThreadData[threadnum].virialSumN3 += virial * 0.5;
169 } else {
170 // for non-newton3 the division is in the post-processing step.
171 _aosThreadData[threadnum].upotSumNoN3 += upot;
172 _aosThreadData[threadnum].virialSumNoN3 += virial;
173 }
174 }
175 // for non-newton3 the second particle will be considered in a separate calculation
176 if (newton3 and j.isOwned()) {
177 _aosThreadData[threadnum].upotSumN3 += upot * 0.5;
178 _aosThreadData[threadnum].virialSumN3 += virial * 0.5;
179 }
180 }

391 const svfloat64_t &z1, const svbool_t &pg, const int64_t *const __restrict ownedStatePtr2,
392 const double *const __restrict x2ptr, const double *const __restrict y2ptr,

449 inline void applyForces(const size_t j, const svuint64_t &index, const bool ownedStateIisOwned,
450 double *const __restrict fx2ptr, double *const __restrict fy2ptr,

452 svfloat64_t &virialSumX, svfloat64_t &virialSumY, svfloat64_t &virialSumZ,
453 svfloat64_t &upotSum, const svfloat64_t &drx, const svfloat64_t &dry, const svfloat64_t &drz,
454 const double *const __restrict x2ptr, const double *const __restrict y2ptr,
455 const double *const __restrict z2ptr, const svint64_t &ownedStateJ, const svbool_t &pgC,
456 const svfloat64_t &epsilon24s, const svfloat64_t &shift6s, const svfloat64_t &lj6,
457 const svfloat64_t &fac) {
458 const svfloat64_t fx = svmul_x(pgC, drx, fac);
459 const svfloat64_t fy = svmul_x(pgC, dry, fac);
460 const svfloat64_t fz = svmul_x(pgC, drz, fac);
461
462 fxacc = svadd_f64_m(pgC, fxacc, fx);
463 fyacc = svadd_f64_m(pgC, fyacc, fy);
464 fzacc = svadd_f64_m(pgC, fzacc, fz);
465
466 if (newton3) {
467 const svfloat64_t fx2 = (indexed) ? svld1_gather_index(pgC, fx2ptr, index) : svld1_f64(pgC, &fx2ptr[j]);
468 const svfloat64_t fy2 = (indexed) ? svld1_gather_index(pgC, fy2ptr, index) : svld1_f64(pgC, &fy2ptr[j]);
469 const svfloat64_t fz2 = (indexed) ? svld1_gather_index(pgC, fz2ptr, index) : svld1_f64(pgC, &fz2ptr[j]);
470
471 const svfloat64_t fx2new = svsub_x(pgC, fx2, fx);
472 const svfloat64_t fy2new = svsub_x(pgC, fy2, fy);
473 const svfloat64_t fz2new = svsub_x(pgC, fz2, fz);
474
475 if constexpr (indexed) {
476 svst1_scatter_index(pgC, &fx2ptr[0], index, fx2new);
477 svst1_scatter_index(pgC, &fy2ptr[0], index, fy2new);
478 svst1_scatter_index(pgC, &fz2ptr[0], index, fz2new);
479 } else {
480 svst1(pgC, &fx2ptr[j], fx2new);
481 svst1(pgC, &fy2ptr[j], fy2new);

484 }
485
486 if (calculateGlobals) {
487 // Global Virial
488 const svfloat64_t virialX = svmul_x(pgC, fx, drx);
489 const svfloat64_t virialY = svmul_x(pgC, fy, dry);
490 const svfloat64_t virialZ = svmul_x(pgC, fz, drz);
491
492 // Global Potential
493 const svfloat64_t lj12m6 = svnmls_x(pgC, lj6, lj6, lj6);
494 const svfloat64_t upot = svmad_x(pgC, epsilon24s, lj12m6, shift6s);

765 _virialSum = {0., 0., 0.};

771
776 void endTraversal(bool newton3) final {
777 using namespace autopas::utils::ArrayMath::literals;
778
779 if (_postProcessed) {
780 throw utils::ExceptionHandler::AutoPasException(
781 "Already postprocessed, endTraversal(bool newton3) was called twice without calling initTraversal().");
782 }
783 if (calculateGlobals) {
784 // We distinguish between non-newton3 and newton3 functor calls. Newton3 calls are accumulated directly.
785 // Non-newton3 calls are accumulated temporarily and later divided by 2.
786 double upotSumNoN3Acc = 0;
787 std::array<double, 3> virialSumNoN3Acc = {0, 0, 0};
788 for (size_t i = 0; i < _aosThreadData.size(); ++i) {
789 upotSumNoN3Acc += _aosThreadData[i].upotSumNoN3;
790 _upotSum += _aosThreadData[i].upotSumN3;
791
792 virialSumNoN3Acc += _aosThreadData[i].virialSumNoN3;
793 _virialSum += _aosThreadData[i].virialSumN3;
794 }
795 // if the newton3 optimization is disabled we have added every energy contribution twice, so we divide by 2
796 // here.
797 upotSumNoN3Acc *= 0.5;
798 virialSumNoN3Acc *= 0.5;
799
800 _upotSum += upotSumNoN3Acc;
801 _virialSum += virialSumNoN3Acc;
802
803 // we have always calculated 6*upot, so we divide by 6 here!
804 _upotSum /= 6.;
805 _postProcessed = true;
806
807 AutoPasLog(TRACE, "Final potential energy {}", _upotSum);
808 AutoPasLog(TRACE, "Final virial {}", _virialSum[0] + _virialSum[1] + _virialSum[2]);
809 }
