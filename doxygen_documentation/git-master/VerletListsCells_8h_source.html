126  void rebuildNeighborLists(TraversalInterface *traversal) override {
127  bool useNewton3 = traversal->getUseNewton3();
128  this->_verletBuiltNewton3 = useNewton3;
129 
130  // Initialize a neighbor list for each cell.
131  _neighborLists.clear();
132  auto &cells = this->_linkedCells.getCells();
133  size_t cellsSize = cells.size();
134  _neighborLists.resize(cellsSize);
135  for (size_t cellIndex = 0; cellIndex < cellsSize; ++cellIndex) {
136  _neighborLists[cellIndex].reserve(cells[cellIndex].numParticles());
137  size_t particleIndexWithinCell = 0;
138  for (auto iter = cells[cellIndex].begin(); iter.isValid(); ++iter, ++particleIndexWithinCell) {
139  Particle *particle = &*iter;
140  _neighborLists[cellIndex].emplace_back(particle, std::vector<Particle *>());
141  // In a cell with N particles, reserve space for 5N neighbors.
142  // 5 is an empirically determined magic number that provides good speed.
143  _neighborLists[cellIndex].back().second.reserve(cells[cellIndex].numParticles() * 5);
144  _particleToCellMap[particle] = std::make_pair(cellIndex, particleIndexWithinCell);
145  }
146  }
