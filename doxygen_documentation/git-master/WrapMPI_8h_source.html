705int AutoPas_MPI_Error_string(int errorcode, char *string, int *resultlen) {
706 static const std::map<int, const char *> errorStrings = {
707 {AUTOPAS_MPI_SUCCESS, "MPI_SUCCESS: no errors"},
708 {AUTOPAS_MPI_ERR_ARG, "MPI_ERR_ARG: invalid argument of some other kind"},
709 {AUTOPAS_MPI_ERR_COMM, "MPI_ERR_COMM: invalid communicator"},
710 {AUTOPAS_MPI_ERR_RANK, "MPI_ERR_RANK: invalid rank"},
711 {AUTOPAS_MPI_ERR_REQUEST, "MPI_ERR_REQUEST: invalid Request"},
712 {AUTOPAS_MPI_ERR_TYPE, "MPI_ERR_TYPE: invalid datatype"},
713 };
714 snprintf(string, AUTOPAS_MPI_MAX_ERROR_STRING, "%s", errorStrings.at(errorcode));
715 *resultlen = strnlen(string, AUTOPAS_MPI_MAX_ERROR_STRING);
716 return AUTOPAS_MPI_SUCCESS;

822inline int AutoPas_MPI_Allgather(void *buffer_send, int count_send, AutoPas_MPI_Datatype datatype_send,
823 void *buffer_recv, int count_recv, AutoPas_MPI_Datatype datatype_recv,
824 AutoPas_MPI_Comm comm) {
825 if (buffer_send != AUTOPAS_MPI_IN_PLACE) {
826 for (long i = 0; i < (count_recv / count_send); i++)
827 // offsets from pointers are of type ptrdiff_t which is an alias for long. Hence, i should be long.
828 memcpy(static_cast<char *>(buffer_recv) + (i * count_send * sizeof(datatype_send)), buffer_send,
829 count_send * sizeof(datatype_send));
830 }
831 return AUTOPAS_MPI_SUCCESS;
832}
