562int AutoPas_MPI_Error_string(int errorcode, char *string, int *resultlen) {
563 static const std::map<int, const char *> errorStrings = {
564 {AUTOPAS_MPI_SUCCESS, "MPI_SUCCESS: no errors"},
565 {AUTOPAS_MPI_ERR_ARG, "MPI_ERR_ARG: invalid argument of some other kind"},
566 {AUTOPAS_MPI_ERR_COMM, "MPI_ERR_COMM: invalid communicator"},
567 {AUTOPAS_MPI_ERR_RANK, "MPI_ERR_RANK: invalid rank"},
568 {AUTOPAS_MPI_ERR_REQUEST, "MPI_ERR_REQUEST: invalid Request"},
569 {AUTOPAS_MPI_ERR_TYPE, "MPI_ERR_TYPE: invalid datatype"},
570 };
571 snprintf(string, AUTOPAS_MPI_MAX_ERROR_STRING, "%s", errorStrings.at(errorcode));
572 *resultlen = strnlen(string, AUTOPAS_MPI_MAX_ERROR_STRING);
573 return AUTOPAS_MPI_SUCCESS;
574}
