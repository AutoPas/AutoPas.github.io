519int AutoPas_MPI_Error_string(int errorcode, char *string, int *resultlen) {
520 static const std::map<int, const char *> errorStrings = {
521 {AUTOPAS_MPI_SUCCESS, "MPI_SUCCESS: no errors"},
522 {AUTOPAS_MPI_ERR_ARG, "MPI_ERR_ARG: invalid argument of some other kind"},
523 {AUTOPAS_MPI_ERR_COMM, "MPI_ERR_COMM: invalid communicator"},
524 {AUTOPAS_MPI_ERR_RANK, "MPI_ERR_RANK: invalid rank"},
525 {AUTOPAS_MPI_ERR_REQUEST, "MPI_ERR_REQUEST: invalid Request"},
526 {AUTOPAS_MPI_ERR_TYPE, "MPI_ERR_TYPE: invalid datatype"},
527 };
528 snprintf(string, AUTOPAS_MPI_MAX_ERROR_STRING, "%s", errorStrings.at(errorcode));
529 *resultlen = strnlen(string, AUTOPAS_MPI_MAX_ERROR_STRING);
530 return AUTOPAS_MPI_SUCCESS;
531}
