405int AutoPas_MPI_Error_string(int errorcode, char *string, int *resultlen) {
406 static const std::map<int, const char *> errorStrings = {
407 {AUTOPAS_MPI_SUCCESS, "MPI_SUCCESS: no errors"},
408 {AUTOPAS_MPI_ERR_ARG, "MPI_ERR_ARG: invalid argument of some other kind"},
409 {AUTOPAS_MPI_ERR_COMM, "MPI_ERR_COMM: invalid communicator"},
410 {AUTOPAS_MPI_ERR_RANK, "MPI_ERR_RANK: invalid rank"},
411 {AUTOPAS_MPI_ERR_REQUEST, "MPI_ERR_REQUEST: invalid Request"},
412 {AUTOPAS_MPI_ERR_TYPE, "MPI_ERR_TYPE: invalid datatype"},
413 };
414 snprintf(string, AUTOPAS_MPI_MAX_ERROR_STRING, "%s", errorStrings.at(errorcode));
415 *resultlen = strnlen(string, AUTOPAS_MPI_MAX_ERROR_STRING);
416 return AUTOPAS_MPI_SUCCESS;
417}
