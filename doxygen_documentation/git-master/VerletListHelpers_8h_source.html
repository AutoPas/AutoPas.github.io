62 void AoSFunctor(Particle &i, Particle &j, bool /*newton3*/) override {
63 if (i.isDummy() or j.isDummy()) {
64 return;
65 }
66 auto dist = utils::ArrayMath::sub(i.getR(), j.getR());
67
68 double distsquare = utils::ArrayMath::dot(dist, dist);
69 if (distsquare < _interactionLengthSquared) {
70 // this is thread safe, only if particle i is accessed by only one
71 // thread at a time. which is ensured, as particle i resides in a
72 // specific cell and each cell is only accessed by one thread at a time
73 // (ensured by traversals)
74 // also the list is not allowed to be resized!
75
76 _verletListsAoS.at(&i).push_back(&j);
77 // no newton3 here, as AoSFunctor(j,i) will also be called if newton3 is disabled.
78 }
79 }

86 void SoAFunctorSingle(SoAView<SoAArraysType> soa, bool newton3) override {
87 if (soa.getNumParticles() == 0) return;
88
89 auto **const __restrict ptrptr = soa.template begin<Particle::AttributeNames::ptr>();
90 const double *const __restrict xptr = soa.template begin<Particle::AttributeNames::posX>();
91 const double *const __restrict yptr = soa.template begin<Particle::AttributeNames::posY>();
92 const double *const __restrict zptr = soa.template begin<Particle::AttributeNames::posZ>();
93
94 size_t numPart = soa.getNumParticles();
95 for (unsigned int i = 0; i < numPart; ++i) {
96 auto &currentList = _verletListsAoS.at(ptrptr[i]);
97
98 for (unsigned int j = i + 1; j < numPart; ++j) {
99 const double drx = xptr[i] - xptr[j];
100 const double dry = yptr[i] - yptr[j];
101 const double drz = zptr[i] - zptr[j];
102
103 const double drx2 = drx * drx;
104 const double dry2 = dry * dry;
105 const double drz2 = drz * drz;
106
107 const double dr2 = drx2 + dry2 + drz2;
108
109 if (dr2 < _interactionLengthSquared) {
110 currentList.push_back(ptrptr[j]);
111 if (not newton3) {
112 // we need this here, as SoAFunctorSingle will only be called once for both newton3=true and false.
113 _verletListsAoS.at(ptrptr[j]).push_back(ptrptr[i]);
114 }
115 }
116 }
