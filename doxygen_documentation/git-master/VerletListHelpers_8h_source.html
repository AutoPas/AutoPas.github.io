80  void AoSFunctor(Particle &i, Particle &j, bool /*newton3*/) override {

85 
86  double distsquare = utils::ArrayMath::dot(dist, dist);
87  if (distsquare < _interactionLengthSquared) {

92  // also the list is not allowed to be resized!

94  _verletListsAoS.at(&i).push_back(&j);
95  // no newton3 here, as AoSFunctor(j,i) will also be called if newton3 is disabled.

118  const double dry = yptr[i] - yptr[j];
