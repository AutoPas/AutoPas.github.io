57 void AoSFunctor(Particle &i, Particle &j, bool newton3) override {
58 if (i.isDummy() or j.isDummy()) {
59 return;
60 }
61 auto dist = utils::ArrayMath::sub(i.getR(), j.getR());
62 double distsquare = utils::ArrayMath::dot(dist, dist);
63 if (distsquare < _cutoffskinsquared) {
64 // this is thread safe, only if particle i is accessed by only one
65 // thread at a time. which is ensured, as particle i resides in a
66 // specific cell and each cell is only accessed by one thread at a time
67 // (ensured by traversals)
68 // also the list is not allowed to be resized!
69
70 auto &[cellIndex, particleIndex] = _particleToCellMap[&i];
71 auto &[cellIndexNeighbor, particleIndexNeighbor] = _particleToCellMap[&j];
72
73 // if cell1 hasn't interacted with cell2 yet and there is no mapping from global to relative index for cell2,
74 // add one
75 auto iter = _globalToLocalIndex[cellIndex].find(cellIndexNeighbor);
76 if (iter == _globalToLocalIndex[cellIndex].end()) {
77 _globalToLocalIndex[cellIndex].insert({cellIndexNeighbor, _globalToLocalIndex[cellIndex].size()});
78 }
79 auto secondCellIndexInFirst = _globalToLocalIndex[cellIndex].at(cellIndexNeighbor);
80 _neighborLists[cellIndex][secondCellIndexInFirst][particleIndex].second.push_back(&j);
81 }

87 void SoAFunctorSingle(SoAView<SoAArraysType> soa, bool newton3) override {
88 if (soa.getNumberOfParticles() == 0) return;
89
90 auto **const __restrict__ ptrptr = soa.template begin<Particle::AttributeNames::ptr>();
91 double *const __restrict__ xptr = soa.template begin<Particle::AttributeNames::posX>();
92 double *const __restrict__ yptr = soa.template begin<Particle::AttributeNames::posY>();
93 double *const __restrict__ zptr = soa.template begin<Particle::AttributeNames::posZ>();
94
95 // index of cell1 is particleToCellMap of ptr1ptr, same for 2
96 auto cell = _particleToCellMap.at(ptrptr[0]).first;
97
98 auto iter = _globalToLocalIndex[cell].find(cell);
99 if (iter == _globalToLocalIndex[cell].end()) {
100 _globalToLocalIndex[cell].insert({cell, _globalToLocalIndex[cell].size()});
101 }
102 auto localCell2Index = _globalToLocalIndex[cell].at(cell);
103
104 auto &currentList = _neighborLists[cell][localCell2Index];
105
106 // iterating over particle indices and accessing currentList at index i works
107 // because the particles are iterated in the same order they are loaded in
108 // which is the same order they were initialized when building the aosNeighborList
109 size_t numPart = soa.getNumberOfParticles();
110 for (unsigned int i = 0; i < numPart; ++i) {
111 for (unsigned int j = i + 1; j < numPart; ++j) {
112 const double drx = xptr[i] - xptr[j];
113 const double dry = yptr[i] - yptr[j];
114 const double drz = zptr[i] - zptr[j];
115
116 const double drx2 = drx * drx;
117 const double dry2 = dry * dry;
118 const double drz2 = drz * drz;
119
120 const double dr2 = drx2 + dry2 + drz2;
121
122 if (dr2 < _cutoffskinsquared) {
123 currentList[i].second.push_back(ptrptr[j]);
124 if (not newton3) {
125 // we need this here, as SoAFunctorSingle will only be called once for both newton3=true and false.
126 currentList[j].second.push_back(ptrptr[i]);
127 }
128 }
129 }
130 }
