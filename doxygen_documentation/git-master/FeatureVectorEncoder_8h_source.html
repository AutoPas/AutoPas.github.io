119  [[nodiscard]] Eigen::VectorXd oneHotEncode(const FeatureVector &vec) const {
120  std::vector<double> data;
121  data.reserve(_oneHotDims);
122 
123  auto [discreteValues, continuousValues] = convertToTunable(vec);
124 
125  // discrete values are encoded using one-hot-encoding
126  for (size_t i = 0; i < tunableDiscreteDims; ++i) {
127  for (int listIndex = 0; listIndex < _discreteRestrictions[i]; ++listIndex) {
128  data.push_back(listIndex == discreteValues[i] ? 1. : 0.);
129  }
130  }

133  for (size_t i = 0; i < tunableContinuousDims; ++i) {
134  data.push_back(continuousValues[i]);
135  }

145  [[nodiscard]] FeatureVector oneHotDecode(const Eigen::VectorXd &vec) {
146  if (static_cast<size_t>(vec.size()) != _oneHotDims) {
147  utils::ExceptionHandler::exception("FeatureVectorEncoder.oneHotDecode: Expected size {}, got {}", _oneHotDims,
148  vec.size());
149  }
150 
151  size_t pos = 0;
152 
153  DiscreteDimensionType discreteValues;
154 
155  // extract each one-hot-encoded discrete option
156  for (size_t i = 0; i < tunableDiscreteDims; ++i) {
157  // for each dimension get index whose value equals 1.
158  std::optional<int> value = {};
159  for (int listIndex = 0; listIndex < _discreteRestrictions[i]; ++listIndex) {
160  if (vec[pos++] == 1.) {
161  if (value) {
162  utils::ExceptionHandler::exception(
163  "FeatureVectorEncoder.oneHotDecode: Vector encodes more than one option for tunable dimension {}. "
164  "(More than one value equals 1.)",
165  i);
166  }
167  value = listIndex;
168  }
169  }
170  if (not value) {
171  utils::ExceptionHandler::exception(
172  "FeatureVectorEncoder.oneHotDecode: Vector encodes no option for tunable dimension {}. (All values equal "
173  "0.)",
174  i);
175  }
176  discreteValues[i] = *value;
177  }
178 
179  ContinuousDimensionType continuousValues;
180  for (size_t i = 0; i < tunableContinuousDims; ++i) {
181  continuousValues[i] = vec[pos++];
182  }
183 
184  return convertFromTunable(discreteValues, continuousValues);
