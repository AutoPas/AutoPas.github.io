121 [[nodiscard]] Eigen::VectorXd oneHotEncode(const FeatureVector &vec) const {
122 std::vector<double> data;
123 data.reserve(_oneHotDims);
124
125 auto [discreteValues, continuousValues] = convertToTunable(vec);
126
127 // discrete values are encoded using one-hot-encoding
128 for (size_t i = 0; i < tunableDiscreteDims; ++i) {
129 for (int listIndex = 0; listIndex < _discreteRestrictions[i]; ++listIndex) {
130 data.push_back(listIndex == discreteValues[i] ? 1. : 0.);
131 }
132 }
133
134 // continuous values are simply copied
135 for (size_t i = 0; i < tunableContinuousDims; ++i) {
136 data.push_back(continuousValues[i]);
137 }
138
139 return Eigen::Map<Eigen::VectorXd>(data.data(), _oneHotDims);
140 }

147 [[nodiscard]] FeatureVector oneHotDecode(const Eigen::VectorXd &vec) {
148 if (static_cast<size_t>(vec.size()) != _oneHotDims) {
149 utils::ExceptionHandler::exception("FeatureVectorEncoder.oneHotDecode: Expected size {}, got {}", _oneHotDims,
150 vec.size());
151 }
152
153 size_t pos = 0;
154
155 DiscreteDimensionType discreteValues;
156
157 // extract each one-hot-encoded discrete option
158 for (size_t i = 0; i < tunableDiscreteDims; ++i) {
159 // for each dimension get index whose value equals 1.
160 std::optional<int> value = {};
161 for (int listIndex = 0; listIndex < _discreteRestrictions[i]; ++listIndex) {
162 if (vec[pos++] == 1.) {
163 if (value) {
164 utils::ExceptionHandler::exception(
165 "FeatureVectorEncoder.oneHotDecode: Vector encodes more than one option for tunable dimension {}. "
166 "(More than one value equals 1.)",
167 i);
168 }
169 value = listIndex;
170 }
171 }
172 if (not value) {
173 utils::ExceptionHandler::exception(
174 "FeatureVectorEncoder.oneHotDecode: Vector encodes no option for tunable dimension {}. (All values equal "
175 "0.)",
176 i);
177 }
178 discreteValues[i] = *value;
179 }
180
181 ContinuousDimensionType continuousValues;
182 for (size_t i = 0; i < tunableContinuousDims; ++i) {
183 continuousValues[i] = vec[pos++];
184 }
185
186 return convertFromTunable(discreteValues, continuousValues);
187 }

240 [[nodiscard]] std::vector<std::pair<Eigen::VectorXi, double>> clusterNeighboursManhattan1(
241 const Eigen::VectorXi &target) {
242 std::vector<std::pair<Eigen::VectorXi, double>> result;
243 // neighbours should contain #(possible values for each dimension) - #dimensions (initial vector is skipped once per
244 // dimension)
245 result.reserve(
246 std::accumulate(_discreteRestrictions.begin(), _discreteRestrictions.end(), -_discreteRestrictions.size()));
247
248 // for each dimension
249 for (int i = 0; i < target.size(); ++i) {
250 // initial value
251 auto init = target[i];
252
253 // for each possible value of that dimension
254 for (int x = 0; x < _discreteRestrictions[i]; ++x) {
255 // skip initial value
256 if (x != init) {
257 auto neighbour = target;
258 neighbour[i] = x;
259 result.emplace_back(std::move(neighbour), 1.);
260 }
261 }
262 }
263 return result;
264 }

272 [[nodiscard]] std::vector<std::pair<Eigen::VectorXi, double>> clusterNeighboursManhattan1Container(
273 const Eigen::VectorXi &target) {
274 std::vector<std::pair<Eigen::VectorXi, double>> result;
275 // neighbours should contain #(possible values for each dimension) - #dimensions (initial vector is skipped once per
276 // dimension)
277 result.reserve(
278 std::accumulate(_discreteRestrictions.begin(), _discreteRestrictions.end(), -_discreteRestrictions.size()));
279
280 auto targetContainer = std::get<0>(
281 _containerTraversalEstimatorOptions[target[static_cast<int>(DiscreteIndices::containerTraversalEstimator)]]);
282
283 // for each dimension
284 for (int i = 0; i < target.size(); ++i) {
285 // initial value
286 auto init = target[i];
287
288 // for each possible value of that dimension
289 for (int x = 0; x < _discreteRestrictions[i]; ++x) {
290 // skip initial value
291 if (x != init) {
292 auto neighbour = target;
293 neighbour[i] = x;
294 double weight = 1.;
295 // check if container changed
296 if (i == static_cast<int>(DiscreteIndices::containerTraversalEstimator)) {
297 auto xContainer = std::get<0>(_containerTraversalEstimatorOptions[x]);
298 if (targetContainer != xContainer) {
299 // assign lower weight
300 weight = 0.5;
301 }
302 }
303
304 result.emplace_back(std::move(neighbour), weight);
305 }
306 }
307 }
308 return result;

317 std::vector<FeatureVector> lhsSampleFeatures(size_t n, Random &rng) const {
318 // create n samples for each discrete dimension.
319 std::array<std::vector<size_t>, tunableDiscreteDims> lhsDiscreteSamples;
320 for (size_t i = 0; i < tunableDiscreteDims; ++i) {
321 lhsDiscreteSamples[i] = rng.uniformSample(0, _discreteRestrictions[i] - 1, n);
322 }
323
324 // create n samples for each continuous dimension.
325 std::array<std::vector<double>, tunableContinuousDims> lhsContinuousSamples;
326 for (size_t i = 0; i < tunableContinuousDims; ++i) {
327 lhsContinuousSamples[i] = _continuousRestrictions[i]->uniformSample(n, rng);
328 }
329
330 // create FeatureVectors from raw samples
331 std::vector<FeatureVector> result;
332 for (size_t i = 0; i < n; ++i) {
333 DiscreteDimensionType discreteValues;
334 for (size_t d = 0; d < tunableDiscreteDims; ++d) {
335 discreteValues[d] = lhsDiscreteSamples[d][i];
336 }
337
338 ContinuousDimensionType continuousValues;
339 for (size_t c = 0; c < tunableContinuousDims; ++c) {
340 continuousValues[c] = lhsContinuousSamples[c][i];
341 }
342
343 result.emplace_back(convertFromTunable(discreteValues, continuousValues));
344 }
345
346 return result;
347 }

408 [[nodiscard]] FeatureVector convertFromTunable(const DiscreteDimensionType &discreteValues,
