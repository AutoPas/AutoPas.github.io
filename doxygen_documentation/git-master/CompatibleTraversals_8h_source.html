110static inline const std::set<TraversalOption> &allCompatibleTraversals(ContainerOption containerOption) {
111 switch (containerOption) {
112 case ContainerOption::linkedCells: {
113 return allLCCompatibleTraversals();
114 }
115 case ContainerOption::directSum: {
116 return allDSCompatibleTraversals();
117 }
118 case ContainerOption::verletClusterLists: {
119 return allVCLCompatibleTraversals();
120 }
121 case ContainerOption::verletLists: {
122 return allVLCompatibleTraversals();
123 }
124 case ContainerOption::verletListsCells: {
125 return allVLCCompatibleTraversals();
126 }
127 case ContainerOption::varVerletListsAsBuild: {
128 return allVarVLAsBuildCompatibleTraversals();
129 }
130 case ContainerOption::linkedCellsReferences: {
131 return allRLCCompatibleTraversals();
132 }
133 case ContainerOption::pairwiseVerletLists: {
134 return allVLPCompatibleTraversals();
135 }
136 }
137
138 autopas::utils::ExceptionHandler::exception("CompatibleTraversals: Unknown container option {}!",
139 containerOption.to_string());
140
141 static const std::set<TraversalOption> s{};
142 return s;
143}

150static inline std::set<ContainerOption> allCompatibleContainers(TraversalOption traversalOption) {
151 std::set<ContainerOption> result{};
152
153 for (const auto &container : ContainerOption::getAllOptions()) {
154 auto allCompatible = compatibleTraversals::allCompatibleTraversals(container);
155 if (allCompatible.find(traversalOption) != allCompatible.end()) {
156 result.insert(container);
157 }
158 }
159
160 return result;
161}
