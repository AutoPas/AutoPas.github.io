27std::set<TraversalOption> filterAllOptions(const std::string &prefix) {
28 const auto allOpts = TraversalOption::getAllOptions();
29 std::set<TraversalOption> retSet;
30 // If the lambda condition holds (=prefix matches) copy the option in the return set.
31 std::copy_if(allOpts.begin(), allOpts.end(), std::inserter(retSet, retSet.begin()),
32 [&](const auto &option) { return option.to_string().substr(0, prefix.length()) == prefix; });
33 return retSet;
34}

207[[maybe_unused]] static std::set<ContainerOption> allCompatibleContainers(TraversalOption traversalOption) {
208 std::set<ContainerOption> result{};
209
210 for (const auto &container : ContainerOption::getAllOptions()) {
211 auto allCompatible = compatibleTraversals::allCompatibleTraversals(container);
212 if (allCompatible.find(traversalOption) != allCompatible.end()) {
213 result.insert(container);
214 }
215 }
216
217 return result;
218}
