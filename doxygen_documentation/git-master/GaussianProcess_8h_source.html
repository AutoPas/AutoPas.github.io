149 [[nodiscard]] double predictMean(const Vector &input) const {
150 if (static_cast<size_t>(input.size()) != _dims) {
151 utils::ExceptionHandler::exception(
152 "GaussianProcess.predictMean: size of input {} does not match specified dimensions {}", input.size(), _dims);
153 }
154
155 double result = 0.;
156 if (_inputs.empty()) {
157 // no evidence
158 for (const auto &hyper : _hypers) {
159 result += hyper.score * hyper.mean;
160 }
161 } else {
162 for (const auto &hyper : _hypers) {
163 result += hyper.score * (hyper.mean + kernelVector(input, hyper.theta, hyper.dimScales).dot(hyper.weights));
164 }
165 }
166
167 return result;
168 }

187 [[nodiscard]] double predictVar(const Vector &input) const {
188 if (static_cast<size_t>(input.size()) != _dims) {
189 utils::ExceptionHandler::exception(
190 "GaussianProcess.predictVar: size of input {} does not match specified dimensions {}", input.size(), _dims);
191 }
192
193 double result = 0.;
194 if (_inputs.empty()) {
195 // no evidence
196 return getDefaultVar();
197 } else {
198 for (const auto &hyper : _hypers) {
199 auto kVec = kernelVector(input, hyper.theta, hyper.dimScales);
200 result += hyper.score * (kernel(input, input, hyper.theta, hyper.dimScales) - kVec.dot(hyper.covMatInv * kVec));
201 }
202 }
203
204 return result;

277 [[nodiscard]] static std::tuple<std::vector<double>, std::vector<double>, std::vector<Eigen::VectorXd>>
278 generateHyperparameterSamples(size_t sampleSize, Random &rng, size_t dims, double sigma, double evidenceMinValue,
279 double evidenceMaxValue) {
280 // range of mean
281 // inside bounds of evidence outputs
282 NumberInterval<double> meanRange(evidenceMinValue, evidenceMaxValue);
283 // range of theta
284 // max sample stddev: (max - min)
285 // max stddev from zero: abs(min) & abs(max)
286 double thetaMax = std::pow(
287 std::max({evidenceMaxValue - evidenceMinValue, std::abs(evidenceMinValue), std::abs(evidenceMaxValue)}), 2);
288 // at least sigma
289 thetaMax = std::max(thetaMax, sigma);
290 NumberInterval<double> thetaRange(sigma, thetaMax);
291 // range of dimScale
292 // Assuming most distances are greater equal 1.
293 // For a dimScale d > 5 + ln(thetaMax): theta * exp(-d r) < 1%.
294 // So choosing a greater dimScale may lead to many kernels close to zero.
295 // But if needed the upper bound can be increased.
296 NumberInterval<double> dimScaleRange(0., 5. + std::max(0., std::log(thetaMax)));
297
298 // generate mean
299 auto sample_means = meanRange.uniformSample(sampleSize, rng);
300
301 // generate theta
302 auto sample_thetas = thetaRange.uniformSample(sampleSize, rng);
303
304 // generate dimScale
305 std::vector<std::vector<double>> sample_dimScaleData;
306 sample_dimScaleData.reserve(dims);
307 for (size_t d = 0; d < dims; ++d) {
308 sample_dimScaleData.emplace_back(dimScaleRange.uniformSample(sampleSize, rng));
309 }
310 // convert dimScales to Vectors
311 std::vector<Eigen::VectorXd> sample_dimScales;
312 sample_dimScales.reserve(sampleSize);
313 for (size_t t = 0; t < sampleSize; ++t) {
314 std::vector<double> dimScaleData;
315 dimScaleData.reserve(dims);
316 for (size_t d = 0; d < dims; ++d) {
317 dimScaleData.push_back(sample_dimScaleData[d][t]);
318 }
319 sample_dimScales.emplace_back(utils::Math::makeVectorXd(dimScaleData));
320 }
321
322 return std::make_tuple(sample_means, sample_thetas, sample_dimScales);
323 }

364 void normalizeHyperparameters() {
365 // sort by score
366 std::sort(_hypers.begin(), _hypers.end(),
367 [](const GaussianHyperparameters &h1, const GaussianHyperparameters &h2) { return h1.score > h2.score; });
368
369 // only keep hp_size highest scores
370 if (_hypers.size() > hp_size) {
371 _hypers.erase(_hypers.begin() + hp_size, _hypers.end());
372 }
373
374 // normalize scores
375 double scoreSum = 0.;
376 for (auto &hyper : _hypers) {
377 scoreSum += hyper.score;
378 }
379 if (scoreSum > 0) {
380 for (auto &hyper : _hypers) {
381 hyper.score /= scoreSum;
382 }
383 } else {
384 // all scores are 0
385 double uniformProbability = 1. / _hypers.size();
386 for (auto &hyper : _hypers) {
387 hyper.score = uniformProbability;
388 }
389 }

428 [[nodiscard]] static inline double kernel(const Vector &input1, const Vector &input2, double theta,
429 const Eigen::VectorXd &dimScale) {
430 double dot = 0;
431 for (int i = 0; i < input1.size(); ++i) {
432 double dist = input1[i] - input2[i];
433 dist *= dist * dimScale[i];
434 dot += dist;
435 }
436 return theta * std::exp(-dot);
437 }
