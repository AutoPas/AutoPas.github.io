124  void AoSFunctor(Particle &i, Particle &j, bool newton3) override {
125  if (i.isDummy() or j.isDummy()) {
126  return;
127  }
128  auto sigmasquare = _sigmaSquareAoS;
129  auto epsilon24 = _epsilon24AoS;
130  auto shift6 = _shift6AoS;
131  if constexpr (useMixing) {
132  sigmasquare = _PPLibrary->mixingSigmaSquare(i.getTypeId(), j.getTypeId());
133  epsilon24 = _PPLibrary->mixing24Epsilon(i.getTypeId(), j.getTypeId());
134  if constexpr (applyShift) {
135  shift6 = _PPLibrary->mixingShift6(i.getTypeId(), j.getTypeId());
136  }
137  }
138  auto dr = utils::ArrayMath::sub(i.getR(), j.getR());
139  double dr2 = utils::ArrayMath::dot(dr, dr);
140 
141  if (dr2 > _cutoffsquareAoS) {
142  return;
143  }
144 
145  double invdr2 = 1. / dr2;
146  double lj6 = sigmasquare * invdr2;
147  lj6 = lj6 * lj6 * lj6;
148  double lj12 = lj6 * lj6;
149  double lj12m6 = lj12 - lj6;
150  double fac = epsilon24 * (lj12 + lj12m6) * invdr2;
151  auto f = utils::ArrayMath::mulScalar(dr, fac);
152  i.addF(f);
153  if (newton3) {
154  // only if we use newton 3 here, we want to
155  j.subF(f);
156  }
157  if (calculateGlobals) {
158  auto virial = utils::ArrayMath::mul(dr, f);
159  double upot = epsilon24 * lj12m6 + shift6;
160 
161  const int threadnum = autopas_get_thread_num();
162  // for non-newton3 the division is in the post-processing step.
163  if (newton3) {
164  upot *= 0.5;
165  virial = utils::ArrayMath::mulScalar(virial, (double)0.5);
166  }
167  if (i.isOwned()) {
168  _aosThreadData[threadnum].upotSum += upot;
169  _aosThreadData[threadnum].virialSum = utils::ArrayMath::add(_aosThreadData[threadnum].virialSum, virial);
170  }
171  // for non-newton3 the second particle will be considered in a separate calculation
172  if (newton3 and j.isOwned()) {
173  _aosThreadData[threadnum].upotSum += upot;
174  _aosThreadData[threadnum].virialSum = utils::ArrayMath::add(_aosThreadData[threadnum].virialSum, virial);
175  }
176  }

512 
513  const __m256d drx = _mm256_sub_pd(x1, x2);
514  const __m256d dry = _mm256_sub_pd(y1, y2);
515  const __m256d drz = _mm256_sub_pd(z1, z2);
516 
517  const __m256d drx2 = _mm256_mul_pd(drx, drx);
518  const __m256d dry2 = _mm256_mul_pd(dry, dry);
519  const __m256d drz2 = _mm256_mul_pd(drz, drz);
520 
521  const __m256d dr2PART = _mm256_add_pd(drx2, dry2);
522  const __m256d dr2 = _mm256_add_pd(dr2PART, drz2);
523 
524  // _CMP_LE_OS == Less-Equal-then (ordered, signaling)
525  // signaling = throw error if NaN is encountered
526  // dr2 <= _cutoffsquare ? 0xFFFFFFFFFFFFFFFF : 0
527  const __m256d cutoffMask = _mm256_cmp_pd(dr2, _cutoffsquare, _CMP_LE_OS);
528 
529  // This requires that dummy is zero (otherwise when loading using a mask the owned state will not be zero)
530  const __m256i ownedStateJ = remainderIsMasked
531  ? _mm256_castpd_si256(_mm256_maskload_pd(
532  reinterpret_cast<double const *>(&ownedStatePtr2[j]), _masks[rest - 1]))
533  : _mm256_loadu_si256(reinterpret_cast<const __m256i *>(&ownedStatePtr2[j]));
534  // This requires that dummy is the first entry in OwnershipState!
535  const __m256d dummyMask = _mm256_cmp_pd(_mm256_castsi256_pd(ownedStateJ), _zero, _CMP_NEQ_UQ);
536  const __m256d cutoffDummyMask = _mm256_and_pd(cutoffMask, dummyMask);
537 
538  // if everything is masked away return from this function.
539  if (_mm256_movemask_pd(cutoffDummyMask) == 0) {
540  return;
541  }
542 
543  const __m256d invdr2 = _mm256_div_pd(_one, dr2);
544  const __m256d lj2 = _mm256_mul_pd(sigmaSquares, invdr2);
545  const __m256d lj4 = _mm256_mul_pd(lj2, lj2);
546  const __m256d lj6 = _mm256_mul_pd(lj2, lj4);
547  const __m256d lj12 = _mm256_mul_pd(lj6, lj6);
548  const __m256d lj12m6 = _mm256_sub_pd(lj12, lj6);
549  const __m256d lj12m6alj12 = _mm256_add_pd(lj12m6, lj12);
550  const __m256d lj12m6alj12e = _mm256_mul_pd(lj12m6alj12, epsilon24s);
551  const __m256d fac = _mm256_mul_pd(lj12m6alj12e, invdr2);
552 
553  const __m256d facMasked =
554  remainderIsMasked ? _mm256_and_pd(fac, _mm256_and_pd(cutoffDummyMask, _mm256_castsi256_pd(_masks[rest - 1])))
555  : _mm256_and_pd(fac, cutoffDummyMask);
556 
557  const __m256d fx = _mm256_mul_pd(drx, facMasked);
558  const __m256d fy = _mm256_mul_pd(dry, facMasked);
559  const __m256d fz = _mm256_mul_pd(drz, facMasked);
560 
561  fxacc = _mm256_add_pd(fxacc, fx);
562  fyacc = _mm256_add_pd(fyacc, fy);
563  fzacc = _mm256_add_pd(fzacc, fz);
564 
565  // if newton 3 is used subtract fD from particle j
566  if (newton3) {
567  const __m256d fx2 =
568  remainderIsMasked ? _mm256_maskload_pd(&fx2ptr[j], _masks[rest - 1]) : _mm256_loadu_pd(&fx2ptr[j]);
569  const __m256d fy2 =
570  remainderIsMasked ? _mm256_maskload_pd(&fy2ptr[j], _masks[rest - 1]) : _mm256_loadu_pd(&fy2ptr[j]);
571  const __m256d fz2 =
572  remainderIsMasked ? _mm256_maskload_pd(&fz2ptr[j], _masks[rest - 1]) : _mm256_loadu_pd(&fz2ptr[j]);
573 
574  const __m256d fx2new = _mm256_sub_pd(fx2, fx);
575  const __m256d fy2new = _mm256_sub_pd(fy2, fy);
576  const __m256d fz2new = _mm256_sub_pd(fz2, fz);
577 
578  remainderIsMasked ? _mm256_maskstore_pd(&fx2ptr[j], _masks[rest - 1], fx2new)
579  : _mm256_storeu_pd(&fx2ptr[j], fx2new);
580  remainderIsMasked ? _mm256_maskstore_pd(&fy2ptr[j], _masks[rest - 1], fy2new)
581  : _mm256_storeu_pd(&fy2ptr[j], fy2new);
582  remainderIsMasked ? _mm256_maskstore_pd(&fz2ptr[j], _masks[rest - 1], fz2new)
583  : _mm256_storeu_pd(&fz2ptr[j], fz2new);
584  }
585 
586  if (calculateGlobals) {
587  // Global Virial
588  const __m256d virialX = _mm256_mul_pd(fx, drx);
589  const __m256d virialY = _mm256_mul_pd(fy, dry);
590  const __m256d virialZ = _mm256_mul_pd(fz, drz);
591 
592  // Global Potential
593  const __m256d upot = wrapperFMA(epsilon24s, lj12m6, shift6s);
594 
595  const __m256d upotMasked =
596  remainderIsMasked ? _mm256_and_pd(upot, _mm256_and_pd(cutoffDummyMask, _mm256_castsi256_pd(_masks[rest - 1])))
597  : _mm256_and_pd(upot, cutoffDummyMask);
598 
599  __m256d ownedMaskI =
600  _mm256_cmp_pd(_mm256_castsi256_pd(ownedStateI), _mm256_castsi256_pd(_ownedStateOwnedMM256i), _CMP_EQ_UQ);
601  __m256d energyFactor = _mm256_blendv_pd(_zero, _one, ownedMaskI);
602  if constexpr (newton3) {
603  __m256d ownedMaskJ =
604  _mm256_cmp_pd(_mm256_castsi256_pd(ownedStateJ), _mm256_castsi256_pd(_ownedStateOwnedMM256i), _CMP_EQ_UQ);
605  energyFactor = _mm256_add_pd(energyFactor, _mm256_blendv_pd(_zero, _one, ownedMaskJ));
606  }
607  *upotSum = wrapperFMA(energyFactor, upotMasked, *upotSum);
608  *virialSumX = wrapperFMA(energyFactor, virialX, *virialSumX);
609  *virialSumY = wrapperFMA(energyFactor, virialY, *virialSumY);
610  *virialSumZ = wrapperFMA(energyFactor, virialZ, *virialSumZ);
611  }
612  }
613 #endif

665  const auto y1 = _mm256_broadcast_sd(&yptr[indexFirst]);
666  const auto z1 = _mm256_broadcast_sd(&zptr[indexFirst]);
667  // ownedStatePtr contains int64_t, so we broadcast these to make an __m256i.
668  // _mm256_set1_epi64x broadcasts a 64-bit integer, we use this instruction to have 4 values!
669  __m256i ownedStateI = _mm256_set1_epi64x(static_cast<int64_t>(ownedStatePtr[indexFirst]));

672  alignas(64) std::array<double, vecLength> y2tmp{};

674  alignas(64) std::array<double, vecLength> fx2tmp{};
675  alignas(64) std::array<double, vecLength> fy2tmp{};
676  alignas(64) std::array<double, vecLength> fz2tmp{};

680  // load 4 neighbors
681  size_t j = 0;
682  // If b is a power of 2 the following holds:
683  // a & ~(b -1) == a - (a mod b)
684  for (; j < (neighborList.size() & ~(vecLength - 1)); j += vecLength) {
685  // AVX2 variant:
686  // create buffer for 4 interaction particles
687  // and fill buffers via gathering
688  // const __m256d x2tmp = _mm256_i64gather_pd(&xptr[j], _vindex, 1);
689  // const __m256d y2tmp = _mm256_i64gather_pd(&yptr[j], _vindex, 1);
690  // const __m256d z2tmp = _mm256_i64gather_pd(&zptr[j], _vindex, 1);
691  // const __m256d fx2tmp = _mm256_i64gather_pd(&fxptr[j], _vindex, 1);

693  // const __m256d fz2tmp = _mm256_i64gather_pd(&fzptr[j], _vindex, 1);
694  // const __m256i typeID2tmp = _mm256_i64gather_epi64(&typeIDptr[j], _vindex, 1);
695 
696  for (size_t vecIndex = 0; vecIndex < vecLength; ++vecIndex) {
697  x2tmp[vecIndex] = xptr[neighborList[j + vecIndex]];
698  y2tmp[vecIndex] = yptr[neighborList[j + vecIndex]];
699  z2tmp[vecIndex] = zptr[neighborList[j + vecIndex]];
700  fx2tmp[vecIndex] = fxptr[neighborList[j + vecIndex]];
701  fy2tmp[vecIndex] = fyptr[neighborList[j + vecIndex]];
702  fz2tmp[vecIndex] = fzptr[neighborList[j + vecIndex]];
703  typeID2tmp[vecIndex] = typeIDptr[neighborList[j + vecIndex]];
704  ownedStates2tmp[vecIndex] = ownedStatePtr[neighborList[j + vecIndex]];
705  }
706 
707  SoAKernel<newton3, false>(0, ownedStateI, reinterpret_cast<const int64_t *>(ownedStates2tmp.data()), x1, y1, z1,
708  x2tmp.data(), y2tmp.data(), z2tmp.data(), fx2tmp.data(), fy2tmp.data(), fz2tmp.data(),
709  &typeIDptr[indexFirst], typeID2tmp.data(), fxacc, fyacc, fzacc, &virialSumX,
710  &virialSumY, &virialSumZ, &upotSum, 0);
711 
712  if (newton3) {
713  for (size_t vecIndex = 0; vecIndex < vecLength; ++vecIndex) {
714  fxptr[neighborList[j + vecIndex]] = fx2tmp[vecIndex];
715  fyptr[neighborList[j + vecIndex]] = fy2tmp[vecIndex];
716  fzptr[neighborList[j + vecIndex]] = fz2tmp[vecIndex];
717  }
718  }
719  }
720  // If b is a power of 2 the following holds:
721  // a & (b -1) == a mod b
722  const int rest = (int)(neighborList.size() & (vecLength - 1));
723  if (rest > 0) {
724  // AVX2 variant:
725  // create buffer for 4 interaction particles
726  // and fill buffers via gathering
727  // TODO: use masked load because there will not be enough data left for the whole gather
728  // const __m256d x2tmp = _mm256_i64gather_pd(&xptr[j], _vindex, 1);
729  // const __m256d y2tmp = _mm256_i64gather_pd(&yptr[j], _vindex, 1);
730  // const __m256d z2tmp = _mm256_i64gather_pd(&zptr[j], _vindex, 1);
731  // const __m256d fx2tmp = _mm256_i64gather_pd(&fxptr[j], _vindex, 1);
732  // const __m256d fy2tmp = _mm256_i64gather_pd(&fyptr[j], _vindex, 1);
733  // const __m256d fz2tmp = _mm256_i64gather_pd(&fzptr[j], _vindex, 1);
734  // const __m256d typeID2tmp = _mm256_i64gather_pd(&typeIDptr[j], _vindex, 1);
735 
736  for (size_t vecIndex = 0; vecIndex < rest; ++vecIndex) {
737  x2tmp[vecIndex] = xptr[neighborList[j + vecIndex]];
738  y2tmp[vecIndex] = yptr[neighborList[j + vecIndex]];
739  z2tmp[vecIndex] = zptr[neighborList[j + vecIndex]];
740  fx2tmp[vecIndex] = fxptr[neighborList[j + vecIndex]];
741  fy2tmp[vecIndex] = fyptr[neighborList[j + vecIndex]];
742  fz2tmp[vecIndex] = fzptr[neighborList[j + vecIndex]];
743  typeID2tmp[vecIndex] = typeIDptr[neighborList[j + vecIndex]];
744  ownedStates2tmp[vecIndex] = ownedStatePtr[neighborList[j + vecIndex]];

748  x2tmp.data(), y2tmp.data(), z2tmp.data(), fx2tmp.data(), fy2tmp.data(), fz2tmp.data(),
749  &typeIDptr[indexFirst], typeID2tmp.data(), fxacc, fyacc, fzacc, &virialSumX, &virialSumY,
750  &virialSumZ, &upotSum, rest);
751 
752  if (newton3) {
753  for (size_t vecIndex = 0; vecIndex < rest; ++vecIndex) {
754  fxptr[neighborList[j + vecIndex]] = fx2tmp[vecIndex];
755  fyptr[neighborList[j + vecIndex]] = fy2tmp[vecIndex];
756  fzptr[neighborList[j + vecIndex]] = fz2tmp[vecIndex];
757  }
758  }
759  }

761  // horizontally reduce fDacc to sumfD
762  const __m256d hSumfxfy = _mm256_hadd_pd(fxacc, fyacc);
763  const __m256d hSumfz = _mm256_hadd_pd(fzacc, fzacc);
764

766  const __m128d hSumfzLow = _mm256_extractf128_pd(hSumfz, 0);

775  const double sumfy = sumfxfyVEC[1];
776  const double sumfz = _mm_cvtsd_f64(sumfzVEC);

779  fyptr[indexFirst] += sumfy;
780  fzptr[indexFirst] += sumfz;

786  const __m256d hSumVirialxy = _mm256_hadd_pd(virialSumX, virialSumY);

788  const __m128d hSumVirialxyHigh = _mm256_extractf128_pd(hSumVirialxy, 1);

798  double globals[4];

800  _mm_store_pd(&globals[2], hSumVirialzUpotVec);

803  // we assume newton3 to be enabled in this function call, thus we multiply by two if the value of newton3 is
804  // false, since for newton3 disabled we divide by two later on.
805  factor *= newton3 ? .5 : 1.;
806  // In case we have a non-cell-wise owned state, we have multiplied everything by two, so we divide it by 2 again.
807  _aosThreadData[threadnum].virialSum[0] += globals[0] * factor;
808  _aosThreadData[threadnum].virialSum[1] += globals[1] * factor;
809  _aosThreadData[threadnum].virialSum[2] += globals[2] * factor;

811  }

813 #endif // __AVX__

879  void endTraversal(bool newton3) override {
880  if (_postProcessed) {
881  throw utils::ExceptionHandler::AutoPasException(
882  "Already postprocessed, endTraversal(bool newton3) was called twice without calling initTraversal().");
883  }
884 
885  if (calculateGlobals) {
886  for (size_t i = 0; i < _aosThreadData.size(); ++i) {
887  _upotSum += _aosThreadData[i].upotSum;
888  _virialSum = utils::ArrayMath::add(_virialSum, _aosThreadData[i].virialSum);
889  }
890  if (not newton3) {
891  // if the newton3 optimization is disabled we have added every energy contribution twice, so we divide by 2
892  // here.
893  _upotSum *= 0.5;
894  _virialSum = utils::ArrayMath::mulScalar(_virialSum, 0.5);
895  }
896  // we have always calculated 6*upot, so we divide by 6 here!
897  _upotSum /= 6.;
898  _postProcessed = true;
899  }
