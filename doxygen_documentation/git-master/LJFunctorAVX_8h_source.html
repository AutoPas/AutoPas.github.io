124 inline void AoSFunctor(Particle &i, Particle &j, bool newton3) final {
125 using namespace autopas::utils::ArrayMath::literals;
126 if (i.isDummy() or j.isDummy()) {
127 return;
128 }
129 auto sigmaSquared = _sigmaSquaredAoS;
130 auto epsilon24 = _epsilon24AoS;
131 auto shift6 = _shift6AoS;
132 if constexpr (useMixing) {
133 sigmaSquared = _PPLibrary->getMixingSigmaSquared(i.getTypeId(), j.getTypeId());
134 epsilon24 = _PPLibrary->getMixing24Epsilon(i.getTypeId(), j.getTypeId());
135 if constexpr (applyShift) {
136 shift6 = _PPLibrary->getMixingShift6(i.getTypeId(), j.getTypeId());
137 }
138 }
139 auto dr = i.getR() - j.getR();
140 double dr2 = autopas::utils::ArrayMath::dot(dr, dr);
141
142 if (dr2 > _cutoffSquaredAoS) {
143 return;
144 }
145
146 double invdr2 = 1. / dr2;
147 double lj6 = sigmaSquared * invdr2;
148 lj6 = lj6 * lj6 * lj6;
149 double lj12 = lj6 * lj6;
150 double lj12m6 = lj12 - lj6;
151 double fac = epsilon24 * (lj12 + lj12m6) * invdr2;
152 auto f = dr * fac;
153 i.addF(f);
154 if (newton3) {
155 // only if we use newton 3 here, we want to
156 j.subF(f);
157 }
158 if (calculateGlobals) {
159 // We always add the full contribution for each owned particle and divide the sums by 2 in endTraversal().
160 // Potential energy has an additional factor of 6, which is also handled in endTraversal().
161
162 auto virial = dr * f;
163 double potentialEnergy6 = epsilon24 * lj12m6 + shift6;
164
165 const int threadnum = autopas::autopas_get_thread_num();
166 if (i.isOwned()) {
167 _aosThreadData[threadnum].potentialEnergySum += potentialEnergy6;
168 _aosThreadData[threadnum].virialSum += virial;
169 }
170 // for non-newton3 the second particle will be considered in a separate calculation
171 if (newton3 and j.isOwned()) {
172 _aosThreadData[threadnum].potentialEnergySum += potentialEnergy6;
173 _aosThreadData[threadnum].virialSum += virial;
174 }
175 }

469 inline void SoAKernel(const size_t j, const __m256i ownedStateI, const int64_t *const __restrict ownedStatePtr2,
470 const __m256d &x1, const __m256d &y1, const __m256d &z1, const double *const __restrict x2ptr,
471 const double *const __restrict y2ptr, const double *const __restrict z2ptr,
472 double *const __restrict fx2ptr, double *const __restrict fy2ptr,
473 double *const __restrict fz2ptr, const size_t *const typeID1ptr, const size_t *const typeID2ptr,
474 __m256d &fxacc, __m256d &fyacc, __m256d &fzacc, __m256d *virialSumX, __m256d *virialSumY,
475 __m256d *virialSumZ, __m256d *potentialEnergySum, const unsigned int rest = 0) {
476 __m256d epsilon24s = _epsilon24;
477 __m256d sigmaSquareds = _sigmaSquared;
478 __m256d shift6s = _shift6;
479 if (useMixing) {
480 // the first argument for set lands in the last bits of the register
481 epsilon24s = _mm256_set_pd(
482 not remainderIsMasked or rest > 3 ? _PPLibrary->getMixing24Epsilon(*typeID1ptr, *(typeID2ptr + 3)) : 0,
483 not remainderIsMasked or rest > 2 ? _PPLibrary->getMixing24Epsilon(*typeID1ptr, *(typeID2ptr + 2)) : 0,
484 not remainderIsMasked or rest > 1 ? _PPLibrary->getMixing24Epsilon(*typeID1ptr, *(typeID2ptr + 1)) : 0,
485 _PPLibrary->getMixing24Epsilon(*typeID1ptr, *(typeID2ptr + 0)));
486 sigmaSquareds = _mm256_set_pd(
487 not remainderIsMasked or rest > 3 ? _PPLibrary->getMixingSigmaSquared(*typeID1ptr, *(typeID2ptr + 3)) : 0,
488 not remainderIsMasked or rest > 2 ? _PPLibrary->getMixingSigmaSquared(*typeID1ptr, *(typeID2ptr + 2)) : 0,
489 not remainderIsMasked or rest > 1 ? _PPLibrary->getMixingSigmaSquared(*typeID1ptr, *(typeID2ptr + 1)) : 0,
490 _PPLibrary->getMixingSigmaSquared(*typeID1ptr, *(typeID2ptr + 0)));
491 if constexpr (applyShift) {
492 shift6s = _mm256_set_pd(
493 (not remainderIsMasked or rest > 3) ? _PPLibrary->getMixingShift6(*typeID1ptr, *(typeID2ptr + 3)) : 0,
494 (not remainderIsMasked or rest > 2) ? _PPLibrary->getMixingShift6(*typeID1ptr, *(typeID2ptr + 2)) : 0,
495 (not remainderIsMasked or rest > 1) ? _PPLibrary->getMixingShift6(*typeID1ptr, *(typeID2ptr + 1)) : 0,
496 _PPLibrary->getMixingShift6(*typeID1ptr, *(typeID2ptr + 0)));
497 }
498 }
499
500 const __m256d x2 = remainderIsMasked ? _mm256_maskload_pd(&x2ptr[j], _masks[rest - 1]) : _mm256_loadu_pd(&x2ptr[j]);
501 const __m256d y2 = remainderIsMasked ? _mm256_maskload_pd(&y2ptr[j], _masks[rest - 1]) : _mm256_loadu_pd(&y2ptr[j]);
502 const __m256d z2 = remainderIsMasked ? _mm256_maskload_pd(&z2ptr[j], _masks[rest - 1]) : _mm256_loadu_pd(&z2ptr[j]);
503
504 const __m256d drx = _mm256_sub_pd(x1, x2);
505 const __m256d dry = _mm256_sub_pd(y1, y2);
506 const __m256d drz = _mm256_sub_pd(z1, z2);
507
508 const __m256d drx2 = _mm256_mul_pd(drx, drx);
509 const __m256d dry2 = _mm256_mul_pd(dry, dry);
510 const __m256d drz2 = _mm256_mul_pd(drz, drz);
511
512 const __m256d dr2PART = _mm256_add_pd(drx2, dry2);
513 const __m256d dr2 = _mm256_add_pd(dr2PART, drz2);
514
515 // _CMP_LE_OS == Less-Equal-then (ordered, signaling)
516 // signaling = throw error if NaN is encountered
517 // dr2 <= _cutoffSquared ? 0xFFFFFFFFFFFFFFFF : 0
518 const __m256d cutoffMask = _mm256_cmp_pd(dr2, _cutoffSquared, _CMP_LE_OS);
519
520 // This requires that dummy is zero (otherwise when loading using a mask the owned state will not be zero)
521 const __m256i ownedStateJ = remainderIsMasked
522 ? _mm256_castpd_si256(_mm256_maskload_pd(
523 reinterpret_cast<double const *>(&ownedStatePtr2[j]), _masks[rest - 1]))
524 : _mm256_loadu_si256(reinterpret_cast<const __m256i *>(&ownedStatePtr2[j]));
525 // This requires that dummy is the first entry in OwnershipState!
526 const __m256d dummyMask = _mm256_cmp_pd(_mm256_castsi256_pd(ownedStateJ), _zero, _CMP_NEQ_UQ);
527 const __m256d cutoffDummyMask = _mm256_and_pd(cutoffMask, dummyMask);
528
529 // if everything is masked away return from this function.
530 if (_mm256_movemask_pd(cutoffDummyMask) == 0) {
531 return;
532 }
533
534 const __m256d invdr2 = _mm256_div_pd(_one, dr2);
535 const __m256d lj2 = _mm256_mul_pd(sigmaSquareds, invdr2);
536 const __m256d lj4 = _mm256_mul_pd(lj2, lj2);
537 const __m256d lj6 = _mm256_mul_pd(lj2, lj4);
538 const __m256d lj12 = _mm256_mul_pd(lj6, lj6);
539 const __m256d lj12m6 = _mm256_sub_pd(lj12, lj6);
540 const __m256d lj12m6alj12 = _mm256_add_pd(lj12m6, lj12);
541 const __m256d lj12m6alj12e = _mm256_mul_pd(lj12m6alj12, epsilon24s);
542 const __m256d fac = _mm256_mul_pd(lj12m6alj12e, invdr2);
543
544 const __m256d facMasked =
545 remainderIsMasked ? _mm256_and_pd(fac, _mm256_and_pd(cutoffDummyMask, _mm256_castsi256_pd(_masks[rest - 1])))
546 : _mm256_and_pd(fac, cutoffDummyMask);
547
548 const __m256d fx = _mm256_mul_pd(drx, facMasked);
549 const __m256d fy = _mm256_mul_pd(dry, facMasked);
550 const __m256d fz = _mm256_mul_pd(drz, facMasked);
551
552 fxacc = _mm256_add_pd(fxacc, fx);
553 fyacc = _mm256_add_pd(fyacc, fy);
554 fzacc = _mm256_add_pd(fzacc, fz);
555
556 // if newton 3 is used subtract fD from particle j
557 if constexpr (newton3) {
558 const __m256d fx2 =
559 remainderIsMasked ? _mm256_maskload_pd(&fx2ptr[j], _masks[rest - 1]) : _mm256_loadu_pd(&fx2ptr[j]);
560 const __m256d fy2 =
561 remainderIsMasked ? _mm256_maskload_pd(&fy2ptr[j], _masks[rest - 1]) : _mm256_loadu_pd(&fy2ptr[j]);
562 const __m256d fz2 =
563 remainderIsMasked ? _mm256_maskload_pd(&fz2ptr[j], _masks[rest - 1]) : _mm256_loadu_pd(&fz2ptr[j]);
564
565 const __m256d fx2new = _mm256_sub_pd(fx2, fx);
566 const __m256d fy2new = _mm256_sub_pd(fy2, fy);
567 const __m256d fz2new = _mm256_sub_pd(fz2, fz);
568
569 remainderIsMasked ? _mm256_maskstore_pd(&fx2ptr[j], _masks[rest - 1], fx2new)
570 : _mm256_storeu_pd(&fx2ptr[j], fx2new);
571 remainderIsMasked ? _mm256_maskstore_pd(&fy2ptr[j], _masks[rest - 1], fy2new)
572 : _mm256_storeu_pd(&fy2ptr[j], fy2new);
573 remainderIsMasked ? _mm256_maskstore_pd(&fz2ptr[j], _masks[rest - 1], fz2new)
574 : _mm256_storeu_pd(&fz2ptr[j], fz2new);
575 }
576
577 if constexpr (calculateGlobals) {
578 // Global Virial
579 const __m256d virialX = _mm256_mul_pd(fx, drx);
580 const __m256d virialY = _mm256_mul_pd(fy, dry);
581 const __m256d virialZ = _mm256_mul_pd(fz, drz);
582
583 // Global Potential
584 const __m256d potentialEnergy = wrapperFMA(epsilon24s, lj12m6, shift6s);
585
586 const __m256d potentialEnergyMasked =
587 remainderIsMasked
588 ? _mm256_and_pd(potentialEnergy, _mm256_and_pd(cutoffDummyMask, _mm256_castsi256_pd(_masks[rest - 1])))
589 : _mm256_and_pd(potentialEnergy, cutoffDummyMask);
590
591 __m256d ownedMaskI =
592 _mm256_cmp_pd(_mm256_castsi256_pd(ownedStateI), _mm256_castsi256_pd(_ownedStateOwnedMM256i), _CMP_EQ_UQ);
593 __m256d energyFactor = _mm256_blendv_pd(_zero, _one, ownedMaskI);
594 if constexpr (newton3) {
595 __m256d ownedMaskJ =
596 _mm256_cmp_pd(_mm256_castsi256_pd(ownedStateJ), _mm256_castsi256_pd(_ownedStateOwnedMM256i), _CMP_EQ_UQ);
597 energyFactor = _mm256_add_pd(energyFactor, _mm256_blendv_pd(_zero, _one, ownedMaskJ));
598 }
599 *potentialEnergySum = wrapperFMA(energyFactor, potentialEnergyMasked, *potentialEnergySum);
600 *virialSumX = wrapperFMA(energyFactor, virialX, *virialSumX);
601 *virialSumY = wrapperFMA(energyFactor, virialY, *virialSumY);
602 *virialSumZ = wrapperFMA(energyFactor, virialZ, *virialSumZ);
603 }
604 }

627 template <bool newton3>
628 inline void SoAFunctorVerletImpl(autopas::SoAView<SoAArraysType> soa, const size_t indexFirst,
629 const std::vector<size_t, autopas::AlignedAllocator<size_t>> &neighborList) {
630#ifdef __AVX__
631 const auto *const __restrict ownedStatePtr = soa.template begin<Particle::AttributeNames::ownershipState>();
632 if (ownedStatePtr[indexFirst] == autopas::OwnershipState::dummy) {
633 return;
634 }
635
636 const auto *const __restrict xptr = soa.template begin<Particle::AttributeNames::posX>();
637 const auto *const __restrict yptr = soa.template begin<Particle::AttributeNames::posY>();
638 const auto *const __restrict zptr = soa.template begin<Particle::AttributeNames::posZ>();
639
640 auto *const __restrict fxptr = soa.template begin<Particle::AttributeNames::forceX>();
641 auto *const __restrict fyptr = soa.template begin<Particle::AttributeNames::forceY>();
642 auto *const __restrict fzptr = soa.template begin<Particle::AttributeNames::forceZ>();
643
644 const auto *const __restrict typeIDptr = soa.template begin<Particle::AttributeNames::typeId>();
645
646 // accumulators
647 __m256d virialSumX = _mm256_setzero_pd();
648 __m256d virialSumY = _mm256_setzero_pd();
649 __m256d virialSumZ = _mm256_setzero_pd();
650 __m256d potentialEnergySum = _mm256_setzero_pd();
651 __m256d fxacc = _mm256_setzero_pd();
652 __m256d fyacc = _mm256_setzero_pd();
653 __m256d fzacc = _mm256_setzero_pd();
654
655 // broadcast particle 1
656 const auto x1 = _mm256_broadcast_sd(&xptr[indexFirst]);
657 const auto y1 = _mm256_broadcast_sd(&yptr[indexFirst]);
658 const auto z1 = _mm256_broadcast_sd(&zptr[indexFirst]);
659 // ownedStatePtr contains int64_t, so we broadcast these to make an __m256i.
660 // _mm256_set1_epi64x broadcasts a 64-bit integer, we use this instruction to have 4 values!
661 __m256i ownedStateI = _mm256_set1_epi64x(static_cast<int64_t>(ownedStatePtr[indexFirst]));
662
663 alignas(64) std::array<double, vecLength> x2tmp{};
664 alignas(64) std::array<double, vecLength> y2tmp{};
665 alignas(64) std::array<double, vecLength> z2tmp{};
666 alignas(64) std::array<double, vecLength> fx2tmp{};
667 alignas(64) std::array<double, vecLength> fy2tmp{};
668 alignas(64) std::array<double, vecLength> fz2tmp{};
669 alignas(64) std::array<size_t, vecLength> typeID2tmp{};
670 alignas(64) std::array<autopas::OwnershipState, vecLength> ownedStates2tmp{};
671
672 // load 4 neighbors
673 size_t j = 0;
674 // Loop over all neighbors as long as we can fill full vectors
675 // (until `neighborList.size() - neighborList.size() % vecLength`)
676 //
677 // If b is a power of 2 the following holds:
678 // a & ~(b - 1) == a - (a mod b)
679 for (; j < (neighborList.size() & ~(vecLength - 1)); j += vecLength) {
680 // AVX2 variant:
681 // create buffer for 4 interaction particles
682 // and fill buffers via gathering
683 // const __m256d x2tmp = _mm256_i64gather_pd(&xptr[j], _vindex, 1);
684 // const __m256d y2tmp = _mm256_i64gather_pd(&yptr[j], _vindex, 1);
685 // const __m256d z2tmp = _mm256_i64gather_pd(&zptr[j], _vindex, 1);
686 // const __m256d fx2tmp = _mm256_i64gather_pd(&fxptr[j], _vindex, 1);
687 // const __m256d fy2tmp = _mm256_i64gather_pd(&fyptr[j], _vindex, 1);
688 // const __m256d fz2tmp = _mm256_i64gather_pd(&fzptr[j], _vindex, 1);
689 // const __m256i typeID2tmp = _mm256_i64gather_epi64(&typeIDptr[j], _vindex, 1);
690
691 for (size_t vecIndex = 0; vecIndex < vecLength; ++vecIndex) {
692 x2tmp[vecIndex] = xptr[neighborList[j + vecIndex]];
693 y2tmp[vecIndex] = yptr[neighborList[j + vecIndex]];
694 z2tmp[vecIndex] = zptr[neighborList[j + vecIndex]];
695 if constexpr (newton3) {
696 fx2tmp[vecIndex] = fxptr[neighborList[j + vecIndex]];
697 fy2tmp[vecIndex] = fyptr[neighborList[j + vecIndex]];
698 fz2tmp[vecIndex] = fzptr[neighborList[j + vecIndex]];
699 }
700 typeID2tmp[vecIndex] = typeIDptr[neighborList[j + vecIndex]];
701 ownedStates2tmp[vecIndex] = ownedStatePtr[neighborList[j + vecIndex]];
702 }
703
704 SoAKernel<newton3, false>(0, ownedStateI, reinterpret_cast<const int64_t *>(ownedStates2tmp.data()), x1, y1, z1,
705 x2tmp.data(), y2tmp.data(), z2tmp.data(), fx2tmp.data(), fy2tmp.data(), fz2tmp.data(),
706 &typeIDptr[indexFirst], typeID2tmp.data(), fxacc, fyacc, fzacc, &virialSumX,
707 &virialSumY, &virialSumZ, &potentialEnergySum, 0);
708
709 if constexpr (newton3) {
710 for (size_t vecIndex = 0; vecIndex < vecLength; ++vecIndex) {
711 fxptr[neighborList[j + vecIndex]] = fx2tmp[vecIndex];
712 fyptr[neighborList[j + vecIndex]] = fy2tmp[vecIndex];
713 fzptr[neighborList[j + vecIndex]] = fz2tmp[vecIndex];
714 }
715 }
716 }
717 // Remainder loop
718 // If b is a power of 2 the following holds:
719 // a & (b - 1) == a mod b
720 const auto rest = static_cast<int>(neighborList.size() & (vecLength - 1));
721 if (rest > 0) {
722 // AVX2 variant:
723 // create buffer for 4 interaction particles
724 // and fill buffers via gathering
725 // TODO: use masked load because there will not be enough data left for the whole gather
726 // const __m256d x2tmp = _mm256_i64gather_pd(&xptr[j], _vindex, 1);
727 // const __m256d y2tmp = _mm256_i64gather_pd(&yptr[j], _vindex, 1);
728 // const __m256d z2tmp = _mm256_i64gather_pd(&zptr[j], _vindex, 1);
729 // const __m256d fx2tmp = _mm256_i64gather_pd(&fxptr[j], _vindex, 1);
730 // const __m256d fy2tmp = _mm256_i64gather_pd(&fyptr[j], _vindex, 1);
731 // const __m256d fz2tmp = _mm256_i64gather_pd(&fzptr[j], _vindex, 1);
732 // const __m256d typeID2tmp = _mm256_i64gather_pd(&typeIDptr[j], _vindex, 1);
733
734 for (size_t vecIndex = 0; vecIndex < rest; ++vecIndex) {
735 x2tmp[vecIndex] = xptr[neighborList[j + vecIndex]];
736 y2tmp[vecIndex] = yptr[neighborList[j + vecIndex]];
737 z2tmp[vecIndex] = zptr[neighborList[j + vecIndex]];
738 // if newton3 is used we need to load f of particle j so the kernel can update it too
739 if constexpr (newton3) {
740 fx2tmp[vecIndex] = fxptr[neighborList[j + vecIndex]];
741 fy2tmp[vecIndex] = fyptr[neighborList[j + vecIndex]];
742 fz2tmp[vecIndex] = fzptr[neighborList[j + vecIndex]];
743 }
744 typeID2tmp[vecIndex] = typeIDptr[neighborList[j + vecIndex]];
745 ownedStates2tmp[vecIndex] = ownedStatePtr[neighborList[j + vecIndex]];
746 }
747
748 SoAKernel<newton3, true>(0, ownedStateI, reinterpret_cast<const int64_t *>(ownedStates2tmp.data()), x1, y1, z1,
749 x2tmp.data(), y2tmp.data(), z2tmp.data(), fx2tmp.data(), fy2tmp.data(), fz2tmp.data(),
750 &typeIDptr[indexFirst], typeID2tmp.data(), fxacc, fyacc, fzacc, &virialSumX, &virialSumY,
751 &virialSumZ, &potentialEnergySum, rest);
752
753 if constexpr (newton3) {
754 for (size_t vecIndex = 0; vecIndex < rest; ++vecIndex) {
755 fxptr[neighborList[j + vecIndex]] = fx2tmp[vecIndex];
756 fyptr[neighborList[j + vecIndex]] = fy2tmp[vecIndex];
757 fzptr[neighborList[j + vecIndex]] = fz2tmp[vecIndex];
758 }
759 }
760 }

762 // horizontally reduce fDacc to sumfD
763 const __m256d hSumfxfy = _mm256_hadd_pd(fxacc, fyacc);
764 const __m256d hSumfz = _mm256_hadd_pd(fzacc, fzacc);
765
766 const __m128d hSumfxfyLow = _mm256_extractf128_pd(hSumfxfy, 0);
767 const __m128d hSumfzLow = _mm256_extractf128_pd(hSumfz, 0);
768
769 const __m128d hSumfxfyHigh = _mm256_extractf128_pd(hSumfxfy, 1);
770 const __m128d hSumfzHigh = _mm256_extractf128_pd(hSumfz, 1);
771
772 const __m128d sumfxfyVEC = _mm_add_pd(hSumfxfyLow, hSumfxfyHigh);
773 const __m128d sumfzVEC = _mm_add_pd(hSumfzLow, hSumfzHigh);
774
775 const double sumfx = sumfxfyVEC[0];
776 const double sumfy = sumfxfyVEC[1];
777 const double sumfz = _mm_cvtsd_f64(sumfzVEC);
778
779 fxptr[indexFirst] += sumfx;
780 fyptr[indexFirst] += sumfy;
781 fzptr[indexFirst] += sumfz;
782
783 if constexpr (calculateGlobals) {
784 const int threadnum = autopas::autopas_get_thread_num();
785
786 // horizontally reduce virialSumX and virialSumY
787 const __m256d hSumVirialxy = _mm256_hadd_pd(virialSumX, virialSumY);
788 const __m128d hSumVirialxyLow = _mm256_extractf128_pd(hSumVirialxy, 0);
789 const __m128d hSumVirialxyHigh = _mm256_extractf128_pd(hSumVirialxy, 1);
790 const __m128d hSumVirialxyVec = _mm_add_pd(hSumVirialxyHigh, hSumVirialxyLow);
791
792 // horizontally reduce virialSumZ and potentialEnergySum
793 const __m256d hSumVirialzPotentialEnergy = _mm256_hadd_pd(virialSumZ, potentialEnergySum);
794 const __m128d hSumVirialzPotentialEnergyLow = _mm256_extractf128_pd(hSumVirialzPotentialEnergy, 0);
795 const __m128d hSumVirialzPotentialEnergyHigh = _mm256_extractf128_pd(hSumVirialzPotentialEnergy, 1);
796 const __m128d hSumVirialzPotentialEnergyVec =
797 _mm_add_pd(hSumVirialzPotentialEnergyHigh, hSumVirialzPotentialEnergyLow);
798
799 // globals = {virialX, virialY, virialZ, potentialEnergy}
800 double globals[4];
801 _mm_store_pd(&globals[0], hSumVirialxyVec);
802 _mm_store_pd(&globals[2], hSumVirialzPotentialEnergyVec);
803
804 _aosThreadData[threadnum].virialSum[0] += globals[0];
805 _aosThreadData[threadnum].virialSum[1] += globals[1];
806 _aosThreadData[threadnum].virialSum[2] += globals[2];
807 _aosThreadData[threadnum].potentialEnergySum += globals[3];
808 }
809 // interact with i with 4 neighbors
810#endif // __AVX__

852 _potentialEnergySum = 0.;
853 _virialSum = {0., 0., 0.};

864 void endTraversal(bool newton3) final {
865 using namespace autopas::utils::ArrayMath::literals;
866
867 if (_postProcessed) {
868 throw autopas::utils::ExceptionHandler::AutoPasException(
869 "Already postprocessed, endTraversal(bool newton3) was called twice without calling initTraversal().");
870 }
871 if (calculateGlobals) {
872 for (size_t i = 0; i < _aosThreadData.size(); ++i) {
873 _potentialEnergySum += _aosThreadData[i].potentialEnergySum;
874 _virialSum += _aosThreadData[i].virialSum;
875 }
876 // For each interaction, we added the full contribution for both particles. Divide by 2 here, so that each
877 // contribution is only counted once per pair.
878 _potentialEnergySum *= 0.5;
879 _virialSum *= 0.5;
880
881 // We have always calculated 6*potentialEnergy, so we divide by 6 here!
882 _potentialEnergySum /= 6.;
883 _postProcessed = true;
884
885 AutoPasLog(TRACE, "Final potential energy {}", _potentialEnergySum);
886 AutoPasLog(TRACE, "Final virial {}", _virialSum[0] + _virialSum[1] + _virialSum[2]);
887 }
