303 void PredictiveTuning::linePrediction() {
304  for (const auto &configuration : _searchSpace) {
305  auto &functionParams = _configurationPredictionFunctionParams[configuration];
306  // if configuration was not tested in last tuning phase reuse prediction function.
307  // also make sure prediction function is of the right type (=correct amount of params)
308  if (_lastTest[configuration] != (_tuningPhaseCounter - 1) and functionParams.size() == 2) {
309  const auto delta = _firstIterationOfTuningPhase - _traversalTimesStorage[configuration].back().first;
310 
311  // gradient * delta + last point
312  _configurationPredictions[configuration] = functionParams[0] * delta + functionParams[1];
313  } else
314  // if there is enough evidence calculate new prediction function
315  if (const auto &traversalValues = _traversalTimesStorage[configuration];
316  traversalValues.size() >= _evidenceFirstPrediction) {
317  const auto &[traversal1Iteration, traversal1Time] = traversalValues[traversalValues.size() - 1];
318  const auto &[traversal2Iteration, traversal2Time] = traversalValues[traversalValues.size() - 2];
319 
320  const auto gradient = (traversal1Time - traversal2Time) / (traversal1Iteration - traversal2Iteration);
321  const auto delta = _firstIterationOfTuningPhase - traversal1Iteration;
322 
323  // time1 + (time1 - time2) / (iteration1 - iteration2) / tuningPhase - iteration1)
324  _configurationPredictions[configuration] = traversal1Time + gradient * delta;
325  functionParams.clear();
326  functionParams.emplace_back(gradient);
327  functionParams.emplace_back(traversal1Time);
328  } else {
329  // When a configuration was not yet tested twice.
330  _configurationPredictions[configuration] = std::numeric_limits<long unsigned int>::max();
331  _tooLongNotTestedSearchSpace.emplace(configuration);
332  }
333  }

336 void PredictiveTuning::linearRegression() {
337  for (const auto &configuration : _searchSpace) {
338  auto &functionParams = _configurationPredictionFunctionParams[configuration];
339  // if configuration was not tested in last tuning phase reuse prediction function.
340  // checks is _configurationPredictionFunctionParams has a linear function saved and only then it can reuse it
341  if ((_lastTest[configuration] != (_tuningPhaseCounter - 1)) and functionParams.size() == 2) {
342  // if configuration was not tested in last tuning phase reuse prediction function.
343  // gradient * iteration + y-intercept
344  _configurationPredictions[configuration] = functionParams[0] * _firstIterationOfTuningPhase + functionParams[1];
345  } else if (const auto &traversalValues = _traversalTimesStorage[configuration];
346  traversalValues.size() >= _evidenceFirstPrediction) {
347  size_t iterationMultTime = 0, iterationSum = 0, iterationSquareSum = 0, timeSum = 0;
348 
349  for (auto i = traversalValues.size() - _evidenceFirstPrediction; i < traversalValues.size(); i++) {
350  const auto &[iteration, time] = traversalValues[i];
351  iterationMultTime += iteration * time;
352  iterationSum += iteration;
353  iterationSquareSum += iteration * iteration;
354  timeSum += time;
355  }
356 
357  // cast integer to decimal because this division contains small numbers which would cause precision lose
358  const double iterationMeanValue =
359  static_cast<double>(iterationSum) / static_cast<double>(_evidenceFirstPrediction);
360  const auto timeMeanValue = timeSum / _evidenceFirstPrediction;
361 
362  // cast unsigned to signed because this difference can be negative
363  const auto numerator = static_cast<long>(iterationMultTime) - static_cast<long>(iterationSum * timeMeanValue);
364  const auto denominator =
365  static_cast<double>(iterationSquareSum) - static_cast<double>(iterationSum * iterationMeanValue);
366 
367  // ((Sum iteration_i * time_i) - n * iterationMeanValue * timeMeanValue) / ((Sum iteration_i^2) - n *
368  // iterationMeanValue ^ 2)
369  const auto gradient = numerator / denominator;
370  const auto yIntercept = timeMeanValue - gradient * iterationMeanValue;
371 
372  _configurationPredictions[configuration] = gradient * _firstIterationOfTuningPhase + yIntercept;
373  functionParams.clear();
374  functionParams.emplace_back(gradient);
375  functionParams.emplace_back(yIntercept);
376 
377  } else {
378  // When a configuration was not yet tested twice.
379  _configurationPredictions[configuration] = std::numeric_limits<long unsigned int>::max();
380  _tooLongNotTestedSearchSpace.emplace(configuration);
381  }
382  }

385 void PredictiveTuning::lagrangePolynomial() {
386  for (const auto &configuration : _searchSpace) {
387  const auto &traversalValues = _traversalTimesStorage[configuration];
388  if (traversalValues.size() >= _evidenceFirstPrediction) {
389  const auto lengthTTS = traversalValues.size() - 1;
390  long prediction = 0;
391  // calculates lagrange basis functions L_i and multiplies it with the tested time of p_i
392  for (unsigned int i = 0; i < _evidenceFirstPrediction; i++) {
393  long numerator = 1;
394  long denominator = 1;
395  const auto &[pointIIteration, pointITime] = traversalValues[lengthTTS - i];
396  for (unsigned int j = 0; j < _evidenceFirstPrediction; j++) {
397  if (i != j) {
398  // cast unsigned to signed before subtraction, because this difference can be negative
399  numerator *= static_cast<long>(_firstIterationOfTuningPhase) - traversalValues[lengthTTS - j].first;
400  denominator *= static_cast<long>(pointIIteration) - traversalValues[lengthTTS - j].first;
401  // TODO: Look if this is sufficient or if there is a better solution
402  // Should only be needed when the denominator overflows and therefore gets set to zero
403  if (denominator == 0) {
404  denominator = std::numeric_limits<long>::max();
405  break;
406  }
407  }
408  }
409  prediction += numerator * pointITime / denominator;
410  }
411  _configurationPredictions[configuration] = prediction;
412  } else {
413  // When a configuration was not yet tested twice.
414  _configurationPredictions[configuration] = std::numeric_limits<long unsigned int>::max();
415  _tooLongNotTestedSearchSpace.emplace(configuration);
416  }

420 void PredictiveTuning::newtonPolynomial() {
421  for (const auto &configuration : _searchSpace) {
422  auto &functionParams = _configurationPredictionFunctionParams[configuration];
423  // if configuration was not tested in last tuning phase reuse prediction function.
424  if ((_lastTest[configuration] != (_tuningPhaseCounter - 1)) and functionParams.size() == _evidenceFirstPrediction) {
425  const auto &traversalValues = _traversalTimesStorage[configuration];
426  auto lengthTTS = traversalValues.size() - 1;
427  long prediction = 0;
428  for (unsigned int i = 0; i < _evidenceFirstPrediction; i++) {
429  auto interimValue = functionParams[i];
430  for (unsigned int j = 0; j < i; j++) {
431  // cast to double, because _configurationPredictionFunctionParams contains doubles
432  interimValue *= static_cast<double>(_firstIterationOfTuningPhase) -
433  traversalValues[lengthTTS - _evidenceFirstPrediction + j].first;
434  }
435  prediction += interimValue;
436  }
437  _configurationPredictions[configuration] = prediction;
438  } else if (const auto &traversalValues = _traversalTimesStorage[configuration];
439  traversalValues.size() >= _evidenceFirstPrediction) {
440  std::vector<std::vector<double>> interimCalculation(_evidenceFirstPrediction);
441  std::vector<size_t> iterationValues(_evidenceFirstPrediction);
442  std::vector<double> coefficients(_evidenceFirstPrediction);
443  const auto lengthTTS = traversalValues.size();
444  auto lengthIthColumn = _evidenceFirstPrediction;
445 
446  // calculates a column of the newton interpolation
447  for (unsigned int i = 0; i < _evidenceFirstPrediction; i++) {
448  std::vector<double> ithColumn(lengthIthColumn);
449  for (unsigned int j = 0; j < lengthIthColumn; j++) {
450  if (i == 0) {
451  auto &[iteration, runTime] = traversalValues[lengthTTS - _evidenceFirstPrediction + j];
452  ithColumn[j] = runTime;
453  iterationValues[j] = iteration;
454  } else {
455  // cast integer to decimal because this division contains small numbers which would cause precision lose
456  ithColumn[j] = (interimCalculation[i - 1][j + 1] - interimCalculation[i - 1][j]) /
457  static_cast<double>(iterationValues[j + i] - iterationValues[j]);
458  }
459  }
460  interimCalculation[i] = ithColumn;
461  coefficients[i] = ithColumn.front();
462  --lengthIthColumn;
463  }
464 
465  size_t prediction = 0;
466  for (unsigned int i = 0; i < _evidenceFirstPrediction; i++) {
467  auto interimValue = coefficients[i];
468  for (unsigned int j = 0; j < i; j++) {
469  // cast to double, because coefficients contains doubles
470  interimValue *= static_cast<double>(_firstIterationOfTuningPhase - iterationValues[j]);
471  }
472  prediction += interimValue;
473  }
474 
475  _configurationPredictions[configuration] = prediction;
476  functionParams = coefficients;
477  } else {
478  // When a configuration was not yet tested twice.
479  _configurationPredictions[configuration] = _predictionErrorValue;
480  _tooLongNotTestedSearchSpace.emplace(configuration);
481  }
482  }

486  // since this function is called only when the whole _optimalSearchSpace is invalid
487  // mark all of it as invalid and remove it from the predictions
488  for (const auto &configuration : _optimalSearchSpace) {
489  _configurationPredictions.erase(configuration);
490  _validSearchSpace.erase(configuration);
491  }
492 
493  _optimalSearchSpace.clear();
494 
495  // shortcut if only one valid configuration is left
496  if (_validSearchSpace.size() == 1) {
497  _optimalSearchSpace = _validSearchSpace;
498  _currentConfig = _optimalSearchSpace.begin();
499  return;
500  }
501 
502  // checks if there are any configurations left that can be tested
503  if (_configurationPredictions.empty()) {
504  autopas::utils::ExceptionHandler::exception(
505  "PredictiveTuning::reselectOptimalSearchSpace() : No valid configuration could be found");
506  }
507 
508  // get optimum of remaining predictions
509  const auto &[optimalConfig, optimalRuntime] = *getOptimum(_configurationPredictions);
510 
511  // abort if optimum is invalid
512  if (_validSearchSpace.count(optimalConfig) == 0) {
513  autopas::utils::ExceptionHandler::exception(
514  "PredictiveTuning::reselectOptimalSearchSpace() : No valid optimal configuration could be found");
515  }
516 
517  // rebuild _optimalSearchSpace from valid configurations that have similar performance as the optimum
518  _optimalSearchSpace.emplace(optimalConfig);
519  const auto absoluteOptimumRange = _relativeOptimumRange * optimalRuntime;
520  for (const auto &configuration : _validSearchSpace) {
521  // Adds configurations that are within the _relativeOptimumRange
522  if (_configurationPredictions[configuration] <= absoluteOptimumRange) {
523  _optimalSearchSpace.emplace(configuration);
524  _tooLongNotTestedSearchSpace.erase(configuration);
525  }
526  }
527 
528  // sanity check
529  if (_optimalSearchSpace.empty()) {
530  autopas::utils::ExceptionHandler::exception(
531  "PredictiveTuning::reselectOptimalSearchSpace() : No possible configuration prediction found!");
532  }
533 
534  _currentConfig = _optimalSearchSpace.begin();

571 void PredictiveTuning::selectOptimalConfiguration() {
572  if (_optimalSearchSpace.size() == 1) {
573  _currentConfig = _optimalSearchSpace.begin();
574  } else if (_searchSpace.size() == 1) {
575  _currentConfig = _searchSpace.begin();
576  } else {
577  // select the tested traversal times for the current tuning phase
578  std::unordered_map<Configuration, size_t, ConfigHash> traversalTimes{};
579  // as long as there is too few evidence there is no optimal search space
580  if (_optimalSearchSpace.empty()) {
581  traversalTimes.reserve(_searchSpace.size());
582  for (const auto &configuration : _searchSpace) {
583  // check if this config was tests in the current phase
584  if (_lastTest[configuration] == _tuningPhaseCounter) {
585  traversalTimes[configuration] = _traversalTimesStorage[configuration].back().second;
586  }
587  }
588  } else {
589  traversalTimes.reserve(_optimalSearchSpace.size() + _tooLongNotTestedSearchSpace.size());
590  for (const auto &configuration : _optimalSearchSpace) {
591  if (_lastTest[configuration] == _tuningPhaseCounter) {
592  traversalTimes[configuration] = _traversalTimesStorage[configuration].back().second;
593  }
594  }
595  for (const auto &configuration : _tooLongNotTestedSearchSpace) {
596  if (_lastTest[configuration] == _tuningPhaseCounter) {
597  traversalTimes[configuration] = _traversalTimesStorage[configuration].back().second;
598  }
599  }
600  }
601 
602  // Time measure strategy
603  if (traversalTimes.empty()) {
604  utils::ExceptionHandler::exception(
605  "PredictiveTuning::selectOptimalConfiguration() : Trying to determine fastest configuration without any "
606  "measurements! "
607  "Either selectOptimalConfiguration was called too early or no applicable configurations were found");
608  }
609 
610  const auto &[optimalConfig, _] = *getOptimum(traversalTimes);
611 
612  _currentConfig = _searchSpace.find(optimalConfig);
613 
614  // sanity check
615  if (_currentConfig == _searchSpace.end() or _currentConfig == _optimalSearchSpace.end() or
616  _currentConfig == _tooLongNotTestedSearchSpace.end()) {
617  autopas::utils::ExceptionHandler::exception(
618  "PredictiveTuning::selectOptimalConfiguration() : Optimal configuration not found in any set of "
619  "configurations!");
620  }
621 
622  // if a blacklist range is set blacklist the really bad stuff
623  if (_relativeBlacklistRange != 0) {
624  blacklistBadConfigurations();
625  }
626  }
627 }

629 void PredictiveTuning::blacklistBadConfigurations() {
630  const auto optimumTime = _traversalTimesStorage[getCurrentConfiguration()].back().second;
631  const auto blacklistThreshold = _relativeBlacklistRange * optimumTime;
632  for (auto configurationIter = _searchSpace.begin(); configurationIter != _searchSpace.end();) {
633  // blacklist if the configuration was tested in this phase and is above the threshold
634  if (_lastTest[*configurationIter] == _tuningPhaseCounter and
635  _traversalTimesStorage[*configurationIter].back().second > blacklistThreshold) {
636  _traversalTimesStorage.erase(*configurationIter);
637  _lastTest.erase(*configurationIter);
638  configurationIter = _searchSpace.erase(configurationIter);
639  } else {
640  configurationIter++;
641  }
642  }
