305 void PredictiveTuning::linePrediction() {
306  for (const auto &configuration : _searchSpace) {
307  // if configuration was not tested in last tuning phase reuse prediction function.
308  // checks is _configurationPredictionFunction has a linear function saved and only then it can reuse it
309  if (_lastTest[configuration] != (_tuningPhaseCounter - 1) and
310  _configurationPredictionFunction[configuration].size() == 2) {
311  const auto delta = _iterationBeginTuningPhase - _traversalTimesStorage[configuration].back().first;
312 
313  // gradient * delta + last point
314  _configurationPredictions[configuration] = _configurationPredictionFunction[configuration][0] * delta +
315  _configurationPredictionFunction[configuration][1];
316  predictionOutput(configuration);
317  } else if (_traversalTimesStorage[configuration].size() >= _evidenceFirstPrediction) {
318  const auto &vector = _traversalTimesStorage[configuration];
319  const auto &[traversal1Iteration, traversal1Time] = vector[vector.size() - 1];
320  const auto &[traversal2Iteration, traversal2Time] = vector[vector.size() - 2];
321 
322  const auto gradient = (traversal1Time - traversal2Time) / (traversal1Iteration - traversal2Iteration);
323  const auto delta = _iterationBeginTuningPhase - traversal1Iteration;
324 
325  // time1 + (time1 - time2) / (iteration1 - iteration2) / tuningPhase - iteration1)
326  _configurationPredictions[configuration] = traversal1Time + gradient * delta;
327  _configurationPredictionFunction[configuration].clear();
328  _configurationPredictionFunction[configuration].emplace_back(gradient);
329  _configurationPredictionFunction[configuration].emplace_back(traversal1Time);
330  predictionOutput(configuration);
331  } else {
332  // When a configuration was not yet tested twice.
333  _configurationPredictions[configuration] = std::numeric_limits<long unsigned int>::max();
334  _tooLongNotTestedSearchSpace.emplace(configuration);
335  AutoPasLog(debug, "No traversal time prediction for {}", configuration.toString());
336  }
337  }

340 void PredictiveTuning::linearRegression() {
341  for (const auto &configuration : _searchSpace) {
342  // if configuration was not tested in last tuning phase reuse prediction function.
343  // checks is _configurationPredictionFunction has a linear function saved and only then it can reuse it
344  if ((_lastTest[configuration] != (_tuningPhaseCounter - 1)) and
345  _configurationPredictionFunction[configuration].size() == 2) {
346  // if configuration was not tested in last tuning phase reuse prediction function.
347  // gradient * iteration + y-intercept
348  _configurationPredictions[configuration] =
349  _configurationPredictionFunction[configuration][0] * _iterationBeginTuningPhase +
350  _configurationPredictionFunction[configuration][1];
351  predictionOutput(configuration);
352  } else if (_traversalTimesStorage[configuration].size() >= _evidenceFirstPrediction) {
353  size_t iterationMultTime = 0, iterationSum = 0, iterationSquareSum = 0, timeSum = 0;
354 
355  for (auto i = _traversalTimesStorage[configuration].size() - _evidenceFirstPrediction;
356  i < _traversalTimesStorage[configuration].size(); i++) {
357  const auto &[iteration, time] = _traversalTimesStorage[configuration][i];
358  iterationMultTime += iteration * time;
359  iterationSum += iteration;
360  iterationSquareSum += iteration * iteration;
361  timeSum += time;
362  }
363 
364  // cast integer to decimal because this division contains small numbers which would cause precision lose
365  const double iterationMeanValue =
366  static_cast<double>(iterationSum) / static_cast<double>(_evidenceFirstPrediction);
367  const auto timeMeanValue = timeSum / _evidenceFirstPrediction;
368 
369  // cast unsigned to signed because this difference can be negative
370  const auto numerator = static_cast<long>(iterationMultTime) - static_cast<long>(iterationSum * timeMeanValue);
371  const auto denominator =
372  static_cast<double>(iterationSquareSum) - static_cast<double>(iterationSum * iterationMeanValue);
373 
374  // ((Sum iteration_i * time_i) - n * iterationMeanValue * timeMeanValue) / ((Sum iteration_i^2) - n *
375  // iterationMeanValue ^ 2)
376  const auto gradient = numerator / denominator;
377  const auto yIntercept = timeMeanValue - gradient * iterationMeanValue;
378 
379  _configurationPredictions[configuration] = gradient * _iterationBeginTuningPhase + yIntercept;
380  _configurationPredictionFunction[configuration].clear();
381  _configurationPredictionFunction[configuration].emplace_back(gradient);
382  _configurationPredictionFunction[configuration].emplace_back(yIntercept);
383 
384  predictionOutput(configuration);
385  } else {
386  // When a configuration was not yet tested twice.
387  _configurationPredictions[configuration] = std::numeric_limits<long unsigned int>::max();
388  _tooLongNotTestedSearchSpace.emplace(configuration);
389  AutoPasLog(debug, "No traversal time prediction for {}", configuration.toString());
390  }
391  }

394 void PredictiveTuning::lagrangePolynomial() {
395  for (const auto &configuration : _searchSpace) {
396  if (_traversalTimesStorage[configuration].size() >= _evidenceFirstPrediction) {
397  const auto lengthTTS = _traversalTimesStorage[configuration].size() - 1;
398  long prediction = 0;
399  // calculates lagrange basis functions L_i and multiplies it with the tested time of p_i
400  for (unsigned int i = 0; i < _evidenceFirstPrediction; i++) {
401  long numerator = 1;
402  long denominator = 1;
403  const auto &[pointIIteration, pointITime] = _traversalTimesStorage[configuration][lengthTTS - i];
404  for (unsigned int j = 0; j < _evidenceFirstPrediction; j++) {
405  if (i != j) {
406  // cast unsigned to signed because this difference can be negative
407  numerator *= static_cast<long>(_iterationBeginTuningPhase) -
408  _traversalTimesStorage[configuration][lengthTTS - j].first;
409  denominator *=
410  static_cast<long>(pointIIteration) - _traversalTimesStorage[configuration][lengthTTS - j].first;
411  // TODO: Look if this is sufficient or if there is a better solution
412  // Should only be needed when the denominator overflows and therefore gets set to zero
413  if (denominator == 0) {
414  denominator = std::numeric_limits<long>::max();
415  break;
416  }
417  }
418  }
419  prediction += numerator * pointITime / denominator;
420  }
421  _configurationPredictions[configuration] = prediction;
422  predictionOutput(configuration);
423  } else {
424  // When a configuration was not yet tested twice.
425  _configurationPredictions[configuration] = std::numeric_limits<long unsigned int>::max();

432 void PredictiveTuning::newtonPolynomial() {
433  for (const auto &configuration : _searchSpace) {
434  // if configuration was not tested in last tuning phase reuse prediction function.
435  if ((_lastTest[configuration] != (_tuningPhaseCounter - 1)) and
436  _configurationPredictionFunction[configuration].size() == _evidenceFirstPrediction) {
437  auto lengthTTS = _traversalTimesStorage[configuration].size() - 1;
438  long prediction = 0;
439  for (unsigned int i = 0; i < _evidenceFirstPrediction; i++) {
440  auto interimValue = _configurationPredictionFunction[configuration][i];
441  for (unsigned int j = 0; j < i; j++) {
442  // cast to double, because _configurationPredictionFunction contains doubles
443  interimValue *= static_cast<double>(_iterationBeginTuningPhase) -
444  _traversalTimesStorage[configuration][lengthTTS - _evidenceFirstPrediction + j].first;
445  }
446  prediction += interimValue;
447  }
448  _configurationPredictions[configuration] = prediction;
449  predictionOutput(configuration);
450  } else if (_traversalTimesStorage[configuration].size() >= _evidenceFirstPrediction) {
451  std::vector<std::vector<double>> interimCalculation;
452  std::vector<unsigned long> iterationValues;
453  std::vector<double> coefficients;
454  const auto lengthTTS = _traversalTimesStorage[configuration].size();
455  auto lengthIthColumn = _evidenceFirstPrediction;
456 
457  // calculates a column of the newton interpolation
458  for (unsigned int i = 0; i < _evidenceFirstPrediction; i++) {
459  std::vector<double> ithColumn;
460  for (unsigned int j = 0; j < lengthIthColumn; j++) {
461  if (i == 0) {
462  ithColumn.emplace_back(
463  _traversalTimesStorage[configuration][lengthTTS - _evidenceFirstPrediction + j].second);
464  iterationValues.emplace_back(
465  _traversalTimesStorage[configuration][lengthTTS - _evidenceFirstPrediction + j].first);
466  } else {
467  // cast integer to decimal because this division contains small numbers which would cause precision lose
468  ithColumn.emplace_back((interimCalculation[i - 1][j + 1] - interimCalculation[i - 1][j]) /
469  static_cast<double>(iterationValues[j + i] - iterationValues[j]));
470  }
471  }
472  interimCalculation.emplace_back(ithColumn);
473  coefficients.emplace_back(ithColumn[0]);
474  lengthIthColumn--;
475  }
476 
477  size_t prediction = 0;
478  for (unsigned int i = 0; i < _evidenceFirstPrediction; i++) {
479  auto interimValue = coefficients[i];
480  for (unsigned int j = 0; j < i; j++) {
481  // cast to double, because coefficients contains doubles
482  interimValue *= static_cast<double>(_iterationBeginTuningPhase - iterationValues[j]);
483  }
484  prediction += interimValue;
485  }
486 
487  _configurationPredictions[configuration] = prediction;
488  predictionOutput(configuration);
489  _configurationPredictionFunction[configuration] = coefficients;
490  } else {
491  // When a configuration was not yet tested twice.
492  _configurationPredictions[configuration] = std::numeric_limits<long unsigned int>::max();
493  _tooLongNotTestedSearchSpace.emplace(configuration);
494  AutoPasLog(debug, "No traversal time prediction for {}", configuration.toString());
495  }
496  }

556 bool PredictiveTuning::tune(bool currentInvalid) {
557  if (not currentInvalid) {
558  _validConfigurationFound = true;
559  }
560 
561  // repeat as long as traversals are not applicable or we run out of configs
562  ++_currentConfig;
563 
564  if (_currentConfig == _searchSpace.end() or _currentConfig == _optimalSearchSpace.end()) {
565  if (_validConfigurationFound) {

586 void PredictiveTuning::selectOptimalConfiguration() {
587  if (_optimalSearchSpace.size() == 1) {
588  _currentConfig = _optimalSearchSpace.begin();
589  AutoPasLog(debug, "Selected Configuration {}", _currentConfig->toString());
590  return;
591  }
592  if (_searchSpace.size() == 1) {
593  _currentConfig = _searchSpace.begin();
594  AutoPasLog(debug, "Selected Configuration {}", _currentConfig->toString());
595  return;
596  }
597 
598  // select the tested traversal times for the current tuning phase
599  std::unordered_map<Configuration, size_t, ConfigHash> traversalTimes;
600  // In the first couple iterations tune iterates through _searchSpace until predictions are made
601  if (_optimalSearchSpace.empty()) {
602  for (const auto &configuration : _searchSpace) {
603  if (_lastTest[configuration] == _tuningPhaseCounter) {
604  traversalTimes[configuration] = _traversalTimesStorage[configuration].back().second;
605  }
606  }
607  } else {
608  for (const auto &configuration : _optimalSearchSpace) {
609  if (_lastTest[configuration] == _tuningPhaseCounter) {
610  traversalTimes[configuration] = _traversalTimesStorage[configuration].back().second;
611  }
612  }
613  for (const auto &configuration : _tooLongNotTestedSearchSpace) {
614  if (_lastTest[configuration] == _tuningPhaseCounter) {
615  traversalTimes[configuration] = _traversalTimesStorage[configuration].back().second;
616  }
617  }
618  }
619 
620  // Time measure strategy
621  if (traversalTimes.empty()) {
622  utils::ExceptionHandler::exception(
623  "PredictiveTuning: Trying to determine fastest configuration without any measurements! "
624  "Either selectOptimalConfiguration was called too early or no applicable configurations were found");
625  }
626 
627  const auto optimum = getOptimum(traversalTimes);
628 
629  _currentConfig = _searchSpace.find(optimum->first);
630 
631  // sanity check
632  if (_currentConfig == _searchSpace.end() or _currentConfig == _optimalSearchSpace.end() or
633  _currentConfig == _tooLongNotTestedSearchSpace.end()) {
634  autopas::utils::ExceptionHandler::exception(
635  "PredictiveTuning: Optimal configuration not found in list of configurations!");
636  }
637 
638  if (_relativeBlacklistRange != 0 and not _notTestedYet.empty()) {
639  blacklistBadConfigurations();
640  }
641 
642  AutoPasLog(debug, "Selected Configuration {}", _currentConfig->toString());
643 }

645 void PredictiveTuning::blacklistBadConfigurations() {
646  auto optimumTime = static_cast<double>(_traversalTimesStorage[getCurrentConfiguration()].back().second);
647  auto configurationIter = _notTestedYet.begin();
648  while (configurationIter != _notTestedYet.end()) {
649  if (_lastTest[*configurationIter] == _tuningPhaseCounter) {
650  if (_traversalTimesStorage[*configurationIter].back().second / optimumTime > _relativeBlacklistRange) {
651  _searchSpace.erase(*configurationIter);
652  _traversalTimesStorage.erase(*configurationIter);
653  _lastTest.erase(*configurationIter);
654  }
655  configurationIter = _notTestedYet.erase(configurationIter);
656  } else {
657  configurationIter++;
658  }
659  }
