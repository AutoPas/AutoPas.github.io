322void PredictiveTuning::linePrediction() {
323 for (const auto &configuration : _searchSpace) {
324 auto &functionParams = _configurationPredictionFunctionParams[configuration];
325 // if configuration was not tested in last tuning phase reuse prediction function.
326 // also make sure prediction function is of the right type (=correct amount of params)
327 if (_lastTest[configuration] != (_tuningPhaseCounter - 1) and functionParams.size() == 2) {
328 const auto delta = _firstIterationOfTuningPhase - _traversalTimesStorage[configuration].back().first;
329
330 // gradient * delta + last point
331 const double newValue = utils::Math::safeMul(functionParams[0], static_cast<double>(delta)) + functionParams[1];
332 // check if multiplication overflowed and then explicitly set error value. No cast to avoid rounding errors.
333 _configurationPredictions[configuration] =
334 newValue == std::numeric_limits<double>::max() ? _predictionOverflowValue : static_cast<long>(newValue);
335 } else
336 // if there is enough evidence calculate new prediction function
337 if (const auto &traversalValues = _traversalTimesStorage[configuration];
338 traversalValues.size() >= _evidenceFirstPrediction) {
339 const auto &[traversal1Iteration, traversal1Time] = traversalValues[traversalValues.size() - 1];
340 const auto &[traversal2Iteration, traversal2Time] = traversalValues[traversalValues.size() - 2];
341
342 const long gradient = static_cast<long>(traversal1Time - traversal2Time) /
343 static_cast<long>(traversal1Iteration - traversal2Iteration);
344 const long delta = static_cast<long>(_firstIterationOfTuningPhase) - static_cast<long>(traversal1Iteration);
345
346 const long change = utils::Math::safeMul(gradient, delta);
347
348 // this might overflow so use safeAdd.
349 const long newValue =
350 utils::Math::safeAdd(traversal1Time, change, _predictionUnderflowValue, _predictionOverflowValue);
351 // Do not accept values smaller zero.
352 _configurationPredictions[configuration] = newValue < 0 ? _predictionUnderflowValue : newValue;
353
354 functionParams.clear();
355 functionParams.emplace_back(gradient);
356 functionParams.emplace_back(traversal1Time);
357 } else {
358 // When a configuration was not yet tested twice.
359 _configurationPredictions[configuration] = _predictionErrorValue;
360 _tooLongNotTestedSearchSpace.emplace(configuration);
361 }
362 }

365void PredictiveTuning::linearRegression() {
366 for (const auto &configuration : _searchSpace) {
367 auto &functionParams = _configurationPredictionFunctionParams[configuration];
368 // if configuration was not tested in last tuning phase reuse prediction function.
369 // checks is _configurationPredictionFunctionParams has a linear function saved and only then it can reuse it
370 if ((_lastTest[configuration] != (_tuningPhaseCounter - 1)) and functionParams.size() == 2) {
371 // if configuration was not tested in last tuning phase reuse prediction function.
372 // gradient * iteration + y-intercept
373 const double newValue =
374 utils::Math::safeMul(functionParams[0], static_cast<double>(_firstIterationOfTuningPhase)) +
375 functionParams[1];
376 // check if multiplication overflowed and then explicitly set error value. No cast to avoid rounding errors.
377 _configurationPredictions[configuration] =
378 newValue == std::numeric_limits<double>::max() ? _predictionOverflowValue : static_cast<long>(newValue);
379 } else if (const auto &traversalValues = _traversalTimesStorage[configuration];
380 traversalValues.size() >= _evidenceFirstPrediction) {
381 // we need signed types because calculation of the gradient might have negative result
382 long iterationMultTime = 0;
383 long timeSum = 0;
384 size_t iterationSum = 0;
385 size_t iterationSquareSum = 0;
386
387 bool numericOverflow = false;
388 for (auto i = traversalValues.size() - _evidenceFirstPrediction; i < traversalValues.size(); i++) {
389 const auto &[iteration, time] = traversalValues[i];
390 long iterationMultTimeI = utils::Math::safeMul(static_cast<long>(iteration), time);
391 iterationMultTime = utils::Math::safeAdd(iterationMultTime, iterationMultTimeI);
392 // if any of the safe operations overflow we can directly move to the next config
393 if (iterationMultTime == std::numeric_limits<decltype(iterationMultTime)>::max()) {
394 numericOverflow = true;
395 break;
396 }
397 iterationSum += iteration;
398 iterationSquareSum += iteration * iteration;
399 timeSum += time;
400 }
401
402 // if there is an overflow the actual prediction will also overflow so abort and continue.
403 if (numericOverflow) {
404 _configurationPredictions[configuration] = _predictionOverflowValue;
405 continue;
406 }
407
408 // cast integer to decimal because this division contains small numbers which would cause precision lose
409 const double iterationMeanValue =
410 static_cast<double>(iterationSum) / static_cast<double>(_evidenceFirstPrediction);
411 const long timeMeanValue = timeSum / _evidenceFirstPrediction;
412
413 const auto numerator = static_cast<double>(iterationMultTime - static_cast<long>(iterationSum) * timeMeanValue);
414 const double denominator =
415 static_cast<double>(iterationSquareSum) - static_cast<double>(iterationSum) * iterationMeanValue;
416
417 // ((Sum iteration_i * time_i) - n * iterationMeanValue * timeMeanValue) / ((Sum iteration_i^2) - n *
418 // iterationMeanValue ^ 2)
419 const auto gradient = numerator / denominator;
420
421 const auto change =
422 static_cast<long>(utils::Math::safeMul(gradient, (_firstIterationOfTuningPhase - iterationMeanValue)));
423 // check if prediction runs into over or underflow.
424 const long newValue =
425 utils::Math::safeAdd(change, timeMeanValue, _predictionUnderflowValue, _predictionOverflowValue);
426 // Do not accept values smaller zero.
427 _configurationPredictions[configuration] = newValue < 0 ? _predictionUnderflowValue : newValue;
428
429 functionParams.clear();
430 functionParams.emplace_back(gradient);
431 const auto yIntercept = static_cast<double>(timeMeanValue) - gradient * iterationMeanValue;
432 functionParams.emplace_back(yIntercept);
433
434 } else {
435 // When a configuration was not yet tested twice.
436 _configurationPredictions[configuration] = _predictionErrorValue;
437 _tooLongNotTestedSearchSpace.emplace(configuration);
438 }
439 }

442void PredictiveTuning::newtonPolynomial() {
443 for (const auto &configuration : _searchSpace) {
444 auto &functionParams = _configurationPredictionFunctionParams[configuration];
445 bool numericOverflow = false;
446
447 // if configuration was not tested in last tuning phase reuse prediction function.
448 if ((_lastTest[configuration] != (_tuningPhaseCounter - 1)) and functionParams.size() == _evidenceFirstPrediction) {
449 const auto &traversalValues = _traversalTimesStorage[configuration];
450 auto lengthTTS = traversalValues.size() - 1;
451 long prediction = 0;
452
453 for (unsigned int i = 0; i < _evidenceFirstPrediction; i++) {
454 auto interimValue = functionParams[i];
455 for (unsigned int j = 0; j < i; j++) {
456 // cast to double, because _configurationPredictionFunctionParams contains doubles
457 auto factor = static_cast<double>(_firstIterationOfTuningPhase) -
458 traversalValues[lengthTTS - _evidenceFirstPrediction + j].first;
459 // check if interimValue *= factor will overflow
460 if (interimValue > 0 and factor > std::numeric_limits<double>::max() / interimValue) {
461 numericOverflow = true;
462 break;
463 }
464 interimValue *= factor;
465 }
466 // check if prediction += interimValue will overflow
467 if (numericOverflow or static_cast<decltype(prediction)>(interimValue) >
468 (std::numeric_limits<decltype(prediction)>::max() - prediction)) {
469 numericOverflow = true;
470 }
471 // if the inner loop overflowed no point in continuing the outer one.
472 if (numericOverflow) {
473 break;
474 }
475
476 prediction += static_cast<decltype(prediction)>(interimValue);
477 }
478
479 if (numericOverflow) {
480 _configurationPredictions[configuration] = _predictionOverflowValue;
481 } else if (prediction < 0) {
482 _configurationPredictions[configuration] = _predictionUnderflowValue;
483 } else {
484 _configurationPredictions[configuration] = prediction;
485 }
486 } else if (const auto &traversalValues = _traversalTimesStorage[configuration];
487 traversalValues.size() >= _evidenceFirstPrediction) {
488 std::vector<std::vector<double>> interimCalculation(_evidenceFirstPrediction);
489 std::vector<size_t> iterationValues(_evidenceFirstPrediction);
490 std::vector<double> coefficients(_evidenceFirstPrediction);
491 const auto lengthTTS = traversalValues.size();
492 auto lengthIthColumn = _evidenceFirstPrediction;
493
494 // calculates a column of the newton interpolation
495 for (unsigned int i = 0; i < _evidenceFirstPrediction; i++) {
496 std::vector<double> ithColumn(lengthIthColumn);
497 for (unsigned int j = 0; j < lengthIthColumn; j++) {
498 if (i == 0) {
499 const auto &[iteration, runTime] = traversalValues[lengthTTS - _evidenceFirstPrediction + j];
500 ithColumn[j] = runTime;
501 iterationValues[j] = iteration;
502 } else {
503 // cast integer to decimal because this division contains small numbers which would cause precision lose
504 ithColumn[j] = (interimCalculation[i - 1][j + 1] - interimCalculation[i - 1][j]) /
505 static_cast<double>(iterationValues[j + i] - iterationValues[j]);
506 }
507 }
508 interimCalculation[i] = ithColumn;
509 coefficients[i] = ithColumn.front();
510 --lengthIthColumn;
511 }
512
513 long prediction = 0;
514 for (unsigned int i = 0; i < _evidenceFirstPrediction; i++) {
515 auto interimValue = coefficients[i];
516 for (unsigned int j = 0; j < i; j++) {
517 // cast to double, because coefficients contains doubles
518 auto factor = static_cast<double>(_firstIterationOfTuningPhase - iterationValues[j]);
519 // check if interimValue *= factor will overflow
520 if (interimValue > 0 and factor > std::numeric_limits<double>::max() / interimValue) {
521 numericOverflow = true;
522 break;
523 }
524 interimValue *= factor;
525 }
526 // check if prediction += interimValue will overflow
527 if (numericOverflow or interimValue > static_cast<double>(std::numeric_limits<decltype(prediction)>::max()) or
528 static_cast<decltype(prediction)>(interimValue) >
529 (std::numeric_limits<decltype(prediction)>::max() - prediction)) {
530 numericOverflow = true;
531 }
532 // if the inner loop overflowed no point in continuing the outer one.
533 if (numericOverflow) {
534 break;
535 }
536
537 prediction += interimValue;
538 }
539
540 if (numericOverflow) {
541 _configurationPredictions[configuration] = _predictionOverflowValue;
542 } else if (prediction < 0) {
543 _configurationPredictions[configuration] = _predictionUnderflowValue;
544 } else {
545 _configurationPredictions[configuration] = prediction;
546 }
547
548 functionParams = coefficients;
549 } else {
550 // When a configuration was not yet tested twice.
551 _configurationPredictions[configuration] = _predictionErrorValue;
552 _tooLongNotTestedSearchSpace.emplace(configuration);
553 }
554 }

557void PredictiveTuning::reselectOptimalSearchSpace() {
558 // since this function is called only when the whole _optimalSearchSpace is invalid
559 // mark all of it as invalid and remove it from the predictions
560 for (const auto &configuration : _optimalSearchSpace) {
561 _configurationPredictions.erase(configuration);
562 _validSearchSpace.erase(configuration);
563 }
564
565 _optimalSearchSpace.clear();
566
567 // shortcut if only one valid configuration is left
568 if (_validSearchSpace.size() == 1) {
569 _optimalSearchSpace = _validSearchSpace;
570 _currentConfig = _optimalSearchSpace.begin();
571 return;
572 }
573
574 // checks if there are any configurations left that can be tested
575 if (_configurationPredictions.empty()) {
576 autopas::utils::ExceptionHandler::exception(
577 "PredictiveTuning::reselectOptimalSearchSpace() : No valid configuration could be found");
578 }
579
580 // get optimum of remaining predictions
581 const auto &[optimalConfig, optimalRuntime] = *getOptimum(_configurationPredictions);
582
583 // abort if optimum is invalid
584 if (_validSearchSpace.count(optimalConfig) == 0) {
585 autopas::utils::ExceptionHandler::exception(
586 "PredictiveTuning::reselectOptimalSearchSpace() : No valid optimal configuration could be found");
587 }
588
589 // rebuild _optimalSearchSpace from valid configurations that have similar performance as the optimum
590 _optimalSearchSpace.emplace(optimalConfig);
591 const auto absoluteOptimumRange = _relativeOptimumRange * optimalRuntime;
592 for (const auto &configuration : _validSearchSpace) {
593 // Adds configurations that are within the _relativeOptimumRange
594 if (_configurationPredictions[configuration] <= absoluteOptimumRange) {
595 _optimalSearchSpace.emplace(configuration);
596 _tooLongNotTestedSearchSpace.erase(configuration);
597 }
598 }
599
600 // sanity check
601 if (_optimalSearchSpace.empty()) {
602 autopas::utils::ExceptionHandler::exception(
603 "PredictiveTuning::reselectOptimalSearchSpace() : No possible configuration prediction found!");
604 }
605
606 _currentConfig = _optimalSearchSpace.begin();
607}

609bool PredictiveTuning::tune(bool currentInvalid) {
610 if (not currentInvalid) {
611 _validConfigurationFound = true;
612 }
613
614 ++_currentConfig;
615
616 // if we tested every config that needed testing (and are not iterating through _tooLongNotTestedSearchSpace)...
617 if (_currentConfig == _searchSpace.end() or _currentConfig == _optimalSearchSpace.end()) {
618 // ... and at least one config was valid
619 if (_validConfigurationFound) {
620 // ... and there is nothing that needs re-testing
621 if (_tooLongNotTestedSearchSpace.empty()) {
622 // finalize this tuning phase and settle on a configuration
623 selectOptimalConfiguration();
624 ++_tuningPhaseCounter;
625 return false;
626 } else {
627 _currentConfig = _tooLongNotTestedSearchSpace.begin();
628 return true;
629 }
630 } else {
631 reselectOptimalSearchSpace();
632 }
633 } else if (_currentConfig == _tooLongNotTestedSearchSpace.end()) {
634 // if we reach this point, we really tested everything there is to test
635 selectOptimalConfiguration();
636 ++_tuningPhaseCounter;
637 return false;
638 }
639
640 return true;
641}

643void PredictiveTuning::selectOptimalConfiguration() {
644 if (_optimalSearchSpace.size() == 1) {
645 _currentConfig = _optimalSearchSpace.begin();
646 } else if (_searchSpace.size() == 1) {
647 _currentConfig = _searchSpace.begin();
648 } else {
649 // select the tested traversal times for the current tuning phase
650 std::unordered_map<Configuration, long, ConfigHash> traversalTimes{};
651 // as long as there is too few evidence there is no optimal search space
652 if (_optimalSearchSpace.empty()) {
653 traversalTimes.reserve(_searchSpace.size());
654 for (const auto &configuration : _searchSpace) {
655 // check if this config was tests in the current phase
656 if (_lastTest[configuration] == _tuningPhaseCounter) {
657 traversalTimes[configuration] = _traversalTimesStorage[configuration].back().second;
658 }
659 }
660 } else {
661 traversalTimes.reserve(_optimalSearchSpace.size() + _tooLongNotTestedSearchSpace.size());
662 for (const auto &configuration : _optimalSearchSpace) {
663 if (_lastTest[configuration] == _tuningPhaseCounter) {
664 traversalTimes[configuration] = _traversalTimesStorage[configuration].back().second;
665 }
666 }
667 for (const auto &configuration : _tooLongNotTestedSearchSpace) {
668 if (_lastTest[configuration] == _tuningPhaseCounter) {
669 traversalTimes[configuration] = _traversalTimesStorage[configuration].back().second;
670 }
671 }
672 }
673
674 // Time measure strategy
675 if (traversalTimes.empty()) {
676 utils::ExceptionHandler::exception(
677 "PredictiveTuning::selectOptimalConfiguration() : Trying to determine fastest configuration without any "
678 "measurements! "
679 "Either selectOptimalConfiguration was called too early or no applicable configurations were found");
680 }
681
682 const auto &[optimalConfig, _] = *getOptimum(traversalTimes);
683
684 _currentConfig = _searchSpace.find(optimalConfig);
685
686 // sanity check
687 if (_currentConfig == _searchSpace.end() or _currentConfig == _optimalSearchSpace.end() or
688 _currentConfig == _tooLongNotTestedSearchSpace.end()) {
689 autopas::utils::ExceptionHandler::exception(
690 "PredictiveTuning::selectOptimalConfiguration() : Optimal configuration not found in any set of "
691 "configurations!");
692 }
693
694 // if a blacklist range is set blacklist the really bad stuff
695 if (_relativeBlacklistRange != 0) {
696 blacklistBadConfigurations();
697 }
698 }

701void PredictiveTuning::blacklistBadConfigurations() {
702 const auto optimumTime = _traversalTimesStorage[getCurrentConfiguration()].back().second;
703 const auto blacklistThreshold = _relativeBlacklistRange * optimumTime;
704 for (auto configurationIter = _searchSpace.begin(); configurationIter != _searchSpace.end();) {
705 // blacklist if the configuration was tested in this phase and is above the threshold
706 if (_lastTest[*configurationIter] == _tuningPhaseCounter and
707 _traversalTimesStorage[*configurationIter].back().second > blacklistThreshold) {
708 _traversalTimesStorage.erase(*configurationIter);
709 _lastTest.erase(*configurationIter);
710 configurationIter = _searchSpace.erase(configurationIter);
711 } else {
712 configurationIter++;
713 }
714 }
