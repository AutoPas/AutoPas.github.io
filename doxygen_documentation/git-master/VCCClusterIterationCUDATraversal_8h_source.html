81  std::vector<std::vector<std::array<double, 6>>> &boundingBoxes, int interactionCellRadius,
82  double distance) override {

85  const size_t cellsSize = cells.size();
86  _neighborCellIds->clear();
87  _neighborCellIds->resize(cellsSize, {});
88 
89  // iterate over all cells within interaction radius in xy plane
90  for (size_t i = 0; i < cellsSize; ++i) {
91  auto pos = utils::ThreeDimensionalMapping::oneToThreeD(i, this->_cellsPerDimension);
92  for (int x = -interactionCellRadius; x <= interactionCellRadius; ++x) {
93  if (0 <= (pos[0] + x) and (pos[0] + x) < this->_cellsPerDimension[0]) {
94  for (int y = -interactionCellRadius; y <= interactionCellRadius; ++y) {
95  if (0 <= (pos[1] + y) and (pos[1] + y) < this->_cellsPerDimension[1]) {
96  // current neighbor cell
97  auto other = utils::ThreeDimensionalMapping::threeToOneD(pos[0] + x, pos[1] + y, (unsigned long)0,
98  this->_cellsPerDimension);
99  // only one way interaction when using newton3
100  if (useNewton3 and other > i) {
101  continue;
102  }
103  // iterate through clusters in own cell
104  for (size_t ownClusterId = 0; ownClusterId < boundingBoxes[i].size(); ++ownClusterId) {
105  (*_neighborCellIds)[i].resize(boundingBoxes[i].size(), {});
106  const std::array<double, 6> ownBox = boundingBoxes[i][ownClusterId];
107 
108  // find range of clusters in other cell within range of own cluster
109  auto start = std::find_if(boundingBoxes[other].begin(), boundingBoxes[other].end(),
110  [this, ownBox, distance](const std::array<double, 6> &otherbox) {
111  return getMinDist(ownBox, otherbox) < distance;
112  });
113  auto end = std::find_if(start, boundingBoxes[other].end(),
114  [this, ownBox, distance](const std::array<double, 6> &otherbox) {
115  return getMinDist(ownBox, otherbox) > distance;
116  });
117 
118  const size_t size = end - start;
119 
120  if (start != end) {
121  (*_neighborCellIds)[i][ownClusterId].reserve(size);
122  auto indexStart = start - boundingBoxes[other].begin();
123  if (other == i) {
124  // add clusters to neighbor list when within same cell
125  for (size_t k = 0; k < size; ++k) {
126  if (useNewton3) {
127  if (indexStart + k > ownClusterId) {
128  (*_neighborCellIds)[i][ownClusterId].push_back(std::make_pair(other, indexStart + k));
129  }
130  } else {
131  if (indexStart + k != ownClusterId) {
132  (*_neighborCellIds)[i][ownClusterId].push_back(std::make_pair(other, indexStart + k));
133  }
134  }
135  }
136  } else {
137  // add clusters to neighbor list in the form [mycell][mycluster] pair(othercell, othercluster)
138  for (size_t k = 0; k < size; ++k) {
139  (*_neighborCellIds)[i][ownClusterId].push_back(std::make_pair(other, indexStart + k));
140  }
141  }
142  }
143  }
144  }
145  }
146  }
147  }
148  }
149  // Make neighbor matrix for GPU by linearizing _neighborCellIds
150  size_t neighborMatrixDim = 0;
151  for (auto &cell : *_neighborCellIds) {
152  for (auto &cluster : cell) {
153  neighborMatrixDim = std::max(neighborMatrixDim, cluster.size());
154  }
155  }
156 
157  ++neighborMatrixDim;
158  if (not useNewton3) {
159  ++neighborMatrixDim;
160  }
161  *_neighborMatrixDim = neighborMatrixDim;

163  std::vector<size_t> cellSizePartSums(cellsSize + 1, 0);
164  for (size_t i = 0; i < cellsSize; ++i) {
165  cellSizePartSums[i + 1] = boundingBoxes[i].size() + cellSizePartSums[i];
166  }
167 
168  std::vector<unsigned int> neighborMatrix(cellSizePartSums.back() * neighborMatrixDim,
169  std::numeric_limits<unsigned int>::max());
170 
171  for (size_t cell = 0; cell < cellsSize; ++cell) {
172  for (size_t cluster = 0; cluster < (*_neighborCellIds)[cell].size(); ++cluster) {
173  size_t i = 0;
174  for (auto &neighbors : (*_neighborCellIds)[cell][cluster]) {
175  neighborMatrix[(cellSizePartSums[cell] + cluster) * neighborMatrixDim + i] =
176  cellSizePartSums[neighbors.first] + neighbors.second;

178  }
179  if (not useNewton3) {

250  [[nodiscard]] inline double getMinDist(const std::array<double, 6> &box1, const std::array<double, 6> &box2) const {
251  double sqrDist = 0;
252  for (int i = 0; i < 3; ++i) {
253  if (box2[i + 3] < box1[i]) {
254  double d = box2[i + 3] - box1[i];
255  sqrDist += d * d;
256  } else if (box2[i] > box1[i + 3]) {
257  double d = box2[i] - box1[i + 3];
258  sqrDist += d * d;
259  }
260  }
261  return sqrt(sqrDist);
262  }
