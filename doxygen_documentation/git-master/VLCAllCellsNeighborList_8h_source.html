108 void generateSoAFromAoS(LinkedCells<Particle> &linkedCells) override {
109 _soaNeighborList.clear();
110
111 // particle pointer to global index of particle
112 std::unordered_map<Particle *, size_t> particleToIndex;
113 particleToIndex.reserve(linkedCells.getNumParticles());
114 size_t i = 0;
115 for (auto iter = linkedCells.begin(IteratorBehavior::ownedOrHaloOrDummy); iter.isValid(); ++iter, ++i) {
116 particleToIndex[&(*iter)] = i;
117 }
118
119 _soaNeighborList.resize(linkedCells.getCells().size());
120
121 // iterate over cells
122 for (size_t firstCellIndex = 0; firstCellIndex < _aosNeighborList.size(); ++firstCellIndex) {
123 const auto &aosCurrentCell = _aosNeighborList[firstCellIndex];
124 auto &soaCurrentCell = _soaNeighborList[firstCellIndex];
125 soaCurrentCell.reserve(aosCurrentCell.capacity());
126
127 // iterate over pairs of particle and neighbor list
128 for (const auto &aosParticleAndParticleList : aosCurrentCell) {
129 Particle *currentParticle = aosParticleAndParticleList.first;
130 // global index of current particle
131 size_t currentParticleGlobalIndex = particleToIndex.at(currentParticle);
132
133 // create SoA neighbor list for current particle
134 std::vector<size_t, autopas::AlignedAllocator<size_t>> currentSoANeighborList{};
135
136 // fill the SoA neighbor list with the indices of the particles from the corresponding AoS neighbor list
137 for (const auto &neighborOfCurrentParticle : aosParticleAndParticleList.second) {
138 currentSoANeighborList.emplace_back(particleToIndex.at(neighborOfCurrentParticle));
139 }
140
141 // add newly constructed pair of particle index and SoA neighbor list to cell
142 soaCurrentCell.emplace_back(std::make_pair(currentParticleGlobalIndex, currentSoANeighborList));
143 }
144 }
