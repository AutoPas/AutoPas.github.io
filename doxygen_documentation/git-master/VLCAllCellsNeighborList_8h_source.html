67 // Initialize a neighbor list for each cell.
68 _aosNeighborList.clear();
69 this->_internalLinkedCells = &linkedCells;
70 auto &cells = linkedCells.getCells();
71 size_t cellsSize = cells.size();
72 _aosNeighborList.resize(cellsSize);
73 for (size_t cellIndex = 0; cellIndex < cellsSize; ++cellIndex) {
74 auto &cell = cells[cellIndex];
75 _aosNeighborList[cellIndex].reserve(cell.size());
76 size_t particleIndexWithinCell = 0;
77 for (auto iter = cell.begin(); iter != cell.end(); ++iter, ++particleIndexWithinCell) {
78 Particle *particle = &*iter;
79 _aosNeighborList[cellIndex].emplace_back(particle, std::vector<Particle *>());
80 // In a cell with N particles, reserve space for 5N neighbors.
81 // 5 is an empirically determined magic number that provides good speed.
82 _aosNeighborList[cellIndex].back().second.reserve(cell.size() * 5);
83 _particleToCellMap[particle] = std::make_pair(cellIndex, particleIndexWithinCell);

115 void generateSoAFromAoS(LinkedCells<Particle> &linkedCells) override {
116 _soaNeighborList.clear();
117
118 // particle pointer to global index of particle
119 std::unordered_map<Particle *, size_t> particleToIndex;
120 particleToIndex.reserve(linkedCells.size());
121 size_t i = 0;
122 for (auto iter = linkedCells.begin(IteratorBehavior::ownedOrHaloOrDummy); iter.isValid(); ++iter, ++i) {
123 particleToIndex[&(*iter)] = i;
124 }
125
126 _soaNeighborList.resize(linkedCells.getCells().size());
127
128 // iterate over cells
129 for (size_t firstCellIndex = 0; firstCellIndex < _aosNeighborList.size(); ++firstCellIndex) {
130 const auto &aosCurrentCell = _aosNeighborList[firstCellIndex];
131 auto &soaCurrentCell = _soaNeighborList[firstCellIndex];
132 soaCurrentCell.reserve(aosCurrentCell.capacity());
133
134 // iterate over pairs of particle and neighbor list
135 for (const auto &aosParticleAndParticleList : aosCurrentCell) {
136 Particle *currentParticle = aosParticleAndParticleList.first;
137 // global index of current particle
138 size_t currentParticleGlobalIndex = particleToIndex.at(currentParticle);
139
140 // create SoA neighbor list for current particle
141 std::vector<size_t, autopas::AlignedAllocator<size_t>> currentSoANeighborList{};
142
143 // fill the SoA neighbor list with the indices of the particles from the corresponding AoS neighbor list
144 for (const auto &neighborOfCurrentParticle : aosParticleAndParticleList.second) {
145 currentSoANeighborList.emplace_back(particleToIndex.at(neighborOfCurrentParticle));
146 }
147
148 // add newly constructed pair of particle index and SoA neighbor list to cell
149 soaCurrentCell.emplace_back(std::make_pair(currentParticleGlobalIndex, currentSoANeighborList));
150 }
151 }
