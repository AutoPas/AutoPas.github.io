209 void deleteHaloParticles() override {
210 // Step 1: Remove particles from _particlesToAdd
211 for (auto &particleVec : _particlesToAdd) {
212 for (size_t j = 0; j < particleVec.size();) {
213 if (particleVec[j].isHalo()) {
214 particleVec[j] = particleVec[particleVec.size() - 1];
215 particleVec.pop_back();
216 } else {
217 ++j;
218 }
219 }
220 }
221 // Step 2: Remove particles from _towers
222 bool deletedSomething = false;
223#ifdef AUTOPAS_OPENMP
224#pragma omp parallel for reduction(|| : deletedSomething)
225#endif
226 for (size_t i = 0; i < _towers.size(); ++i) {
227 auto &tower = _towers[i];
228 const auto towerSize = tower.size();
229 auto numTailDummies = tower.getNumTailDummyParticles();
230 // iterate over all non-tail dummies.
231 for (size_t j = 0; j < towerSize - numTailDummies;) {
232 if (tower[j].isHalo()) {
233 // swap-"delete"
234 tower[j] = tower[towerSize - 1 - numTailDummies];
235 // Since we can't pop the moved particle here mark it for deletion.
236 internal::markParticleAsDeleted(tower[towerSize - 1 - numTailDummies]);
237 ++numTailDummies;
238 deletedSomething = true;
239 } else {
240 ++j;
241 }
242 }
243 // if anything was marked for deletion actually delete it now.
244 if (deletedSomething) {
245 tower.deleteDummyParticles();
246 }
247 }
248 if (deletedSomething) {
249 _isValid.store(ValidityState::invalid, std::memory_order::memory_order_relaxed);
250 }

282 std::tuple<const Particle *, size_t, size_t> getParticleImpl(size_t cellIndex, size_t particleIndex,
283 IteratorBehavior iteratorBehavior,
284 const std::array<double, 3> &boxMin,
285 const std::array<double, 3> &boxMax) const {
286 // in this context cell == tower
287 // catching the edge case that towers are not yet built -> Iterator jumps to additional vectors
288 if (_towers.empty()) {
289 return {nullptr, 0, 0};
290 }
291
292 // first and last relevant cell index
293 const auto [startCellIndex, endCellIndex] = [&]() -> std::tuple<size_t, size_t> {
294 if constexpr (regionIter) {
295 return {getTowerIndex1DAtPosition(boxMin), getTowerIndex1DAtPosition(boxMax)};
296 } else {
297 // whole range of cells
298 return {0, this->_towers.size() - 1};
299 }
300 }();
301
302 // if we are at the start of an iteration ...
303 if (cellIndex == 0 and particleIndex == 0) {
304 cellIndex =
305 startCellIndex + ((iteratorBehavior & IteratorBehavior::forceSequential) ? 0 : autopas_get_thread_num());
306 }
307 // abort if the start index is already out of bounds
308 if (cellIndex >= this->_towers.size()) {
309 return {nullptr, 0, 0};
310 }
311 // check the data behind the indices
312 if (particleIndex >= this->_towers[cellIndex].size() or
313 not containerIteratorUtils::particleFulfillsIteratorRequirements<regionIter>(
314 this->_towers[cellIndex][particleIndex], iteratorBehavior, boxMin, boxMax)) {
315 // either advance them to something interesting or invalidate them.
316 std::tie(cellIndex, particleIndex) =
317 advanceIteratorIndices<regionIter>(cellIndex, particleIndex, iteratorBehavior, boxMin, boxMax, endCellIndex);
318 }
319
320 // shortcut if the given index doesn't exist
321 if (cellIndex > endCellIndex) {
322 return {nullptr, 0, 0};
323 }
324 const Particle *retPtr = &this->_towers[cellIndex][particleIndex];
325
326 return {retPtr, cellIndex, particleIndex};
327 }

1160 this->template traverseClusters<false>(
1161 [&numClusterPairs](auto &cluster) { numClusterPairs += cluster.getNeighbors().size(); });
1162
1163 constexpr int minNumClusterPairsPerThread = 1000;
1164 auto numThreads =
1165 std::clamp(static_cast<int>(numClusterPairs / minNumClusterPairsPerThread), 1, autopas_get_max_threads());
1166
1167 size_t numClusterPairsPerThread =
1168 std::max(static_cast<unsigned long>(std::ceil(static_cast<double>(numClusterPairs) / numThreads)), 1ul);
1169 if (numClusterPairsPerThread * numThreads < numClusterPairs) {
1170 autopas::utils::ExceptionHandler::exception(
1171 "VerletClusterLists::calculateClusterThreadPartition(): numClusterPairsPerThread ({}) * numThreads ({})={} "
1172 "should always "
1173 "be at least the amount of Cluster Pairs ({})!",

1175 }

1185 void fillClusterRanges(size_t numClusterPairsPerThread, int numThreads) {
1186 if (numClusterPairsPerThread < 1) {
1187 autopas::utils::ExceptionHandler::exception(
1188 "VerletClusterLists::fillClusterRanges(): numClusterPairsPerThread({}) is less than one, this is not "
1189 "supported "
1190 "and will lead to errors!",
1191 numClusterPairsPerThread);
1192 }
1193 _clusterThreadPartition.resize(numThreads);
1194
1195 size_t currentThread = 0;
1196 size_t currentNumClustersToAdd = 0;
1197 size_t numClusterPairsTotal = 0;
1198 bool threadIsInitialized = false;
1199 // Iterate over the clusters of all towers
1200 for (size_t currentTowerIndex = 0; currentTowerIndex < _towers.size(); currentTowerIndex++) {
1201 auto &currentTower = _towers[currentTowerIndex];
1202 for (size_t currentClusterInTower = 0; currentClusterInTower < currentTower.getNumClusters();
1203 currentClusterInTower++) {
1204 auto &currentCluster = currentTower.getCluster(currentClusterInTower);
1205
1206 // If on a new thread, start with the clusters for this thread here.
1207 if (not threadIsInitialized) {
1208 _clusterThreadPartition[currentThread] = {currentTowerIndex, currentClusterInTower, 0};
1209 threadIsInitialized = true;
1210 }
1211
1212 currentNumClustersToAdd++;
1213 numClusterPairsTotal += currentCluster.getNeighbors().size();
1214
1215 // If the thread is finished, write number of clusters and start new thread.
1216 if (numClusterPairsTotal >= numClusterPairsPerThread * (currentThread + 1)) {
1217 // Add the number of clusters for the finished thread.
1218 _clusterThreadPartition[currentThread].numClusters += currentNumClustersToAdd;
1219 currentNumClustersToAdd = 0;
1220 // Go to next thread!
1221 currentThread++;
1222 // if we are already at the end of all threads, go back to last thread!
1223 // this is a safety precaution and should not really matter.
1224 if (currentThread >= numThreads) {
1225 --currentThread;
1226 threadIsInitialized = true;
1227 } else {
1228 threadIsInitialized = false;
1229 }
1230 }
1231 }
1232 }
1233 if (not threadIsInitialized) {
1234 _clusterThreadPartition[currentThread] = {0, 0, 0};
1235 }
1236 // Make sure the last cluster range contains the rest of the clusters, even if there is not the perfect number left.
1237 if (currentNumClustersToAdd != 0) {
1238 _clusterThreadPartition[currentThread].numClusters += currentNumClustersToAdd;
1239 }
1240 // Theoretically, some threads may still remain. This ensures that their numClusters are set to 0.
1241 while (++currentThread < numThreads) {
1242 _clusterThreadPartition[currentThread] = {0, 0, 0};
1243 }
1244 }

1270 [[nodiscard]] std::tuple<size_t, size_t> advanceIteratorIndices(size_t cellIndex, size_t particleIndex,
1271 IteratorBehavior iteratorBehavior,
1272 const std::array<double, 3> &boxMin,
1273 const std::array<double, 3> &boxMax,
1274 size_t endCellIndex) const {

1276 const size_t stride = (iteratorBehavior & IteratorBehavior::forceSequential) ? 1 : autopas_get_num_threads();
1277
1278 // helper function to determine if the cell can even contain particles of interest to the iterator
1279 auto towerIsRelevant = [&]() -> bool {
1280 // special case: Towers are not yet built, then the tower acting as buffer is always relevant.
1281 if (_towers.size() == 1) {
1282 return true;
1283 }
1284 bool isRelevant = true;
1285 if constexpr (regionIter) {
1286 // is the cell in the region?
1287 const auto [towerLowCorner, towerHighCorner] = getTowerBoundingBox(cellIndex);
1288 // particles can move over cell borders. Calculate the volume this cell's particles can be.
1289 const auto towerLowCornerSkin = utils::ArrayMath::subScalar(towerLowCorner, this->getVerletSkin() * 0.5);
1290 const auto towerHighCornerSkin = utils::ArrayMath::addScalar(towerHighCorner, this->getVerletSkin() * 0.5);
1291 isRelevant = utils::boxesOverlap(towerLowCornerSkin, towerHighCornerSkin, boxMin, boxMax);
1292 }
1293 return isRelevant;
1294 };
1295
1296 do {
1297 // advance to the next particle
1298 ++particleIndex;
1299 // If this breaches the end of a cell, find the next non-empty cell and reset particleIndex.
1300
1301 // If cell has wrong type, or there are no more particles in this cell jump to the next
1302 while (not towerIsRelevant() or particleIndex >= this->_towers[cellIndex].size()) {
1303 cellIndex += stride;
1304 particleIndex = 0;
1305
1306 // If we notice that there is nothing else to look at set invalid values, so we get a nullptr next time and
1307 // break.

1309 return {std::numeric_limits<size_t>::max(), particleIndex};
1310 }
1311 }
1312 } while (not containerIteratorUtils::particleFulfillsIteratorRequirements<regionIter>(
1313 this->_towers[cellIndex][particleIndex], iteratorBehavior, boxMin, boxMax));
1314
1315 // the indices returned at this point should always be valid
1316 return {cellIndex, particleIndex};
1317 }
