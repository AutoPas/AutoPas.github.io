111 BalancedTraversal::EstimatorFunction getLoadEstimatorFunction() {
112 switch (this->_loadEstimator) {
113 case LoadEstimatorOption::neighborListLength: {
114 return [&](const std::array<unsigned long, 3> &cellsPerDimension,
115 const std::array<unsigned long, 3> &lowerCorner, const std::array<unsigned long, 3> &upperCorner) {
116 // the neighborListLength function defined for verletListsCells in not compatible with this container.
117 unsigned long sum = 0;
118 for (unsigned long x = lowerCorner[0]; x <= upperCorner[0]; x++) {
119 for (unsigned long y = lowerCorner[1]; y <= upperCorner[1]; y++) {
120 unsigned long cellLoad = 0;
121 auto &tower = _towerBlock.getTowerByIndex2D(x, y);
122 for (auto &cluster : tower.getClusters()) {
123 if (cluster.getNeighbors()) {
124 cellLoad += cluster.getNeighbors()->size();

126 }
127 sum += cellLoad;
128 }
129 }

209 void deleteHaloParticles() override {
210 // Step 1: Remove particles from _particlesToAdd
211 for (auto &particleVec : _particlesToAdd) {
212 for (size_t j = 0; j < particleVec.size();) {
213 if (particleVec[j].isHalo()) {
214 particleVec[j] = particleVec[particleVec.size() - 1];
215 particleVec.pop_back();
216 } else {
217 ++j;
218 }
219 }
220 }
221 // Step 2: Remove particles from _towers
222 bool deletedSomething = false;
223 AUTOPAS_OPENMP(parallel for reduction(|| : deletedSomething))
224 // Thanks to clang 13 this has to be a index based loop instead of a range based
225 for (size_t i = 0; i < _towerBlock.size(); ++i) {
226 auto &tower = _towerBlock[i];
227 const auto towerSize = tower.size();
228 auto numTailDummies = tower.getNumTailDummyParticles();
229 // iterate over all non-tail dummies. Avoid underflows.
230 for (size_t j = 0; numTailDummies < towerSize and j < towerSize - numTailDummies;) {
231 if (tower[j].isHalo()) {
232 // swap-"delete"
233 tower[j] = tower[towerSize - 1 - numTailDummies];
234 // Since we can't pop the moved particle here mark it for deletion.
235 internal::markParticleAsDeleted(tower[towerSize - 1 - numTailDummies]);
236 ++numTailDummies;
237 deletedSomething = true;
238 } else {
239 ++j;
240 }
241 }
242 // if anything was marked for deletion actually delete it now.
243 if (deletedSomething) {
244 tower.deleteDummyParticles();
245 }
246 }
247 if (deletedSomething) {
248 _isValid.store(ValidityState::invalid, std::memory_order::memory_order_relaxed);
249 }

282 IteratorBehavior iteratorBehavior,
283 const std::array<double, 3> &boxMin,
284 const std::array<double, 3> &boxMax) const {
285 using namespace autopas::utils::ArrayMath::literals;
286
287 // in this context cell == tower
288 // catching the edge case that towers are not yet built -> Iterator jumps to additional vectors
289 if (_towerBlock.empty()) {
290 return {nullptr, 0, 0};
291 }
292
293 std::array<double, 3> boxMinWithSafetyMargin = boxMin;
294 std::array<double, 3> boxMaxWithSafetyMargin = boxMax;
295 if constexpr (regionIter) {
296 // We extend the search box for cells here since particles might have moved
297 boxMinWithSafetyMargin -= (this->_skinPerTimestep * static_cast<double>(this->getStepsSinceLastRebuild()));
298 boxMaxWithSafetyMargin += (this->_skinPerTimestep * static_cast<double>(this->getStepsSinceLastRebuild()));
299 }
300
301 // first and last relevant cell index
302 const auto [startCellIndex, endCellIndex] = [&]() -> std::tuple<size_t, size_t> {
303 if constexpr (regionIter) {
304 // We extend the search box for cells here since particles might have moved
305 return {_towerBlock.getTowerIndex1DAtPosition(boxMinWithSafetyMargin),
306 _towerBlock.getTowerIndex1DAtPosition(boxMaxWithSafetyMargin)};
307 } else {
308 if (not(iteratorBehavior & IteratorBehavior::halo)) {
309 // only potentially owned region
310 return {_towerBlock.getFirstOwnedTowerIndex(), _towerBlock.getLastOwnedTowerIndex()};
311 } else {
312 // whole range of cells
313 return {0, _towerBlock.size() - 1};
314 }
315 }
316 }();
317
318 // if we are at the start of an iteration ...
319 if (cellIndex == 0 and particleIndex == 0) {
320 cellIndex =
321 startCellIndex + ((iteratorBehavior & IteratorBehavior::forceSequential) ? 0 : autopas_get_thread_num());
322 }
323 // abort if the start index is already out of bounds
324 if (cellIndex >= _towerBlock.size()) {
325 return {nullptr, 0, 0};
326 }
327 // check the data behind the indices
328 if (particleIndex >= _towerBlock[cellIndex].getNumActualParticles() or
329 not containerIteratorUtils::particleFulfillsIteratorRequirements<regionIter>(
330 _towerBlock[cellIndex][particleIndex], iteratorBehavior, boxMin, boxMax)) {
331 // either advance them to something interesting or invalidate them.
332 std::tie(cellIndex, particleIndex) =
333 advanceIteratorIndices<regionIter>(cellIndex, particleIndex, iteratorBehavior, boxMin, boxMax,
334 boxMinWithSafetyMargin, boxMaxWithSafetyMargin, endCellIndex);
335 }
336 // shortcut if the given index doesn't exist
337 if (cellIndex > endCellIndex) {
338 return {nullptr, 0, 0};
339 }
340 const Particle *retPtr = &_towerBlock[cellIndex][particleIndex];
341
342 return {retPtr, cellIndex, particleIndex};
343 }

358 if (keepNeighborListsValid) {
359 return autopas::LeavingParticleCollector::collectParticlesAndMarkNonOwnedAsDummy(*this);

361 // First delete all halo particles.
362 this->deleteHaloParticles();

377 for (auto iter = this->begin(IteratorBehavior::owned); iter.isValid(); ++iter) {

1057 void calculateClusterThreadPartition() {
1058 size_t numClusterPairs = 0;
1059 this->template traverseClusters<false>(
1060 [&numClusterPairs](auto &cluster) { numClusterPairs += cluster.getNeighbors()->size(); });
1061
1062 constexpr int minNumClusterPairsPerThread = 1000;
1063 auto numThreads =
1064 std::clamp(static_cast<int>(numClusterPairs / minNumClusterPairsPerThread), 1, autopas_get_max_threads());
1065
1066 size_t numClusterPairsPerThread =
1067 std::max(static_cast<unsigned long>(std::ceil(static_cast<double>(numClusterPairs) / numThreads)), 1ul);
1068 if (numClusterPairsPerThread * numThreads < numClusterPairs) {
1069 autopas::utils::ExceptionHandler::exception(
1070 "VerletClusterLists::calculateClusterThreadPartition(): numClusterPairsPerThread ({}) * numThreads ({})={} "
1071 "should always "
1072 "be at least the amount of Cluster Pairs ({})!",

1084 void fillClusterRanges(size_t numClusterPairsPerThread, int numThreads) {
1085 if (numClusterPairsPerThread < 1) {
1086 autopas::utils::ExceptionHandler::exception(
1087 "VerletClusterLists::fillClusterRanges(): numClusterPairsPerThread({}) is less than one, this is not "
1088 "supported "
1089 "and will lead to errors!",
1090 numClusterPairsPerThread);
1091 }
1092 _clusterThreadPartition.resize(numThreads);
1093
1094 size_t currentThread = 0;
1095 size_t currentNumClustersToAdd = 0;
1096 size_t numClusterPairsTotal = 0;
1097 bool threadIsInitialized = false;
1098 // Iterate over the clusters of all towers
1099 for (size_t currentTowerIndex = 0; currentTowerIndex < _towerBlock.size(); currentTowerIndex++) {
1100 auto &currentTower = _towerBlock[currentTowerIndex];
1101 const auto firstOwnedClusterIndex =
1102 std::distance(currentTower.getClusters().begin(), currentTower.getFirstOwnedCluster());
1103 const auto firstTailHaloClusterIndex =
1104 std::distance(currentTower.getClusters().begin(), currentTower.getFirstTailHaloCluster());
1105 for (size_t currentClusterInTower = firstOwnedClusterIndex; currentClusterInTower < firstTailHaloClusterIndex;
1106 ++currentClusterInTower) {
1107 auto &currentCluster = currentTower.getCluster(currentClusterInTower);
1108
1109 // If on a new thread, start with the clusters for this thread here.
1110 if (not threadIsInitialized) {
1111 _clusterThreadPartition[currentThread] = {currentTowerIndex, currentClusterInTower, 0};
1112 threadIsInitialized = true;
1113 }
1114
1115 currentNumClustersToAdd++;
1116 numClusterPairsTotal += currentCluster.getNeighbors()->size();
1117
1118 // If the thread is finished, write number of clusters and start new thread.
1119 if (numClusterPairsTotal >= numClusterPairsPerThread * (currentThread + 1)) {
1120 // Add the number of clusters for the finished thread.
1121 _clusterThreadPartition[currentThread].numClusters += currentNumClustersToAdd;
1122 currentNumClustersToAdd = 0;
1123 // Go to next thread!
1124 currentThread++;
1125 // if we are already at the end of all threads, go back to last thread!
1126 // this is a safety precaution and should not really matter.
1127 if (currentThread >= numThreads) {
1128 --currentThread;
1129 threadIsInitialized = true;
1130 } else {
1131 threadIsInitialized = false;
1132 }
1133 }
1134 }
1135 }
1136 if (not threadIsInitialized) {
1137 _clusterThreadPartition[currentThread] = {0, 0, 0};
1138 }
1139 // Make sure the last cluster range contains the rest of the clusters, even if there is not the perfect number left.
1140 if (currentNumClustersToAdd != 0) {
1141 _clusterThreadPartition[currentThread].numClusters += currentNumClustersToAdd;
1142 }
1143 // Theoretically, some threads may still remain. This ensures that their numClusters are set to 0.
1144 while (++currentThread < numThreads) {
1145 _clusterThreadPartition[currentThread] = {0, 0, 0};
1146 }
1147 }

1174 template <bool regionIter>
1175 [[nodiscard]] std::tuple<size_t, size_t> advanceIteratorIndices(
1176 size_t cellIndex, size_t particleIndex, IteratorBehavior iteratorBehavior, const std::array<double, 3> &boxMin,
1177 const std::array<double, 3> &boxMax, const std::array<double, 3> &boxMinWithSafetyMargin,
1178 const std::array<double, 3> &boxMaxWithSafetyMargin, size_t endCellIndex) const {
1179 // Finding the indices for the next particle
1180 const size_t stride = (iteratorBehavior & IteratorBehavior::forceSequential) ? 1 : autopas_get_num_threads();
1181
1182 // helper function to determine if the cell can even contain particles of interest to the iterator
1183 auto towerIsRelevant = [&]() -> bool {
1184 // special case: Towers are not yet built, then the tower acting as buffer is always relevant.
1185 if (_towerBlock.size() == 1) {
1186 return true;
1187 }
1188 bool isRelevant = true;
1189 if constexpr (regionIter) {
1190 // is the cell in the region?
1191 const auto [towerLowCorner, towerHighCorner] = _towerBlock.getTowerBoundingBox(cellIndex);
1192 isRelevant =
1193 utils::boxesOverlap(towerLowCorner, towerHighCorner, boxMinWithSafetyMargin, boxMaxWithSafetyMargin);
1194 }
1195 return isRelevant;
1196 };
1197
1198 do {
1199 // advance to the next particle
1200 ++particleIndex;
1201 // If this breaches the end of a cell, find the next non-empty cell and reset particleIndex.
1202
1203 // If cell has wrong type, or there are no more particles in this cell jump to the next
1204 while (not towerIsRelevant() or particleIndex >= _towerBlock[cellIndex].getNumActualParticles()) {
1205 cellIndex += stride;
1206 particleIndex = 0;
1207
1208 // If we notice that there is nothing else to look at set invalid values, so we get a nullptr next time and
1209 // break.
1210 if (cellIndex > endCellIndex) {
1211 return {std::numeric_limits<size_t>::max(), particleIndex};
1212 }
1213 }
1214 } while (not containerIteratorUtils::particleFulfillsIteratorRequirements<regionIter>(
1215 _towerBlock[cellIndex][particleIndex], iteratorBehavior, boxMin, boxMax));
1216
1217 // the indices returned at this point should always be valid
1218 return {cellIndex, particleIndex};
1219 }
