102 BalancedTraversal::EstimatorFunction getLoadEstimatorFunction() {
103 switch (this->_loadEstimator) {
104 case LoadEstimatorOption::neighborListLength: {
105 return [&](const std::array<unsigned long, 3> &cellsPerDimension,
106 const std::array<unsigned long, 3> &lowerCorner, const std::array<unsigned long, 3> &upperCorner) {
107 // the neighborListLength function defined for verletListsCells in not compatible with this container.
108 unsigned long sum = 0;
109 for (unsigned long x = lowerCorner[0]; x <= upperCorner[0]; x++) {
110 for (unsigned long y = lowerCorner[1]; y <= upperCorner[1]; y++) {

112 auto &tower = getTowerAtCoordinates(x, y);

650 void calculateClusterThreadPartition() {
651 size_t numClusterPairs = 0;
652 this->template traverseClusters<false>(
653 [&numClusterPairs](auto &cluster) { numClusterPairs += cluster.getNeighbors().size(); });
654
655 constexpr int minNumClusterPairsPerThread = 1000;
656 auto numThreads =
657 std::clamp(static_cast<int>(numClusterPairs / minNumClusterPairsPerThread), 1, autopas_get_max_threads());
658
659 size_t numClusterPairsPerThread =
660 std::max(static_cast<unsigned long>(std::ceil(static_cast<double>(numClusterPairs) / numThreads)), 1ul);
661 if (numClusterPairsPerThread * numThreads < numClusterPairs) {
662 autopas::utils::ExceptionHandler::exception(
663 "VerletClusterLists::calculateClusterThreadPartition(): numClusterPairsPerThread ({}) * numThreads ({})={} "
664 "should always "
665 "be at least the amount of Cluster Pairs ({})!",
666 numClusterPairsPerThread, numThreads, numClusterPairsPerThread * numThreads, numClusterPairs);
667 }
668 fillClusterRanges(numClusterPairsPerThread, numThreads);
669 }

677 void fillClusterRanges(size_t numClusterPairsPerThread, int numThreads) {
678 if (numClusterPairsPerThread < 1) {
679 autopas::utils::ExceptionHandler::exception(
680 "VerletClusterLists::fillClusterRanges(): numClusterPairsPerThread({}) is less than one, this is not "
681 "supported "
682 "and will lead to errors!",
683 numClusterPairsPerThread);
684 }
685 _clusterThreadPartition.resize(numThreads);
686
687 size_t currentThread = 0;
688 size_t currentNumClustersToAdd = 0;
689 size_t numClusterPairsTotal = 0;
690 bool threadIsInitialized = false;
691 // Iterate over the clusters of all towers
692 for (size_t currentTowerIndex = 0; currentTowerIndex < _towers.size(); currentTowerIndex++) {
693 auto &currentTower = _towers[currentTowerIndex];
694 for (size_t currentClusterInTower = 0; currentClusterInTower < currentTower.getNumClusters();
695 currentClusterInTower++) {
696 auto &currentCluster = currentTower.getCluster(currentClusterInTower);
697
698 // If on a new thread, start with the clusters for this thread here.
699 if (not threadIsInitialized) {
700 _clusterThreadPartition[currentThread] = {currentTowerIndex, currentClusterInTower, 0};
701 threadIsInitialized = true;
702 }
703
704 currentNumClustersToAdd++;
705 numClusterPairsTotal += currentCluster.getNeighbors().size();
706
707 // If the thread is finished, write number of clusters and start new thread.
708 if (numClusterPairsTotal >= numClusterPairsPerThread * (currentThread + 1)) {
709 // Add the number of clusters for the finished thread.
710 _clusterThreadPartition[currentThread].numClusters += currentNumClustersToAdd;
711 currentNumClustersToAdd = 0;
712 // Go to next thread!
713 currentThread++;
714 // if we are already at the end of all threads, go back to last thread!
715 // this is a safety precaution and should not really matter.
716 if (currentThread >= numThreads) {
717 --currentThread;
718 threadIsInitialized = true;
719 } else {
720 threadIsInitialized = false;
721 }
722 }
723 }
724 }
725 if (not threadIsInitialized) {
726 _clusterThreadPartition[currentThread] = {0, 0, 0};
727 }
728 // Make sure the last cluster range contains the rest of the clusters, even if there is not the perfect number left.
729 if (currentNumClustersToAdd != 0) {
730 _clusterThreadPartition[currentThread].numClusters += currentNumClustersToAdd;
731 }
732 // Theoretically, some threads may still remain. This ensures that their numClusters are set to 0.
733 while (++currentThread < numThreads) {
734 _clusterThreadPartition[currentThread] = {0, 0, 0};
735 }
736 }

762 void prepareContainerForIterator(const IteratorBehavior &behavior) {
763 // For good openmp scalability we want the particles to be sorted into the clusters, so we do this!
764
765 // If multiple asynchronous iterators are used the container must already be valid.
766 // Otherwise it is impossible to decide which thread needs to rebuild the tower structure and which need to wait.
767 if (behavior & IteratorBehavior::forceSequential) {
768 if (_isValid == ValidityState::invalid) {
769 autopas::utils::ExceptionHandler::exception(
770 "VerletClusterLists::prepareContainerForIterator(): Parallel iterators with behavior containing "
771 "forceSequential encountered, but the container is invalid.");
772 }
773 } else {
774 // Only one thread is allowed to rebuild the towers, so we do an omp single here.
775 // This single is only possible when we do not force sequential mode as otherwise not all threads might pass
776 // through here and we end up waiting for them forever.
777#ifdef AUTOPAS_OPENMP
778#pragma omp single
779#endif
780 if (_isValid == ValidityState::invalid) {
781 rebuildTowersAndClusters();
782 }
783 // there is an implicit barrier at end of single!
784 }

794 [[nodiscard]] auto getRegionIteratorHelper(const std::array<double, 3> &lowerCorner,
795 const std::array<double, 3> &higherCorner,
796 IteratorBehavior behavior) const {
797 // Check all cells, as dummy particles are outside the domain they are only found if the search region is outside
798 // the domain.
799 const auto lowerCornerInBounds = utils::ArrayMath::max(lowerCorner, _haloBoxMin);
800 const auto upperCornerInBounds = utils::ArrayMath::min(higherCorner, _haloBoxMax);
801
802 if (not _builder) {
803 // if no builder exists the clusters have not been built yet and all particles are stored in the first tower.
804 return std::make_tuple(lowerCornerInBounds, upperCornerInBounds, std::vector<size_t>{0});
805 }
806
807 // Find towers intersecting the search region
808 auto firstTowerCoords = _builder->getTowerCoordinates(lowerCornerInBounds);
809 auto firstTowerIndex = _builder->towerIndex2DTo1D(firstTowerCoords[0], firstTowerCoords[1]);
810 auto lastTowerCoords = _builder->getTowerCoordinates(upperCornerInBounds);
811 auto lastTowerIndex = _builder->towerIndex2DTo1D(lastTowerCoords[0], lastTowerCoords[1]);
812
813 std::array<size_t, 2> towersOfInterstPerDim;
814 for (size_t dim = 0; dim < towersOfInterstPerDim.size(); ++dim) {
815 // use ternary operators instead of abs because these are unsigned values
816 towersOfInterstPerDim[dim] = firstTowerCoords[dim] > lastTowerCoords[dim]
817 ? firstTowerCoords[dim] - lastTowerCoords[dim]
818 : lastTowerCoords[dim] - firstTowerCoords[dim];
819 // +1 because we want to include first AND last
820 towersOfInterstPerDim[dim] += 1;
821 // sanity check
822 towersOfInterstPerDim[dim] = std::max(towersOfInterstPerDim[dim], static_cast<size_t>(1));
823 }
824
825 std::vector<size_t> towersOfInterest(towersOfInterstPerDim[0] * towersOfInterstPerDim[1]);
826
827 auto towersOfInterestIterator = towersOfInterest.begin();
828 for (size_t i = 0; i < towersOfInterstPerDim[1]; ++i) {
829 std::iota(towersOfInterestIterator, towersOfInterestIterator + towersOfInterstPerDim[0],
830 std::min(firstTowerIndex, lastTowerIndex) + i * _towersPerDim[0]);
831 towersOfInterestIterator += towersOfInterstPerDim[0];
832 }
833
834 return std::make_tuple(lowerCornerInBounds, upperCornerInBounds, towersOfInterest);
