602  this->template traverseClusters<false>(
603  [&numClusterPairs](auto &cluster) { numClusterPairs += cluster.getNeighbors().size(); });
604

607  std::clamp(static_cast<int>(numClusterPairs / minNumClusterPairsPerThread), 1, autopas_get_max_threads());

610  std::max(static_cast<unsigned long>(std::ceil(static_cast<double>(numClusterPairs) / numThreads)), 1ul);

627  void fillClusterRanges(size_t numClusterPairsPerThread, int numThreads) {
628  if (numClusterPairsPerThread < 1) {
629  autopas::utils::ExceptionHandler::exception(
630  "VerletClusterLists::fillClusterRanges(): numClusterPairsPerThread({}) is less than one, this is not "
631  "supported "
632  "and will lead to errors!",
633  numClusterPairsPerThread);
634  }
635  _clusterThreadPartition.resize(numThreads);
636 
637  size_t currentThread = 0;
638  size_t currentNumClustersToAdd = 0;
639  size_t numClusterPairsTotal = 0;
640  bool threadIsInitialized = false;
641  // Iterate over the clusters of all towers
642  for (size_t currentTowerIndex = 0; currentTowerIndex < _towers.size(); currentTowerIndex++) {
643  auto &currentTower = _towers[currentTowerIndex];
644  for (size_t currentClusterInTower = 0; currentClusterInTower < currentTower.getNumClusters();
645  currentClusterInTower++) {
646  auto &currentCluster = currentTower.getCluster(currentClusterInTower);
647 
648  // If on a new thread, start with the clusters for this thread here.
649  if (not threadIsInitialized) {
650  _clusterThreadPartition[currentThread] = {currentTowerIndex, currentClusterInTower, 0};
651  threadIsInitialized = true;
652  }
653 
654  currentNumClustersToAdd++;
655  numClusterPairsTotal += currentCluster.getNeighbors().size();
656 
657  // If the thread is finished, write number of clusters and start new thread.
658  if (numClusterPairsTotal >= numClusterPairsPerThread * (currentThread + 1)) {
659  // Add the number of clusters for the finished thread.
660  _clusterThreadPartition[currentThread].numClusters += currentNumClustersToAdd;
661  currentNumClustersToAdd = 0;
662  // Go to next thread!
663  currentThread++;
664  // if we are already at the end of all threads, go back to last thread!
665  // this is a safety precaution and should not really matter.
666  if (currentThread >= numThreads) {
667  --currentThread;
668  threadIsInitialized = true;
669  } else {
670  threadIsInitialized = false;
671  }
672  }
673  }
674  }
675  if (not threadIsInitialized) {
676  _clusterThreadPartition[currentThread] = {0, 0, 0};
677  }
678  // Make sure the last cluster range contains the rest of the clusters, even if there is not the perfect number left.
679  if (currentNumClustersToAdd != 0) {
680  _clusterThreadPartition[currentThread].numClusters += currentNumClustersToAdd;
681  }
682  // Theoretically, some threads may still remain. This ensures that their numClusters are set to 0.
683  while (++currentThread < numThreads) {
684  _clusterThreadPartition[currentThread] = {0, 0, 0};
685  }
686  }

706  [[nodiscard]] auto getRegionIteratorHelper(const std::array<double, 3> &lowerCorner,
707  const std::array<double, 3> &higherCorner,
708  IteratorBehavior behavior) const {
709  // Check all cells, as dummy particles are outside the domain they are only found if the search region is outside
710  // the domain.
711  const auto lowerCornerInBounds = utils::ArrayMath::max(lowerCorner, _haloBoxMin);
712  const auto upperCornerInBounds = utils::ArrayMath::min(higherCorner, _haloBoxMax);
713 
714  if (not _builder) {
715  // if no builder exists the clusters have not been built yet and all particles are stored in the first tower.
716  return std::make_tuple(lowerCornerInBounds, upperCornerInBounds, std::vector<size_t>{0});
717  }
718 
719  // Find towers intersecting the search region
720  auto firstTowerCoords = _builder->getTowerCoordinates(lowerCornerInBounds);
721  auto firstTowerIndex = _builder->towerIndex2DTo1D(firstTowerCoords[0], firstTowerCoords[1]);
722  auto lastTowerCoords = _builder->getTowerCoordinates(upperCornerInBounds);
723  auto lastTowerIndex = _builder->towerIndex2DTo1D(lastTowerCoords[0], lastTowerCoords[1]);
724 
725  std::array<size_t, 2> towersOfInterstPerDim;
726  for (size_t dim = 0; dim < towersOfInterstPerDim.size(); ++dim) {
727  // use ternary operators instead of abs because these are unsigned values
728  towersOfInterstPerDim[dim] = firstTowerCoords[dim] > lastTowerCoords[dim]
729  ? firstTowerCoords[dim] - lastTowerCoords[dim]
730  : lastTowerCoords[dim] - firstTowerCoords[dim];
731  // +1 because we want to include first AND last
732  towersOfInterstPerDim[dim] += 1;
733  // sanity check
734  towersOfInterstPerDim[dim] = std::max(towersOfInterstPerDim[dim], static_cast<size_t>(1));
735  }
736 
737  std::vector<size_t> towersOfInterest(towersOfInterstPerDim[0] * towersOfInterstPerDim[1]);
738 
739  auto towersOfInterestIterator = towersOfInterest.begin();
740  for (size_t i = 0; i < towersOfInterstPerDim[1]; ++i) {
741  std::iota(towersOfInterestIterator, towersOfInterestIterator + towersOfInterstPerDim[0],
742  std::min(firstTowerIndex, lastTowerIndex) + i * _towersPerDim[0]);
743  towersOfInterestIterator += towersOfInterstPerDim[0];
744  }
745 
746  return std::make_tuple(lowerCornerInBounds, upperCornerInBounds, towersOfInterest);
747  }
