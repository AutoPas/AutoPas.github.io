651  void calculateClusterThreadPartition() {
652  size_t numClusterPairs = 0;
653  this->template traverseClusters<false>(
654  [&numClusterPairs](auto &cluster) { numClusterPairs += cluster.getNeighbors().size(); });
655 
656  constexpr int minNumClusterPairsPerThread = 1000;
657  auto numThreads =
658  std::clamp(static_cast<int>(numClusterPairs / minNumClusterPairsPerThread), 1, autopas_get_max_threads());

660  size_t numClusterPairsPerThread =
661  std::max(static_cast<unsigned long>(std::ceil(static_cast<double>(numClusterPairs) / numThreads)), 1ul);
662  if (numClusterPairsPerThread * numThreads < numClusterPairs) {
663  autopas::utils::ExceptionHandler::exception(

665  "should always "

678  void fillClusterRanges(size_t numClusterPairsPerThread, int numThreads) {
679  if (numClusterPairsPerThread < 1) {
680  autopas::utils::ExceptionHandler::exception(
681  "VerletClusterLists::fillClusterRanges(): numClusterPairsPerThread({}) is less than one, this is not "
682  "supported "
683  "and will lead to errors!",
684  numClusterPairsPerThread);
685  }
686  _clusterThreadPartition.resize(numThreads);
687 
688  size_t currentThread = 0;
689  size_t currentNumClustersToAdd = 0;
690  size_t numClusterPairsTotal = 0;
691  bool threadIsInitialized = false;
692  // Iterate over the clusters of all towers
693  for (size_t currentTowerIndex = 0; currentTowerIndex < _towers.size(); currentTowerIndex++) {
694  auto &currentTower = _towers[currentTowerIndex];
695  for (size_t currentClusterInTower = 0; currentClusterInTower < currentTower.getNumClusters();
696  currentClusterInTower++) {
697  auto &currentCluster = currentTower.getCluster(currentClusterInTower);
698 
699  // If on a new thread, start with the clusters for this thread here.
700  if (not threadIsInitialized) {
701  _clusterThreadPartition[currentThread] = {currentTowerIndex, currentClusterInTower, 0};
702  threadIsInitialized = true;
703  }
704 
705  currentNumClustersToAdd++;
706  numClusterPairsTotal += currentCluster.getNeighbors().size();
707 
708  // If the thread is finished, write number of clusters and start new thread.
709  if (numClusterPairsTotal >= numClusterPairsPerThread * (currentThread + 1)) {
710  // Add the number of clusters for the finished thread.
711  _clusterThreadPartition[currentThread].numClusters += currentNumClustersToAdd;
712  currentNumClustersToAdd = 0;
713  // Go to next thread!
714  currentThread++;
715  // if we are already at the end of all threads, go back to last thread!
716  // this is a safety precaution and should not really matter.
717  if (currentThread >= numThreads) {
718  --currentThread;
719  threadIsInitialized = true;
720  } else {
721  threadIsInitialized = false;
722  }
723  }
724  }
725  }
726  if (not threadIsInitialized) {
727  _clusterThreadPartition[currentThread] = {0, 0, 0};
728  }
729  // Make sure the last cluster range contains the rest of the clusters, even if there is not the perfect number left.
730  if (currentNumClustersToAdd != 0) {
731  _clusterThreadPartition[currentThread].numClusters += currentNumClustersToAdd;
732  }
733  // Theoretically, some threads may still remain. This ensures that their numClusters are set to 0.
734  while (++currentThread < numThreads) {
735  _clusterThreadPartition[currentThread] = {0, 0, 0};
736  }
737  }

762  [[nodiscard]] auto getRegionIteratorHelper(const std::array<double, 3> &lowerCorner,
763  const std::array<double, 3> &higherCorner,
764  IteratorBehavior behavior) const {
765  // Check all cells, as dummy particles are outside the domain they are only found if the search region is outside
766  // the domain.
767  const auto lowerCornerInBounds = utils::ArrayMath::max(lowerCorner, _haloBoxMin);
768  const auto upperCornerInBounds = utils::ArrayMath::min(higherCorner, _haloBoxMax);
769 
770  if (not _builder) {
771  // if no builder exists the clusters have not been built yet and all particles are stored in the first tower.
772  return std::make_tuple(lowerCornerInBounds, upperCornerInBounds, std::vector<size_t>{0});
773  }
774 
775  // Find towers intersecting the search region
776  auto firstTowerCoords = _builder->getTowerCoordinates(lowerCornerInBounds);
777  auto firstTowerIndex = _builder->towerIndex2DTo1D(firstTowerCoords[0], firstTowerCoords[1]);
778  auto lastTowerCoords = _builder->getTowerCoordinates(upperCornerInBounds);
779  auto lastTowerIndex = _builder->towerIndex2DTo1D(lastTowerCoords[0], lastTowerCoords[1]);
780 
781  std::array<size_t, 2> towersOfInterstPerDim;
782  for (size_t dim = 0; dim < towersOfInterstPerDim.size(); ++dim) {
783  // use ternary operators instead of abs because these are unsigned values
784  towersOfInterstPerDim[dim] = firstTowerCoords[dim] > lastTowerCoords[dim]
785  ? firstTowerCoords[dim] - lastTowerCoords[dim]
786  : lastTowerCoords[dim] - firstTowerCoords[dim];
787  // +1 because we want to include first AND last
788  towersOfInterstPerDim[dim] += 1;
789  // sanity check
790  towersOfInterstPerDim[dim] = std::max(towersOfInterstPerDim[dim], static_cast<size_t>(1));
791  }
792 
793  std::vector<size_t> towersOfInterest(towersOfInterstPerDim[0] * towersOfInterstPerDim[1]);
794 
795  auto towersOfInterestIterator = towersOfInterest.begin();
796  for (size_t i = 0; i < towersOfInterstPerDim[1]; ++i) {
797  std::iota(towersOfInterestIterator, towersOfInterestIterator + towersOfInterstPerDim[0],
798  std::min(firstTowerIndex, lastTowerIndex) + i * _towersPerDim[0]);
799  towersOfInterestIterator += towersOfInterstPerDim[0];
