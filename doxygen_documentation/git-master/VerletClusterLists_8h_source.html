100 BalancedTraversal::EstimatorFunction getLoadEstimatorFunction() {
101 switch (this->_loadEstimator) {
102 case LoadEstimatorOption::neighborListLength: {
103 return [&](const std::array<unsigned long, 3> &cellsPerDimension,
104 const std::array<unsigned long, 3> &lowerCorner, const std::array<unsigned long, 3> &upperCorner) {
105 // the neighborListLength function defined for verletListsCells in not compatible with this container.
106 unsigned long sum = 0;
107 for (unsigned long x = lowerCorner[0]; x <= upperCorner[0]; x++) {
108 for (unsigned long y = lowerCorner[1]; y <= upperCorner[1]; y++) {
109 unsigned long cellLoad = 0;
110 auto &tower = getTowerAtCoordinates(x, y);
111 for (auto &cluster : tower.getClusters()) {
112 cellLoad += cluster.getNeighbors().size();
113 }
114 sum += cellLoad;
115 }
116 }
117 return sum;
118 };
119 }
120 case LoadEstimatorOption::none:
121 [[fallthrough]];
122 default: {
123 return

125 const std::array<unsigned long, 3> &upperCorner) { return 1; };

799 void calculateClusterThreadPartition() {
800 size_t numClusterPairs = 0;
801 this->template traverseClusters<false>(
802 [&numClusterPairs](auto &cluster) { numClusterPairs += cluster.getNeighbors().size(); });
803
804 constexpr int minNumClusterPairsPerThread = 1000;
805 auto numThreads =
806 std::clamp(static_cast<int>(numClusterPairs / minNumClusterPairsPerThread), 1, autopas_get_max_threads());
807
808 size_t numClusterPairsPerThread =
809 std::max(static_cast<unsigned long>(std::ceil(static_cast<double>(numClusterPairs) / numThreads)), 1ul);
810 if (numClusterPairsPerThread * numThreads < numClusterPairs) {
811 autopas::utils::ExceptionHandler::exception(
812 "VerletClusterLists::calculateClusterThreadPartition(): numClusterPairsPerThread ({}) * numThreads ({})={} "
813 "should always "
814 "be at least the amount of Cluster Pairs ({})!",
815 numClusterPairsPerThread, numThreads, numClusterPairsPerThread * numThreads, numClusterPairs);
816 }
817 fillClusterRanges(numClusterPairsPerThread, numThreads);
818 }

826 void fillClusterRanges(size_t numClusterPairsPerThread, int numThreads) {
827 if (numClusterPairsPerThread < 1) {
828 autopas::utils::ExceptionHandler::exception(
829 "VerletClusterLists::fillClusterRanges(): numClusterPairsPerThread({}) is less than one, this is not "
830 "supported "
831 "and will lead to errors!",
832 numClusterPairsPerThread);
833 }
834 _clusterThreadPartition.resize(numThreads);
835
836 size_t currentThread = 0;
837 size_t currentNumClustersToAdd = 0;
838 size_t numClusterPairsTotal = 0;
839 bool threadIsInitialized = false;
840 // Iterate over the clusters of all towers
841 for (size_t currentTowerIndex = 0; currentTowerIndex < _towers.size(); currentTowerIndex++) {
842 auto &currentTower = _towers[currentTowerIndex];
843 for (size_t currentClusterInTower = 0; currentClusterInTower < currentTower.getNumClusters();
844 currentClusterInTower++) {
845 auto &currentCluster = currentTower.getCluster(currentClusterInTower);
846
847 // If on a new thread, start with the clusters for this thread here.
848 if (not threadIsInitialized) {
849 _clusterThreadPartition[currentThread] = {currentTowerIndex, currentClusterInTower, 0};
850 threadIsInitialized = true;
851 }
852
853 currentNumClustersToAdd++;
854 numClusterPairsTotal += currentCluster.getNeighbors().size();
855
856 // If the thread is finished, write number of clusters and start new thread.
857 if (numClusterPairsTotal >= numClusterPairsPerThread * (currentThread + 1)) {
858 // Add the number of clusters for the finished thread.
859 _clusterThreadPartition[currentThread].numClusters += currentNumClustersToAdd;
860 currentNumClustersToAdd = 0;
861 // Go to next thread!
862 currentThread++;
863 // if we are already at the end of all threads, go back to last thread!
864 // this is a safety precaution and should not really matter.
865 if (currentThread >= numThreads) {
866 --currentThread;
867 threadIsInitialized = true;
868 } else {
869 threadIsInitialized = false;
870 }
871 }
872 }
873 }
874 if (not threadIsInitialized) {
875 _clusterThreadPartition[currentThread] = {0, 0, 0};
876 }
877 // Make sure the last cluster range contains the rest of the clusters, even if there is not the perfect number left.
878 if (currentNumClustersToAdd != 0) {
879 _clusterThreadPartition[currentThread].numClusters += currentNumClustersToAdd;
880 }
881 // Theoretically, some threads may still remain. This ensures that their numClusters are set to 0.
882 while (++currentThread < numThreads) {
883 _clusterThreadPartition[currentThread] = {0, 0, 0};
884 }
885 }

911 void prepareContainerForIteration(const IteratorBehavior &behavior) {
912 // For good openmp scalability we want the particles to be sorted into the clusters, so we do this!
913
914 // If multiple asynchronous iterators are used the container must already be valid.
915 // Otherwise it is impossible to decide which thread needs to rebuild the tower structure and which need to wait.
916 if (behavior & IteratorBehavior::forceSequential) {
917 if (_isValid == ValidityState::invalid) {
918 autopas::utils::ExceptionHandler::exception(
919 "VerletClusterLists::prepareContainerForIteration(): Parallel iterators with behavior containing "
920 "forceSequential encountered, but the container is invalid.");
921 }
922 } else {
923 // Only one thread is allowed to rebuild the towers, so we do an omp single here.
924 // This single is only possible when we do not force sequential mode as otherwise not all threads might pass
925 // through here and we end up waiting for them forever.
926#ifdef AUTOPAS_OPENMP
927#pragma omp single
928#endif
929 if (_isValid == ValidityState::invalid) {
930 rebuildTowersAndClusters();
931 }
932 // there is an implicit barrier at end of single!
933 }

943 [[nodiscard]] auto getRegionIteratorHelper(const std::array<double, 3> &lowerCorner,
944 const std::array<double, 3> &higherCorner,
945 IteratorBehavior behavior) const {
946 // Check all cells, as dummy particles are outside the domain they are only found if the search region is outside
947 // the domain.
948 const auto lowerCornerInBounds = utils::ArrayMath::max(lowerCorner, _haloBoxMin);
949 const auto upperCornerInBounds = utils::ArrayMath::min(higherCorner, _haloBoxMax);
950
951 if (not _builder) {
952 // if no builder exists the clusters have not been built yet and all particles are stored in the first tower.
953 return std::make_tuple(lowerCornerInBounds, upperCornerInBounds, std::vector<size_t>{0});
954 }
955
956 // Find towers intersecting the search region
957 auto firstTowerCoords = _builder->getTowerCoordinates(lowerCornerInBounds);
958 auto firstTowerIndex = _builder->towerIndex2DTo1D(firstTowerCoords[0], firstTowerCoords[1]);
959 auto lastTowerCoords = _builder->getTowerCoordinates(upperCornerInBounds);
960 auto lastTowerIndex = _builder->towerIndex2DTo1D(lastTowerCoords[0], lastTowerCoords[1]);
961
962 std::array<size_t, 2> towersOfInterstPerDim;
963 for (size_t dim = 0; dim < towersOfInterstPerDim.size(); ++dim) {
964 // use ternary operators instead of abs because these are unsigned values
965 towersOfInterstPerDim[dim] = firstTowerCoords[dim] > lastTowerCoords[dim]
966 ? firstTowerCoords[dim] - lastTowerCoords[dim]
967 : lastTowerCoords[dim] - firstTowerCoords[dim];
968 // +1 because we want to include first AND last
969 towersOfInterstPerDim[dim] += 1;
970 // sanity check
971 towersOfInterstPerDim[dim] = std::max(towersOfInterstPerDim[dim], static_cast<size_t>(1));
972 }
973
974 std::vector<size_t> towersOfInterest(towersOfInterstPerDim[0] * towersOfInterstPerDim[1]);
975
976 auto towersOfInterestIterator = towersOfInterest.begin();
977 for (size_t i = 0; i < towersOfInterstPerDim[1]; ++i) {
978 std::iota(towersOfInterestIterator, towersOfInterestIterator + towersOfInterstPerDim[0],
979 std::min(firstTowerIndex, lastTowerIndex) + i * _towersPerDim[0]);
980 towersOfInterestIterator += towersOfInterstPerDim[0];
981 }
982
983 return std::make_tuple(lowerCornerInBounds, upperCornerInBounds, towersOfInterest);
984 }
