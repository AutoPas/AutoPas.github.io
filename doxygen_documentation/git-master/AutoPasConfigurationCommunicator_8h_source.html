93template <class Particle>
94void distributeRanksInBuckets(AutoPas_MPI_Comm comm, AutoPas_MPI_Comm *bucket,
95 const std::shared_ptr<autopas::ParticleContainerInterface<Particle>> &container,
96 const std::pair<double, double> smoothedHomogeneityAndMaxDensity,
97 double MPITuningMaxDifferenceForBucket, double MPITuningWeightForMaxDensity) {
98 int rank;
99 AutoPas_MPI_Comm_rank(comm, &rank);
100 int commSize;
101 AutoPas_MPI_Comm_size(comm, &commSize);
102
103 std::vector<double> similarityMetrics(commSize);
104 double similarityMetric =
105 smoothedHomogeneityAndMaxDensity.first + MPITuningWeightForMaxDensity * smoothedHomogeneityAndMaxDensity.second;
106
107 // debug print for evaluation
108 AutoPasLog(debug, "similarityMetric of rank: " + std::to_string(rank) + " is: " + std::to_string(similarityMetric));
109 AutoPasLog(debug, "smoothedHomogeneity of rank: " + std::to_string(rank) +
110 " is: " + std::to_string(smoothedHomogeneityAndMaxDensity.first));
111 AutoPasLog(debug, "smoothedMaxDensity of rank: " + std::to_string(rank) +
112 " is: " + std::to_string(smoothedHomogeneityAndMaxDensity.second));
113
114 // get all the similarityMetrics of the other ranks
115 AutoPas_MPI_Allgather(&similarityMetric, 1, AUTOPAS_MPI_DOUBLE, similarityMetrics.data(), 1, AUTOPAS_MPI_DOUBLE,
116 comm);
117
118 // sort all values
119 std::sort(similarityMetrics.begin(), similarityMetrics.end());
120
121 // calculate absolute differences between neighbouring values
122 std::vector<double> differences;
123 std::adjacent_difference(similarityMetrics.begin(), similarityMetrics.end(), std::back_inserter(differences));
124
125 // convert differences to percentage changes
126 std::transform(differences.begin(), differences.end(), similarityMetrics.begin(), differences.begin(),
127 std::divides<>());
128
129 int current_bucket = 0;
130 int my_bucket = 0;
131
132 for (int i = 0; (size_t)i < similarityMetrics.size(); i++) {
133 // if a difference exceeds MPITuningMaxDifferenceForBucket, start a new bucket
134 if (differences[i] > MPITuningMaxDifferenceForBucket) current_bucket++;
135
136 // debug print for evaluation
137 AutoPasLog(debug, "I am rank: " + std::to_string(rank) + " bucket: " + std::to_string(current_bucket) +
138 " new value: " + std::to_string(similarityMetrics[i]));
139 if (similarityMetrics[i] == similarityMetric) my_bucket = current_bucket;
140 }
141 // split MPI_Comm in as many new communications as there are groups with similar scenarios
142 AutoPas_MPI_Comm_split(comm, my_bucket, rank, bucket);
143}
