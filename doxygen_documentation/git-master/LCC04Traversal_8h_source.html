97template <class ParticleCell, class PairwiseFunctor, DataLayoutOption::Value dataLayout, bool useNewton3>
98constexpr auto LCC04Traversal<ParticleCell, PairwiseFunctor, dataLayout, useNewton3>::computeOffsets32Pack() const {
99 using std::make_pair;
100 using utils::ThreeDimensionalMapping::threeToOneD;
101
102 std::array<std::array<long, 3>, 32> cellOffsets32Pack = {};
103
104 unsigned int i = 0;
105 long z = 0l;
106 cellOffsets32Pack[i++] = {1l, 1l, z};
107 cellOffsets32Pack[i++] = {1l, 2l, z};
108 cellOffsets32Pack[i++] = {2l, 1l, z};
109 cellOffsets32Pack[i++] = {2l, 2l, z};
110
111 // z = 1ul; z = 2ul
112 for (z = 1l; z < 3l; ++z) {
113 for (long y = 0l; y < 4l; y++) {
114 for (long x = 0l; x < 4l; x++) {
115 if ((x == 0l and y == 0l) or (x == 3l and y == 0l) or (x == 0l and y == 3l) or (x == 3l and y == 3l)) {
116 continue;
117 }
118 cellOffsets32Pack[i++] = {x, y, z};
119 }
120 }
121 }
122
123 z = 3ul;
124 cellOffsets32Pack[i++] = {1l, 1l, z};
125 cellOffsets32Pack[i++] = {1l, 2l, z};
126 cellOffsets32Pack[i++] = {2l, 1l, z};
127 cellOffsets32Pack[i++] = {2l, 2l, z};
128
130 if (i != 32) {
131 utils::ExceptionHandler::exception("Internal error: Wrong number of offsets (expected: 32, actual: {})", i);
132 }
133
134 return cellOffsets32Pack;
135}

149void LCC04Traversal<ParticleCell, PairwiseFunctor, dataLayout, useNewton3>::processBasePack32(
150 std::vector<ParticleCell> &cells, const std::array<long, 3> &base3DIndex) {
151 using utils::ThreeDimensionalMapping::threeToOneD;
152 std::array<long, 3> index;
153 const std::array<long, 3> signedDims = utils::ArrayUtils::static_cast_copy_array<long>(this->_cellsPerDimension);
154
155 for (auto Offset32Pack : _cellOffsets32Pack) {
156 // compute 3D index
157 bool isIn = true;
158 for (int d = 0; d < 3; ++d) {
159 index[d] = base3DIndex[d] + Offset32Pack[d];
160 isIn &= (index[d] >= 0l) and (index[d] < _end[d]);
161 }
162
163 if (isIn) {
164 const unsigned long ulIndex = threeToOneD(index, signedDims);
165 _cellHandler.processBaseCell(cells, ulIndex);
166 }
167 }

198template <class ParticleCell, class PairwiseFunctor, DataLayoutOption::Value dataLayout, bool useNewton3>
199void LCC04Traversal<ParticleCell, PairwiseFunctor, dataLayout, useNewton3>::traverseSingleColor(
200 std::vector<ParticleCell> &cells, int color) {
201 // we need to traverse one body-centered cubic (BCC) grid, which consists of two cartesian grids
202
203 // colors 0 and 2 form one cartesian grid
204 // colors 1 and 3 form another cartesian grid, whose origin is shifted by (2,2,2)
205
206 // determine a starting point of one of the grids
207 std::array<long, 3> startOfThisColor{};
208
209 switch (color % 2) {
210 case 0:
211 // colours 0 and 2
212 startOfThisColor = {-2l, -2l, -2l};
213 break;
214 case 1:
215 // colours 1 and 3
216 startOfThisColor = {0l, 0l, 0l};
217 break;
218 }
219
220 // calculate whether the calculated starting point is part of the color
221 long correctParity = parity(startOfThisColor[0], startOfThisColor[1], startOfThisColor[2]);
222 if (color >= 2) {
223 correctParity += 4;
224 }
225
226 // to fix intel64 icpc compiler complaints about perfectly nested loop (tested with version 19.0.4.20190416).
227 const long startX = startOfThisColor[0], endX = _end[0];
228 const long startY = startOfThisColor[1], endY = _end[1];
229 const long startZ = startOfThisColor[2], endZ = _end[2];
230
231// first cartesian grid
232// grids are interlinked: one grid fills the gaps in the other grid
233#if defined(AUTOPAS_OPENMP)
234#pragma omp for schedule(dynamic, 1) collapse(3) nowait
235#endif
236 for (long z = startZ; z < endZ; z += 4) {
237 for (long y = startY; y < endY; y += 4) {
238 for (long x = startX; x < endX; x += 4) {
239 const long par = parity(x, y, z);
240
241 if (par != correctParity) {
242 continue;
243 }
244
245 const std::array<long, 3> base3DIndex = {x, y, z};
246 processBasePack32(cells, base3DIndex);
247 }
248 }
249 }
