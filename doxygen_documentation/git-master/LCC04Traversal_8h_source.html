102  using std::make_pair;
103  using utils::ThreeDimensionalMapping::threeToOneD;
104 
105  std::array<std::array<long, 3>, 32> cellOffsets32Pack = {};
106 
107  unsigned int i = 0;
108  long z = 0l;
109  cellOffsets32Pack[i++] = {1l, 1l, z};
110  cellOffsets32Pack[i++] = {1l, 2l, z};
111  cellOffsets32Pack[i++] = {2l, 1l, z};
112  cellOffsets32Pack[i++] = {2l, 2l, z};
113 
114  // z = 1ul; z = 2ul
115  for (z = 1l; z < 3l; ++z) {
116  for (long y = 0l; y < 4l; y++) {
117  for (long x = 0l; x < 4l; x++) {
118  if ((x == 0l and y == 0l) or (x == 3l and y == 0l) or (x == 0l and y == 3l) or (x == 3l and y == 3l)) {
119  continue;
120  }
121  cellOffsets32Pack[i++] = {x, y, z};
122  }
123  }
124  }
125 
126  z = 3ul;
127  cellOffsets32Pack[i++] = {1l, 1l, z};
128  cellOffsets32Pack[i++] = {1l, 2l, z};
129  cellOffsets32Pack[i++] = {2l, 1l, z};
130  cellOffsets32Pack[i++] = {2l, 2l, z};
131 
133  if (i != 32) {
134  utils::ExceptionHandler::exception("Internal error: Wrong number of offsets (expected: 32, actual: {})", i);
135  }

201 template <class ParticleCell, class PairwiseFunctor, DataLayoutOption::Value dataLayout, bool useNewton3>
202 void LCC04Traversal<ParticleCell, PairwiseFunctor, dataLayout, useNewton3>::traverseSingleColor(
203  std::vector<ParticleCell> &cells, int color) {
204  // we need to traverse one body-centered cubic (BCC) grid, which consists of two cartesian grids
205 
206  // colors 0 and 2 form one cartesian grid
207  // colors 1 and 3 form another cartesian grid, whose origin is shifted by (2,2,2)
208 
209  // determine a starting point of one of the grids
210  std::array<long, 3> startOfThisColor{};
211 
212  switch (color % 2) {
213  case 0:
214  // colours 0 and 2
215  startOfThisColor = {-2l, -2l, -2l};
216  break;
217  case 1:
218  // colours 1 and 3
219  startOfThisColor = {0l, 0l, 0l};
220  break;
221  }
222 
223  // calculate whether the calculated starting point is part of the color
224  long correctParity = parity(startOfThisColor[0], startOfThisColor[1], startOfThisColor[2]);
225  if (color >= 2) {
226  correctParity += 4;
227  }
228 
229  // to fix intel64 icpc compiler complaints about perfectly nested loop (tested with version 19.0.4.20190416).
230  const long startX = startOfThisColor[0], endX = _end[0];
231  const long startY = startOfThisColor[1], endY = _end[1];
232  const long startZ = startOfThisColor[2], endZ = _end[2];
233 
234 // first cartesian grid
235 // grids are interlinked: one grid fills the gaps in the other grid
236 #if defined(AUTOPAS_OPENMP)
237 #pragma omp for schedule(dynamic, 1) collapse(3) nowait
238 #endif
239  for (long z = startZ; z < endZ; z += 4) {
240  for (long y = startY; y < endY; y += 4) {
241  for (long x = startX; x < endX; x += 4) {
242  const long par = parity(x, y, z);
243 
244  if (par != correctParity) {
245  continue;
246  }
247 
248  const std::array<long, 3> base3DIndex = {x, y, z};
249  processBasePack32(cells, base3DIndex);
250  }
251  }
252  }
253 }
