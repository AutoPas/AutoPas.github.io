398  // get the dimensions and volumes of the box

403  boxSize[d] = _boxMaxWithHalo[d] - _boxMinWithHalo[d];

407  // get all particles and clear clusters

409 
410  for (size_t i = 0; i < this->_cells.size(); ++i) {
411  for (auto it = this->_cells[i].begin(); it != this->_cells[i].end(); ++it) {
412  Particle &p = *it;
