82 auto rebuildTowersAndClusters() {
83 using namespace autopas::utils::ArrayMath::literals;
84
85 auto invalidParticles = collectAllParticlesFromTowers();
86 invalidParticles.push_back(std::move(_particlesToAdd));
87 _particlesToAdd.clear();
88
89 // count particles by accumulating tower sizes
90 size_t numParticles = std::accumulate(std::begin(invalidParticles), std::end(invalidParticles), 0,
91 [](auto acc, auto &v) { return acc + v.size(); });
92
93 auto boxSizeWithHalo = _haloBoxMax - _haloBoxMin;
94
95 _towerSideLength = estimateOptimalGridSideLength(numParticles, boxSizeWithHalo, _clusterSize);
96 _towerSideLengthReciprocal = 1 / _towerSideLength;
97 _interactionLengthInTowers = static_cast<int>(std::ceil(_interactionLength * _towerSideLengthReciprocal));
98
99 _towersPerDim = calculateTowersPerDim(boxSizeWithHalo, _towerSideLengthReciprocal);
100 size_t numTowers = _towersPerDim[0] * _towersPerDim[1];
101
102 // resize to number of towers. Cannot use resize since towers are not default constructable.
103 _towers.clear();
104 _towers.reserve(numTowers);
105
106 // create towers and make an estimate for how many particles memory needs to be allocated
107 // 2.7 seems high but gave the best performance when testing
108 const auto sizeEstimation =
109 static_cast<size_t>((static_cast<double>(numParticles) / static_cast<double>(numTowers)) * 2.7);
110 for (int i = 0; i < numTowers; ++i) {
111 _towers.emplace_back(ClusterTower<Particle>(_clusterSize));
112 _towers[i].reserve(sizeEstimation);
113 }
114
115 sortParticlesIntoTowers(invalidParticles);
116
117 // generate clusters and count them
118 size_t numClusters = 0;
119 for (auto &tower : _towers) {
120 numClusters += tower.generateClusters();
121 }
122
123 return std::make_tuple(_towerSideLength, _interactionLengthInTowers, _towersPerDim, numClusters);
124 }

152 [[nodiscard]] static double estimateOptimalGridSideLength(size_t numParticles, const std::array<double, 3> &boxSize,
153 size_t clusterSize) {
154 const double volume = boxSize[0] * boxSize[1] * boxSize[2];
155 if (numParticles > 0) {
156 // estimate particle density
157 const double density = static_cast<double>(numParticles) / volume;
158
159 return std::cbrt(static_cast<double>(clusterSize) / density);
160 } else {
161 return std::max(boxSize[0], boxSize[1]);
162 }
163 }

173 std::array<size_t, 2> towersPerDim{};
174 for (int d = 0; d < 2; d++) {
175 towersPerDim[d] = static_cast<size_t>(std::ceil(boxSize[d] * towerSideLengthReciprocal));
176 // at least one cell

287 template <class FunType>
288 void iterateNeighborTowers(const int towerIndexX, const int towerIndexY, const int minNeighborIndexX,
289 const int maxNeighborIndexX, const int minNeighborIndexY, const int maxNeighborIndexY,
290 const bool useNewton3, FunType function) {
291 auto &tower = getTower(towerIndexX, towerIndexY);
292 // for all neighbor towers
293 for (int neighborIndexY = minNeighborIndexY; neighborIndexY <= maxNeighborIndexY; neighborIndexY++) {
294 double distBetweenTowersY = std::max(0, std::abs(towerIndexY - neighborIndexY) - 1) * _towerSideLength;
295
296 for (int neighborIndexX = minNeighborIndexX; neighborIndexX <= maxNeighborIndexX; neighborIndexX++) {
297 if (useNewton3 and not isForwardNeighbor(towerIndexX, towerIndexY, neighborIndexX, neighborIndexY)) {
298 continue;
299 }
300
301 double distBetweenTowersX = std::max(0, std::abs(towerIndexX - neighborIndexX) - 1) * _towerSideLength;
302
303 // calculate distance in xy-plane
304 auto distBetweenTowersXYsqr = distBetweenTowersX * distBetweenTowersX + distBetweenTowersY * distBetweenTowersY;
305 // skip if already longer than interactionLength
306 if (distBetweenTowersXYsqr <= _interactionLengthSqr) {
307 auto &neighborTower = getTower(neighborIndexX, neighborIndexY);
308
309 function(tower, neighborTower, distBetweenTowersXYsqr, useNewton3);
310 }
311 }
312 }

345 bool isForwardNeighbor(const int towerIndexX, const int towerIndexY, const int neighborIndexX,
346 const int neighborIndexY) {
347 const auto interactionCellTowerIndex1D = get1DInteractionCellIndexForTower(towerIndexX, towerIndexY);
348 const auto interactionCellNeighborIndex1D = get1DInteractionCellIndexForTower(neighborIndexX, neighborIndexY);
349
350 if (interactionCellNeighborIndex1D > interactionCellTowerIndex1D) {
351 return true;
352 } else if (interactionCellNeighborIndex1D < interactionCellTowerIndex1D) {
353 return false;
354 } // else if (interactionCellNeighborIndex1D == interactionCellTowerIndex1D) ...
355
356 const auto towerIndex1D = towerIndex2DTo1D(towerIndexX, towerIndexY);
357 const auto neighborIndex1D = towerIndex2DTo1D(neighborIndexX, neighborIndexY);
358
359 return neighborIndex1D >= towerIndex1D;
360 }

374 void calculateNeighborsBetweenTowers(internal::ClusterTower<Particle> &towerA,
375 internal::ClusterTower<Particle> &towerB, double distBetweenTowersXYsqr,
376 bool useNewton3) {
377 const bool isSameTower = (&towerA == &towerB);
378 for (size_t clusterIndexInTowerA = 0; clusterIndexInTowerA < towerA.getNumClusters(); clusterIndexInTowerA++) {
379 // if we are within one tower depending on newton3 only look at forward neighbors
380 auto startClusterIndexInTowerB = isSameTower and useNewton3 ? clusterIndexInTowerA + 1 : 0;
381 auto &clusterA = towerA.getCluster(clusterIndexInTowerA);
382 auto [clusterABoxBottom, clusterABoxTop, clusterAContainsParticles] = clusterA.getZMinMax();
383
384 if (clusterAContainsParticles) {
385 for (size_t clusterIndexInTowerB = startClusterIndexInTowerB; clusterIndexInTowerB < towerB.getNumClusters();
386 clusterIndexInTowerB++) {
387 // a cluster cannot be a neighbor to itself
388 // If newton3 is true this is not possible because of the choice of the start index.
389 if (not useNewton3 and isSameTower and clusterIndexInTowerA == clusterIndexInTowerB) {
390 continue;
391 }
392 auto &clusterB = towerB.getCluster(clusterIndexInTowerB);
393 auto [clusterBBoxBottom, clusterBBoxTop, clusterBcontainsParticles] = clusterB.getZMinMax();
394 if (clusterBcontainsParticles) {
395 double distZ = bboxDistance(clusterABoxBottom, clusterABoxTop, clusterBBoxBottom, clusterBBoxTop);
396 if (distBetweenTowersXYsqr + distZ * distZ <= _interactionLengthSqr) {
397 clusterA.addNeighbor(clusterB);
398 }
399 }
400 }
401 }
402 }

415 if (max1 < min2) {
416 return min2 - max1;
417 } else if (min1 > max2) {
418 return min1 - max2;
419 } else {
420 return 0;

443 [[nodiscard]] std::array<size_t, 2> getTowerCoordinates(const std::array<double, 3> &location) const {
444 std::array<size_t, 2> towerIndex2D{};
445
446 for (int dim = 0; dim < 2; dim++) {
447 const auto towerDimIndex =
448 static_cast<long int>(floor((location[dim] - _haloBoxMin[dim]) * _towerSideLengthReciprocal));
449 const auto towerDimIndexNonNegative = static_cast<size_t>(std::max(towerDimIndex, 0l));
450 const auto towerDimIndexNonLargerValue = std::min(towerDimIndexNonNegative, _towersPerDim[dim] - 1);
451 towerIndex2D[dim] = towerDimIndexNonLargerValue;
453 // flag manager
454 if (location[dim] >= _haloBoxMax[dim]) {
455 towerIndex2D[dim] = _towersPerDim[dim] - 1;
456 } else if (location[dim] < _haloBoxMin[dim]) {
457 towerIndex2D[dim] = 0;
458 }
459 }
460
461 return towerIndex2D;
462 }
