66 size_t rebuildTowersAndClusters() {
67 using namespace autopas::utils::ArrayMath::literals;
68 // get rid of dummies
69 for (auto &tower : _towerBlock) {
70 tower.deleteDummyParticles();
71 }
72
73 // count particles by accumulating tower sizes
74 const size_t numParticles = std::accumulate(_towerBlock.begin(), _towerBlock.end(), _particlesToAdd.size(),
75 [](auto acc, const auto &tower) {
76 // actually we want only the number of owned or halo particles
77 // but dummies were just deleted.
78 return acc + tower.getNumActualParticles();
79 });
80
81 // calculate new number of towers and their size
82 const auto boxSizeWithHalo = _towerBlock.getHaloBoxMax() - _towerBlock.getHaloBoxMin();
83 const auto numTowersOld = _towerBlock.size();
84 const auto [towerSideLength, numTowersPerDim] =
85 _towerBlock.estimateOptimalGridSideLength(numParticles, _clusterSize);
86 const auto numTowersNew = numTowersPerDim[0] * numTowersPerDim[1];
87
88 // collect all particles that are now not in the right tower anymore
89 std::vector<std::vector<Particle>> invalidParticles{};
90 // Reserve for:
91 // - _particlesToAdd (+1)
92 // - surplus towers (+max(0, numTowersNew - numTowersOld))
93 // - particles out of bounds of new towers (+numTowersNew)
94 invalidParticles.reserve(1 + std::max(0, static_cast<int>(numTowersNew) - static_cast<int>(numTowersOld)) +
95 numTowersNew);
96 // collect all particles that are not yet assigned to towers
97 invalidParticles.push_back(std::move(_particlesToAdd));
98 _particlesToAdd.clear();
99 // if we have less towers than before, collect all particles from the unused towers.
100 for (size_t i = numTowersNew; i < numTowersOld; ++i) {
101 invalidParticles.push_back(std::move(_towerBlock[i].particleVector()));
102 }
103
104 // resize to number of towers.
105 // Attention! This uses the dummy constructor so we still need to set the desired cluster size.
106 _towerBlock.resize(towerSideLength, numTowersPerDim);
107
108 // after resizing the towers we collect all the particles that are out of bounds
109 const auto collectedParticlesFromTowers = collectOutOfBoundsParticlesFromTowers();
110 invalidParticles.insert(invalidParticles.end(), collectedParticlesFromTowers.begin(),
111 collectedParticlesFromTowers.end());
112
113 // create more towers if needed and make an estimate for how many particles memory needs to be allocated
114 // Factor is more or less a random guess.
115 // Historically 2.7 used to be good but in other tests there was no significant difference to lower values.
116 const auto sizeEstimation =
117 static_cast<size_t>((static_cast<double>(numParticles) / static_cast<double>(numTowersNew)) * 1.2);
118 for (auto &tower : _towerBlock) {
119 // Set potentially new towers to the desired cluster size
120 tower.setClusterSize(_clusterSize);
121 tower.reserve(sizeEstimation);
122 }
123
124 sortParticlesIntoTowers(invalidParticles);
125
126 // estimate the number of clusters by particles divided by cluster size + one extra per tower.
127 _neighborListsBuffer.reserveNeighborLists(numParticles / _clusterSize + numTowersNew);
128 // generate clusters and count them
129 size_t numClusters = 0;
130 for (auto &tower : _towerBlock) {
131 numClusters += tower.generateClusters();
132 for (auto clusterIter = _newton3 ? tower.getClusters().begin() : tower.getFirstOwnedCluster();
133 clusterIter < (_newton3 ? tower.getClusters().end() : tower.getFirstTailHaloCluster()); ++clusterIter) {
134 // VCL stores the references to the lists in the clusters, therefore there is no need to create a
135 // cluster -> list lookup structure in the buffer structure
136 const auto listID = _neighborListsBuffer.getNewNeighborList();
137 clusterIter->setNeighborList(&(_neighborListsBuffer.template getNeighborListRef<false>(listID)));
138 }
139 }
140
141 return numClusters;
142 }

237 const int maxTowerIndexX = _towerBlock.getTowersPerDim()[0] - 1;
238 const int maxTowerIndexY = _towerBlock.getTowersPerDim()[1] - 1;
239 const auto numTowersPerInteractionLength = _towerBlock.getNumTowersPerInteractionLength();
240 // for all towers
242 AUTOPAS_OPENMP(parallel for schedule(dynamic) collapse(2))
243 for (int towerIndexY = 0; towerIndexY <= maxTowerIndexY; towerIndexY++) {
244 for (int towerIndexX = 0; towerIndexX <= maxTowerIndexX; towerIndexX++) {
245 // calculate extent of interesting tower 2D indices

247 const int minY = std::max(towerIndexY - numTowersPerInteractionLength, 0);
248 const int maxX = std::min(towerIndexX + numTowersPerInteractionLength, maxTowerIndexX);
249 const int maxY = std::min(towerIndexY + numTowersPerInteractionLength, maxTowerIndexY);

278 void iterateNeighborTowers(const int towerIndexX, const int towerIndexY, const int minNeighborIndexX,
279 const int maxNeighborIndexX, const int minNeighborIndexY, const int maxNeighborIndexY,
280 FunType function) {
281 auto &tower = _towerBlock.getTowerByIndex2D(towerIndexX, towerIndexY);
282 // for all neighbor towers
283 for (int neighborIndexY = minNeighborIndexY; neighborIndexY <= maxNeighborIndexY; neighborIndexY++) {
284 double distBetweenTowersY =
285 std::max(0, std::abs(towerIndexY - neighborIndexY) - 1) * _towerBlock.getTowerSideLength()[1];
286
287 for (int neighborIndexX = minNeighborIndexX; neighborIndexX <= maxNeighborIndexX; neighborIndexX++) {
288 if (_newton3 and not isForwardNeighbor(towerIndexX, towerIndexY, neighborIndexX, neighborIndexY)) {
289 continue;
290 }
291
292 double distBetweenTowersX =
293 std::max(0, std::abs(towerIndexX - neighborIndexX) - 1) * _towerBlock.getTowerSideLength()[0];
294
295 // calculate distance in xy-plane
296 auto distBetweenTowersXYsqr = distBetweenTowersX * distBetweenTowersX + distBetweenTowersY * distBetweenTowersY;
297 // skip if already longer than interactionLength
298 if (distBetweenTowersXYsqr <= _interactionLengthSqr) {
299 auto &neighborTower = _towerBlock.getTowerByIndex2D(neighborIndexX, neighborIndexY);
300
301 function(tower, neighborTower);
302 }
303 }
304 }

365 void calculateNeighborsBetweenTowers(internal::ClusterTower<Particle> &towerA,
366 internal::ClusterTower<Particle> &towerB) {
367 using autopas::utils::ArrayMath::boxDistanceSquared;
368
369 const auto interactionLengthFracOfDomainZ =
370 _towerBlock.getInteractionLength() / (_towerBlock.getHaloBoxMax()[2] - _towerBlock.getHaloBoxMin()[2]);
371 const bool isSameTower = (&towerA == &towerB);
372 // This heuristic seems to find a good middle ground between not too much memory allocated and no additional
373 // allocations when calling clusterA.addNeighbor(clusterB)
374 const auto neighborListReserveHeuristicFactor = (interactionLengthFracOfDomainZ * 2.1) / _clusterSize;
375 const auto isHaloCluster = [](const auto &clusterIter, const auto &tower) {
376 return clusterIter < tower.getFirstOwnedCluster() or clusterIter >= tower.getFirstTailHaloCluster();
377 };
378 // iterate over all clusters from tower A. In newton3 mode go over all of them, otherwise only owned.
379 for (auto clusterIterA = _newton3 ? towerA.getClusters().begin() : towerA.getFirstOwnedCluster();
380 clusterIterA < (_newton3 ? towerA.getClusters().end() : towerA.getFirstTailHaloCluster()); ++clusterIterA) {
381 if (not clusterIterA->empty()) {
382 clusterIterA->getNeighbors()->reserve((towerA.getNumActualParticles() + 8 * towerB.getNumActualParticles()) *
383 neighborListReserveHeuristicFactor);
384
385 // if we are within one tower depending on newton3 only look at forward neighbors
386 // clusterIterB can't be const because it will potentially be added as a non-const neighbor
387 for (auto clusterIterB = isSameTower and _newton3 ? clusterIterA + 1 : towerB.getClusters().begin();
388 clusterIterB < towerB.getClusters().end(); ++clusterIterB) {
389 // a cluster cannot be a neighbor to itself
390 if (&*clusterIterA == &*clusterIterB) {
391 continue;
392 }
393 // never do halo-halo interactions
394 if (isHaloCluster(clusterIterA, towerA) and isHaloCluster(clusterIterB, towerB)) {
395 continue;
396 }
397 if (not clusterIterB->empty()) {
398 const auto [aMin, aMax] = clusterIterA->getBoundingBox();
399 const auto [bMin, bMax] = clusterIterB->getBoundingBox();
400 const auto boxDistSquared = boxDistanceSquared(aMin, aMax, bMin, bMax);
401 if (boxDistSquared <= _interactionLengthSqr) {
402 clusterIterA->addNeighbor(*clusterIterB);
403 }
404 }
405 }
406 }
407 }
