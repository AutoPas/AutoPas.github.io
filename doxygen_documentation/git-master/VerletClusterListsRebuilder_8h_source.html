79 auto rebuildTowersAndClusters() {
80 auto invalidParticles = collectAllParticlesFromTowers();
81 invalidParticles.push_back(std::move(_particlesToAdd));
82 _particlesToAdd.clear();
83
84 // count particles by accumulating tower sizes
85 size_t numParticles = std::accumulate(std::begin(invalidParticles), std::end(invalidParticles), 0,
86 [](auto acc, auto &v) { return acc + v.size(); });
87
88 auto boxSizeWithHalo = utils::ArrayMath::sub(_haloBoxMax, _haloBoxMin);
89
90 _towerSideLength = estimateOptimalGridSideLength(numParticles, boxSizeWithHalo, _clusterSize);
91 _towerSideLengthReciprocal = 1 / _towerSideLength;
92 _interactionLengthInTowers = static_cast<int>(std::ceil(_interactionLength * _towerSideLengthReciprocal));
93
94 _towersPerDim = calculateTowersPerDim(boxSizeWithHalo, _towerSideLengthReciprocal);
95 size_t numTowers = _towersPerDim[0] * _towersPerDim[1];
96
97 // resize to number of towers. Cannot use resize since towers are not default constructable.
98 _towers.clear();
99 _towers.reserve(numTowers);
100
101 // create towers and make an estimate for how many particles memory needs to be allocated
102 // 2.7 seems high but gave the best performance when testing
103 const size_t sizeEstimation = (static_cast<double>(numParticles) / numTowers) * 2.7;
104 for (int i = 0; i < numTowers; ++i) {
105 _towers.emplace_back(ClusterTower<Particle>(_clusterSize));
106 _towers[i].reserve(sizeEstimation);
107 }
108
109 sortParticlesIntoTowers(invalidParticles);
110
111 // generate clusters and count them
112 size_t numClusters = 0;
113 for (auto &tower : _towers) {
114 numClusters += tower.generateClusters();
115 }
116
117 return std::make_tuple(_towerSideLength, _interactionLengthInTowers, _towersPerDim, numClusters);
118 }

145 [[nodiscard]] static double estimateOptimalGridSideLength(size_t numParticles, std::array<double, 3> boxSize,
146 size_t clusterSize) {
147 double volume = boxSize[0] * boxSize[1] * boxSize[2];
148 if (numParticles > 0) {
149 // estimate particle density
150 double density = numParticles / volume;
151
152 return std::cbrt(clusterSize / density);
153 } else {
154 return std::max(boxSize[0], boxSize[1]);
155 }

166 std::array<size_t, 2> towersPerDim{};
167 for (int d = 0; d < 2; d++) {
168 towersPerDim[d] = static_cast<size_t>(std::ceil(boxSize[d] * towerSideLengthReciprocal));
169 // at least one cell

280 template <class FunType>
281 void iterateNeighborTowers(const int towerIndexX, const int towerIndexY, const int minNeighborIndexX,
282 const int maxNeighborIndexX, const int minNeighborIndexY, const int maxNeighborIndexY,
283 const bool useNewton3, FunType function) {
284 auto &tower = getTower(towerIndexX, towerIndexY);
285 // for all neighbor towers
286 for (int neighborIndexY = minNeighborIndexY; neighborIndexY <= maxNeighborIndexY; neighborIndexY++) {
287 double distBetweenTowersY = std::max(0, std::abs(towerIndexY - neighborIndexY) - 1) * _towerSideLength;
288
289 for (int neighborIndexX = minNeighborIndexX; neighborIndexX <= maxNeighborIndexX; neighborIndexX++) {
290 if (useNewton3 and not isForwardNeighbor(towerIndexX, towerIndexY, neighborIndexX, neighborIndexY)) {
291 continue;
292 }
293
294 double distBetweenTowersX = std::max(0, std::abs(towerIndexX - neighborIndexX) - 1) * _towerSideLength;
295
296 // calculate distance in xy-plane
297 auto distBetweenTowersXYsqr = distBetweenTowersX * distBetweenTowersX + distBetweenTowersY * distBetweenTowersY;
298 // skip if already longer than interactionLength
299 if (distBetweenTowersXYsqr <= _interactionLengthSqr) {
300 auto &neighborTower = getTower(neighborIndexX, neighborIndexY);
301
302 function(tower, neighborTower, distBetweenTowersXYsqr, useNewton3);
303 }
304 }
305 }

337 bool isForwardNeighbor(const int towerIndexX, const int towerIndexY, const int neighborIndexX,
338 const int neighborIndexY) {
339 auto interactionCellTowerIndex1D = get1DInteractionCellIndexForTower(towerIndexX, towerIndexY);
340 auto interactionCellNeighborIndex1D = get1DInteractionCellIndexForTower(neighborIndexX, neighborIndexY);
341
342 if (interactionCellNeighborIndex1D > interactionCellTowerIndex1D) {
343 return true;
344 } else if (interactionCellNeighborIndex1D < interactionCellTowerIndex1D) {
345 return false;
346 } // else if (interactionCellNeighborIndex1D == interactionCellTowerIndex1D) ...
347
348 auto towerIndex1D = towerIndex2DTo1D(towerIndexX, towerIndexY);
349 auto neighborIndex1D = towerIndex2DTo1D(neighborIndexX, neighborIndexY);
350
351 return neighborIndex1D >= towerIndex1D;
352 }

366 void calculateNeighborsBetweenTowers(internal::ClusterTower<Particle> &towerA,
367 internal::ClusterTower<Particle> &towerB, double distBetweenTowersXYsqr,
368 bool useNewton3) {
369 const bool isSameTower = (&towerA == &towerB);
370 for (size_t clusterIndexInTowerA = 0; clusterIndexInTowerA < towerA.getNumClusters(); clusterIndexInTowerA++) {
371 // if we are within one tower depending on newton3 only look at forward neighbors
372 auto startClusterIndexInTowerB = isSameTower and useNewton3 ? clusterIndexInTowerA + 1 : 0;
373 auto &clusterA = towerA.getCluster(clusterIndexInTowerA);
374 auto [clusterABoxBottom, clusterABoxTop, clusterAContainsParticles] = clusterA.getZMinMax();
375
376 if (clusterAContainsParticles) {
377 for (size_t clusterIndexInTowerB = startClusterIndexInTowerB; clusterIndexInTowerB < towerB.getNumClusters();
378 clusterIndexInTowerB++) {
379 // a cluster cannot be a neighbor to itself
380 // If newton3 is true this is not possible because of the choice of the start index.
381 if (not useNewton3 and isSameTower and clusterIndexInTowerA == clusterIndexInTowerB) {
382 continue;
383 }
384 auto &clusterB = towerB.getCluster(clusterIndexInTowerB);
385 auto [clusterBBoxBottom, clusterBBoxTop, clusterBcontainsParticles] = clusterB.getZMinMax();
386 if (clusterBcontainsParticles) {
387 double distZ = bboxDistance(clusterABoxBottom, clusterABoxTop, clusterBBoxBottom, clusterBBoxTop);
388 if (distBetweenTowersXYsqr + distZ * distZ <= _interactionLengthSqr) {
389 clusterA.addNeighbor(clusterB);
390 }
391 }
392 }
393 }
394 }

435 std::array<size_t, 2> getTowerCoordinates(std::array<double, 3> location) {
436 std::array<size_t, 2> towerIndex2D{};
437
438 for (int dim = 0; dim < 2; dim++) {
439 const auto towerDimIndex =
440 static_cast<long int>(floor((location[dim] - _haloBoxMin[dim]) * _towerSideLengthReciprocal));
441 const auto towerDimIndexNonNegative = static_cast<size_t>(std::max(towerDimIndex, 0l));
442 const auto towerDimIndexNonLargerValue = std::min(towerDimIndexNonNegative, _towersPerDim[dim] - 1);
443 towerIndex2D[dim] = towerDimIndexNonLargerValue;
445 // flag manager
446 if (location[dim] >= _haloBoxMax[dim]) {
447 towerIndex2D[dim] = _towersPerDim[dim] - 1;
448 } else if (location[dim] < _haloBoxMin[dim]) {
449 towerIndex2D[dim] = 0;
450 }
451 }
452
453 return towerIndex2D;
454 }
