66 size_t rebuildTowersAndClusters() {
67 using namespace autopas::utils::ArrayMath::literals;
68 // get rid of dummies
69 for (auto &tower : _towerBlock) {
70 tower.deleteDummyParticles();
71 }
72
73 // count particles by accumulating tower sizes
74 const size_t numParticles = std::accumulate(_towerBlock.begin(), _towerBlock.end(), _particlesToAdd.size(),
75 [](auto acc, const auto &tower) {
76 // actually we want only the number of owned or halo particles
77 // but dummies were just deleted.
78 return acc + tower.getNumActualParticles();
79 });
80
81 // calculate new number of towers and their size
82 const auto boxSizeWithHalo = _towerBlock.getHaloBoxMax() - _towerBlock.getHaloBoxMin();
83 const auto numTowersOld = _towerBlock.size();
84 const auto [towerSideLength, numTowersPerDim] =
85 _towerBlock.estimateOptimalGridSideLength(numParticles, _clusterSize);
86 const auto numTowersNew = numTowersPerDim[0] * numTowersPerDim[1];
87
88 // collect all particles that are now not in the right tower anymore
89 auto invalidParticles = collectOutOfBoundsParticlesFromTowers();
90 // collect all remaining particles that are not yet assigned to towers
91 invalidParticles.push_back(std::move(_particlesToAdd));
92 _particlesToAdd.clear();
93 // if we have less towers than before, collect all particles from the unused towers.
94 for (size_t i = numTowersNew; i < numTowersOld; ++i) {
95 invalidParticles.push_back(std::move(_towerBlock[i].particleVector()));
96 }
97
98 // resize to number of towers.
99 // Attention! This uses the dummy constructor so we still need to set the desired cluster size.
100 _towerBlock.resize(towerSideLength, numTowersPerDim);
101
102 // create more towers if needed and make an estimate for how many particles memory needs to be allocated
103 // Factor is more or less a random guess.
104 // Historically 2.7 used to be good but in other tests there was no significant difference to lower values.
105 const auto sizeEstimation =
106 static_cast<size_t>((static_cast<double>(numParticles) / static_cast<double>(numTowersNew)) * 1.2);
107 for (auto &tower : _towerBlock) {
108 // Set potentially new towers to the desired cluster size
109 tower.setClusterSize(_clusterSize);
110 tower.reserve(sizeEstimation);
111 }
112
113 sortParticlesIntoTowers(invalidParticles);
114
115 // estimate the number of clusters by particles divided by cluster size + one extra per tower.
116 _neighborListsBuffer.reserveNeighborLists(numParticles / _clusterSize + numTowersNew);
117 // generate clusters and count them
118 size_t numClusters = 0;
119 for (auto &tower : _towerBlock) {
120 numClusters += tower.generateClusters();
121 for (auto clusterIter = _newton3 ? tower.getClusters().begin() : tower.getFirstOwnedCluster();
122 clusterIter < (_newton3 ? tower.getClusters().end() : tower.getFirstTailHaloCluster()); ++clusterIter) {
123 // VCL stores the references to the lists in the clusters, therefore there is no need to create a
124 // cluster -> list lookup structure in the buffer structure
125 const auto listID = _neighborListsBuffer.getNewNeighborList();
126 clusterIter->setNeighborList(&(_neighborListsBuffer.template getNeighborListRef<false>(listID)));
127 }
128 }
129
130 return numClusters;
131 }

266 template <class FunType>
267 void iterateNeighborTowers(const int towerIndexX, const int towerIndexY, const int minNeighborIndexX,
268 const int maxNeighborIndexX, const int minNeighborIndexY, const int maxNeighborIndexY,
269 FunType function) {
270 auto &tower = _towerBlock.getTowerByIndex2D(towerIndexX, towerIndexY);
271 // for all neighbor towers
272 for (int neighborIndexY = minNeighborIndexY; neighborIndexY <= maxNeighborIndexY; neighborIndexY++) {
273 double distBetweenTowersY =
274 std::max(0, std::abs(towerIndexY - neighborIndexY) - 1) * _towerBlock.getTowerSideLength()[1];
275
276 for (int neighborIndexX = minNeighborIndexX; neighborIndexX <= maxNeighborIndexX; neighborIndexX++) {
277 if (_newton3 and not isForwardNeighbor(towerIndexX, towerIndexY, neighborIndexX, neighborIndexY)) {
278 continue;
279 }
280
281 double distBetweenTowersX =
282 std::max(0, std::abs(towerIndexX - neighborIndexX) - 1) * _towerBlock.getTowerSideLength()[0];
283
284 // calculate distance in xy-plane
285 auto distBetweenTowersXYsqr = distBetweenTowersX * distBetweenTowersX + distBetweenTowersY * distBetweenTowersY;
286 // skip if already longer than interactionLength
287 if (distBetweenTowersXYsqr <= _interactionLengthSqr) {
288 auto &neighborTower = _towerBlock.getTowerByIndex2D(neighborIndexX, neighborIndexY);
289
290 function(tower, neighborTower);
291 }
292 }
293 }

354 void calculateNeighborsBetweenTowers(internal::ClusterTower<Particle> &towerA,
355 internal::ClusterTower<Particle> &towerB) {
356 using autopas::utils::ArrayMath::boxDistanceSquared;
357
358 const auto interactionLengthFracOfDomainZ =
359 _towerBlock.getInteractionLength() / (_towerBlock.getHaloBoxMax()[2] - _towerBlock.getHaloBoxMin()[2]);
360 const bool isSameTower = (&towerA == &towerB);
361 // This heuristic seems to find a good middle ground between not too much memory allocated and no additional
362 // allocations when calling clusterA.addNeighbor(clusterB)
363 const auto neighborListReserveHeuristicFactor = (interactionLengthFracOfDomainZ * 2.1) / _clusterSize;
364 const auto isHaloCluster = [](const auto &clusterIter, const auto &tower) {
365 return clusterIter < tower.getFirstOwnedCluster() or clusterIter >= tower.getFirstTailHaloCluster();
366 };
367 // iterate over all clusters from tower A. In newton3 mode go over all of them, otherwise only owned.
368 for (auto clusterIterA = _newton3 ? towerA.getClusters().begin() : towerA.getFirstOwnedCluster();
369 clusterIterA < (_newton3 ? towerA.getClusters().end() : towerA.getFirstTailHaloCluster()); ++clusterIterA) {
370 if (not clusterIterA->empty()) {
371 clusterIterA->getNeighbors()->reserve((towerA.getNumActualParticles() + 8 * towerB.getNumActualParticles()) *
372 neighborListReserveHeuristicFactor);
373
374 // if we are within one tower depending on newton3 only look at forward neighbors
375 // clusterIterB can't be const because it will potentially be added as a non-const neighbor
376 for (auto clusterIterB = isSameTower and _newton3 ? clusterIterA + 1 : towerB.getClusters().begin();
377 clusterIterB < towerB.getClusters().end(); ++clusterIterB) {
378 // a cluster cannot be a neighbor to itself
379 if (&*clusterIterA == &*clusterIterB) {
380 continue;
381 }
382 // never do halo-halo interactions
383 if (isHaloCluster(clusterIterA, towerA) and isHaloCluster(clusterIterB, towerB)) {
384 continue;
385 }
386 if (not clusterIterB->empty()) {
387 const auto [aMin, aMax] = clusterIterA->getBoundingBox();
388 const auto [bMin, bMax] = clusterIterB->getBoundingBox();
389 const auto boxDistSquared = boxDistanceSquared(aMin, aMax, bMin, bMax);
390 if (boxDistSquared <= _interactionLengthSqr) {
391 clusterIterA->addNeighbor(*clusterIterB);
392 }
393 }
394 }
395 }
396 }
