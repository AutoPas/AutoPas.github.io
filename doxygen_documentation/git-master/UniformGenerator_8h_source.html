96template <class Container, class Particle, class HaloAddFunction>
97void UniformGenerator::fillWithHaloParticles(Container &container, const Particle &defaultParticle, double haloWidth,
98 unsigned long numParticles, const HaloAddFunction &haloAddFunction,
99 unsigned int seed) {
100 std::mt19937 generator(seed);
101
102 auto haloBoxMin = container.getBoxMin();
103 auto haloBoxMax = container.getBoxMax();
104
105 // increase the box size not exactly by the width to make it exclusive
106 for (unsigned int i = 0; i < 3; ++i) {
107 haloBoxMin[i] -= haloWidth * .99;
108 haloBoxMax[i] += haloWidth * .99;
109 }
110
111 for (unsigned long i = defaultParticle.getID(); i < defaultParticle.getID() + numParticles; ++i) {
112 auto pos = randomPosition(generator, haloBoxMin, haloBoxMax);
113 // we only want to add particles not in the actual box
114 while (autopas::utils::inBox(pos, container.getBoxMin(), container.getBoxMax())) {
115 pos = randomPosition(generator, haloBoxMin, haloBoxMax);
116 }
117 Particle particle(defaultParticle);
118 particle.setR(pos);
119 particle.setID(i);
120 haloAddFunction(container, particle);
121 }
122}
