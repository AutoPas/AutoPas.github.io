42 inline void AoSFunctor(Particle &i, Particle &j, bool newton3 = true) override {
43 if (i.isDummy() or j.isDummy()) {
44 return;
45 }
46 const std::array<double, 3> dr = utils::ArrayMath::sub(j.getR(), i.getR()); // ep_j[j].pos - ep_i[i].pos;
47 const double density =
48 j.getMass() * SPHKernels::W(dr, i.getSmoothingLength()); // ep_j[j].mass * W(dr, ep_i[i].smth)
49 i.addDensity(density);
50 if (newton3) {
51 // Newton 3:
52 // W is symmetric in dr, so no -dr needed, i.e. we can reuse dr
53 const double density2 = i.getMass() * SPHKernels::W(dr, j.getSmoothingLength());
54 j.addDensity(density2);
55 }
56 }

88 size_t numParticles = soa.getNumberOfParticles();
89 for (unsigned int i = 0; i < numParticles; ++i) {
90 // checks whether particle i is owned.
91 if (ownedStatePtr[i] == OwnershipState::dummy) {
92 continue;
93 }
94
95 double densacc = 0.;
96// icpc vectorizes this.
97// g++ only with -ffast-math or -funsafe-math-optimizations
98#pragma omp simd reduction(+ : densacc)
99 for (unsigned int j = i + 1; j < numParticles; ++j) {
100 const double drx = xptr[i] - xptr[j];
101 const double dry = yptr[i] - yptr[j];
102 const double drz = zptr[i] - zptr[j];
103
104 const double drx2 = drx * drx;
105 const double dry2 = dry * dry;
106 const double drz2 = drz * drz;
107
108 const double dr2 = drx2 + dry2 + drz2;
109
110 // if second particle is a dummy, we skip the interaction.
111 const bool mask = ownedStatePtr[j] != OwnershipState::dummy;
112
113 const double density = mask ? massptr[j] * SPHKernels::W(dr2, smthptr[i]) : 0.;
114 densacc += density;
115
116 // Newton 3:
117 // W is symmetric in dr, so no -dr needed, i.e. we can reuse dr
118 const double density2 = mask ? massptr[i] * SPHKernels::W(dr2, smthptr[j]) : 0.;
119 densityptr[j] += density2;
120 }
121
122 densityptr[i] += densacc;
