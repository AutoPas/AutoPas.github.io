48  inline void AoSFunctor(Particle &i, Particle &j, bool newton3 = true) override {
49  if (i.isDummy() or j.isDummy()) {
50  return;
51  }
52  const std::array<double, 3> dr = utils::ArrayMath::sub(j.getR(), i.getR()); // ep_j[j].pos - ep_i[i].pos;
53  const double density =
54  j.getMass() * SPHKernels::W(dr, i.getSmoothingLength()); // ep_j[j].mass * W(dr, ep_i[i].smth)
55  i.addDensity(density);
56  if (newton3) {
57  // Newton 3:
58  // W is symmetric in dr, so no -dr needed, i.e. we can reuse dr
59  const double density2 = i.getMass() * SPHKernels::W(dr, j.getSmoothingLength());
60  j.addDensity(density2);
61  }
62  }

108  const double drz = zptr[i] - zptr[j];

112  const double drz2 = drz * drz;
