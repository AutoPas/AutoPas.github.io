42 inline void AoSFunctor(Particle &i, Particle &j, bool newton3 = true) override {
43 using namespace autopas::utils::ArrayMath::literals;
44
45 if (i.isDummy() or j.isDummy()) {
46 return;
47 }
48 const std::array<double, 3> dr = j.getR() - i.getR(); // ep_j[j].pos - ep_i[i].pos;
49 const double density =
50 j.getMass() * SPHKernels::W(dr, i.getSmoothingLength()); // ep_j[j].mass * W(dr, ep_i[i].smth)
51 i.addDensity(density);
52 if (newton3) {
53 // Newton 3:
54 // W is symmetric in dr, so no -dr needed, i.e. we can reuse dr
55 const double density2 = i.getMass() * SPHKernels::W(dr, j.getSmoothingLength());
56 j.addDensity(density2);
57 }

84 double *const __restrict densityptr = soa.template begin<Particle::AttributeNames::density>();

88 const auto *const __restrict ownedStatePtr = soa.template begin<Particle::AttributeNames::ownershipState>();
89
90 size_t numParticles = soa.size();
91 for (unsigned int i = 0; i < numParticles; ++i) {
92 // checks whether particle i is owned.
93 if (ownedStatePtr[i] == autopas::OwnershipState::dummy) {
94 continue;
95 }
96
97 double densacc = 0.;
98// icpc vectorizes this.
99// g++ only with -ffast-math or -funsafe-math-optimizations
100#pragma omp simd reduction(+ : densacc)
101 for (unsigned int j = i + 1; j < numParticles; ++j) {
102 const double drx = xptr[i] - xptr[j];
103 const double dry = yptr[i] - yptr[j];
104 const double drz = zptr[i] - zptr[j];
105
106 const double drx2 = drx * drx;
107 const double dry2 = dry * dry;
108 const double drz2 = drz * drz;
109
110 const double dr2 = drx2 + dry2 + drz2;
111
112 // if second particle is a dummy, we skip the interaction.
113 const bool mask = ownedStatePtr[j] != autopas::OwnershipState::dummy;
114
115 const double density = mask ? massptr[j] * SPHKernels::W(dr2, smthptr[i]) : 0.;
116 densacc += density;
117
118 // Newton 3:
119 // W is symmetric in dr, so no -dr needed, i.e. we can reuse dr
120 const double density2 = mask ? massptr[i] * SPHKernels::W(dr2, smthptr[j]) : 0.;
121 densityptr[j] += density2;
122 }
123
124 densityptr[i] += densacc;
