47  inline void AoSFunctor(Particle &i, Particle &j, bool newton3 = true) override {
48  if (i.isDummy() or j.isDummy()) {
49  return;
50  }
51  const std::array<double, 3> dr = utils::ArrayMath::sub(j.getR(), i.getR()); // ep_j[j].pos - ep_i[i].pos;
52  const double density =
53  j.getMass() * SPHKernels::W(dr, i.getSmoothingLength()); // ep_j[j].mass * W(dr, ep_i[i].smth)
54  i.addDensity(density);
55  if (newton3) {
56  // Newton 3:
57  // W is symmetric in dr, so no -dr needed, i.e. we can reuse dr
58  const double density2 = i.getMass() * SPHKernels::W(dr, j.getSmoothingLength());
59  j.addDensity(density2);
60  }
