171inline void LCC04SoACellHandler<ParticleCell, PairwiseFunctor, dataLayout, useNewton3>::processBaseCell(
172 std::vector<ParticleCell> &cells, unsigned long x, unsigned long y, unsigned long z) {
173 const unsigned long baseIndex = utils::ThreeDimensionalMapping::threeToOneD(x, y, z, _cellsPerDimension);
174
175 // get all information for current thread
176 const auto threadID = static_cast<size_t>(autopas_get_thread_num());
177 auto &currentSlice = _currentSlices[threadID * _cacheOffset];
178 auto &combinationSlice = _combinationSlices[threadID];
179 auto &combinationSlicesOffsets = _combinationSlicesOffsets[threadID];
180
181 const size_t numSlices = _baseOffsets.size();
182
183 // First cell needs to initialize whole buffer
184 if (x == 0ul) {
185 currentSlice = 0ul;
186 for (unsigned int slice = 0ul; slice < numSlices; slice++) {
187 writeCellIntoBuffer(cells, baseIndex, combinationSlice, combinationSlicesOffsets, slice, slice);
188 }
189 } else {
190 writeCellIntoBuffer(cells, baseIndex, combinationSlice, combinationSlicesOffsets, currentSlice, numSlices - 1ul);
191
192 ++currentSlice %= numSlices;
193 }
194
195 // compute interactions
196 for (unsigned long slice = 0; slice < numSlices; slice++) {
197 for (auto const &[offset1, interval] : _offsets[(slice + currentSlice) % numSlices]) {
198 ParticleCell *cell1 = nullptr;
199 size_t cell1ViewStart = 0;
200 size_t cell1ViewEnd;
201
202 // special cases (cell1 one is also stored in a combination slice)
203 // base cell
204 if (offset1 == 0ul) {
205 const auto numParticlesBaseCell = cells[baseIndex].numParticles();
206 if (numParticlesBaseCell == 0) {
207 continue;
208 }
209 // two cases interval in current stripe
210 cell1 = &combinationSlice[currentSlice];
211 auto stripeView = cell1->_particleSoABuffer.constructView(0, numParticlesBaseCell);
212 if (slice == currentSlice) {
213 // Process stripe with itself
214 _pairwiseFunctor->SoAFunctorSingle(stripeView, useNewton3);
215
216 auto restView = cell1->_particleSoABuffer.constructView(numParticlesBaseCell,
217 cell1->_particleSoABuffer.getNumParticles());
218 _pairwiseFunctor->SoAFunctorPair(stripeView, restView, useNewton3);
219 if (not useNewton3) {
220 _pairwiseFunctor->SoAFunctorPair(restView, stripeView, useNewton3);
221 }
222 cell1ViewEnd = cell1->_particleSoABuffer.getNumParticles();
223 continue;
224 } else {
225 // interval in other stripe
226 cell1ViewEnd = numParticlesBaseCell;
227 }
228 } else if (offset1 == _baseOffsets.front().back()) {
229 cell1 = &combinationSlice[currentSlice];
230 cell1ViewStart = combinationSlicesOffsets[currentSlice][combinationSlicesOffsets[currentSlice].size() - 2];
231 cell1ViewEnd = cell1->_particleSoABuffer.getNumParticles();
232 } else if (offset1 == _baseOffsets.back().front()) {
233 const auto index = (currentSlice + numSlices - 1) % numSlices;
234 if (combinationSlicesOffsets[index][1] == 0) {
235 continue;
236 }
237 cell1 = &combinationSlice[index];
238 cell1ViewEnd = combinationSlicesOffsets[index][1];
239 } else {
240 const unsigned long cellIndex1 = baseIndex + offset1;
241 cell1 = &cells[cellIndex1];
242 cell1ViewEnd = cell1->_particleSoABuffer.getNumParticles();
243 }
244
245 auto &currentCS = combinationSlice[slice];
246 const auto &currentCSOffsets = combinationSlicesOffsets[slice];
247 auto currentCSViewStart = currentCSOffsets[interval.first];
248 auto currentCSViewEnd = currentCSOffsets[interval.second];
249
250 auto cell1View = cell1->_particleSoABuffer.constructView(cell1ViewStart, cell1ViewEnd);
251 auto currentCSView = currentCS._particleSoABuffer.constructView(currentCSViewStart, currentCSViewEnd);
252 _pairwiseFunctor->SoAFunctorPair(cell1View, currentCSView, useNewton3);
253 if (not useNewton3) {
254 _pairwiseFunctor->SoAFunctorPair(currentCSView, cell1View, useNewton3);
255 }
256 }
257 }
258
259 // write information of combined SoA buffers back to cell
260 writeBufferIntoCell(cells, baseIndex, combinationSlice, combinationSlicesOffsets, currentSlice, 0ul);
261
262 // last cell in stripe need to save whole buffer
263 if (x == _cellsPerDimension[0] - _overlap[0] - 1l) {
264 for (unsigned long slice = 1; slice < numSlices; slice++) {
265 const long bufferSlice = (currentSlice + slice) % numSlices;
266 writeBufferIntoCell(cells, baseIndex, combinationSlice, combinationSlicesOffsets, bufferSlice, slice);
267 }
268 }
269}

272inline void LCC04SoACellHandler<ParticleCell, PairwiseFunctor, dataLayout, useNewton3>::writeCellIntoBuffer(
273 const std::vector<ParticleCell> &cells, const unsigned long baseIndex, std::vector<ParticleCell> &combinationSlice,
274 std::vector<std::vector<unsigned long>> &combinationSlicesOffsets, const unsigned int bufferSlice,
275 const unsigned int cellSlice) {
276 // delete old data
277 combinationSlice[bufferSlice]._particleSoABuffer.clear();
278 combinationSlicesOffsets[bufferSlice].clear();
279 // fill buffer and compute partial sums
280 unsigned long sum = 0ul;
281 combinationSlicesOffsets[bufferSlice].push_back(sum);
282 for (const auto offset : _baseOffsets[cellSlice]) {
283 const unsigned long otherIndex = baseIndex + offset;
284 const ParticleCell &otherCell = cells[otherIndex];
285 combinationSlice[bufferSlice]._particleSoABuffer.append(otherCell._particleSoABuffer);
286 sum += otherCell.numParticles();
287 combinationSlicesOffsets[bufferSlice].push_back(sum);
288 }

292inline void LCC04SoACellHandler<ParticleCell, PairwiseFunctor, dataLayout, useNewton3>::writeBufferIntoCell(
293 std::vector<ParticleCell> &cells, const unsigned long baseIndex, std::vector<ParticleCell> &combinationSlice,
294 std::vector<std::vector<unsigned long>> &combinationSlicesOffsets, const unsigned long bufferSlice,
295 const unsigned long cellSlice) {
296 auto &buffer = combinationSlice[bufferSlice]._particleSoABuffer;
297 for (long i = _baseOffsets[cellSlice].size() - 1l; i >= 0; i--) {
298 const auto start = combinationSlicesOffsets[bufferSlice][i];
299 const auto end = combinationSlicesOffsets[bufferSlice][i + 1];
300
301 if (start == end) {
302 continue;
303 }
304 buffer.resizeArrays(end);
305 auto bufferView = buffer.constructView(start, buffer.getNumParticles());
306
307 const unsigned long currentOffset = baseIndex + _baseOffsets[cellSlice][i];
308 // clear old cell buffer
309 cells[currentOffset]._particleSoABuffer.clear();
310 // make sure everything is correct
311 if (bufferView.getNumParticles() != cells[currentOffset].numParticles()) {
312 const auto pos = utils::ThreeDimensionalMapping::oneToThreeD(currentOffset, _cellsPerDimension);
313 AutoPasLog(error,
314 "Particle number in SoA buffer and cell doesn't match. current position: [{} {} {}] is: {} should: {}",
315 pos[0], pos[1], pos[2], buffer.getNumParticles(), cells[currentOffset].numParticles());
316 }
317 // append new cell buffer
318 cells[currentOffset]._particleSoABuffer.append(bufferView);
319 }

323inline void LCC04SoACellHandler<ParticleCell, PairwiseFunctor, dataLayout, useNewton3>::computeOffsets(
324 std::array<unsigned long, 3> cellsPerDimension) {
325 using std::make_pair;
326
327 std::vector<std::vector<std::pair<unsigned long, unsigned long>>> cellPairOffsets;
328
330 // @TODO: Replace following lines with vector to support asymmetric cells
331 const unsigned long ov1 = _overlap[0] + 1;
332 const unsigned long ov1_squared = ov1 * ov1;
334
335 _baseOffsets.resize(ov1);
336
337 std::array<unsigned long, 3> overlap_1 = utils::ArrayMath::addScalar(_overlap, 1ul);
338
339 std::vector<unsigned long> cellOffsets;
340 cellOffsets.reserve(overlap_1[0] * overlap_1[1] * overlap_1[2]);
341
342 cellPairOffsets.clear();
343 cellPairOffsets.resize(overlap_1[0]);
344
345 const auto interactionLengthSquare(this->_interactionLength * this->_interactionLength);
346
347 for (unsigned long x = 0ul; x <= _overlap[0]; ++x) {
348 for (unsigned long y = 0ul; y <= _overlap[1]; ++y) {
349 _baseOffsets[x].push_back(utils::ThreeDimensionalMapping::threeToOneD(x, y, 0ul, cellsPerDimension));
350 for (unsigned long z = 0ul; z <= _overlap[2]; ++z) {
351 cellOffsets.push_back(utils::ThreeDimensionalMapping::threeToOneD(x, y, z, cellsPerDimension));
352 }
353 }
354 }
355 for (unsigned long x = 0ul; x <= _overlap[0]; ++x) {
356 for (unsigned long y = 0ul; y <= _overlap[1]; ++y) {
357 for (unsigned long z = 0ul; z <= _overlap[2]; ++z) {
358 const unsigned long offset = cellOffsets[ov1_squared * x + ov1 * y];
359 // origin
360 {
361 // check whether cell is within cutoff radius
362 auto distVec = utils::ArrayMath::mul({std::max(0.0, x - 1.0), std::max(0.0, y - 1.0), std::max(0.0, z - 1.0)},
363 _cellLength);
364 const auto distSquare = utils::ArrayMath::dot(distVec, distVec);
365 if (distSquare <= interactionLengthSquare) {
366 cellPairOffsets[x].push_back(make_pair(cellOffsets[z], offset));
367 }
368 }
369 // back left
370 if (y != _overlap[1] and z != 0) {
371 // check whether cell is within cutoff radius
372 auto distVec = utils::ArrayMath::mul(
373 {std::max(0.0, x - 1.0), std::max(0.0, _overlap[1] - y - 1.0), std::max(0.0, z - 1.0)}, _cellLength);
374 const auto distSquare = utils::ArrayMath::dot(distVec, distVec);
375 if (distSquare <= interactionLengthSquare) {
376 cellPairOffsets[x].push_back(make_pair(cellOffsets[ov1_squared - ov1 + z], offset));
377 }
378 }
379 // front right
380 if (x != _overlap[0] and (y != 0 or z != 0)) {
381 // check whether cell is within cutoff radius
382 auto distVec = utils::ArrayMath::mul(
383 {std::max(0.0, _overlap[0] - x - 1.0), std::max(0.0, y - 1.0), std::max(0.0, z - 1.0)}, _cellLength);
384 const auto distSquare = utils::ArrayMath::dot(distVec, distVec);
385 if (distSquare <= interactionLengthSquare) {
386 cellPairOffsets[x].push_back(make_pair(cellOffsets[ov1_squared * _overlap[0] + z], offset));
387 }
388 }
389 // back right
390 if (y != _overlap[1] and x != _overlap[0] and z != 0) {
391 // check whether cell is within cutoff radius
392 auto distVec = utils::ArrayMath::mul(
393 {std::max(0.0, _overlap[0] - x - 1.0), std::max(0.0, _overlap[1] - y - 1.0), std::max(0.0, z - 1.0)},
394 _cellLength);
395 const auto distSquare = utils::ArrayMath::dot(distVec, distVec);
396 if (distSquare <= interactionLengthSquare) {
397 cellPairOffsets[x].push_back(make_pair(cellOffsets[ov1_squared * ov1 - ov1 + z], offset));
398 }
399 }
400 }
401 }
402 }
403 setupIntervals(cellPairOffsets);
404}

406template <class ParticleCell, class PairwiseFunctor, DataLayoutOption::Value dataLayout, bool useNewton3>
407inline void LCC04SoACellHandler<ParticleCell, PairwiseFunctor, dataLayout, useNewton3>::setupIntervals(
408 std::vector<std::vector<std::pair<unsigned long, unsigned long>>> &cellPairOffsets) {
409 // Create intervals
410 const unsigned long numStripes = cellPairOffsets.size();
411 _offsets.resize(numStripes);
412
413 // iterate over all stripes
414 for (unsigned long i = 0; i < numStripes; ++i) {
415 auto &currentStripe = cellPairOffsets[i];
416 // Sort
417 std::stable_sort(currentStripe.begin(), currentStripe.end(),
418 [](const auto &a, const auto &b) -> bool { return a.first < b.first; });
419
420 // Collect intervals
421 unsigned long current = currentStripe.front().first;
422 unsigned long startID =
423 std::distance(_baseOffsets[i].begin(),
424 std::find(_baseOffsets[i].begin(), _baseOffsets[i].end(), currentStripe.front().second));
425 unsigned long endID = startID;
426 for (unsigned long j = 0; j < currentStripe.size(); j++) {
427 if (current != currentStripe[j].first) {
428 auto interval = std::make_pair(startID, endID);
429 _offsets[i].push_back(std::make_pair(current, interval));
430 startID = std::distance(_baseOffsets[i].begin(),
431 std::find(_baseOffsets[i].begin(), _baseOffsets[i].end(), currentStripe[j].second));
432 endID = startID;
433 current = currentStripe[j].first;
434 }
435 endID++;
436 }
437 // last interval
438 auto interval = std::make_pair(startID, endID);
439 _offsets[i].push_back(std::make_pair(current, interval));
440 }
441}

447 _combinationSlices.resize(numThreads);
448 const auto cellOffsetsSize = _overlap[0] + 1;

452 std::for_each(_combinationSlicesOffsets.begin(), _combinationSlicesOffsets.end(),
453 [cellOffsetsSize](auto &e) { e.resize(cellOffsetsSize); });
454 _currentSlices.resize(numThreads * _cacheOffset);
