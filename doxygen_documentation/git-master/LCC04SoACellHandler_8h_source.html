183template <class ParticleCell, class PairwiseFunctor>
184inline void LCC04SoACellHandler<ParticleCell, PairwiseFunctor>::processBaseCell(std::vector<ParticleCell> &cells,
185 unsigned long x, unsigned long y,
186 unsigned long z) {
187 const unsigned long baseIndex = utils::ThreeDimensionalMapping::threeToOneD(x, y, z, _cellsPerDimension);
188
189 // get all information for current thread
190 const auto threadID = static_cast<size_t>(autopas_get_thread_num());
191 auto &currentSlice = _currentSlices[threadID * _cacheOffset];
192 auto &combinationSlice = _combinationSlices[threadID];
193 auto &combinationSlicesOffsets = _combinationSlicesOffsets[threadID];
194
195 const size_t numSlices = _baseOffsets.size();
196
197 // First cell needs to initialize whole buffer
198 if (x == 0ul) {
199 currentSlice = 0ul;
200 for (unsigned int slice = 0ul; slice < numSlices; slice++) {
201 writeCellIntoBuffer(cells, baseIndex, combinationSlice, combinationSlicesOffsets, slice, slice);
202 }
203 } else {
204 writeCellIntoBuffer(cells, baseIndex, combinationSlice, combinationSlicesOffsets, currentSlice, numSlices - 1ul);
205
206 ++currentSlice %= numSlices;
207 }
208
209 // compute interactions
210 for (unsigned long slice = 0; slice < numSlices; slice++) {
211 for (auto const &[offset1, interval] : _offsets[(slice + currentSlice) % numSlices]) {
212 ParticleCell *cell1 = nullptr;
213 size_t cell1ViewStart = 0;
214 size_t cell1ViewEnd;
215
216 // special cases (cell1 one is also stored in a combination slice)
217 // base cell
218 if (offset1 == 0ul) {
219 const auto numParticlesBaseCell = cells[baseIndex].size();
220 if (numParticlesBaseCell == 0) {
221 continue;
222 }
223 // two cases interval in current stripe
224 cell1 = &combinationSlice[currentSlice];
225 auto stripeView = cell1->_particleSoABuffer.constructView(0, numParticlesBaseCell);
226 if (slice == currentSlice) {
227 // Process stripe with itself
228 _pairwiseFunctor->SoAFunctorSingle(stripeView, _useNewton3);
229
230 auto restView =
231 cell1->_particleSoABuffer.constructView(numParticlesBaseCell, cell1->_particleSoABuffer.size());
232 _pairwiseFunctor->SoAFunctorPair(stripeView, restView, _useNewton3);
233 if (not _useNewton3) {
234 _pairwiseFunctor->SoAFunctorPair(restView, stripeView, _useNewton3);
235 }
236 cell1ViewEnd = cell1->_particleSoABuffer.size();
237 continue;
238 } else {
239 // interval in other stripe
240 cell1ViewEnd = numParticlesBaseCell;
241 }
242 } else if (offset1 == _baseOffsets.front().back()) {
243 cell1 = &combinationSlice[currentSlice];
244 cell1ViewStart = combinationSlicesOffsets[currentSlice][combinationSlicesOffsets[currentSlice].size() - 2];
245 cell1ViewEnd = cell1->_particleSoABuffer.size();
246 } else if (offset1 == _baseOffsets.back().front()) {
247 const auto index = (currentSlice + numSlices - 1) % numSlices;
248 if (combinationSlicesOffsets[index][1] == 0) {
249 continue;
250 }
251 cell1 = &combinationSlice[index];
252 cell1ViewEnd = combinationSlicesOffsets[index][1];
253 } else {
254 const unsigned long cellIndex1 = baseIndex + offset1;
255 cell1 = &cells[cellIndex1];
256 cell1ViewEnd = cell1->_particleSoABuffer.size();
257 }
258
259 auto &currentCS = combinationSlice[slice];
260 const auto &currentCSOffsets = combinationSlicesOffsets[slice];
261 auto currentCSViewStart = currentCSOffsets[interval.first];
262 auto currentCSViewEnd = currentCSOffsets[interval.second];
263
264 auto cell1View = cell1->_particleSoABuffer.constructView(cell1ViewStart, cell1ViewEnd);
265 auto currentCSView = currentCS._particleSoABuffer.constructView(currentCSViewStart, currentCSViewEnd);
266 _pairwiseFunctor->SoAFunctorPair(cell1View, currentCSView, _useNewton3);
267 if (not _useNewton3) {
268 _pairwiseFunctor->SoAFunctorPair(currentCSView, cell1View, _useNewton3);
269 }
270 }
271 }
272
273 // write information of combined SoA buffers back to cell
274 writeBufferIntoCell(cells, baseIndex, combinationSlice, combinationSlicesOffsets, currentSlice, 0ul);
275
276 // last cell in stripe need to save whole buffer
277 if (x == _cellsPerDimension[0] - _overlap[0] - 1l) {
278 for (unsigned long slice = 1; slice < numSlices; slice++) {
279 const long bufferSlice = (currentSlice + slice) % numSlices;
280 writeBufferIntoCell(cells, baseIndex, combinationSlice, combinationSlicesOffsets, bufferSlice, slice);
281 }
282 }
283}

286inline void LCC04SoACellHandler<ParticleCell, PairwiseFunctor>::writeCellIntoBuffer(
287 const std::vector<ParticleCell> &cells, const unsigned long baseIndex, std::vector<ParticleCell> &combinationSlice,
288 std::vector<std::vector<unsigned long>> &combinationSlicesOffsets, const unsigned int bufferSlice,
289 const unsigned int cellSlice) {
290 // delete old data
291 combinationSlice[bufferSlice]._particleSoABuffer.clear();
292 combinationSlicesOffsets[bufferSlice].clear();
293 // fill buffer and compute partial sums
294 unsigned long sum = 0ul;
295 combinationSlicesOffsets[bufferSlice].push_back(sum);
296 for (const auto offset : _baseOffsets[cellSlice]) {
297 const unsigned long otherIndex = baseIndex + offset;
298 const ParticleCell &otherCell = cells[otherIndex];
299 combinationSlice[bufferSlice]._particleSoABuffer.append(otherCell._particleSoABuffer);
300 sum += otherCell.size();
301 combinationSlicesOffsets[bufferSlice].push_back(sum);
302 }

305template <class ParticleCell, class PairwiseFunctor>
306inline void LCC04SoACellHandler<ParticleCell, PairwiseFunctor>::writeBufferIntoCell(
307 std::vector<ParticleCell> &cells, const unsigned long baseIndex, std::vector<ParticleCell> &combinationSlice,
308 std::vector<std::vector<unsigned long>> &combinationSlicesOffsets, const unsigned long bufferSlice,
309 const unsigned long cellSlice) {
310 auto &buffer = combinationSlice[bufferSlice]._particleSoABuffer;
311 for (long i = _baseOffsets[cellSlice].size() - 1l; i >= 0; i--) {
312 const auto start = combinationSlicesOffsets[bufferSlice][i];
313 const auto end = combinationSlicesOffsets[bufferSlice][i + 1];
314
315 if (start == end) {
316 continue;
317 }
318 buffer.resizeArrays(end);
319 auto bufferView = buffer.constructView(start, buffer.size());
320
321 const unsigned long currentOffset = baseIndex + _baseOffsets[cellSlice][i];
322 // clear old cell buffer
323 cells[currentOffset]._particleSoABuffer.clear();
324 // make sure everything is correct
325 if (bufferView.size() != cells[currentOffset].size()) {
326 const auto pos = utils::ThreeDimensionalMapping::oneToThreeD(currentOffset, _cellsPerDimension);
327 AutoPasLog(ERROR,
328 "Particle number in SoA buffer and cell doesn't match. current position: [{} {} {}] is: {} should: {}",
329 pos[0], pos[1], pos[2], buffer.size(), cells[currentOffset].size());
330 }
331 // append new cell buffer
332 cells[currentOffset]._particleSoABuffer.append(bufferView);
333 }

337inline void LCC04SoACellHandler<ParticleCell, PairwiseFunctor>::computeOffsets(
338 const std::array<unsigned long, 3> &cellsPerDimension) {
339 using namespace autopas::utils::ArrayMath::literals;
340 using std::make_pair;
341
342 std::vector<std::vector<std::pair<unsigned long, unsigned long>>> cellPairOffsets;
343
345 // @TODO: Replace following lines with vector to support asymmetric cells
346 const unsigned long ov1 = _overlap[0] + 1;
347 const unsigned long ov1_squared = ov1 * ov1;
349
350 _baseOffsets.resize(ov1);
351
352 const std::array<unsigned long, 3> overlap_1 = _overlap + 1ul;
353
354 std::vector<unsigned long> cellOffsets;
355 cellOffsets.reserve(overlap_1[0] * overlap_1[1] * overlap_1[2]);
356
357 cellPairOffsets.clear();
358 cellPairOffsets.resize(overlap_1[0]);
359
360 const auto interactionLengthSquare(this->_interactionLength * this->_interactionLength);
361
362 for (unsigned long x = 0ul; x <= _overlap[0]; ++x) {
363 for (unsigned long y = 0ul; y <= _overlap[1]; ++y) {
364 _baseOffsets[x].push_back(utils::ThreeDimensionalMapping::threeToOneD(x, y, 0ul, cellsPerDimension));
365 for (unsigned long z = 0ul; z <= _overlap[2]; ++z) {
366 cellOffsets.push_back(utils::ThreeDimensionalMapping::threeToOneD(x, y, z, cellsPerDimension));
367 }
368 }
369 }
370 for (unsigned long x = 0ul; x <= _overlap[0]; ++x) {
371 for (unsigned long y = 0ul; y <= _overlap[1]; ++y) {
372 for (unsigned long z = 0ul; z <= _overlap[2]; ++z) {
373 const unsigned long offset = cellOffsets[ov1_squared * x + ov1 * y];
374 // origin
375 {
376 // check whether cell is within cutoff radius
377 const auto distVec =
378 std::array<double, 3>{std::max(0.0, x - 1.0), std::max(0.0, y - 1.0), std::max(0.0, z - 1.0)} *
379 _cellLength;
380 const auto distSquare = utils::ArrayMath::dot(distVec, distVec);
381 if (distSquare <= interactionLengthSquare) {
382 cellPairOffsets[x].push_back(make_pair(cellOffsets[z], offset));
383 }
384 }
385 // back left
386 if (y != _overlap[1] and z != 0) {
387 // check whether cell is within cutoff radius
388 const auto distVec = std::array<double, 3>{std::max(0.0, x - 1.0), std::max(0.0, _overlap[1] - y - 1.0),
389 std::max(0.0, z - 1.0)} *
390 _cellLength;
391 const auto distSquare = utils::ArrayMath::dot(distVec, distVec);
392 if (distSquare <= interactionLengthSquare) {
393 cellPairOffsets[x].push_back(make_pair(cellOffsets[ov1_squared - ov1 + z], offset));
394 }
395 }
396 // front right
397 if (x != _overlap[0] and (y != 0 or z != 0)) {
398 // check whether cell is within cutoff radius
399 const auto distVec = std::array<double, 3>{std::max(0.0, _overlap[0] - x - 1.0), std::max(0.0, y - 1.0),
400 std::max(0.0, z - 1.0)} *
401 _cellLength;
402 const auto distSquare = utils::ArrayMath::dot(distVec, distVec);
403 if (distSquare <= interactionLengthSquare) {
404 cellPairOffsets[x].push_back(make_pair(cellOffsets[ov1_squared * _overlap[0] + z], offset));
405 }
406 }
407 // back right
408 if (y != _overlap[1] and x != _overlap[0] and z != 0) {
409 // check whether cell is within cutoff radius
410 const auto distVec = std::array<double, 3>{std::max(0.0, _overlap[0] - x - 1.0),
411 std::max(0.0, _overlap[1] - y - 1.0), std::max(0.0, z - 1.0)} *
412 _cellLength;
413 const auto distSquare = utils::ArrayMath::dot(distVec, distVec);
414 if (distSquare <= interactionLengthSquare) {
415 cellPairOffsets[x].push_back(make_pair(cellOffsets[ov1_squared * ov1 - ov1 + z], offset));
416 }
417 }
418 }
419 }
420 }
421 setupIntervals(cellPairOffsets);
422}

424template <class ParticleCell, class PairwiseFunctor>
425inline void LCC04SoACellHandler<ParticleCell, PairwiseFunctor>::setupIntervals(
426 std::vector<std::vector<std::pair<unsigned long, unsigned long>>> &cellPairOffsets) {
427 // Create intervals
428 const unsigned long numStripes = cellPairOffsets.size();
429 _offsets.resize(numStripes);
430
431 // iterate over all stripes
432 for (unsigned long i = 0; i < numStripes; ++i) {
433 auto &currentStripe = cellPairOffsets[i];
434 // Sort
435 std::stable_sort(currentStripe.begin(), currentStripe.end(),
436 [](const auto &a, const auto &b) -> bool { return a.first < b.first; });
437
438 // Collect intervals
439 unsigned long current = currentStripe.front().first;
440 unsigned long startID =
441 std::distance(_baseOffsets[i].begin(),
442 std::find(_baseOffsets[i].begin(), _baseOffsets[i].end(), currentStripe.front().second));
443 unsigned long endID = startID;
444 for (unsigned long j = 0; j < currentStripe.size(); j++) {
445 if (current != currentStripe[j].first) {
446 auto interval = std::make_pair(startID, endID);
447 _offsets[i].push_back(std::make_pair(current, interval));
448 startID = std::distance(_baseOffsets[i].begin(),
449 std::find(_baseOffsets[i].begin(), _baseOffsets[i].end(), currentStripe[j].second));
450 endID = startID;
451 current = currentStripe[j].first;
452 }
453 endID++;
454 }
455 // last interval
456 auto interval = std::make_pair(startID, endID);
457 _offsets[i].push_back(std::make_pair(current, interval));
458 }
459}

462void LCC04SoACellHandler<ParticleCell, PairwiseFunctor>::resizeBuffers() {
463 const auto numThreads = static_cast<size_t>(autopas_get_max_threads());
464 if (_combinationSlices.size() != numThreads) {
465 _combinationSlices.resize(numThreads);
466 const auto cellOffsetsSize = _overlap[0] + 1;
467 std::for_each(_combinationSlices.begin(), _combinationSlices.end(),
468 [cellOffsetsSize](auto &e) { e.resize(cellOffsetsSize); });
469 _combinationSlicesOffsets.resize(numThreads);
470 std::for_each(_combinationSlicesOffsets.begin(), _combinationSlicesOffsets.end(),
471 [cellOffsetsSize](auto &e) { e.resize(cellOffsetsSize); });
472 _currentSlices.resize(numThreads * _cacheOffset);
473 }
