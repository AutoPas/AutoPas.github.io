107void SetSearchSpaceBasedTuningStrategy::populateSearchSpace(
108 const std::set<ContainerOption> &allowedContainerOptions, const std::set<double> &allowedCellSizeFactors,
109 const std::set<TraversalOption> &allowedTraversalOptions,
110 const std::set<LoadEstimatorOption> &allowedLoadEstimatorOptions,
111 const std::set<DataLayoutOption> &allowedDataLayoutOptions, const std::set<Newton3Option> &allowedNewton3Options) {
112 // generate all potential configs
113 for (const auto &containerOption : allowedContainerOptions) {
114 // get all traversals of the container and restrict them to the allowed ones
115 const std::set<TraversalOption> &allContainerTraversals =
116 compatibleTraversals::allCompatibleTraversals(containerOption);
117 std::set<TraversalOption> allowedAndApplicable;
118 std::set_intersection(allowedTraversalOptions.begin(), allowedTraversalOptions.end(),
119 allContainerTraversals.begin(), allContainerTraversals.end(),
120 std::inserter(allowedAndApplicable, allowedAndApplicable.begin()));
121
122 for (const auto &cellSizeFactor : allowedCellSizeFactors)
123 for (const auto &traversalOption : allowedAndApplicable) {
124 // if load estimators are not applicable LoadEstimatorOption::none is returned.
125 const std::set<LoadEstimatorOption> allowedAndApplicableLoadEstimators =
126 loadEstimators::getApplicableLoadEstimators(containerOption, traversalOption, allowedLoadEstimatorOptions);
127 for (const auto &loadEstimatorOption : allowedAndApplicableLoadEstimators) {
128 for (const auto &dataLayoutOption : allowedDataLayoutOptions) {
129 for (const auto &newton3Option : allowedNewton3Options) {
130 _searchSpace.emplace(containerOption, cellSizeFactor, traversalOption, loadEstimatorOption,
131 dataLayoutOption, newton3Option);
132 }
133 }
134 }
135 }
136 }
137
138 AutoPasLog(debug, "Points in search space: {}", _searchSpace.size());
139
140 if (_searchSpace.empty()) {
141 autopas::utils::ExceptionHandler::exception("No valid configurations could be created.");
142 }
143
144 _currentConfig = _searchSpace.begin();
145}

155void SetSearchSpaceBasedTuningStrategy::removeN3Option(Newton3Option badNewton3Option) {
156 for (auto ssIter = _searchSpace.begin(); ssIter != _searchSpace.end();) {
157 if (ssIter->newton3 == badNewton3Option) {
158 // change current config to the next non-deleted
159 if (ssIter == _currentConfig) {
160 ssIter = _searchSpace.erase(ssIter);
161 _currentConfig = ssIter;
162 } else {
163 ssIter = _searchSpace.erase(ssIter);
164 }
165 } else {
166 ++ssIter;
167 }
168 }
169
170 if (this->searchSpaceIsEmpty()) {
171 utils::ExceptionHandler::exception(
172 "Removing all configurations with Newton 3 {} caused the search space to be empty!", badNewton3Option);
173 }
174}
