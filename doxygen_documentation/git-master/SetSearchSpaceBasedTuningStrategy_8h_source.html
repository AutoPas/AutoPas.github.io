105 void SetSearchSpaceBasedTuningStrategy::populateSearchSpace(
106  const std::set<ContainerOption> &allowedContainerOptions, const std::set<double> &allowedCellSizeFactors,
107  const std::set<TraversalOption> &allowedTraversalOptions,
108  const std::set<LoadEstimatorOption> &allowedLoadEstimatorOptions,
109  const std::set<DataLayoutOption> &allowedDataLayoutOptions, const std::set<Newton3Option> &allowedNewton3Options) {
110  // generate all potential configs
111  for (const auto &containerOption : allowedContainerOptions) {
112  // get all traversals of the container and restrict them to the allowed ones
113  const std::set<TraversalOption> &allContainerTraversals =
114  compatibleTraversals::allCompatibleTraversals(containerOption);
115  std::set<TraversalOption> allowedAndApplicable;
116  std::set_intersection(allowedTraversalOptions.begin(), allowedTraversalOptions.end(),
117  allContainerTraversals.begin(), allContainerTraversals.end(),
118  std::inserter(allowedAndApplicable, allowedAndApplicable.begin()));
119 
120  for (const auto &cellSizeFactor : allowedCellSizeFactors)
121  for (const auto &traversalOption : allowedAndApplicable) {
122  // if load estimators are not applicable LoadEstimatorOption::none is returned.
123  const std::set<LoadEstimatorOption> allowedAndApplicableLoadEstimators =
124  loadEstimators::getApplicableLoadEstimators(containerOption, traversalOption, allowedLoadEstimatorOptions);
125  for (const auto &loadEstimatorOption : allowedAndApplicableLoadEstimators) {
126  for (const auto &dataLayoutOption : allowedDataLayoutOptions) {
127  for (const auto &newton3Option : allowedNewton3Options) {
128  _searchSpace.emplace(containerOption, cellSizeFactor, traversalOption, loadEstimatorOption,
129  dataLayoutOption, newton3Option);
130  }
131  }
132  }
133  }
134  }
135 
136  AutoPasLog(debug, "Points in search space: {}", _searchSpace.size());
137 
138  if (_searchSpace.empty()) {
139  autopas::utils::ExceptionHandler::exception("No valid configurations could be created.");
140  }
141 
142  _currentConfig = _searchSpace.begin();
143 }

154 void SetSearchSpaceBasedTuningStrategy::removeN3Option(Newton3Option badNewton3Option) {
155  for (auto ssIter = _searchSpace.begin(); ssIter != _searchSpace.end();) {
156  if (ssIter->newton3 == badNewton3Option) {
157  // change current config to the next non-deleted
158  if (ssIter == _currentConfig) {
159  ssIter = _searchSpace.erase(ssIter);
160  _currentConfig = ssIter;
161  } else {
162  ssIter = _searchSpace.erase(ssIter);
163  }
164  } else {
165  ++ssIter;
166  }
167  }
168 
169  if (this->searchSpaceIsEmpty()) {
170  utils::ExceptionHandler::exception(
171  "Removing all configurations with Newton 3 {} caused the search space to be empty!", badNewton3Option);
172  }
