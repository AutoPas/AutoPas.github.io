214void ActiveHarmony::fetchConfiguration() {
215 TraversalOption traversalOption = fetchTuningParameter(traversalOptionName, _allowedTraversalOptions);
216 DataLayoutOption dataLayoutOption = fetchTuningParameter(dataLayoutOptionName, _allowedDataLayoutOptions);
217 Newton3Option newton3Option = fetchTuningParameter(newton3OptionName, _allowedNewton3Options);
218 ContainerOption containerOption = *compatibleTraversals::allCompatibleContainers(traversalOption).begin();
219 auto applicableLoadEstimators =
220 loadEstimators::getApplicableLoadEstimators(containerOption, traversalOption, _allowedLoadEstimatorOptions);
221 LoadEstimatorOption loadEstimatorOption = fetchTuningParameter(loadEstimatorOptionName, applicableLoadEstimators);
222
223 double cellSizeFactor = 0;
224 if (_allowedCellSizeFactors->isFinite()) {
225 if (_allowedCellSizeFactors->size() == 1) {
226 cellSizeFactor = _allowedCellSizeFactors->getMin();
227 } else if (_allowedCellSizeFactors->size() > 1) {
228 cellSizeFactor = std::strtod(ah_get_enum(htask, cellSizeFactorsName), nullptr);
229 }
230 } else {
231 cellSizeFactor = ah_get_real(htask, cellSizeFactorsName);
232 }
233
234 _currentConfig = Configuration(containerOption, cellSizeFactor, traversalOption, loadEstimatorOption,
235 dataLayoutOption, newton3Option);
236}

244bool ActiveHarmony::tune(bool currentInvalid) {
245 if (searchSpaceIsTrivial()) {
246 fetchConfiguration();
247 return false;
248 } else if (searchSpaceIsEmpty()) {
249 _currentConfig = Configuration();
250 return false;
251 }
252 if (currentInvalid) {
253 if (ah_converged(htask)) {
254 AutoPasLog(debug, "Active Harmony converged to invalid configuration; resetting active-harmony server.");
255 resetHarmony();
256 } else {
257 invalidateConfiguration();
258 }
259 }
260
261 // get configurations from server until new configuration with valid newton3 option is found
262 bool skipConfig;
263 do {
264 skipConfig = false;
265 if (ah_fetch(htask) < 0) {
266 utils::ExceptionHandler::exception("ActiveHarmony::tune: Error fetching values from server");
267 }
268 fetchConfiguration();
269 if (_traversalTimes.find(_currentConfig) != _traversalTimes.end()) {
270 // we already know the performance for this config
271 addEvidence(_traversalTimes[_currentConfig], 0);
272 skipConfig = true;
273 }
274
275 auto converged = ah_converged(htask);
276 if (converged) {
277 // set configuration to optimum
278 AutoPasLog(debug, "ActiveHarmony::tune: Reached converged state.");
279 if (ah_best(htask) != 0) {
280 utils::ExceptionHandler::exception("ActiveHarmony::tune: Error fetching best point.");
281 }
282 fetchConfiguration();
283 AutoPasLog(debug, "ActiveHarmony::tune: Selected optimal configuration {}.", _currentConfig.toString());
284 return false;
285 }
286
287 if (_nonLocalServer) {
288 // When using a non-local server, it is possible that only tested configurations are fetched before the search
289 // converges.
290 // Because this is difficult to test for, the loop is simply ignored for non-local servers.
291 return true;
292 }
293 } while (skipConfig);
294 return true;
295}

332void ActiveHarmony::resetHarmony() {
333 int rank, commSize;
334 AutoPas_MPI_Comm_size(_comm, &commSize);
335 AutoPas_MPI_Comm_rank(_comm, &rank);
336
337 if (_mpiStrategyOption == MPIStrategyOption::divideAndConquer) {
338 AutoPas_MPI_Barrier(_comm);
339 }
340 // free memory
341 if (htask != nullptr) {
342 ah_leave(htask);
343 ah_kill(htask);
344 }
345 if (hdesc != nullptr) {
346 ah_close(hdesc);
347 ah_free(hdesc);
348 }
349
350 if (searchSpaceIsTrivial() or searchSpaceIsEmpty()) {
351 AutoPasLog(debug, "Search space is {}; skipping harmony initialization.",
352 searchSpaceIsTrivial() ? "trivial" : "empty");
353 // Barrier in case the search space is bigger for some ranks
354 if (_mpiStrategyOption == MPIStrategyOption::divideAndConquer) {
355 AutoPas_MPI_Barrier(_comm);
356 }
357 } else {
358 // initiate the tuning session
359 hdesc = ah_alloc();
360 if (hdesc == nullptr) {
361 utils::ExceptionHandler::exception("ActiveHarmony::reset: Error allocating Harmony descriptor: {}", ah_error());
362 }
363
364 if (ah_connect(hdesc, nullptr, 0) != 0) {
365 utils::ExceptionHandler::exception("ActiveHarmony::reset: Error connecting to Harmony session: {}", ah_error());
366 }
367
368 if (not _nonLocalServer or rank == 0) {
369 hdef_t *hdef = ah_def_alloc();
370 if (hdef == nullptr) {
371 utils::ExceptionHandler::exception("ActiveHarmony::reset: Error allocating definition descriptor: {}",
372 ah_error());
373 }
374 setupTuningParameters(commSize, hdef);
375 // task initialization
376 htask = ah_start(hdesc, hdef);
377 ah_def_free(hdef);
378 if (htask == nullptr) {
379 utils::ExceptionHandler::exception("ActiveHarmony::reset: Error starting task.");
380 }
381
382 if (_mpiStrategyOption == MPIStrategyOption::divideAndConquer) {
383 AutoPas_MPI_Barrier(_comm);
384 }
385 } else {
386 // only join a session if using the divideAndConquer mpi strategy, we are not rank 0 and a server is specified.
387 if (_mpiStrategyOption == MPIStrategyOption::divideAndConquer) {
388 AutoPas_MPI_Barrier(_comm);
389 }
390
391 // Everybody else may now join the master's new Harmony search.
392 htask = ah_join(hdesc, "AutoPas");
393 if (htask == nullptr) {
394 utils::ExceptionHandler::exception("ActiveHarmony::reset: Error joining session");
395 }
396 }
397 }
398
399 tune(false);
400}

421void ActiveHarmony::setupTuningParameters(int commSize, hdef_t *hdef) {
422 if (ah_def_name(hdef, "AutoPas") != 0) {
423 utils::ExceptionHandler::exception("ActiveHarmony::reset: Error setting search name: {}", ah_error());
424 }
425
426 if (_allowedCellSizeFactors->isFinite()) { // finite cell-size factors => define parameter as enum
427 if (_allowedCellSizeFactors->size() == 1) {
428 AutoPasLog(debug, "ActiveHarmony::reset: Skipping trivial parameter {}", cellSizeFactorsName);
429 } else if (_allowedCellSizeFactors->size() > 1) {
430 AutoPasLog(debug, "ActiveHarmony::reset: Finite cell-size factors; defining parameter as enum");
431 if (ah_def_enum(hdef, cellSizeFactorsName, nullptr) != 0) {
432 utils::ExceptionHandler::exception("ActiveHarmony::configureTuningParameter: Error defining enum \"{}\"",
433 cellSizeFactorsName);
434 }
435 for (auto cellSizeFactor : _allowedCellSizeFactors->getAll()) {
436 if (ah_def_enum_value(hdef, cellSizeFactorsName, std::to_string(cellSizeFactor).c_str()) != 0) {
437 utils::ExceptionHandler::exception(
438 "ActiveHarmony::configureTuningParameter: Error defining enum value for enum \"{}\"",
439 cellSizeFactorsName);
440 }
441 }
442 }
443 } else { // infinite cell-size factors => define parameter as real
444 AutoPasLog(debug, "ActiveHarmony::reset: Infinite cell-size factors; defining parameter as real");
445 if (ah_def_real(hdef, cellSizeFactorsName, _allowedCellSizeFactors->getMin(), _allowedCellSizeFactors->getMax(),
446 (_allowedCellSizeFactors->getMax() - _allowedCellSizeFactors->getMin()) / cellSizeSamples,
447 nullptr) != 0) {
448 utils::ExceptionHandler::exception("ActiveHarmony::reset: Error defining real \"{}\"", cellSizeFactorsName);
449 }
450 }
451 // set up other parameters
452 configureTuningParameter(hdef, traversalOptionName, _allowedTraversalOptions);
453 configureTuningParameter(hdef, dataLayoutOptionName, _allowedDataLayoutOptions);
454 configureTuningParameter(hdef, newton3OptionName, _allowedNewton3Options);
455 configureTuningParameter(hdef, loadEstimatorOptionName, _allowedLoadEstimatorOptions);
456
457 // use ActiveHarmony's implementation of the Nelder-Mead method
458 ah_def_strategy(hdef, "pro.so");
459 // set the size of the initial simplex (as portion of the total search space)
460 ah_def_cfg(hdef, "INIT_RADIUS", "0.7");
461
462 if (_mpiStrategyOption == MPIStrategyOption::divideAndConquer) {
463 // set the size of the tuning session
464 char numbuf[12];
465 snprintf(numbuf, sizeof(numbuf), "%d", commSize);
466 ah_def_cfg(hdef, "CLIENT_COUNT", numbuf);
467 }
