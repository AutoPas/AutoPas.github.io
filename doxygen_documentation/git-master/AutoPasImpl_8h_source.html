33AutoPas<Particle>::AutoPas(std::ostream &logOutputStream) {
34 // count the number of autopas instances. This is needed to ensure that the autopas
35 // logger is not unregistered while other instances are still using it.

41 // The logger is normally only flushed on successful program termination.

62template <class Particle>
63void AutoPas<Particle>::init() {
64 AutoPasLog(INFO, "AutoPas Version: {}", AutoPas_VERSION);
65 AutoPasLog(INFO, "Compiled with : {}", utils::CompileInfo::getCompilerInfo());
66 if (_autoTunerInfo.maxSamples % _verletRebuildFrequency != 0) {
67 AutoPasLog(WARN,
68 "Number of samples ({}) is not a multiple of the rebuild frequency ({}). This can lead to problems "
69 "when multiple AutoPas instances interact (e.g. via MPI).",
70 _autoTunerInfo.maxSamples, _verletRebuildFrequency);
71 }
72
73 if (_tuningStrategyFactoryInfo.autopasMpiCommunicator == AUTOPAS_MPI_COMM_NULL) {
74 AutoPas_MPI_Comm_dup(AUTOPAS_MPI_COMM_WORLD, &_tuningStrategyFactoryInfo.autopasMpiCommunicator);
75 } else {
76 _externalMPICommunicator = true;
77 }
78 if (std::find(_tuningStrategyOptions.begin(), _tuningStrategyOptions.end(),
79 TuningStrategyOption::mpiDivideAndConquer) != _tuningStrategyOptions.end()) {
80 _tuningStrategyFactoryInfo.mpiDivideAndConquer = true;
81 }
82
83 // If an interval was given for the cell size factor, change it to the relevant values.
84 // Don't modify _allowedCellSizeFactors to preserve the initial (type) information.
85 const auto cellSizeFactors = [&]() -> NumberSetFinite<double> {
86 if (const auto *csfIntervalPtr = dynamic_cast<NumberInterval<double> *>(_allowedCellSizeFactors.get())) {
87 const auto interactionLength =
88 _logicHandlerInfo.cutoff * _logicHandlerInfo.verletSkinPerTimestep * _verletRebuildFrequency;
89 const auto boxLengthX = _logicHandlerInfo.boxMax[0] - _logicHandlerInfo.boxMin[0];
90 return {SearchSpaceGenerators::calculateRelevantCsfs(*csfIntervalPtr, interactionLength, boxLengthX)};
91 } else {
92 // in this case _allowedCellSizeFactors is a finite set
93 return {_allowedCellSizeFactors->getAll()};
94 }
95 }();
96 const auto searchSpace =
97 SearchSpaceGenerators::cartesianProduct(_allowedContainers, _allowedTraversals, _allowedLoadEstimators,
98 _allowedDataLayouts, _allowedNewton3Options, &cellSizeFactors);
99
100 AutoTuner::TuningStrategiesListType tuningStrategies;
101 tuningStrategies.reserve(_tuningStrategyOptions.size());
102 for (const auto &strategy : _tuningStrategyOptions) {
103 tuningStrategies.emplace_back(TuningStrategyFactory::generateTuningStrategy(
104 searchSpace, strategy, _tuningStrategyFactoryInfo, _outputSuffix));
105 }
106 if (_useTuningStrategyLoggerProxy) {
107 tuningStrategies.emplace_back(std::make_unique<TuningStrategyLogger>(_outputSuffix));
108 }
109 _autoTuner = std::make_unique<autopas::AutoTuner>(tuningStrategies, searchSpace, _autoTunerInfo,
110 _verletRebuildFrequency, _outputSuffix);
111
112 _logicHandler = std::make_unique<std::remove_reference_t<decltype(*_logicHandler)>>(
113 *_autoTuner, _logicHandlerInfo, _verletRebuildFrequency, _outputSuffix);
114}

117template <class Functor>
118bool AutoPas<Particle>::iteratePairwise(Functor *f) {
119 static_assert(not std::is_same<Functor, autopas::Functor<Particle, Functor>>::value,
120 "The static type of Functor in iteratePairwise is not allowed to be autopas::Functor. Please use the "
121 "derived type instead, e.g. by using a dynamic_cast.");
122 if (f->getCutoff() > this->getCutoff()) {
123 utils::ExceptionHandler::exception("Functor cutoff ({}) must not be larger than container cutoff ({})",
124 f->getCutoff(), this->getCutoff());
125 }
126 return _logicHandler->iteratePairwisePipeline(f);
127}

129template <class Particle>
130size_t AutoPas<Particle>::getNumberOfParticles(IteratorBehavior behavior) const {
131 size_t numParticles{0};
132 if (behavior & IteratorBehavior::owned) {
133 numParticles += _logicHandler->getNumberOfParticlesOwned();
134 }
135 if (behavior & IteratorBehavior::halo) {
136 numParticles += _logicHandler->getNumberOfParticlesHalo();
137 }
138 // non fatal sanity check whether the behavior contained anything else
139 if (behavior & ~(IteratorBehavior::ownedOrHalo)) {
140 utils::ExceptionHandler::exception(
141 "AutoPas::getNumberOfParticles() does not support iterator behaviors other than owned or halo.");
142 }
143
144 return numParticles;
145}

182template <class Particle>
183template <class Collection, class F>
184void AutoPas<Particle>::addParticlesIf(Collection &&particles, F predicate) {
185 std::vector<char> predicateMask(particles.size());
186 int numTrue = 0;
187#ifdef AUTOPAS_OPENMP
188#pragma omp parallel for reduction(+ : numTrue)
189#endif
190 for (auto i = 0; i < particles.size(); ++i) {
191 if (predicate(particles[i])) {
192 predicateMask[i] = static_cast<char>(true);
193 ++numTrue;
194 } else {
195 predicateMask[i] = static_cast<char>(false);
196 }
197 }
198
199 addParticlesAux(numTrue, 0, particles.size(), [&](auto i) {
200 if (predicateMask[i]) {
201 addParticle(particles[i]);
202 }
203 });
204}

211template <class Particle>

213 const std::array<double, 3> &boxMax) {
214 if (_allowedCellSizeFactors->isInterval()) {
215 AutoPasLog(WARN,
216 "The allowed Cell Size Factors are a continuous interval but internally only those values that "
217 "yield unique numbers of cells are used. Resizing does not cause these values to be recalculated so "
218 "the same configurations might now yield different and non-unique numbers of cells!");
219 }
220 _logicHandlerInfo.boxMin = boxMin;

222 return _logicHandler->resizeBox(boxMin, boxMax);

282template <class Particle>
283bool AutoPas<Particle>::deleteParticle(Particle &particle) {
284 _logicHandler->decreaseParticleCounter(particle);
285 // if the particle was not found in the logic handler's buffers it must be in the container
286 auto [particleDeleted, refStillValid] = _logicHandler->deleteParticleFromBuffers(particle);
287 if (not particleDeleted) {
288 refStillValid = _logicHandler->getContainer().deleteParticle(particle);
289 }
290 return refStillValid;
291}
