95 template <class Particle>
96 size_t AutoPas<Particle>::getNumberOfParticles(IteratorBehavior behavior) const {
97  size_t numParticles{0};
98  if (behavior & IteratorBehavior::owned) {
99  numParticles += _logicHandler->getNumParticlesOwned();
100  }
101  if (behavior & IteratorBehavior::halo) {
102  numParticles += _logicHandler->getNumParticlesHalo();
103  }
104  // non fatal sanity check whether the behavior contained anything else
105  if (behavior & ~(IteratorBehavior::ownedOrHalo)) {
106  utils::ExceptionHandler::exception(
107  "AutoPas::getNumberOfParticles() does not support iterator behaviors other than owned or halo.");
108  }
109 
110  return numParticles;
111 }
