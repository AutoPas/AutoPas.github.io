25template <class Particle>
26AutoPas<Particle>::AutoPas(std::ostream &logOutputStream) {
27 // count the number of autopas instances. This is needed to ensure that the autopas
28 // logger is not unregistered while other instances are still using it.
29 InstanceCounter::count++;
30 // remove potentially existing logger
31 autopas::Logger::unregister();
32 // initialize the Logger
33 autopas::Logger::create(logOutputStream);
34 // The logger is normally only flushed on successful program termination.
35 // This line ensures flushing when log messages of level warning or more severe are created.
36 autopas::Logger::get()->flush_on(spdlog::level::warn);
37}

55template <class Particle>
56void AutoPas<Particle>::init() {
57 AutoPasLog(info, "AutoPas Version: {}", AutoPas_VERSION);
58 if (_numSamples % _verletRebuildFrequency != 0) {
59 AutoPasLog(warn,
60 "Number of samples ({}) is not a multiple of the rebuild frequency ({}). This can lead to problems "
61 "when multiple AutoPas instances interact (e.g. via MPI).",
62 _numSamples, _verletRebuildFrequency);
63 }
64
65 if (_autopasMPICommunicator == AUTOPAS_MPI_COMM_NULL) {
66 AutoPas_MPI_Comm_dup(AUTOPAS_MPI_COMM_WORLD, &_autopasMPICommunicator);
67 } else {
68 _externalMPICommunicator = true;
69 }
70 auto tuningStrategy = TuningStrategyFactory::generateTuningStrategy(
71 _tuningStrategyOption, _allowedContainers, *_allowedCellSizeFactors, _allowedTraversals, _allowedLoadEstimators,
72 _allowedDataLayouts, _allowedNewton3Options, _maxEvidence, _relativeOptimumRange, _maxTuningPhasesWithoutTest,
73 _relativeBlacklistRange, _evidenceFirstPrediction, _acquisitionFunctionOption, _extrapolationMethodOption,
74 _outputSuffix, _mpiStrategyOption, _autopasMPICommunicator);
75 _autoTuner = std::make_unique<autopas::AutoTuner<Particle>>(
76 _boxMin, _boxMax, _cutoff, _verletSkinPerTimestep, _verletClusterSize, std::move(tuningStrategy),
77 _mpiTuningMaxDifferenceForBucket, _mpiTuningWeightForMaxDensity, _selectorStrategy, _tuningInterval, _numSamples,
78 _verletRebuildFrequency, _outputSuffix);
79 _logicHandler =
80 std::make_unique<std::remove_reference_t<decltype(*_logicHandler)>>(*(_autoTuner.get()), _verletRebuildFrequency);
81}

84template <class Functor>
85bool AutoPas<Particle>::iteratePairwise(Functor *f) {
86 static_assert(not std::is_same<Functor, autopas::Functor<Particle, Functor>>::value,
87 "The static type of Functor in iteratePairwise is not allowed to be autopas::Functor. Please use the "
88 "derived type instead, e.g. by using a dynamic_cast.");
89 if (f->getCutoff() > this->getCutoff()) {
90 utils::ExceptionHandler::exception("Functor cutoff ({}) must not be larger than container cutoff ({})",
91 f->getCutoff(), this->getCutoff());
92 }
93 return _logicHandler->iteratePairwise(f);
94}

96template <class Particle>
97size_t AutoPas<Particle>::getNumberOfParticles(IteratorBehavior behavior) const {
98 size_t numParticles{0};
99 if (behavior & IteratorBehavior::owned) {
100 numParticles += _logicHandler->getNumParticlesOwned();
101 }
102 if (behavior & IteratorBehavior::halo) {
103 numParticles += _logicHandler->getNumParticlesHalo();
104 }
105 // non fatal sanity check whether the behavior contained anything else
106 if (behavior & ~(IteratorBehavior::ownedOrHalo)) {
107 utils::ExceptionHandler::exception(
108 "AutoPas::getNumberOfParticles() does not support iterator behaviors other than owned or halo.");
109 }
110
111 return numParticles;
112}
