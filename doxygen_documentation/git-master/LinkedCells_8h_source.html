182

184  // if not in cell

189  }

192  // the barrier is needed because iterators are not threadsafe w.r.t. addParticle()

196  // if not in halo
197  if (utils::inBox(p.getR(), this->getBoxMin(), this->getBoxMax())) {
198  this->template addParticle<false>(p);

238  // We increase the search region by skin, as particles can move over cell borders.
239  auto startIndex3D =

241  auto stopIndex3D =

243 
244  size_t numCellsOfInterest = (stopIndex3D[0] - startIndex3D[0] + 1) * (stopIndex3D[1] - startIndex3D[1] + 1) *
245  (stopIndex3D[2] - startIndex3D[2] + 1);
246  std::vector<size_t> cellsOfInterest(numCellsOfInterest);
247 
248  int i = 0;

251  for (size_t x = startIndex3D[0]; x <= stopIndex3D[0]; ++x) {
252  cellsOfInterest[i++] =
