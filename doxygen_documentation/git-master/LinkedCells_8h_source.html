162 [[nodiscard]] std::vector<ParticleType> updateContainer(bool keepNeighborListsValid) override {
163 if (keepNeighborListsValid) {
164 return autopas::LeavingParticleCollector::collectParticlesAndMarkNonOwnedAsDummy(*this);

167 this->deleteHaloParticles();
168
169 std::vector<ParticleType> invalidParticles;
170#ifdef AUTOPAS_OPENMP
171#pragma omp parallel
172#endif // AUTOPAS_OPENMP
173 {
174 // private for each thread!
175 std::vector<ParticleType> myInvalidParticles, myInvalidNotOwnedParticles;
176#ifdef AUTOPAS_OPENMP
177#pragma omp for
178#endif // AUTOPAS_OPENMP
179 for (size_t cellId = 0; cellId < this->getCells().size(); ++cellId) {
180 // Delete dummy particles of each cell.
181 this->getCells()[cellId].deleteDummyParticles();
182
183 // if empty
184 if (this->getCells()[cellId].isEmpty()) continue;
185
186 auto [cellLowerCorner, cellUpperCorner] = this->getCellBlock().getCellBoundingBox(cellId);
187
188 for (auto &&pIter = this->getCells()[cellId].begin(); pIter.isValid(); ++pIter) {
189 // if not in cell
190 if (utils::notInBox(pIter->getR(), cellLowerCorner, cellUpperCorner)) {
191 myInvalidParticles.push_back(*pIter);
192 internal::deleteParticle(pIter);
193 }
194 }
195 }
196 // implicit barrier here
197 // the barrier is needed because iterators are not threadsafe w.r.t. addParticle()
198
199 // this loop is executed for every thread and thus parallel. Don't use #pragma omp for here!
200 for (auto &&p : myInvalidParticles) {
201 // if not in halo
202 if (utils::inBox(p.getR(), this->getBoxMin(), this->getBoxMax())) {
203 this->template addParticle<false>(p);
204 } else {
205 myInvalidNotOwnedParticles.push_back(p);
206 }
207 }
208#ifdef AUTOPAS_OPENMP
209#pragma omp critical
210#endif
211 {
212 // merge private vectors to global one.
213 invalidParticles.insert(invalidParticles.end(), myInvalidNotOwnedParticles.begin(),
214 myInvalidNotOwnedParticles.end());
215 }
216 }
217 return invalidParticles;

285 [[nodiscard]] ParticleIteratorWrapper<ParticleType, true> getRegionIterator(const std::array<double, 3> &lowerCorner,
286 const std::array<double, 3> &higherCorner,
287 IteratorBehavior behavior) override {
288 // We increase the search region by skin, as particles can move over cell borders.
289 auto startIndex3D =
290 this->_cellBlock.get3DIndexOfPosition(utils::ArrayMath::subScalar(lowerCorner, this->getSkin()));
291 auto stopIndex3D =
292 this->_cellBlock.get3DIndexOfPosition(utils::ArrayMath::addScalar(higherCorner, this->getSkin()));
293
294 size_t numCellsOfInterest = (stopIndex3D[0] - startIndex3D[0] + 1) * (stopIndex3D[1] - startIndex3D[1] + 1) *
295 (stopIndex3D[2] - startIndex3D[2] + 1);
296 std::vector<size_t> cellsOfInterest(numCellsOfInterest);
297
298 int i = 0;
299 for (size_t z = startIndex3D[2]; z <= stopIndex3D[2]; ++z) {
300 for (size_t y = startIndex3D[1]; y <= stopIndex3D[1]; ++y) {
301 for (size_t x = startIndex3D[0]; x <= stopIndex3D[0]; ++x) {
302 cellsOfInterest[i++] =
303 utils::ThreeDimensionalMapping::threeToOneD({x, y, z}, this->_cellBlock.getCellsPerDimensionWithHalo());
304 }
305 }
306 }
307
308 return ParticleIteratorWrapper<ParticleType, true>(
309 new internal::RegionParticleIterator<ParticleType, ParticleCell, true>(
310 &this->_cells, lowerCorner, higherCorner, cellsOfInterest, &_cellBlock, behavior, nullptr));
311 }
