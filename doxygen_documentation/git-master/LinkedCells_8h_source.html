242 template <bool regionIter>
243 std::tuple<const Particle *, size_t, size_t> getParticleImpl(size_t cellIndex, size_t particleIndex,
244 IteratorBehavior iteratorBehavior,
245 const std::array<double, 3> &boxMin,
246 const std::array<double, 3> &boxMax) const {
247 // first and last relevant cell index
248 const auto [startCellIndex, endCellIndex] = [&]() -> std::tuple<size_t, size_t> {
249 if constexpr (regionIter) {
250 return {_cellBlock.get1DIndexOfPosition(boxMin), _cellBlock.get1DIndexOfPosition(boxMax)};
251 } else {
252 if (not(iteratorBehavior & IteratorBehavior::halo)) {
253 // only potentially owned region
254 return {_cellBlock.getFirstOwnedCellIndex(), _cellBlock.getLastOwnedCellIndex()};
255 } else {
256 // whole range of cells
257 return {0, this->_cells.size() - 1};
258 }
259 }
260 }();
261
262 // if we are at the start of an iteration ...
263 if (cellIndex == 0 and particleIndex == 0) {
264 cellIndex =
265 startCellIndex + ((iteratorBehavior & IteratorBehavior::forceSequential) ? 0 : autopas_get_thread_num());
266 }
267 // abort if the start index is already out of bounds
268 if (cellIndex >= this->_cells.size()) {
269 return {nullptr, 0, 0};
270 }
271 // check the data behind the indices
272 if (particleIndex >= this->_cells[cellIndex].size() or
273 not containerIteratorUtils::particleFulfillsIteratorRequirements<regionIter>(
274 this->_cells[cellIndex][particleIndex], iteratorBehavior, boxMin, boxMax)) {
275 // either advance them to something interesting or invalidate them.
276 std::tie(cellIndex, particleIndex) =
277 advanceIteratorIndices<regionIter>(cellIndex, particleIndex, iteratorBehavior, boxMin, boxMax, endCellIndex);
278 }
279
280 // shortcut if the given index doesn't exist
281 if (cellIndex > endCellIndex) {
282 return {nullptr, 0, 0};
283 }
284 const Particle *retPtr = &this->_cells[cellIndex][particleIndex];
285
286 return {retPtr, cellIndex, particleIndex};
287 }

386 template <typename Lambda>
387 void forEachInRegion(Lambda forEachLambda, const std::array<double, 3> &lowerCorner,
388 const std::array<double, 3> &higherCorner, IteratorBehavior behavior) {
389 using namespace autopas::utils::ArrayMath::literals;
390
391 const auto startIndex3D = this->_cellBlock.get3DIndexOfPosition(lowerCorner - this->getVerletSkin());
392 const auto stopIndex3D = this->_cellBlock.get3DIndexOfPosition(higherCorner + this->getVerletSkin());
393
394 const size_t numCellsOfInterest = (stopIndex3D[0] - startIndex3D[0] + 1) * (stopIndex3D[1] - startIndex3D[1] + 1) *
395 (stopIndex3D[2] - startIndex3D[2] + 1);
396 std::vector<size_t> cellsOfInterest;
397 cellsOfInterest.reserve(numCellsOfInterest);
398
399 const auto &cellsPerDimensionWithHalo = this->_cellBlock.getCellsPerDimensionWithHalo();
400
401 for (size_t z = startIndex3D[2]; z <= stopIndex3D[2]; ++z) {
402 for (size_t y = startIndex3D[1]; y <= stopIndex3D[1]; ++y) {
403 for (size_t x = startIndex3D[0]; x <= stopIndex3D[0]; ++x) {
404 cellsOfInterest.push_back(utils::ThreeDimensionalMapping::threeToOneD({x, y, z}, cellsPerDimensionWithHalo));
405 }
406 }
407 }
408
409 for (auto cellIndex : cellsOfInterest) {
410 if (not _cellBlock.ignoreCellForIteration(cellIndex, behavior)) {
411 getCells()[cellIndex].forEach(forEachLambda, lowerCorner, higherCorner, behavior);
412 }
413 }

486 template <bool regionIter>
487 std::tuple<size_t, size_t> advanceIteratorIndices(size_t cellIndex, size_t particleIndex,
488 IteratorBehavior iteratorBehavior,
489 const std::array<double, 3> &boxMin,
490 const std::array<double, 3> &boxMax, size_t endCellIndex) const {
491 // Finding the indices for the next particle
492 const size_t stride = (iteratorBehavior & IteratorBehavior::forceSequential) ? 1 : autopas_get_num_threads();
493
494 // helper function to determine if the cell can even contain particles of interest to the iterator
495 auto cellIsRelevant = [&]() -> bool {
496 bool isRelevant =
497 // behavior matches possible particle ownership
498 (iteratorBehavior & IteratorBehavior::owned and _cellBlock.cellCanContainOwnedParticles(cellIndex)) or
499 (iteratorBehavior & IteratorBehavior::halo and _cellBlock.cellCanContainHaloParticles(cellIndex));
500 if constexpr (regionIter) {
501 // short circuit if already false
502 if (isRelevant) {
503 // is the cell in the region?
504 const auto [cellLowCorner, cellHighCorner] = _cellBlock.getCellBoundingBox(cellIndex);
505 // particles can move over cell borders. Calculate the volume this cell's particles can be.
506 const auto cellLowCornerSkin = utils::ArrayMath::subScalar(cellLowCorner, this->getVerletSkin() * 0.5);
507 const auto cellHighCornerSkin = utils::ArrayMath::addScalar(cellHighCorner, this->getVerletSkin() * 0.5);
508 isRelevant = utils::boxesOverlap(cellLowCornerSkin, cellHighCornerSkin, boxMin, boxMax);
509 }
510 }
511 return isRelevant;
512 };
513
514 do {
515 // advance to the next particle
516 ++particleIndex;
517 // If this breaches the end of a cell, find the next non-empty cell and reset particleIndex.
518
519 // If cell has wrong type, or there are no more particles in this cell jump to the next
520 while (not cellIsRelevant() or particleIndex >= this->_cells[cellIndex].size()) {
521 // TODO: can this jump be done more efficient if behavior is only halo or owned?
522 // TODO: can this jump be done more efficient for region iters if the cell is outside the region?
523 cellIndex += stride;
524 particleIndex = 0;
525
526 // If we notice that there is nothing else to look at set invalid values, so we get a nullptr next time and
527 // break.
528 if (cellIndex > endCellIndex) {
529 return {std::numeric_limits<decltype(cellIndex)>::max(), std::numeric_limits<decltype(particleIndex)>::max()};
530 }
531 }
532 } while (not containerIteratorUtils::particleFulfillsIteratorRequirements<regionIter>(
533 this->_cells[cellIndex][particleIndex], iteratorBehavior, boxMin, boxMax));
534
535 // the indices returned at this point should always be valid
536 return {cellIndex, particleIndex};
537 }
