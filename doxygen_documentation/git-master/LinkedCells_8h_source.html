242 template <bool regionIter>
243 std::tuple<const Particle *, size_t, size_t> getParticleImpl(size_t cellIndex, size_t particleIndex,
244 IteratorBehavior iteratorBehavior,
245 const std::array<double, 3> &boxMin,
246 const std::array<double, 3> &boxMax) const {
247 using namespace autopas::utils::ArrayMath::literals;
248
249 std::array<double, 3> boxMinWithSafetyMargin = boxMin;
250 std::array<double, 3> boxMaxWithSafetyMargin = boxMax;
251 if constexpr (regionIter) {
252 // We extend the search box for cells here since particles might have moved
253 boxMinWithSafetyMargin -= (this->_skinPerTimestep * static_cast<double>(this->getStepsSinceLastRebuild()));
254 boxMaxWithSafetyMargin +=
255 boxMax + (this->_skinPerTimestep * static_cast<double>(this->getStepsSinceLastRebuild()));
256 }
257
258 // first and last relevant cell index
259 const auto [startCellIndex, endCellIndex] = [&]() -> std::tuple<size_t, size_t> {
260 if constexpr (regionIter) {
261 // We extend the search box for cells here since particles might have moved
262 return {_cellBlock.get1DIndexOfPosition(boxMinWithSafetyMargin),
263 _cellBlock.get1DIndexOfPosition(boxMaxWithSafetyMargin)};
264 } else {
265 if (not(iteratorBehavior & IteratorBehavior::halo)) {
266 // only potentially owned region
267 return {_cellBlock.getFirstOwnedCellIndex(), _cellBlock.getLastOwnedCellIndex()};
268 } else {
269 // whole range of cells
270 return {0, this->_cells.size() - 1};
271 }
272 }
273 }();
274
275 // if we are at the start of an iteration ...
276 if (cellIndex == 0 and particleIndex == 0) {
277 cellIndex =
278 startCellIndex + ((iteratorBehavior & IteratorBehavior::forceSequential) ? 0 : autopas_get_thread_num());
279 }
280 // abort if the start index is already out of bounds
281 if (cellIndex >= this->_cells.size()) {
282 return {nullptr, 0, 0};
283 }
284 // check the data behind the indices
285 if (particleIndex >= this->_cells[cellIndex].size() or
286 not containerIteratorUtils::particleFulfillsIteratorRequirements<regionIter>(
287 this->_cells[cellIndex][particleIndex], iteratorBehavior, boxMin, boxMax)) {
288 // either advance them to something interesting or invalidate them.
289 std::tie(cellIndex, particleIndex) =
290 advanceIteratorIndices<regionIter>(cellIndex, particleIndex, iteratorBehavior, boxMin, boxMax,
291 boxMinWithSafetyMargin, boxMaxWithSafetyMargin, endCellIndex);
292 }
293
294 // shortcut if the given index doesn't exist
295 if (cellIndex > endCellIndex) {
296 return {nullptr, 0, 0};
297 }
298 const Particle *retPtr = &this->_cells[cellIndex][particleIndex];
299
300 return {retPtr, cellIndex, particleIndex};
301 }

401 void forEachInRegion(Lambda forEachLambda, const std::array<double, 3> &lowerCorner,
402 const std::array<double, 3> &higherCorner, IteratorBehavior behavior) {
403 using namespace autopas::utils::ArrayMath::literals;
404
405 const auto startIndex3D = this->_cellBlock.get3DIndexOfPosition(lowerCorner - this->getVerletSkin());
406 const auto stopIndex3D = this->_cellBlock.get3DIndexOfPosition(higherCorner + this->getVerletSkin());
407
408 const size_t numCellsOfInterest = (stopIndex3D[0] - startIndex3D[0] + 1) * (stopIndex3D[1] - startIndex3D[1] + 1) *
409 (stopIndex3D[2] - startIndex3D[2] + 1);
410 std::vector<size_t> cellsOfInterest;
411 cellsOfInterest.reserve(numCellsOfInterest);
412
413 const auto &cellsPerDimensionWithHalo = this->_cellBlock.getCellsPerDimensionWithHalo();
414
415 for (size_t z = startIndex3D[2]; z <= stopIndex3D[2]; ++z) {
416 for (size_t y = startIndex3D[1]; y <= stopIndex3D[1]; ++y) {
417 for (size_t x = startIndex3D[0]; x <= stopIndex3D[0]; ++x) {
418 cellsOfInterest.push_back(utils::ThreeDimensionalMapping::threeToOneD({x, y, z}, cellsPerDimensionWithHalo));
419 }
420 }
421 }
422
423 for (auto cellIndex : cellsOfInterest) {
424 if (not _cellBlock.ignoreCellForIteration(cellIndex, behavior)) {
425 getCells()[cellIndex].forEach(forEachLambda, lowerCorner, higherCorner, behavior);
426 }
427 }

502 template <bool regionIter>
503 std::tuple<size_t, size_t> advanceIteratorIndices(
504 size_t cellIndex, size_t particleIndex, IteratorBehavior iteratorBehavior, const std::array<double, 3> &boxMin,
505 const std::array<double, 3> &boxMax, std::array<double, 3> boxMinWithSafetyMargin,
506 std::array<double, 3> boxMaxWithSafetyMargin, size_t endCellIndex) const {
507 // Finding the indices for the next particle
508 const size_t stride = (iteratorBehavior & IteratorBehavior::forceSequential) ? 1 : autopas_get_num_threads();
509
510 // helper function to determine if the cell can even contain particles of interest to the iterator
511 auto cellIsRelevant = [&]() -> bool {
512 bool isRelevant =
513 // behavior matches possible particle ownership
514 (iteratorBehavior & IteratorBehavior::owned and _cellBlock.cellCanContainOwnedParticles(cellIndex)) or
515 (iteratorBehavior & IteratorBehavior::halo and _cellBlock.cellCanContainHaloParticles(cellIndex));
516 if constexpr (regionIter) {
517 // short circuit if already false
518 if (isRelevant) {
519 // is the cell in the region?
520 const auto [cellLowCorner, cellHighCorner] = _cellBlock.getCellBoundingBox(cellIndex);
521 isRelevant =
522 utils::boxesOverlap(cellLowCorner, cellHighCorner, boxMinWithSafetyMargin, boxMaxWithSafetyMargin);
523 }
524 }
525 return isRelevant;
526 };
527
528 do {
529 // advance to the next particle
530 ++particleIndex;
531 // If this breaches the end of a cell, find the next non-empty cell and reset particleIndex.
532
533 // If cell has wrong type, or there are no more particles in this cell jump to the next
534 while (not cellIsRelevant() or particleIndex >= this->_cells[cellIndex].size()) {
535 // TODO: can this jump be done more efficient if behavior is only halo or owned?
536 // TODO: can this jump be done more efficient for region iters if the cell is outside the region?
537 cellIndex += stride;
538 particleIndex = 0;
539
540 // If we notice that there is nothing else to look at set invalid values, so we get a nullptr next time and
541 // break.
542 if (cellIndex > endCellIndex) {
543 return {std::numeric_limits<decltype(cellIndex)>::max(), std::numeric_limits<decltype(particleIndex)>::max()};
544 }
545 }
546 } while (not containerIteratorUtils::particleFulfillsIteratorRequirements<regionIter>(
547 this->_cells[cellIndex][particleIndex], iteratorBehavior, boxMin, boxMax));
548
549 // the indices returned at this point should always be valid
550 return {cellIndex, particleIndex};
551 }
