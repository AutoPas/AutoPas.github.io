59 std::unique_ptr<OctreeNodeInterface<Particle>> insert(const Particle &p) override {
60 // Check if the size of the new leaves would become smaller than cellSizeFactor*interactionLength
61 std::array<double, 3> splitLeafDimensions = utils::ArrayMath::sub(this->getBoxMax(), this->getBoxMin());
62 splitLeafDimensions = utils::ArrayMath::mulScalar(splitLeafDimensions, 0.5);
63 bool anyNewDimSmallerThanMinSize = false;
64 for (auto d = 0; d < 3; ++d) {
65 auto cellSizeFactor = 1.0;
66 // @todo The condition below should actually be
67 // splitLeafDimensions[d] < (this->_cellSizeFactor * this->_interactionLength)
68 // But with this condition, the TraversalComparison test fails for cell size factor 0.5. Find out why the octree
69 // cannot handle this value. This problem is addressed in https://github.com/AutoPas/AutoPas/issues/658.
70 if (splitLeafDimensions[d] < this->_interactionLength) {
71 anyNewDimSmallerThanMinSize = true;
72 break;
73 }
74 }
75
76 if ((this->_particles.size() < this->_treeSplitThreshold) or anyNewDimSmallerThanMinSize) {
77 this->_particles.push_back(p);
78 return nullptr;
79 } else {
80 std::unique_ptr<OctreeNodeInterface<Particle>> newInner = std::make_unique<OctreeInnerNode<Particle>>(
81 this->getBoxMin(), this->getBoxMax(), this->_parent, this->_treeSplitThreshold, this->_interactionLength,
82 this->_cellSizeFactor);
83 auto ret = newInner->insert(p);
84 if (ret) newInner = std::move(ret);
85 for (auto cachedParticle : this->_particles) {
86 ret = newInner->insert(cachedParticle);
87 if (ret) newInner = std::move(ret);
88 }
89
90 return newInner;
91 }
92 }
