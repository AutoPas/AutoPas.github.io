62 std::unique_ptr<OctreeNodeInterface<Particle>> insert(const Particle &p) override {
63 // Check if the size of the new leaves would become smaller than cellSizeFactor*interactionLength
64 std::array<double, 3> splitLeafDimensions = utils::ArrayMath::sub(this->getBoxMax(), this->getBoxMin());
65 splitLeafDimensions = utils::ArrayMath::mulScalar(splitLeafDimensions, 0.5);
66 bool anyNewDimSmallerThanMinSize = false;
67 for (auto d = 0; d < 3; ++d) {
68 auto cellSizeFactor = 1.0;
69 // @todo The condition below should actually be
70 // splitLeafDimensions[d] < (this->_cellSizeFactor * this->_interactionLength)
71 // But with this condition, the TraversalComparison test fails for cell size factor 0.5. Find out why the octree
72 // cannot handle this value. This problem is addressed in https://github.com/AutoPas/AutoPas/issues/658.
73 if (splitLeafDimensions[d] < this->_interactionLength) {
74 anyNewDimSmallerThanMinSize = true;
75 break;
76 }
77 }
78
79 if ((this->_particles.size() < this->_treeSplitThreshold) or anyNewDimSmallerThanMinSize) {
80 this->_particles.push_back(p);
81 return nullptr;
82 } else {
83 std::unique_ptr<OctreeNodeInterface<Particle>> newInner = std::make_unique<OctreeInnerNode<Particle>>(
84 this->getBoxMin(), this->getBoxMax(), this->_parent, this->_treeSplitThreshold, this->_interactionLength,
85 this->_cellSizeFactor);
86 auto ret = newInner->insert(p);
87 if (ret) newInner = std::move(ret);
88 for (auto cachedParticle : this->_particles) {
89 ret = newInner->insert(cachedParticle);
90 if (ret) newInner = std::move(ret);
91 }
92
93 return newInner;
94 }
95 }
