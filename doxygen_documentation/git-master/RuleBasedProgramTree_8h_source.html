66 [[nodiscard]] bool matches(const Configuration &configuration) const {
67 return (_containers.empty() or contains(_containers, configuration.container)) and
68 (_traversals.empty() or contains(_traversals, configuration.traversal)) and
69 (_loadEstimators.empty() or contains(_loadEstimators, configuration.loadEstimator)) and
70 (_dataLayouts.empty() or contains(_dataLayouts, configuration.dataLayout)) and
71 (_newton3Options.empty() or contains(_newton3Options, configuration.newton3)) and
72 (_cellSizeFactors.empty() or contains(_cellSizeFactors, configuration.cellSizeFactor));

80 optionSetToString(_loadEstimators) + "; " + optionSetToString(_loadEstimators) + "; " +
81 optionSetToString(_dataLayouts) + "; " + optionSetToString(_newton3Options) + "; ";
82 if (not _cellSizeFactors.empty()) {
83 res += std::accumulate(std::next(_cellSizeFactors.begin()), _cellSizeFactors.end(),
84 std::to_string(*_cellSizeFactors.begin()),
85 [](std::string s, double d) { return std::move(s) + ',', std::to_string(d); });
86 }

93 if (not set.empty()) {

95
96 return std::accumulate(std::next(set.begin()), set.end(), set.begin()->to_string(), comma_fold);
97 }
98 return {};

214 ConfigurationOrder(ConfigurationPattern greater, ConfigurationPattern smaller, std::vector<SameProperty> same = {})
215 : greater(std::move(greater)), smaller(std::move(smaller)), sameProperties(std::move(same)) {}

218
222 [[nodiscard]] std::string toString() const {
223 std::string sameString = sameProperties.empty() ? "" : "with same ";
224 for (auto same : sameProperties) {
225 sameString += samePropertyToString(same) + ", ";
226 }
227 return greater.toString() + " >= " + smaller.toString() + sameString;

236 [[nodiscard]] bool haveEqualSameProperties(const Configuration &conf1, const Configuration &conf2) const {
237 for (auto same : sameProperties) {
238 switch (same) {
239 case SameProperty::container:
240 if (conf1.container != conf2.container) return false;
241 break;
242 case SameProperty::traversal:
243 if (conf1.traversal != conf2.traversal) return false;
244 break;
245 case SameProperty::dataLayout:
246 if (conf1.dataLayout != conf2.dataLayout) return false;
247 break;
248 case SameProperty::newton3:
249 if (conf1.newton3 != conf2.newton3) return false;
250 break;
251 case SameProperty::loadEstimator:
252 if (conf1.loadEstimator != conf2.loadEstimator) return false;
253 break;
254 case SameProperty::cellSizeFactor:
255 if (conf1.cellSizeFactor != conf2.cellSizeFactor) return false;
256 break;
257 }
258 }
259 return true;
260 }
261
262 private:

264 switch (same) {

267 case SameProperty::traversal:

573struct If : public Statement {

592 condition->generateCode(context, program);
593 program.instructions.emplace_back(RuleVM::JUMPZERO);
594 context.freeStack(1);
595 auto jumpInstrIdx = program.instructions.size() - 1;
596

608struct RuleBasedProgramTree {
612 std::vector<std::shared_ptr<Statement>> statements;
613
619 [[nodiscard]] RuleVM::Program generateCode(CodeGenerationContext &context) const {
620 RuleVM::Program program;
621 program.instructions.emplace_back(RuleVM::RESERVE, 0ul);
622 auto reserveInstructionIdx = program.instructions.size() - 1;
623 for (const auto &statement : statements) {
624 statement->generateCode(context, program);
625 }
626 program.instructions[reserveInstructionIdx].payload = context.getNumLocalVariables();
627 program.neededStackSize = context.getNumLocalVariables() + context.getMaxStackSize();
628 program.instructions.emplace_back(RuleVM::HALT);
629 return program;
630 }
