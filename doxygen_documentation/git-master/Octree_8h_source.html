63 Octree(std::array<double, 3> boxMin, std::array<double, 3> boxMax, const double cutoff, const double skin,
64 const double cellSizeFactor)
65 : CellBasedParticleContainer<ParticleCell>(boxMin, boxMax, cutoff, skin) {
66 // @todo Obtain this from a configuration, reported in https://github.com/AutoPas/AutoPas/issues/624
67 int unsigned treeSplitThreshold = 16;
68
69 double interactionLength = this->getInteractionLength();
70
71 // Create the octree for the owned particles
72 this->_cells.push_back(
73 OctreeNodeWrapper<Particle>(boxMin, boxMax, treeSplitThreshold, interactionLength, cellSizeFactor));
74
75 // Extend the halo region with cutoff + skin in all dimensions
76 auto haloBoxMin = utils::ArrayMath::subScalar(boxMin, interactionLength);
77 auto haloBoxMax = utils::ArrayMath::addScalar(boxMax, interactionLength);
78 // Create the octree for the halo particles
79 this->_cells.push_back(
80 OctreeNodeWrapper<Particle>(haloBoxMin, haloBoxMax, treeSplitThreshold, interactionLength, cellSizeFactor));
81 }

83 [[nodiscard]] std::vector<ParticleType> updateContainer(bool keepNeighborListValid) override {
84 // invalidParticles: all outside boxMin/Max
85 std::vector<Particle> invalidParticles{};
86
87 if (keepNeighborListValid) {
88 invalidParticles = LeavingParticleCollector::collectParticlesAndMarkNonOwnedAsDummy(*this);
89 } else {
90 // This is a very primitive and inefficient way to rebuild the container:
91
92 // @todo Make this less indirect. (Find a better way to iterate all particles inside the octree to change
93 // this function back to a function that actually copies all particles out of the octree.)
94 // The problem is captured by https://github.com/AutoPas/AutoPas/issues/622
95
96 // 1. Copy all particles out of the container
97 std::vector<Particle *> particleRefs;
98 this->_cells[CellTypes::OWNED].collectAllParticles(particleRefs);
99 std::vector<Particle> particles{};
100 particles.reserve(particleRefs.size());
101
102 for (auto *p : particleRefs) {
103 if (utils::inBox(p->getR(), this->getBoxMin(), this->getBoxMax())) {
104 particles.push_back(*p);
105 } else {
106 invalidParticles.push_back(*p);
107 }
108 }
109
110 // 2. Clear the container
111 this->deleteAllParticles();
112
113 // 3. Insert the particles back into the container
114 for (auto &particle : particles) {
115 addParticleImpl(particle);
116 }
117 }
118
119 return invalidParticles;
120 }
