64 Octree(std::array<double, 3> boxMin, std::array<double, 3> boxMax, const double cutoff, const double skinPerTimestep,
65 const unsigned int rebuildFrequency, const double cellSizeFactor)
66 : CellBasedParticleContainer<ParticleCell>(boxMin, boxMax, cutoff, skinPerTimestep * rebuildFrequency) {

69 // @todo Obtain this from a configuration, reported in https://github.com/AutoPas/AutoPas/issues/624
70 int unsigned treeSplitThreshold = 16;
71
72 double interactionLength = this->getInteractionLength();
73
74 // Create the octree for the owned particles
75 this->_cells.push_back(
76 OctreeNodeWrapper<Particle>(boxMin, boxMax, treeSplitThreshold, interactionLength, cellSizeFactor));
77
78 // Extend the halo region with cutoff + skin in all dimensions
79 auto haloBoxMin = boxMin - interactionLength;
80 auto haloBoxMax = boxMax + interactionLength;
81 // Create the octree for the halo particles
82 this->_cells.push_back(

84 }

86 [[nodiscard]] std::vector<ParticleType> updateContainer(bool keepNeighborListValid) override {
87 // invalidParticles: all outside boxMin/Max
88 std::vector<Particle> invalidParticles{};
89
90 if (keepNeighborListValid) {
91 invalidParticles = LeavingParticleCollector::collectParticlesAndMarkNonOwnedAsDummy(*this);
92 } else {
93 // This is a very primitive and inefficient way to rebuild the container:
94
95 // @todo Make this less indirect. (Find a better way to iterate all particles inside the octree to change
96 // this function back to a function that actually copies all particles out of the octree.)
97 // The problem is captured by https://github.com/AutoPas/AutoPas/issues/622
98
99 // 1. Copy all particles out of the container
100 std::vector<Particle *> particleRefs;
101 this->_cells[CellTypes::OWNED].collectAllParticles(particleRefs);
102 std::vector<Particle> particles{};
103 particles.reserve(particleRefs.size());
104
105 for (auto *p : particleRefs) {
106 if (utils::inBox(p->getR(), this->getBoxMin(), this->getBoxMax())) {
107 particles.push_back(*p);
108 } else {
109 invalidParticles.push_back(*p);
110 }
111 }
112
113 // 2. Clear the container
114 this->deleteAllParticles();
115
116 // 3. Insert the particles back into the container
117 for (auto &particle : particles) {
118 addParticleImpl(particle);
119 }
120 }
121
122 return invalidParticles;
123 }
