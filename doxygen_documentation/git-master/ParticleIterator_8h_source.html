139  inline ParticleIterator<Particle, ParticleCell, modifiable> &operator++() override {
140  if (_additionalParticleVectorToIterateState != AdditionalParticleVectorToIterateState::iterating) {
141  // this iteration is done until a proper particle is found. i.e. one that satisfies the owning state.
142  do {
143  if (_iteratorWithinOneCell.isValid()) {
144  ++_iteratorWithinOneCell;
145  }
146 
147  // don't merge into if-else, _cell_iterator may become invalid after ++
148  if (not _iteratorWithinOneCell.isValid()) {
149  next_non_empty_cell();
150  }
151  } while (not isValid() and _iteratorAcrossCells < _vectorOfCells->end());
152  if (_additionalParticleVectorToIterateState == AdditionalParticleVectorToIterateState::notStarted and
153  _iteratorAcrossCells >= _vectorOfCells->end() and _additionalVectors->size() > _additionalVectorIndex) {
154  _additionalParticleVectorToIterateState = AdditionalParticleVectorToIterateState::iterating;
155  if (isValid()) {
156  // if the first particle in the additional Vector is valid, we return here.
157  return *this;
158  } // otherwise we run into the loop below and ++ further.
159  }
160  }
161  if (_additionalParticleVectorToIterateState == AdditionalParticleVectorToIterateState::iterating) {
162  // Simply increase the counter, as long as we aren't valid.
163  do {
164  ++_additionalVectorPosition;
165  // if we reach the end of this buffer jump to the next
166  if (_additionalVectorPosition >= (*_additionalVectors)[_additionalVectorIndex].size()) {
167  _additionalVectorIndex += (_behavior & IteratorBehavior::forceSequential) ? 1 : autopas_get_num_threads();
168  _additionalVectorPosition = 0;
169  }
170  // continue looking for valid iterator positions until there are no buffers left
171  } while (not isValid() and _additionalVectorIndex < _additionalVectors->size());
172  }
173  return *this;
174  }
