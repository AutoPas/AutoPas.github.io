97 explicit ParticleIterator(CellVecType *cont, size_t offset = 0, CellBorderAndFlagManagerType *flagManager = nullptr,
98 IteratorBehavior behavior = IteratorBehavior::ownedOrHalo,
99 ParticleVecTypeInput *additionalVectorsToIterate = nullptr)
100 : ParticleIterator(cont, flagManager, behavior, additionalVectorsToIterate) {
101 auto myThreadId = autopas_get_thread_num();
102 if (not(_behavior & IteratorBehavior::forceSequential)) {
103 offset += myThreadId;
104 }
105
106 if (offset < cont->size()) {
107 _iteratorAcrossCells += offset;
108 _iteratorWithinOneCell = _iteratorAcrossCells->begin();
109 } else if (_additionalParticleVectorToIterateState == AdditionalParticleVectorToIterateState::notStarted) {
110 _additionalParticleVectorToIterateState = AdditionalParticleVectorToIterateState::iterating;
111 } else {
112 _iteratorAcrossCells = cont->end();
113 AutoPasLog(trace, "More threads than cells. No work left for thread {}!", myThreadId);
114 return;
115 }
116
117 if (not(behavior & IteratorBehavior::ownedOrHalo) and flagManager == nullptr) {
118 utils::ExceptionHandler::exception(
119 "ParticleIterator::ParticleIterator() Behavior is not ownedOrHalo, but flagManager is nullptr!");
120 }
121
122 if (_additionalParticleVectorToIterateState != AdditionalParticleVectorToIterateState::iterating) {
123 if (not _iteratorWithinOneCell.isValid() or not isCellTypeBehaviorCorrect()) {
124 // This ensures that the first cell has a correct cell type behavior. This couldn't be done using operator++.
125 ParticleIterator::next_non_empty_cell();
126 }
127 // The iterator might still be invalid (because the cell is empty or the owned-state of the particle is wrong), so
128 // we check it here!
129 if (not ParticleIterator::isValid() and
130 (this->_iteratorAcrossCells < cont->end() or
131 _additionalParticleVectorToIterateState == AdditionalParticleVectorToIterateState::notStarted)) {
132 // the iterator is invalid + we still have particles/cells, so we increment it!
133 ParticleIterator::operator++();
134 }
135 } else {
136 if (not ParticleIterator::isValid()) {
137 // The iterator is invalid + we still have particles/cells, so we increment it!
138 ParticleIterator::operator++();
139 }
140 }

146 inline ParticleIterator<Particle, ParticleCell, modifiable> &operator++() override {
147 if (_additionalParticleVectorToIterateState != AdditionalParticleVectorToIterateState::iterating) {
148 // this iteration is done until a proper particle is found. i.e. one that satisfies the owning state.
149 do {
150 if (_iteratorWithinOneCell.isValid()) {
151 ++_iteratorWithinOneCell;
152 }
153
154 // don't merge into if-else, _cell_iterator may become invalid after ++
155 if (not _iteratorWithinOneCell.isValid()) {
156 next_non_empty_cell();
157 }
158 } while (not isValid() and _iteratorAcrossCells < _vectorOfCells->end());
159 if (_additionalParticleVectorToIterateState == AdditionalParticleVectorToIterateState::notStarted and
160 _iteratorAcrossCells >= _vectorOfCells->end() and _additionalVectors.size() > _additionalVectorIndex) {
161 _additionalParticleVectorToIterateState = AdditionalParticleVectorToIterateState::iterating;
162 if (isValid()) {
163 // if the first particle in the additional Vector is valid, we return here.
164 return *this;
165 } // otherwise we run into the loop below and ++ further.
166 }
167 }
168 if (_additionalParticleVectorToIterateState == AdditionalParticleVectorToIterateState::iterating) {
169 // Simply increase the counter, as long as we aren't valid.
170 do {
171 if (_additionalVectorIndex >= _additionalVectors.size()) {
172 // return if we get to be invalid!
173 _additionalParticleVectorToIterateState = AdditionalParticleVectorToIterateState::ignore;
174 return *this;
175 }
176 ++_additionalVectorPosition;
177 // if we reach the end of this buffer jump to the next
178 if (_additionalVectorPosition >= _additionalVectors[_additionalVectorIndex]->size()) {
179 _additionalVectorIndex += (_behavior & IteratorBehavior::forceSequential) ? 1 : autopas_get_num_threads();
180 _additionalVectorPosition = 0;
181 }
182 // continue looking for valid iterator positions, while we aren't valid.
183 } while (not isValid());
184 }
185 return *this;
186 }

202 [[nodiscard]] bool isValid() const override {
203 if (_additionalParticleVectorToIterateState == AdditionalParticleVectorToIterateState::iterating) {

232 inline void deleteCurrentParticleImpl() override {
233 if (_additionalParticleVectorToIterateState == AdditionalParticleVectorToIterateState::iterating) {
234 if constexpr (modifiable) {
235 auto &currentAdditionalVector = *_additionalVectors[_additionalVectorIndex];
236 // for the additionalParticleVector, we simply swap the last particle with the current particle and remove the
237 // particle that is now at the back.
238 std::swap(currentAdditionalVector[_additionalVectorPosition], currentAdditionalVector.back());
239 currentAdditionalVector.pop_back();
240 --_additionalVectorPosition;
241 } else {
242 utils::ExceptionHandler::exception("Error: Trying to delete a particle through a const iterator.");
243 }
244 } else if (_iteratorWithinOneCell.isValid()) {
245 if constexpr (modifiable) {
246 internal::deleteParticle(_iteratorWithinOneCell);
247 } else {
248 utils::ExceptionHandler::exception("Error: Trying to delete a particle through a const iterator.");
249 }
250 } else {
251 utils::ExceptionHandler::exception("ParticleIterator: deleting invalid particle");
252 }

273 [[nodiscard]] bool isCellTypeBehaviorCorrect() const {
274 // IMPORTANT: `this->` is necessary here! Without it clang 7, 8 and 9 fail due to an compiler bug:
275 // https://stackoverflow.com/questions/55359614/clang-complains-about-constexpr-function-in-case-for-switch-statement
276 switch (this->_behavior & ~IteratorBehavior::forceSequential) {
277 case IteratorBehavior::ownedOrHaloOrDummy:
278 [[fallthrough]];
279 case IteratorBehavior::ownedOrHalo:
280 return true;
281 case IteratorBehavior::halo:
282 return _flagManager->cellCanContainHaloParticles(_iteratorAcrossCells - _vectorOfCells->begin());
283 case IteratorBehavior::owned:
284 return _flagManager->cellCanContainOwnedParticles(_iteratorAcrossCells - _vectorOfCells->begin());
285 default:
286 utils::ExceptionHandler::exception(
287 "ParticleIterator::isCellTypeBehaviorCorrect() encountered unknown iterator behavior: {}", this->_behavior);
288 return false;
289 }
290 }

296 [[nodiscard]] bool particleHasCorrectOwnershipState() const {
297 // IMPORTANT: `this->` is necessary here! Without it clang 7, 8 and 9 fail due to an compiler bug:
298 // https://stackoverflow.com/questions/55359614/clang-complains-about-constexpr-function-in-case-for-switch-statement
299 switch (this->_behavior & ~IteratorBehavior::forceSequential) {
300 case IteratorBehavior::ownedOrHaloOrDummy:
301 return true;
302 case IteratorBehavior::ownedOrHalo:
303 if (_additionalParticleVectorToIterateState == AdditionalParticleVectorToIterateState::iterating) {
304 return not(*_additionalVectors[_additionalVectorIndex])[_additionalVectorPosition].isDummy();
305 } else {
306 return not _iteratorWithinOneCell->isDummy();
307 }
308 case IteratorBehavior::halo:
309 if (_additionalParticleVectorToIterateState == AdditionalParticleVectorToIterateState::iterating) {
310 return (*_additionalVectors[_additionalVectorIndex])[_additionalVectorPosition].isHalo();
311 } else {
312 return _iteratorWithinOneCell->isHalo();
313 }
314 case IteratorBehavior::owned:
315 if (_additionalParticleVectorToIterateState == AdditionalParticleVectorToIterateState::iterating) {
316 return (*_additionalVectors[_additionalVectorIndex])[_additionalVectorPosition].isOwned();
317 } else {
318 return _iteratorWithinOneCell->isOwned();
319 }
320 default:
321 utils::ExceptionHandler::exception(
322 "ParticleIterator::particleHasCorrectOwnershipState() encountered unknown iterator behavior: {}",
323 this->_behavior);
324 return false;
325 }
