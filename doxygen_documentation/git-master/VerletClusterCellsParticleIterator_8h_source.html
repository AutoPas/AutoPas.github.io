266  inline VerletClusterCellsParticleIterator<Particle, ParticleCell, modifiable> &operator++() override {
267  do {
268  // increase the cell-local iterator
269  ++this->_iteratorWithinOneCell;
270 
271  while (this->_iteratorWithinOneCell == this->_cellEnd) {
272  // Increase _currentRegionIndex by stride if we are at the end of a cell!
273  // Also ensure that it does not go beyond _indicesInRegion.size() - 1.
274  // The last entry of _indicesInRegion is invalid (>= _vectorOfCells->size()), so this is fine!
275  _currentRegionIndex = std::min(_currentRegionIndex + autopas_get_num_threads(), _indicesInRegion.size() - 1);
276  this->_cellId = _indicesInRegion[_currentRegionIndex];
277  if (this->_cellId >= this->_vectorOfCells->size()) {
278  return *this;
279  }
280  if (not modifiable) {
281  // We can't delete particles, so optimizations assuming sorted particles are in order!
282 
283  // Find lowest possible particle in the region.
284  // Sorting in the array is at most off by skin/2 as the container is rebuilt if a particle moves more.
285  // increasing the search area by skin guarantees particles in the region to be found.
286  this->_iteratorWithinOneCell = std::lower_bound(
287  (*this->_vectorOfCells)[this->_cellId]._particles.begin(),
288  (*this->_vectorOfCells)[this->_cellId]._particles.begin() + this->getDummyStartbyIndex(this->_cellId),
289  _startRegion[2] - _skin, [](const Particle &a, const double b) { return a.getR()[2] < b; });
290  this->_cellEnd = std::upper_bound(
291  (*this->_vectorOfCells)[this->_cellId]._particles.begin(),
292  (*this->_vectorOfCells)[this->_cellId]._particles.begin() + this->getDummyStartbyIndex(this->_cellId),
293  _endRegion[2] + _skin, [](const double b, const Particle &a) { return b < a.getR()[2]; });
294  } else {
295  // If the iterator is modifiable, we can delete particles, which breaks the sorted assumption and thus
296  // some particles might be skipped.
297  this->_iteratorWithinOneCell = (*this->_vectorOfCells)[this->_cellId]._particles.begin();
298  this->_cellEnd = this->_iteratorWithinOneCell + this->getDummyStartbyIndex(this->_cellId);
299  }
300  }
301  } while ((not VerletClusterCellsParticleIterator<Particle, ParticleCell, modifiable>::fitsBehavior(
302  *this->_iteratorWithinOneCell)) or
303  utils::notInBox(this->_iteratorWithinOneCell->getR(), _startRegion, _endRegion));
304  return *this;
305  }
