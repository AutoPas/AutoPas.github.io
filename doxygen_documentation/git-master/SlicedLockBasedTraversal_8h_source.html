96 if (this->_dynamic) {
97 omp_set_schedule(omp_sched_dynamic, 1);
98 } else {
99 omp_set_schedule(omp_sched_static, 1);
100 }
101#pragma omp parallel for schedule(runtime) num_threads(numThreads)
102#endif

104 timers[slice].start();
105 array<unsigned long, 3> myStartArray{0, 0, 0};
106 for (size_t i = 0; i < slice; ++i) {
107 myStartArray[this->_dimsPerLength[0]] += this->_sliceThickness[i];
108 }
109
110 // all but the first slice need to lock their starting layers.
111 const unsigned long lockBaseIndex = (slice - 1) * this->_overlapLongestAxis;
112 if (slice > 0) {
113 for (unsigned long i = 0ul; i < this->_overlapLongestAxis; i++) {
114 locks[lockBaseIndex + i].lock();
115 }
116 }

119 const auto dimSlice = myStartArray[this->_dimsPerLength[0]] + sliceOffset;
120 // at the last layers request lock for the starting layer of the next
121 // slice. Does not apply for the last slice.
122 if (slice != numSlices - 1 and dimSlice >= lastLayer - this->_overlapLongestAxis) {
123 locks[((slice + 1) * this->_overlapLongestAxis) - (lastLayer - dimSlice)].lock();
124 }

131 idArray[this->_dimsPerLength[1]] = dimMedium;
132 idArray[this->_dimsPerLength[2]] = dimShort;

135 }
136 }
137 // at the end of the first layers release the lock
138 if (slice > 0 and dimSlice < myStartArray[this->_dimsPerLength[0]] + this->_overlapLongestAxis) {

140 // if lastLayer is reached within overlap area, unlock all following locks
141 // this should never be the case if slice thicknesses are set up properly; thickness should always be
142 // greater than the overlap along the longest axis, or the slices won't be processed in parallel.
143 if (dimSlice == lastLayer - 1) {
144 for (unsigned long i = sliceOffset + 1; i < this->_overlapLongestAxis; ++i) {
145 locks[lockBaseIndex + i].unlock();

147 }
148 } else if (slice != numSlices - 1 and dimSlice == lastLayer - 1) {
149 // clearing of the locks set on the last layers of each slice
150 for (size_t i = (slice * this->_overlapLongestAxis); i < (slice + 1) * this->_overlapLongestAxis; ++i) {
151 locks[i].unlock();
152 }
153 }
154 }
155 threadTimes[slice] = timers[slice].stop();
156 }

158 std::string timesStr;

160 timesStr += std::to_string(t) + ", ";
161 }
162 auto minMax = std::minmax_element(threadTimes.begin(), threadTimes.end());
163 auto avg = std::accumulate(threadTimes.begin(), threadTimes.end(), 0.0) / numSlices;
164 auto variance = std::accumulate(threadTimes.cbegin(), threadTimes.cend(), 0.0,
165 [avg](double a, double b) -> double { return a + std::pow(avg - b, 2.0); }) /
166 numSlices;
167 auto stddev = std::sqrt(variance);
168
169 AutoPasLog(debug, "times per slice: [{}].", timesStr);
170 AutoPasLog(debug, "Difference between longest and shortest time: {:.3G}", *minMax.second - *minMax.first);
171 AutoPasLog(debug, "Ratio between longest and shortest time: {:.3G}", (float)*minMax.second / *minMax.first);
172 AutoPasLog(debug, "avg: {:.3G}, std-deviation: {:.3G} ({:.3G}%)", avg, stddev, 100 * stddev / avg);
