71 virtual void initSliceThickness(unsigned long minSliceThickness) {
72 auto numSlices = this->_cellsPerDimension[_dimsPerLength[0]] / minSliceThickness;
73 _sliceThickness.clear();
74
75 // abort if domain is too small -> cleared _sliceThickness array indicates non applicability
76 if (numSlices < 1) return;
77
78 _sliceThickness.insert(_sliceThickness.begin(), numSlices, minSliceThickness);
79 auto rest = this->_cellsPerDimension[_dimsPerLength[0]] - _sliceThickness[0] * numSlices;
80 // remaining slices to distribute the remaining layers on
81 auto remSlices = std::min(rest, numSlices);
82 for (size_t i = 0; i < remSlices; ++i) {
83 _sliceThickness[i] += rest / (remSlices - i);
84 rest -= rest / (remSlices - i);
85 }
86 if (spaciallyForward) {
87 // decreases last _sliceThickness by _overlapLongestAxis to account for the way we handle base cells
88 _sliceThickness.back() -= _overlapLongestAxis;
89 }
90 }

179inline void SlicedBasedTraversal<ParticleCell, PairwiseFunctor, dataLayout, useNewton3, spaciallyForward>::init(
180 const std::array<unsigned long, 3> &dims) {
181 for (unsigned int d = 0; d < 3; d++) {
182 _overlap[d] = std::ceil(_interactionLength / _cellLength[d]);
183 if (not spaciallyForward) {
184 // there is potentially overlap in both directions.
185 _overlap[d] *= 2;
186 }
187 }
188
189 // find longest dimension
190 auto minMaxElem = std::minmax_element(this->_cellsPerDimension.begin(), this->_cellsPerDimension.end());
191 _dimsPerLength[0] = (int)std::distance(this->_cellsPerDimension.begin(), minMaxElem.second);
192 _dimsPerLength[2] = (int)std::distance(this->_cellsPerDimension.begin(), minMaxElem.first);
193 _dimsPerLength[1] = 3 - (_dimsPerLength[0] + _dimsPerLength[2]);
194
195 _overlapLongestAxis = _overlap[_dimsPerLength[0]];
196}
