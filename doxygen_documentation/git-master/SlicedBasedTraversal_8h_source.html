72 virtual void initSliceThickness(unsigned long minSliceThickness) {
73 auto numSlices = this->_cellsPerDimension[_dimsPerLength[0]] / minSliceThickness;
74 _sliceThickness.clear();
75
76 // abort if domain is too small -> cleared _sliceThickness array indicates non applicability
77 if (numSlices < 1) return;
78
79 _sliceThickness.insert(_sliceThickness.begin(), numSlices, minSliceThickness);
80 auto rest = this->_cellsPerDimension[_dimsPerLength[0]] - _sliceThickness[0] * numSlices;
81 // remaining slices to distribute the remaining layers on
82 auto remSlices = std::min(rest, numSlices);
83 for (size_t i = 0; i < remSlices; ++i) {
84 _sliceThickness[i] += rest / (remSlices - i);
85 rest -= rest / (remSlices - i);
86 }
87 if (_spaciallyForward) {
88 // decreases last _sliceThickness by _overlapLongestAxis to account for the way we handle base cells
89 _sliceThickness.back() -= _overlapLongestAxis;
90 }
91 }

180inline void SlicedBasedTraversal<ParticleCell, PairwiseFunctor>::init(const std::array<unsigned long, 3> &dims) {
181 for (unsigned int d = 0; d < 3; d++) {
182 _overlap[d] = std::ceil(_interactionLength / _cellLength[d]);
183 if (not _spaciallyForward) {
184 // there is potentially overlap in both directions.
185 _overlap[d] *= 2;
186 }
187 }
188
189 // find longest dimension
190 auto minMaxElem = std::minmax_element(this->_cellsPerDimension.begin(), this->_cellsPerDimension.end());
191 _dimsPerLength[0] = (int)std::distance(this->_cellsPerDimension.begin(), minMaxElem.second);
192 _dimsPerLength[2] = (int)std::distance(this->_cellsPerDimension.begin(), minMaxElem.first);
193 _dimsPerLength[1] = 3 - (_dimsPerLength[0] + _dimsPerLength[2]);
194
195 _overlapLongestAxis = _overlap[_dimsPerLength[0]];
196}
