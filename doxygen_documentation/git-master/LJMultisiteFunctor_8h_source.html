178 void AoSFunctor(Particle &particleA, Particle &particleB, bool newton3) final {
179 using namespace autopas::utils::ArrayMath::literals;
180 if (particleA.isDummy() or particleB.isDummy()) {
181 return;
182 }
183
184 // Don't calculate force if particleB outside cutoff of particleA
185 const auto displacementCoM = autopas::utils::ArrayMath::sub(particleA.getR(), particleB.getR());
186 const auto distanceSquaredCoM = autopas::utils::ArrayMath::dot(displacementCoM, displacementCoM);
187
188 if (distanceSquaredCoM > _cutoffSquared) {
189 return;
190 }
191
192 // get number of sites
193 const size_t numSitesA = useMixing ? _PPLibrary->getNumSites(particleA.getTypeId()) : _sitePositionsLJ.size();
194 const size_t numSitesB = useMixing ? _PPLibrary->getNumSites(particleB.getTypeId()) : _sitePositionsLJ.size();
195
196 // get siteIds
197 const std::vector<size_t> siteIdsA =
198 useMixing ? _PPLibrary->getSiteTypes(particleA.getTypeId()) : std::vector<unsigned long>();
199 const std::vector<size_t> siteIdsB =
200 useMixing ? _PPLibrary->getSiteTypes(particleB.getTypeId()) : std::vector<unsigned long>();
201
202 // get unrotated relative site positions
203 const std::vector<std::array<double, 3>> unrotatedSitePositionsA =
204 useMixing ? _PPLibrary->getSitePositions(particleA.getTypeId()) : _sitePositionsLJ;
205 const std::vector<std::array<double, 3>> unrotatedSitePositionsB =
206 useMixing ? _PPLibrary->getSitePositions(particleB.getTypeId()) : _sitePositionsLJ;
207
208 // calculate correctly rotated relative site positions
209 const auto rotatedSitePositionsA =
210 autopas::utils::quaternion::rotateVectorOfPositions(particleA.getQuaternion(), unrotatedSitePositionsA);
211 const auto rotatedSitePositionsB =
212 autopas::utils::quaternion::rotateVectorOfPositions(particleB.getQuaternion(), unrotatedSitePositionsB);
213
214 for (int i = 0; i < numSitesA; i++) {
215 for (int j = 0; j < numSitesB; j++) {
216 const auto displacement = autopas::utils::ArrayMath::add(
217 autopas::utils::ArrayMath::sub(displacementCoM, rotatedSitePositionsB[j]), rotatedSitePositionsA[i]);
218 const auto distanceSquared = autopas::utils::ArrayMath::dot(displacement, displacement);
219
220 const auto sigmaSquared =
221 useMixing ? _PPLibrary->getMixingSigmaSquared(siteIdsA[i], siteIdsB[j]) : _sigmaSquared;
222 const auto epsilon24 = useMixing ? _PPLibrary->getMixing24Epsilon(siteIdsA[i], siteIdsB[j]) : _epsilon24;
223 const auto shift6 =
224 applyShift ? (useMixing ? _PPLibrary->getMixingShift6(siteIdsA[i], siteIdsB[j]) : _shift6) : 0;
225
226 // clang-format off
227 // Calculate potential between sites and thus force
228 // Force = 24 * epsilon * (2*(sigma/distance)^12 - (sigma/distance)^6) * (1/distance)^2 * [x_displacement, y_displacement, z_displacement]
229 // { scalarMultiple } * { displacement }
230 // clang-format on
231 const auto invDistSquared = 1. / distanceSquared;
232 const auto lj2 = sigmaSquared * invDistSquared;
233 const auto lj6 = lj2 * lj2 * lj2;
234 const auto lj12 = lj6 * lj6;
235 const auto lj12m6 = lj12 - lj6; // = LJ potential / (4x epsilon)
236 const auto scalarMultiple = epsilon24 * (lj12 + lj12m6) * invDistSquared;
237 const auto force = autopas::utils::ArrayMath::mulScalar(displacement, scalarMultiple);
238
239 // Add force on site to net force
240 particleA.addF(force);
241 if (newton3) {
242 particleB.subF(force);
243 }
244
245 // Add torque applied by force
246 particleA.addTorque(autopas::utils::ArrayMath::cross(rotatedSitePositionsA[i], force));
247 if (newton3) {
248 particleB.subTorque(autopas::utils::ArrayMath::cross(rotatedSitePositionsB[j], force));
249 }
250
251 if (calculateGlobals) {
252 // We always add the full contribution for each owned particle and divide the sums by 2 in endTraversal().
253 // Potential energy has an additional factor of 6, which is also handled in endTraversal().
254 const auto potentialEnergy6 = epsilon24 * lj12m6 + shift6;
255 const auto virial = displacement * force;
256
257 const auto threadNum = autopas::autopas_get_thread_num();
258
259 if (particleA.isOwned()) {
260 _aosThreadData[threadNum].potentialEnergySum += potentialEnergy6;
261 _aosThreadData[threadNum].virialSum += virial;
262 }
263 // for non-newton3 the second particle will be considered in a separate calculation
264 if (newton3 and particleB.isOwned()) {
265 _aosThreadData[threadNum].potentialEnergySum += potentialEnergy6;
266 _aosThreadData[threadNum].virialSum += virial;
267 }
268 }
269 }
270 }

692 unsigned long getNumFlopsPerKernelCall(size_t molAType, size_t molBType, bool newton3) {
693 // Site-to-site displacement: 6 (3 in the SoA case, but this requires O(N) precomputing site positions)
694 // Site-to-site distance squared: 4
695 // Compute scale: 9
696 // Apply scale to force: With newton3: 6, Without: 3
697 // Apply scale to torque: With newton3 18, Without: 9 (0 in SoA case, with O(N) post computing)
698 // Site-to-site total: With newton3: 33, Without: 26
699 // (SoA total: With N3L: 22, Without N3L: 19)
700 // Above multiplied by number sites of i * number sites of j
701 const unsigned long siteToSiteFlops = newton3 ? 33ul : 26ul;
702 return _PPLibrary->getNumSites(molAType) * _PPLibrary->getNumSites(molBType) * siteToSiteFlops;
703 }

709 void initTraversal() final {
710 _potentialEnergySum = 0;
711 _virialSum = {0., 0., 0.};
712 _postProcessed = false;
713 for (size_t i = 0; i < _aosThreadData.size(); i++) {
714 _aosThreadData[i].setZero();
715 }
716 }
717
722 void endTraversal(bool newton3) final {
723 using namespace autopas::utils::ArrayMath::literals;
724
725 if (_postProcessed) {
726 throw autopas::utils::ExceptionHandler::AutoPasException(
727 "Already postprocessed, endTraversal(bool newton3) was called twice without calling initTraversal().");
728 }
729 if (calculateGlobals) {
730 for (size_t i = 0; i < _aosThreadData.size(); ++i) {
731 _potentialEnergySum += _aosThreadData[i].potentialEnergySum;
732 _virialSum += _aosThreadData[i].virialSum;
733 }
734
735 // For each interaction, we added the full contribution for both particles. Divide by 2 here, so that each
736 // contribution is only counted once per pair.
737 _potentialEnergySum *= 0.5;
738 _virialSum *= 0.5;
739
740 // We have always calculated 6*potentialEnergy, so we divide by 6 here!
741 _potentialEnergySum /= 6.;
742 _postProcessed = true;
743
744 AutoPasLog(DEBUG, "Final potential energy {}", _potentialEnergySum);
745 AutoPasLog(DEBUG, "Final virial {}", _virialSum[0] + _virialSum[1] + _virialSum[2]);
746 }
747 }

808 const auto *const __restrict q2Aptr = soaA.template begin<Particle::AttributeNames::quaternion2>();
809 const auto *const __restrict q3Aptr = soaA.template begin<Particle::AttributeNames::quaternion3>();
810 const auto *const __restrict q0Bptr = soaB.template begin<Particle::AttributeNames::quaternion0>();
811 const auto *const __restrict q1Bptr = soaB.template begin<Particle::AttributeNames::quaternion1>();
812 const auto *const __restrict q2Bptr = soaB.template begin<Particle::AttributeNames::quaternion2>();
813 const auto *const __restrict q3Bptr = soaB.template begin<Particle::AttributeNames::quaternion3>();
814

816 SoAFloatPrecision *const __restrict fyAptr = soaA.template begin<Particle::AttributeNames::forceY>();

818 SoAFloatPrecision *const __restrict fxBptr = soaB.template begin<Particle::AttributeNames::forceX>();
819 SoAFloatPrecision *const __restrict fyBptr = soaB.template begin<Particle::AttributeNames::forceY>();
820 SoAFloatPrecision *const __restrict fzBptr = soaB.template begin<Particle::AttributeNames::forceZ>();
821
822 SoAFloatPrecision *const __restrict txAptr = soaA.template begin<Particle::AttributeNames::torqueX>();
823 SoAFloatPrecision *const __restrict tyAptr = soaA.template begin<Particle::AttributeNames::torqueY>();
824 SoAFloatPrecision *const __restrict tzAptr = soaA.template begin<Particle::AttributeNames::torqueZ>();
825 SoAFloatPrecision *const __restrict txBptr = soaB.template begin<Particle::AttributeNames::torqueX>();
826 SoAFloatPrecision *const __restrict tyBptr = soaB.template begin<Particle::AttributeNames::torqueY>();
827 SoAFloatPrecision *const __restrict tzBptr = soaB.template begin<Particle::AttributeNames::torqueZ>();
828
829 [[maybe_unused]] auto *const __restrict typeptrA = soaA.template begin<Particle::AttributeNames::typeId>();
830 [[maybe_unused]] auto *const __restrict typeptrB = soaB.template begin<Particle::AttributeNames::typeId>();
831
832 SoAFloatPrecision potentialEnergySum = 0.;
833 SoAFloatPrecision virialSumX = 0.;
834 SoAFloatPrecision virialSumY = 0.;
835 SoAFloatPrecision virialSumZ = 0.;
836
837 // local redeclarations to help compilers
838 const SoAFloatPrecision cutoffSquared = _cutoffSquared;
839
840 std::vector<SoAFloatPrecision, autopas::AlignedAllocator<SoAFloatPrecision>> sigmaSquareds;
841 std::vector<SoAFloatPrecision, autopas::AlignedAllocator<SoAFloatPrecision>> epsilon24s;
842 std::vector<SoAFloatPrecision, autopas::AlignedAllocator<SoAFloatPrecision>> shift6s;
843
844 std::vector<SoAFloatPrecision, autopas::AlignedAllocator<SoAFloatPrecision>> exactSitePositionBx;
845 std::vector<SoAFloatPrecision, autopas::AlignedAllocator<SoAFloatPrecision>> exactSitePositionBy;
846 std::vector<SoAFloatPrecision, autopas::AlignedAllocator<SoAFloatPrecision>> exactSitePositionBz;
847
848 // we require arrays for forces for sites to maintain SIMD in site-site calculations
849 std::vector<SoAFloatPrecision, autopas::AlignedAllocator<SoAFloatPrecision>> siteForceBx;
850 std::vector<SoAFloatPrecision, autopas::AlignedAllocator<SoAFloatPrecision>> siteForceBy;
851 std::vector<SoAFloatPrecision, autopas::AlignedAllocator<SoAFloatPrecision>> siteForceBz;
852
853 std::vector<size_t, autopas::AlignedAllocator<size_t>> siteTypesB;
854 std::vector<char, autopas::AlignedAllocator<char>> isSiteOwnedBArr;
855
856 const SoAFloatPrecision const_sigmaSquared = _sigmaSquared;
857 const SoAFloatPrecision const_epsilon24 = _epsilon24;
858 const SoAFloatPrecision const_shift6 = _shift6;
859
860 const auto const_unrotatedSitePositions = _sitePositionsLJ;
861
862 // count number of sites in both SoAs
863 size_t siteCountB = 0;
864 if constexpr (useMixing) {
865 for (size_t mol = 0; mol < soaB.size(); ++mol) {
866 siteCountB += _PPLibrary->getNumSites(typeptrB[mol]);
867 }
868 } else {
869 siteCountB = const_unrotatedSitePositions.size() * soaB.size();
870 }
871
872 // pre-reserve std::vectors
873 exactSitePositionBx.reserve(siteCountB);
874 exactSitePositionBy.reserve(siteCountB);
875 exactSitePositionBz.reserve(siteCountB);
876
877 if constexpr (useMixing) {
878 siteTypesB.reserve(siteCountB);
879 }
880
881 siteForceBx.reserve(siteCountB);
882 siteForceBy.reserve(siteCountB);
883 siteForceBz.reserve(siteCountB);
884
885 if constexpr (calculateGlobals) {
886 // this is only needed for vectorization when calculating globals
887 isSiteOwnedBArr.reserve(siteCountB);
888 }
889
890 if constexpr (useMixing) {
891 siteTypesB.reserve(siteCountB);
892 sigmaSquareds.reserve(siteCountB);
893 epsilon24s.reserve(siteCountB);
894 if constexpr (applyShift) {
895 shift6s.reserve(siteCountB);
896 }
897 }
898
899 // Fill site-wise std::vectors for SIMD
900 if constexpr (useMixing) {
901 size_t siteIndex = 0;
902 for (size_t mol = 0; mol < soaB.size(); ++mol) {
903 const auto rotatedSitePositions = autopas::utils::quaternion::rotateVectorOfPositions(
904 {q0Bptr[mol], q1Bptr[mol], q2Bptr[mol], q3Bptr[mol]}, _PPLibrary->getSitePositions(typeptrB[mol]));
905 const auto siteTypesOfMol = _PPLibrary->getSiteTypes(typeptrB[mol]);
906
907 for (size_t site = 0; site < _PPLibrary->getNumSites(typeptrB[mol]); ++site) {
908 exactSitePositionBx[siteIndex] = rotatedSitePositions[site][0] + xBptr[mol];
909 exactSitePositionBy[siteIndex] = rotatedSitePositions[site][1] + yBptr[mol];
910 exactSitePositionBz[siteIndex] = rotatedSitePositions[site][2] + zBptr[mol];
911 siteTypesB[siteIndex] = siteTypesOfMol[site];
912 siteForceBx[siteIndex] = 0.;
913 siteForceBy[siteIndex] = 0.;
914 siteForceBz[siteIndex] = 0.;
915 if (calculateGlobals) {
916 isSiteOwnedBArr[siteIndex] = ownedStatePtrB[mol] == autopas::OwnershipState::owned;
917 }
918 ++siteIndex;
919 }
920 }
921 } else {
922 size_t siteIndex = 0;
923 for (size_t mol = 0; mol < soaB.size(); mol++) {
924 const auto rotatedSitePositions = autopas::utils::quaternion::rotateVectorOfPositions(
925 {q0Bptr[mol], q1Bptr[mol], q2Bptr[mol], q3Bptr[mol]}, const_unrotatedSitePositions);
926 for (size_t site = 0; site < const_unrotatedSitePositions.size(); ++site) {
927 exactSitePositionBx[siteIndex] = rotatedSitePositions[site][0] + xBptr[mol];
928 exactSitePositionBy[siteIndex] = rotatedSitePositions[site][1] + yBptr[mol];
929 exactSitePositionBz[siteIndex] = rotatedSitePositions[site][2] + zBptr[mol];
930 siteForceBx[siteIndex] = 0.;
931 siteForceBy[siteIndex] = 0.;
932 siteForceBz[siteIndex] = 0.;
933 if (calculateGlobals) {
934 isSiteOwnedBArr[siteIndex] = ownedStatePtrB[mol] == autopas::OwnershipState::owned;
935 }
936 ++siteIndex;
937 }
938 }
939 }
940
941 // main force calculation loop
942 for (size_t molA = 0; molA < soaA.size(); ++molA) {
943 const auto ownedStateA = ownedStatePtrA[molA];
944 if (ownedStateA == autopas::OwnershipState::dummy) {
945 continue;
946 }
947
948 const auto noSitesInMolA =
949 useMixing ? _PPLibrary->getNumSites(typeptrA[molA]) : const_unrotatedSitePositions.size();
950 const auto unrotatedSitePositionsA =
951 useMixing ? _PPLibrary->getSitePositions(typeptrA[molA]) : const_unrotatedSitePositions;
952
953 const auto rotatedSitePositionsA = autopas::utils::quaternion::rotateVectorOfPositions(
954 {q0Aptr[molA], q1Aptr[molA], q2Aptr[molA], q3Aptr[molA]}, unrotatedSitePositionsA);
955
956 // create mask over every mol in cell B (char to keep arrays aligned)
957 std::vector<char, autopas::AlignedAllocator<char>> molMask;
958 molMask.reserve(soaB.size());
959
960#pragma omp simd
961 for (size_t molB = 0; molB < soaB.size(); ++molB) {
962 const auto ownedStateB = ownedStatePtrB[molB];
963
964 const auto displacementCoMX = xAptr[molA] - xBptr[molB];
965 const auto displacementCoMY = yAptr[molA] - yBptr[molB];
966 const auto displacementCoMZ = zAptr[molA] - zBptr[molB];
967
968 const auto distanceSquaredCoMX = displacementCoMX * displacementCoMX;
969 const auto distanceSquaredCoMY = displacementCoMY * displacementCoMY;
970 const auto distanceSquaredCoMZ = displacementCoMZ * displacementCoMZ;
971
972 const auto distanceSquaredCoM = distanceSquaredCoMX + distanceSquaredCoMY + distanceSquaredCoMZ;
973
974 // mask sites of molecules beyond cutoff or if molecule is a dummy
975 molMask[molB] = distanceSquaredCoM <= cutoffSquared and ownedStateB != autopas::OwnershipState::dummy;
976 }
977
978 // generate mask for each site in cell B from molecular mask
979 std::vector<char, autopas::AlignedAllocator<char>> siteMask;
980 siteMask.reserve(siteCountB);
981
982 for (size_t molB = 0; molB < soaB.size(); ++molB) {
983 for (size_t siteB = 0; siteB < _PPLibrary->getNumSites(typeptrB[molB]); ++siteB) {
984 siteMask.emplace_back(molMask[molB]);
985 }
986 }
987
988 // sums used for molA
989 SoAFloatPrecision forceSumX = 0.;
990 SoAFloatPrecision forceSumY = 0.;
991 SoAFloatPrecision forceSumZ = 0.;
992 SoAFloatPrecision torqueSumX = 0.;
993 SoAFloatPrecision torqueSumY = 0.;
994 SoAFloatPrecision torqueSumZ = 0.;
995
996 for (size_t siteA = 0; siteA < noSitesInMolA; ++siteA) {
997 if (useMixing) {
998 // preload sigmas, epsilons, and shifts
999 for (size_t siteB = 0; siteB < siteCountB; ++siteB) {
1000 const auto mixingData =
1001 _PPLibrary->getLJMixingData(_PPLibrary->getSiteTypes(typeptrA[molA])[siteA], siteTypesB[siteB]);
1002 sigmaSquareds[siteB] = mixingData.sigmaSquared;
1003 epsilon24s[siteB] = mixingData.epsilon24;
1004 if (applyShift) {
1005 shift6s[siteB] = mixingData.shift6;
1006 }
1007 }
1008 }
1009
1010 const auto rotatedSitePositionAx = rotatedSitePositionsA[siteA][0];
1011 const auto rotatedSitePositionAy = rotatedSitePositionsA[siteA][1];
1012 const auto rotatedSitePositionAz = rotatedSitePositionsA[siteA][2];
1013
1014 const auto exactSitePositionAx = rotatedSitePositionAx + xAptr[molA];
1015 const auto exactSitePositionAy = rotatedSitePositionAy + yAptr[molA];
1016 const auto exactSitePositionAz = rotatedSitePositionAz + zAptr[molA];
1017
1018#pragma omp simd reduction (+ : forceSumX, forceSumY, forceSumZ, torqueSumX, torqueSumY, torqueSumZ, potentialEnergySum, virialSumX, virialSumY, virialSumZ)
1019 for (size_t siteB = 0; siteB < siteCountB; ++siteB) {
1020 const SoAFloatPrecision sigmaSquared = useMixing ? sigmaSquareds[siteB] : const_sigmaSquared;
1021 const SoAFloatPrecision epsilon24 = useMixing ? epsilon24s[siteB] : const_epsilon24;
1022 const SoAFloatPrecision shift6 = applyShift ? (useMixing ? shift6s[siteB] : const_shift6) : 0;
1023
1024 const auto isSiteOwnedB = !calculateGlobals || isSiteOwnedBArr[siteB];
1025
1026 const auto displacementX = exactSitePositionAx - exactSitePositionBx[siteB];
1027 const auto displacementY = exactSitePositionAy - exactSitePositionBy[siteB];
1028 const auto displacementZ = exactSitePositionAz - exactSitePositionBz[siteB];
1029
1030 const auto distanceSquaredX = displacementX * displacementX;
1031 const auto distanceSquaredY = displacementY * displacementY;
1032 const auto distanceSquaredZ = displacementZ * displacementZ;
1033
1034 const auto distanceSquared = distanceSquaredX + distanceSquaredY + distanceSquaredZ;
1035
1036 const auto invDistSquared = 1. / distanceSquared;
1037 const auto lj2 = sigmaSquared * invDistSquared;
1038 const auto lj6 = lj2 * lj2 * lj2;
1039 const auto lj12 = lj6 * lj6;
1040 const auto lj12m6 = lj12 - lj6;
1041 const auto scalarMultiple = siteMask[siteB] ? epsilon24 * (lj12 + lj12m6) * invDistSquared : 0.;
1042
1043 // calculate forces
1044 const auto forceX = scalarMultiple * displacementX;
1045 const auto forceY = scalarMultiple * displacementY;
1046 const auto forceZ = scalarMultiple * displacementZ;
1047
1048 forceSumX += forceX;
1049 forceSumY += forceY;
1050 forceSumZ += forceZ;
1051
1052 torqueSumX += rotatedSitePositionAy * forceZ - rotatedSitePositionAz * forceY;
1053 torqueSumY += rotatedSitePositionAz * forceX - rotatedSitePositionAx * forceZ;
1054 torqueSumZ += rotatedSitePositionAx * forceY - rotatedSitePositionAy * forceX;
1055
1056 // N3L ( total molecular forces + torques to be determined later )
1057 if constexpr (newton3) {
1058 siteForceBx[siteB] -= forceX;
1059 siteForceBy[siteB] -= forceY;
1060 siteForceBz[siteB] -= forceZ;
1061 }
1062
1063 // globals
1064 if constexpr (calculateGlobals) {
1065 const auto potentialEnergy6 = siteMask[siteB] ? (epsilon24 * lj12m6 + shift6) : 0.;
1066 const auto virialX = displacementX * forceX;
1067 const auto virialY = displacementY * forceY;
1068 const auto virialZ = displacementZ * forceZ;
1069
1070 // We add 6 times the potential energy for each owned particle. The total sum is corrected in
1071 // endTraversal().
1072 const auto ownershipFactor =
1073 newton3 ? (ownedStateA == autopas::OwnershipState::owned ? 1. : 0.) + (isSiteOwnedB ? 1. : 0.)
1074 : (ownedStateA == autopas::OwnershipState::owned ? 1. : 0.);
1075 potentialEnergySum += potentialEnergy6 * ownershipFactor;
1076 virialSumX += virialX * ownershipFactor;
1077 virialSumY += virialY * ownershipFactor;
1078 virialSumZ += virialZ * ownershipFactor;
1079 }
1080 }
1081 }
1082 fxAptr[molA] += forceSumX;
1083 fyAptr[molA] += forceSumY;
1084 fzAptr[molA] += forceSumZ;
1085 txAptr[molA] += torqueSumX;
1086 tyAptr[molA] += torqueSumY;
1087 tzAptr[molA] += torqueSumZ;
1088 }
1089
1090 // reduce the forces on individual sites in SoA B to total forces & torques on whole molecules
1091 if constexpr (useMixing) {
1092 size_t siteIndex = 0;
1093 for (size_t mol = 0; mol < soaB.size(); ++mol) {
1094 const auto rotatedSitePositions = autopas::utils::quaternion::rotateVectorOfPositions(
1095 {q0Bptr[mol], q1Bptr[mol], q2Bptr[mol], q3Bptr[mol]}, _PPLibrary->getSitePositions(typeptrB[mol]));
1096 for (size_t site = 0; site < _PPLibrary->getNumSites(typeptrB[mol]); ++site) {
1097 fxBptr[mol] += siteForceBx[siteIndex];
1098 fyBptr[mol] += siteForceBy[siteIndex];
1099 fzBptr[mol] += siteForceBz[siteIndex];
1100 txBptr[mol] += rotatedSitePositions[site][1] * siteForceBz[siteIndex] -
1101 rotatedSitePositions[site][2] * siteForceBy[siteIndex];
1102 tyBptr[mol] += rotatedSitePositions[site][2] * siteForceBx[siteIndex] -
1103 rotatedSitePositions[site][0] * siteForceBz[siteIndex];
1104 tzBptr[mol] += rotatedSitePositions[site][0] * siteForceBy[siteIndex] -
1105 rotatedSitePositions[site][1] * siteForceBx[siteIndex];
1106 ++siteIndex;
1107 }
1108 }
1109 } else {
1110 size_t siteIndex = 0;
1111 for (size_t mol = 0; mol < soaB.size(); ++mol) {
1112 const auto rotatedSitePositions = autopas::utils::quaternion::rotateVectorOfPositions(
1113 {q0Bptr[mol], q1Bptr[mol], q2Bptr[mol], q3Bptr[mol]}, const_unrotatedSitePositions);
1114 for (size_t site = 0; site < const_unrotatedSitePositions.size(); ++site) {
1115 fxBptr[mol] += siteForceBx[siteIndex];
1116 fyBptr[mol] += siteForceBy[siteIndex];
1117 fzBptr[mol] += siteForceBz[siteIndex];
1118 txBptr[mol] += rotatedSitePositions[site][1] * siteForceBz[siteIndex] -
1119 rotatedSitePositions[site][2] * siteForceBy[siteIndex];
1120 tyBptr[mol] += rotatedSitePositions[site][2] * siteForceBx[siteIndex] -
1121 rotatedSitePositions[site][0] * siteForceBz[siteIndex];
1122 tzBptr[mol] += rotatedSitePositions[site][0] * siteForceBy[siteIndex] -
1123 rotatedSitePositions[site][1] * siteForceBx[siteIndex];
1124 ++siteIndex;
1125 }
1126 }
1127 }
1128 if constexpr (calculateGlobals) {
1129 const auto threadNum = autopas::autopas_get_thread_num();
1130 // SoAFunctorPairImpl obtains the potential energy * 12. For non-newton3, this sum is divided by 12 in
1131 // post-processing. For newton3, this sum is only divided by 6 in post-processing, so must be divided by 2 here.
1132 _aosThreadData[threadNum].potentialEnergySum += potentialEnergySum;
1133 _aosThreadData[threadNum].virialSum[0] += virialSumX;
1134 _aosThreadData[threadNum].virialSum[1] += virialSumY;
1135 _aosThreadData[threadNum].virialSum[2] += virialSumZ;
1136 }
1137 }
